223 cr_init(mt_cairo_render_t *mtcr, render_surf_t *rs)
224 {
225  cairo_format_t cr_fmt;
226 
227  ASSERT(mtcr != NULL);

238  cairo_set_operator(rs->cr, CAIRO_OPERATOR_OVER);
239  return (B_TRUE);
240 errout:
241  cairo_destroy(rs->cr);
242  rs->cr = NULL;
243  cairo_surface_destroy(rs->surf);
244  rs->surf = NULL;
245  return (B_FALSE);

266 /*
267  * Recalculates the absolute cv_timedwait sleep target based on our framerate.
268  */
269 static uint64_t
270 recalc_sleep_time(mt_cairo_render_t *mtcr)
271 {
272  double fps;
273  ASSERT(mtcr != NULL);
274  mutex_enter(&mtcr->lock);
275  fps = mtcr->fps;
276  mutex_exit(&mtcr->lock);
277  if (fps <= 0)
278  return (0);
279  return (microclock() + SEC2USEC(1.0 / fps));
280 }

282 static void
283 mtul_submit_mtcr(mt_cairo_uploader_t *mtul, mt_cairo_render_t *mtcr)
284 {
285  ASSERT(mtul != NULL);
286  ASSERT(mtcr != NULL);
287  ASSERT(mtcr->dirty);
288 
289  mutex_enter(&mtul->lock);
290  if (!list_link_active(&mtcr->mtul_queue_node)) {
291  list_insert_tail(&mtul->queue, mtcr);
292  cv_broadcast(&mtul->cv_queue);
293  }
294  while (mtcr->dirty)
295  cv_wait(&mtul->cv_done, &mtul->lock);
296  /* render_done_cv will be signalled by the uploader */
297  mutex_exit(&mtul->lock);
298 }

310 static void
311 worker_render_once(mt_cairo_render_t *mtcr)
312 {
313  render_surf_t *rs;
314  mt_cairo_uploader_t *mtul;
315 
316  ASSERT(mtcr != NULL);
317  ASSERT(mtcr->render_rs != -1);
318  ASSERT_MUTEX_HELD(&mtcr->lock);
319 
320  if (mtcr->coherent_data != NULL) {
321  size_t sz = mtcr_get_surf_sz(mtcr);
322 
323  rs = &mtcr->rs[0];
324  mutex_exit(&mtcr->lock);
325 
326  mtcr->render_cb(rs->cr, mtcr->w, mtcr->h, mtcr->userinfo);
327  cairo_surface_flush(rs->surf);
328 
329  mutex_enter(&mtcr->lock);
330  memcpy(mtcr->coherent_data,
331  cairo_image_surface_get_data(rs->surf), sz);
332  mtcr->dirty = B_TRUE;
333  mtcr->texed = B_FALSE;
334  mtcr->present_rs = mtcr->render_rs;
335  cv_broadcast(&mtcr->render_done_cv);
336  } else {
337  ASSERT3S(mtcr->render_rs, >=, 0);
338  ASSERT3S(mtcr->render_rs, <, ARRAY_NUM_ELEM(mtcr->rs));
339  rs = &mtcr->rs[mtcr->render_rs];
340  mutex_exit(&mtcr->lock);
341 
342  mtcr->render_cb(rs->cr, mtcr->w, mtcr->h, mtcr->userinfo);
343  cairo_surface_flush(rs->surf);
344 
345  mutex_enter(&mtcr->lock);
346  mtcr->dirty = B_TRUE;
347 
348  mtul = mtcr->mtul;
349  if (mtul != NULL) {
350  ASSERT(!coherent);
351  mutex_exit(&mtcr->lock);
352  /* render_done_cv will be signalled by the uploader */
353  mtul_submit_mtcr(mtul, mtcr);
354  mutex_enter(&mtcr->lock);
355  } else {
356  mtcr->present_rs = !mtcr->render_rs;
357  cv_broadcast(&mtcr->render_done_cv);
358  }
359  mtcr->render_rs = !mtcr->render_rs;
360  }
361 }

363 /*
364  * Main mt_cairo_render_t worker thread. Simply waits around for the
365  * required interval and fires off the rendering callback. This performs
366  * no canvas clearing between calls, so the callback is responsible for
367  * making sure its output canvas looks right.
368  */
369 static void
370 worker(void *arg)
371 {
372  mt_cairo_render_t *mtcr;
373  char name[32];
374  char shortname[7];
375  uint64_t next_time = 0;
376 
377  ASSERT(arg != NULL);
378  mtcr = arg;
379  ASSERT(mtcr->render_cb != NULL);
380 
381  strlcpy(shortname, mtcr->init_filename, sizeof (shortname));
382  snprintf(name, sizeof (name), "mtcr:%s:%d", shortname, mtcr->init_line);
383  thread_set_name(name);
384 
385  mutex_enter(&mtcr->lock);
386  mtcr->render_rs = 0;
387  /*
388  * Render the first frame immediately to make sure we have something
389  * to show ASAP.
390  */
391  if (mtcr->fps > 0)
392  worker_render_once(mtcr);
393  while (!mtcr->shutdown) {
394  if (!mtcr->one_shot_block) {
395  if (mtcr->fps > 0) {
396  if (next_time == 0) {
397  /*
398  * If we were in fps=0 mode before,
399  * this will be zero. So reset the
400  * timer ahead of time to avoid
401  * rendering two consecutive frames.
402  */
403  next_time = recalc_sleep_time(mtcr);
404  }
405  cv_timedwait(&mtcr->cv, &mtcr->lock, next_time);
406  /*
407  * Recalc the next frame time now to maintain
408  * near as possible constant framerate that
409  * isn't affected by the cairo render time.
410  */
411  next_time = recalc_sleep_time(mtcr);
412  } else {
413  cv_wait(&mtcr->cv, &mtcr->lock);
414  next_time = 0;
415  }
416  }
417  if (mtcr->shutdown)
418  break;
419  worker_render_once(mtcr);
420  }
421  mutex_exit(&mtcr->lock);
422 }

424 /*
425  * Initializes the mt_cairo_render state. You should call this before doing
426  * *ANY* Cairo or mt_cairo_render calls. Notably, the font machinery of
427  * Cairo isn't thread-safe before this call, so place a call to this function
428  * at the start of your bootstrap code.
429  */
430 void
431 mt_cairo_render_glob_init(bool_t want_coherent_mem)
432 {
433  if (glob_inited)
434  return;
435  cairo_surface_destroy(cairo_image_surface_create(CAIRO_FORMAT_ARGB32,
436  1, 1));
437  mtcr_main_thread = curthread_id;
438  fdr_find(&drs.viewport, "sim/graphics/view/viewport");
439  fdr_find(&drs.proj_matrix, "sim/graphics/view/projection_matrix");
440  fdr_find(&drs.mv_matrix, "sim/graphics/view/modelview_matrix");
441  fdr_find(&drs.draw_call_type, "sim/graphics/view/draw_call_type");
442  /*
443  * Important caveat: in Zink mode we MUST utilize coherent memory
444  * (or just avoid using an uploader). The uploader tries to map
445  * a buffer using glMapBuffer while holding the mtcr->lock. This
446  * can cause a deadlock, because Zink's internal architecture seems
447  * to sometimes require main thread progress before a background
448  * thread's buffer mapping request can succeed.
449  */
450  coherent = ((want_coherent_mem || glutils_in_zink_mode()) &&
451  GLEW_ARB_buffer_storage);
452  glob_inited = B_TRUE;
453 }

455 static void
456 mtcr_gl_init(mt_cairo_render_t *mtcr)
457 {
458  GLint old_vao = 0;
459  bool_t on_main_thread = (curthread_id == mtcr_main_thread);
460  GLint intfmt, gl_fmt;
461 
462  ASSERT(mtcr != NULL);
463 
464  if (GLEW_VERSION_3_0 && !on_main_thread) {
465  glGetIntegerv(GL_VERTEX_ARRAY_BINDING, &old_vao);
466 
467  glGenVertexArrays(1, &mtcr->vao);
468  glBindVertexArray(mtcr->vao);
469  }
470 
471  glGenBuffers(1, &mtcr->vtx_buf);
472 
473  if (GLEW_VERSION_3_0 && !on_main_thread) {
474  glBindBuffer(GL_ARRAY_BUFFER, mtcr->vtx_buf);
475  glutils_enable_vtx_attr_ptr(VTX_ATTRIB_POS, 3, GL_FLOAT,
476  GL_FALSE, sizeof (vtx_t), offsetof(vtx_t, pos));
477  glutils_enable_vtx_attr_ptr(VTX_ATTRIB_TEX0, 2, GL_FLOAT,
478  GL_FALSE, sizeof (vtx_t), offsetof(vtx_t, tex0));
479  }
480 
481  mtcr->idx_buf = glutils_make_quads_IBO(4);
482  if (GLEW_VERSION_3_0 && !on_main_thread)
483  glBindVertexArray(old_vao);
484  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
485 
486  ASSERT0(mtcr->pbo);
487  glGenBuffers(1, &mtcr->pbo);
488  ASSERT(mtcr->pbo != 0);
489  mtcr_gl_formats(mtcr, &intfmt, &gl_fmt);
490  IF_TEXSZ(TEXSZ_ALLOC_INSTANCE(mt_cairo_render_pbo,
491  mtcr, mtcr->init_filename, mtcr->init_line, gl_fmt,
492  GL_UNSIGNED_BYTE, mtcr->w, mtcr->h));
493 
494  if (coherent) {
495  const GLuint flags = (GL_MAP_WRITE_BIT |
496  GL_MAP_PERSISTENT_BIT | GL_MAP_COHERENT_BIT);
497  size_t sz = mtcr_get_surf_sz(mtcr);;
498 
499  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, mtcr->pbo);
500  glBufferStorage(GL_PIXEL_UNPACK_BUFFER, sz, 0, flags);
501  mtcr->coherent_data = glMapBufferRange(GL_PIXEL_UNPACK_BUFFER,
502  0, sz, flags);
503  if (mtcr->coherent_data == NULL) {
504  logMsg("WARNING: cannot grab coherent memory buffer "
505  "with glMapBufferRange(). You may be running out "
506  "of VRAM. Switching to non-coherent mode for this "
507  "renderer (%s:%d).", mtcr->init_filename,
508  mtcr->init_line);
509  }
510  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
511  }
512  glGenTextures(1, &mtcr->tex);
513  ASSERT(mtcr->tex != 0);
514  glBindTexture(GL_TEXTURE_2D, mtcr->tex);
515  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, mtcr->filter);
516  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, mtcr->filter);
517  if (on_main_thread)
518  XPLMBindTexture2d(0, 0);
519 
520  IF_TEXSZ(TEXSZ_ALLOC_INSTANCE(mt_cairo_render_tex, mtcr,
521  mtcr->init_filename, mtcr->init_line, gl_fmt,
522  GL_UNSIGNED_BYTE, mtcr->w, mtcr->h));
523  mtcr->texed = B_FALSE;
524 }

526 /*
527  * Frees a texture & PBO previously allocated using rs_tex_alloc. If the
528  * texture or PBO have already been freed, this function does nothing.
529  * The targets of the `tex' and `pbo' arguments are set to 0 after freeing.
530  */
531 static void
532 mtcr_gl_fini(mt_cairo_render_t *mtcr)
533 {
534  GLint intfmt, format;
535 
536  ASSERT(mtcr != NULL);
537 
538  if (mtcr->vao != 0) {
539  glDeleteVertexArrays(1, &mtcr->vao);
540  mtcr->vao = 0;
541  }
542  if (mtcr->vtx_buf != 0) {
543  glDeleteBuffers(1, &mtcr->vtx_buf);
544  mtcr->vtx_buf = 0;
545  mtcr->last_draw.pos = NULL_VECT2;
546  }
547  if (mtcr->idx_buf != 0) {
548  glDeleteBuffers(1, &mtcr->idx_buf);
549  mtcr->idx_buf = 0;
550  }
551  mtcr_gl_formats(mtcr, &intfmt, &format);
552  if (mtcr->tex != 0) {
553  glDeleteTextures(1, &mtcr->tex);
554  mtcr->tex = 0;
555  IF_TEXSZ(TEXSZ_FREE_INSTANCE(mt_cairo_render_tex, mtcr,
556  format, GL_UNSIGNED_BYTE, mtcr->w, mtcr->h));
557  }
558  if (mtcr->pbo != 0) {
559  if (mtcr->coherent_data) {
560  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, mtcr->pbo);
561  glUnmapBuffer(GL_PIXEL_UNPACK_BUFFER);
562  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
563  mtcr->coherent_data = NULL;
564  }
565  glDeleteBuffers(1, &mtcr->pbo);
566  mtcr->pbo = 0;
567  IF_TEXSZ(TEXSZ_FREE_INSTANCE(mt_cairo_render_pbo, mtcr,
568  format, GL_UNSIGNED_BYTE, mtcr->w, mtcr->h));
569  }
570  if (mtcr->sync != NULL) {
571  glDeleteSync(mtcr->sync);
572  mtcr->sync = NULL;
573  }
574  if (mtcr->shader != 0 && !mtcr->shader_is_custom) {
575  glDeleteProgram(mtcr->shader);
576  mtcr->shader = 0;
577  }
578 }

598 /*
599  * Creates a new mt_cairo_render_t surface.
600  * @param w Width of the rendered surface (in pixels).
601  * @param h Height of the rendered surface (in pixels).
602  * @param fps Framerate at which the surface should be rendered.
603  * This can be changed at any time later. Pass a zero fps value
604  * to make the renderer only run on-request (see mt_cairo_render_once).
605  * @param init_cb An optional initialization callback that can be
606  * used to initialize private resources needed during rendering.
607  * Called once for every cairo_t instance (two instances for
608  * every mt_cairo_render_t, due to double-buffering). Please note that
609  * this can be called even after calling mt_cairo_render_init, since
610  * mt_cairo_render_set_monochrome will re-allocate the cairo instances.
611  * @param render_cb A mandatory rendering callback. This is called for
612  * each rendered frame.

814 /*
815  * Enables or disables monochrome rendering mode. The default is disabled
816  * (i.e. the image is rendered in RGBA mode). Monochrome rendering in
817  * Cairo is controlled using the alpha channel. If you are using the
818  * default mt_cairo_render compositing shader, the `color' vector
819  * argument to this function sets what RGB color is used in the final
820  * render. Passing any non-NULL vector here enables monochrome mode. Pass
821  * a NULL_VECT3 to disable monochrome mode.
822  * Switching rendering modes stops & restarts the worker thread (when not
823  * in FG mode) and calls any surface fini and init callbacks you passed
824  * mt_cairo_render_init.
825  */
826 void
827 mt_cairo_render_set_monochrome(mt_cairo_render_t *mtcr, vect3_t color)
828 {
829  ASSERT(mtcr != NULL);
830  /*
831  * If the monochrome status hasn't changed, there's no need to
832  * rebuild the surface. Just store the potentially new color.
833  */
834  if (IS_NULL_VECT(mtcr->monochrome) == IS_NULL_VECT(color)) {
835  mtcr->monochrome = color;
836  return;
837  }
838  /*
839  * Stop the worker thread.
840  */
841  if (mtcr->started) {
842  mutex_enter(&mtcr->lock);
843  mtcr->shutdown = B_TRUE;
844  cv_broadcast(&mtcr->cv);
845  mutex_exit(&mtcr->lock);
846  thread_join(&mtcr->thr);
847  }
848  for (size_t i = 0; i < ARRAY_NUM_ELEM(mtcr->rs); i++)
849  cr_destroy(mtcr, &mtcr->rs[i]);
850  mtcr_gl_fini(mtcr);
851 
852  mtcr->monochrome = color;
853  mtcr->render_rs = -1;
854  mtcr->present_rs = -1;
855 
856  mtcr_gl_init(mtcr);
857  VERIFY(cr_init(mtcr, &mtcr->rs[0]));
858  if (mtcr->coherent_data == NULL)
859  VERIFY(cr_init(mtcr, &mtcr->rs[1]));
860  /*
861  * If we were set up to use our own shader, reload it to switch
862  * to the monochrome version.
863  */
864  if (!mtcr->shader_is_custom)
865  set_shader_impl(mtcr, 0, B_TRUE);
866  /*
867  * Restart the worker if not in FG mode.
868  */
869  if (!mtcr->fg_mode) {
870  mtcr->shutdown = B_FALSE;
871  VERIFY(thread_create(&mtcr->thr, worker, mtcr));
872  mtcr->started = B_TRUE;
873  }
874 }

943 /*
944  * Uploads a finished cairo surface render to the provided texture & PBO.
945  * The upload is normally done async via the PBO, but if that fails, the
946  * upload is performed synchronously.
947  */
948 static void
949 rs_upload(mt_cairo_render_t *mtcr, render_surf_t *rs)
950 {
951  void *src, *dest;
952  size_t sz;
953 
954  if (mtcr->coherent_data != NULL)
955  return;
956 
957  ASSERT(mtcr != NULL);
958  ASSERT(rs != NULL);
959  ASSERT(rs->surf != NULL);
960  ASSERT(mtcr->tex != 0);
961  ASSERT(mtcr->pbo != 0);
962 
963  sz = mtcr_get_surf_sz(mtcr);
964  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, mtcr->pbo);
965  /* Buffer respecification - makes sure to orphan the old buffer! */
966  glBufferData(GL_PIXEL_UNPACK_BUFFER, sz, NULL, GL_STREAM_DRAW);
967  src = cairo_image_surface_get_data(rs->surf);
968  dest = glMapBuffer(GL_PIXEL_UNPACK_BUFFER, GL_WRITE_ONLY);
969  if (dest != NULL) {
970  memcpy(dest, src, sz);
971  glUnmapBuffer(GL_PIXEL_UNPACK_BUFFER);
972  /*
973  * We MUSTN'T call glTexImage2D yet, because if we're running
974  * on a background uploader thread, the OpenGL renderer can
975  * break and update the texture with old memory contents.
976  * Don't ask me why, I have NO CLUE why the driver messes
977  * this up. So we need to do the glTexImage2D synchronously
978  * on the rendering thread. That seems to pick up the buffer
979  * orphaning operation correctly.
980  */
981  mtcr->texed = B_FALSE;
982  } else {
983  GLint intfmt, format;
984 
985  mtcr_gl_formats(mtcr, &intfmt, &format);
986  logMsg("Error asynchronously updating mt_cairo_render "
987  "surface %p(%s:%d): glMapBuffer returned NULL",
988  mtcr, mtcr->init_filename, mtcr->init_line);
989  glBindTexture(GL_TEXTURE_2D, mtcr->tex);
990  glTexImage2D(GL_TEXTURE_2D, 0, intfmt, mtcr->w, mtcr->h, 0,
991  format, GL_UNSIGNED_BYTE, src);
992  mtcr->texed = B_TRUE;
993  glBindTexture(GL_TEXTURE_2D, 0);
994  }
995  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
996 }

998 /*
999  * After an MT-uploader async-uploads the new surface data, we still
1000  * need to apply it to the texture itself. Otherwise, it will only
1001  * sit in the orphaned buffer. This must be done from the thread
1002  * which plans to use the texture in actual rendering (otherwise the
1003  * drivers spaz out).
1004  * Careful, any texture binding point used previously is unbound by
1005  * this function. This is to facilitate interop with
1006  * mt_cairo_render_get_tex to avoid leaving bound textures lying
1007  * around.
1008  */
1009 static void
1010 mtcr_tex_apply(mt_cairo_render_t *mtcr, bool_t bind)
1011 {
1012  ASSERT(mtcr != NULL);
1013 
1014  if (!mtcr->texed) {
1015  GLint intfmt, format;
1016 
1017  mtcr_gl_formats(mtcr, &intfmt, &format);
1018  ASSERT(mtcr->tex != 0);
1019  glBindTexture(GL_TEXTURE_2D, mtcr->tex);
1020  ASSERT(mtcr->pbo != 0);
1021  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, mtcr->pbo);
1022  glTexImage2D(GL_TEXTURE_2D, 0, intfmt, mtcr->w, mtcr->h, 0,
1023  format, GL_UNSIGNED_BYTE, NULL);
1024  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
1025  mtcr->texed = B_TRUE;
1026  if (!bind)
1027  glBindTexture(GL_TEXTURE_2D, 0);
1028  } else if (bind) {
1029  glBindTexture(GL_TEXTURE_2D, mtcr->tex);
1030  }
1031 }

1033 /*
1034  * Binds the current render_surf_t's texture to the current OpenGL context.
1035  * This is called from the foreground renderer to start drawing a finished
1036  * render frame.
1037  *
1038  * @return The render_surf_t that was bound, or NULL if none is available
1039  * for display.
1040  */
1041 static bool_t
1042 bind_cur_tex(mt_cairo_render_t *mtcr)
1043 {
1044  ASSERT(mtcr != NULL);
1045  ASSERT_MUTEX_HELD(&mtcr->lock);

1170  * the cairo render (though not necessarily in lockstep with it). If
1171  * a new frame hasn't been rendered yet, this function simply renders
1172  * the old buffer again (or nothing at all if no surface has completed
1173  * internal Cairo rendering). You can guarantee that a surface is ready
1174  * by first calling mt_cairo_render_once_wait.
1175  */
1176 void
1177 mt_cairo_render_draw_subrect_pvm(mt_cairo_render_t *mtcr,
1178  vect2_t src_pos, vect2_t src_sz, vect2_t pos, vect2_t size,
1179  const GLfloat *pvm)
1180 {

1182  bool_t use_vao;
1183  double x1 = src_pos.x, x2 = src_pos.x + src_sz.x;
1184  double y1 = src_pos.y, y2 = src_pos.y + src_sz.y;
1185  bool cull_front = false;
1186  vtx_t vtx_buf[4] = {};
1187 
1188  mutex_enter(&mtcr->lock);
1189  if (!bind_cur_tex(mtcr)) {

1561 /*
1562  * Initializes an asynchronous render surface uploader. This should be
1563  * called from the main X-Plane thread.
1564  *
1565  * BACKGROUND:
1566  *
1567  * An mt_cairo_render_t runs its Cairo rendering operations in a background
1568  * thread. However, to present the final rendered image on screen, the
1569  * resulting image must first be uploaded to the GPU. This requires having
1570  * an OpenGL context bound to the current thread, however, OpenGL contexts
1571  * cannot be shared between threads. The workaround would be to create a
1572  * new context in every mt_cairo_render_t worker thread, however, given that
1573  * there can be dozens of render workers, this would generate dozens of
1574  * OpenGL contexts, which isn't very efficient either. In lieu of a better
1575  * mechanism, mt_cairo_render would simply perform the final image upload
1576  * to the GPU during the mt_cairo_render_draw operation. Which this was
1577  * utilizing pixel-buffer-objects to streamline the uploading process and
1578  * avoid OpenGL pipeline stalls, there was still some memcpy'ing involved,
1579  * which had a non-trivial cost when invoked from the main X-Plane thread.
1580  *
1581  * mt_cairo_uploader_t solves all of these issues. It is a separate
1582  * background worker thread with its own OpenGL context dedicated to doing
1583  * nothing but uploading of finished renders to the GPU. Once created, the
1584  * uploader goes into a wait-sleep, awaiting new work assignments from
1585  * renderers. Once a render is finished, the uploader is notified,
1586  * immediately wakes up and uploads the finished render to the GPU. This
1587  * happens without waiting for the main X-Plane thread to come around
1588  * wanting to draw a particular mt_cairo_render_t. With this infrastructure
1589  * in place, calls to mt_cairo_render_draw no longer block for texture
1590  * uploading.
1591  *
1592  * This machinery isn't automatically enabled on all instances of
1593  * mt_cairo_render_t however. To utilize this mechanism, you should create
1594  * an mt_cairo_uploader and associate it with all your mt_cairo_render
1595  * instances using mt_cairo_render_set_uploader. You generally only ever
1596  * need a single uploader for all renderers you create. Thus, following
