221 static bool_t
222 odb_proc_us_dof_impl(const char *buf, size_t len, add_obst_cb_t cb,
223  void *userinfo)
224 {
225  ASSERT(buf != NULL);
226  ASSERT(cb != NULL);
227 
228  for (const char *line_start = buf, *line_end = buf;
229  line_start < buf + len; line_start = line_end + 1) {
230  char **comps;
231  size_t n_comps;
232  obst_type_t type;
233  obst_light_t light;
234  geo_pos3_t pos;
235  float agl, amsl;
236  unsigned quant;
237  char line[256];
238 
239  line_end = strchr(line_start, '\n');
240  if (line_end == NULL)
241  line_end = buf + len;
242  lacf_strlcpy(line, line_start, MIN((int)sizeof (line),
243  (line_end - line_start) + 1));
244 
245  strip_space(line);
246 
247  comps = strsplit(line, ",", B_FALSE, &n_comps);
248 
249  if (n_comps < 19 || strcmp(comps[0], "OAS") == 0)
250  goto next;
251 
252  quant = atoi(comps[10]);
253  agl = FEET2MET(atof(comps[11]));
254  amsl = FEET2MET(atof(comps[12]));
255  type = dof2type(comps[9]);
256  light = dof2light(comps[13]);
257  pos.lat = atof(comps[5]);
258  pos.lon = atof(comps[6]);
259  pos.elev = amsl - agl;
260 
261  if (!is_valid_lat(pos.lat) || !is_valid_lon(pos.lon) ||
262  agl < 0 || !is_valid_alt_m(agl) || !is_valid_alt_m(amsl) ||
263  quant == 0)
264  goto next;
265 
266  cb(type, pos, agl, light, quant, userinfo);
267 next:
268  free_strlist(comps, n_comps);
269  }
270 
271  return (B_TRUE);
272 }

402 static size_t
403 dl_write(char *ptr, size_t size, size_t nmemb, void *userdata)
404 {
405  dl_info_t *dl_info;
406  size_t bytes = size * nmemb;
407 
408  ASSERT(userdata != NULL);
409  dl_info = userdata;
410 
411  /* Respond to an early termination request */
412  if (!dl_info->odb->refresh_run)
413  return (0);
414 
415  if (dl_info->bufcap < dl_info->bufsz + bytes) {
416  do {
417  dl_info->bufcap += REALLOC_STEP;
418  } while (dl_info->bufcap < dl_info->bufsz + bytes);
419  dl_info->buf = realloc(dl_info->buf, dl_info->bufcap);
420  }
421  memcpy(&dl_info->buf[dl_info->bufsz], ptr, bytes);
422  dl_info->bufsz += bytes;
423 
424  return (bytes);
425 }

427 static bool_t
428 write_tile(odb_t *odb, odb_tile_t *tile, const char *cc)
429 {
430  char subpath[32];
431  char *path, *dirpath, *p;
432  FILE *fp = NULL;
433  bool_t res = B_FALSE;
434 
435  ASSERT(odb != NULL);
436  ASSERT(tile != NULL);
437  ASSERT(cc != NULL);
438 
439  latlon2path(tile->lat, tile->lon, subpath);
440  path = mkpathname(odb->cache_dir, cc, subpath, NULL);
441  dirpath = strdup(path);
442  VERIFY(dirpath != NULL);
443 
444  /* strip the last path component to create the directory */
445  p = strrchr(dirpath, DIRSEP);
446  ASSERT(p != NULL);
447  *p = 0;
448 
449  if (!create_directory_recursive(dirpath))
450  goto errout;
451  fp = fopen(path, "wb");
452  if (fp == NULL) {
453  logMsg("Error writing obstacle database tile %s: %s",
454  path, strerror (errno));
455  goto errout;
456  }
457  for (obst_t *obst = list_head(&tile->obst); obst != NULL;
458  obst = list_next(&tile->obst, obst)) {
459  fprintf(fp, ",,US,,,%f,%f,,,%s,%d,%.0f,%.0f,%c,1A,,,,\n",
460  obst->pos.lat, obst->pos.lon, type2dof(obst->type),
461  obst->quant, MET2FEET(obst->agl),
462  MET2FEET(obst->pos.elev + obst->agl),
463  light2dof(obst->light));
464  }
465 
466  res = B_TRUE;
467 errout:
468  if (fp != NULL)
469  fclose(fp);
470  lacf_free(path);
471  free(dirpath);
472 
473  return (res);
474 }

527 static void
528 odb_refresh_us(odb_t *odb)
529 {
530  CURL *curl;
531  CURLcode res;
532  dl_info_t dl_info = { .odb = odb };
533 
534  thread_set_name("odb-refresh-us");
535 
536  ASSERT(odb != NULL);
537 
538  curl = curl_easy_init();
539  VERIFY(curl != NULL);
540 
541  logMsg("Downloading new obstacle data from \"%s\" for region \"US\"",
542  FAA_DOF_URL);
543  curl_easy_setopt(curl, CURLOPT_URL, FAA_DOF_URL);
544  chart_setup_curl(curl, odb->cainfo);
545  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, dl_write);
546  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &dl_info);
547  curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME, LOW_SPD_TIME);
548  curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT, LOW_SPD_LIM);
549  curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
550  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
551  curl_easy_setopt(curl, CURLOPT_TIMEOUT, DL_TIMEOUT);
552  mutex_enter(&odb->proxy_lock);
553  if (odb->proxy != NULL)
554  curl_easy_setopt(curl, CURLOPT_PROXY, odb->proxy);
555  mutex_exit(&odb->proxy_lock);
556 
557  res = curl_easy_perform(curl);
558 
559  if (res == CURLE_OK && dl_info.bufsz != 0) {
560  size_t len;
561  void *buf = decompress_zip(dl_info.buf, dl_info.bufsz, &len);
562 
563  if (buf != NULL) {
564  char *subpath = mkpathname(odb->cache_dir, "US", NULL);
565 
566  /*
567  * The downloaded DOF is HUUUGE, so DON'T lock here.
568  * add_obst_to_odb will lock the database when it's
569  * ready to add a single obstacle instead.
570  */
571  odb_proc_us_dof_impl(buf, len, add_obst_to_odb, odb);
572 
573  mutex_enter(&odb->tiles_lock);
574 
575  if (file_exists(subpath, NULL))
576  remove_directory(subpath);
577  create_directory_recursive(odb->cache_dir);
578  odb_write_tiles(odb, "US");
579  odb_flush_tiles(odb);
580  write_odb_refresh_date(odb, "US");
581 
582  lacf_free(subpath);
583  odb->refresh_times[ODB_REGION_US] = time(NULL);
584 
585  mutex_exit(&odb->tiles_lock);
586  } else {
587  logMsg("Error updating obstacle database from %s: "
588  "failed to decompress downloaded ZIP file",
589  FAA_DOF_URL);
590  odb->refresh_times[ODB_REGION_US] = -1u;
591  }
592  free(buf);
593  } else {
594  logMsg("Error updating obstacle database from %s: %s",
595  FAA_DOF_URL, curl_easy_strerror(res));
596  odb->refresh_times[ODB_REGION_US] = -1u;
597  }
598 
599  curl_easy_cleanup(curl);
600  free(dl_info.buf);
601 
602  mutex_enter(&odb->refresh_lock);
603  odb->refresh_run = B_FALSE;
604  mutex_exit(&odb->refresh_lock);
605 }
