103 static int
104 tile_compar(const void *a, const void *b)
105 {
106  const odb_tile_t *ta = a, *tb = b;
107 
108  if (ta->lat < tb->lat)
109  return (-1);
110  if (ta->lat > tb->lat)
111  return (1);
112  if (ta->lon < tb->lon)
113  return (-1);
114  if (ta->lon > tb->lon)
115  return (1);
116 
117  return (0);
118 }

222 static bool_t
223 odb_proc_us_dof_impl(const char *buf, size_t len, add_obst_cb_t cb,
224  void *userinfo)
225 {
226  ASSERT(buf != NULL);
227  ASSERT(cb != NULL);
228 
229  for (const char *line_start = buf, *line_end = buf;
230  line_start < buf + len; line_start = line_end + 1) {
231  char **comps;
232  size_t n_comps;
233  obst_type_t type;
234  obst_light_t light;
235  geo_pos3_t pos;
236  float agl, amsl;
237  unsigned quant;
238  char line[256];
239 
240  line_end = strchr(line_start, '\n');
241  if (line_end == NULL)
242  line_end = buf + len;
243  lacf_strlcpy(line, line_start, MIN((int)sizeof (line),
244  (line_end - line_start) + 1));
245 
246  strip_space(line);
247 
248  comps = strsplit(line, ",", B_FALSE, &n_comps);
249 
250  if (n_comps < 19 || strcmp(comps[0], "OAS") == 0)
251  goto next;
252 
253  quant = atoi(comps[10]);
254  agl = FEET2MET(atof(comps[11]));
255  amsl = FEET2MET(atof(comps[12]));
256  type = dof2type(comps[9]);
257  light = dof2light(comps[13]);
258  pos.lat = atof(comps[5]);
259  pos.lon = atof(comps[6]);
260  pos.elev = amsl - agl;
261 
262  if (!is_valid_lat(pos.lat) || !is_valid_lon(pos.lon) ||
263  agl < 0 || !is_valid_alt_m(agl) || !is_valid_alt_m(amsl) ||
264  quant == 0)
265  goto next;
266 
267  cb(type, pos, agl, light, quant, userinfo);
268 next:
269  free_strlist(comps, n_comps);
270  }
271 
272  return (B_TRUE);
273 }

403 static size_t
404 dl_write(char *ptr, size_t size, size_t nmemb, void *userdata)
405 {
406  dl_info_t *dl_info;
407  size_t bytes = size * nmemb;
408 
409  ASSERT(userdata != NULL);
410  dl_info = userdata;
411 
412  /* Respond to an early termination request */
413  if (!dl_info->odb->refresh_run)
414  return (0);
415 
416  if (dl_info->bufcap < dl_info->bufsz + bytes) {
417  do {
418  dl_info->bufcap += REALLOC_STEP;
419  } while (dl_info->bufcap < dl_info->bufsz + bytes);
420  dl_info->buf = realloc(dl_info->buf, dl_info->bufcap);
421  }
422  memcpy(&dl_info->buf[dl_info->bufsz], ptr, bytes);
423  dl_info->bufsz += bytes;
424 
425  return (bytes);
426 }

428 static bool_t
429 write_tile(odb_t *odb, odb_tile_t *tile, const char *cc)
430 {
431  char subpath[32];
432  char *path, *dirpath, *p;
433  FILE *fp = NULL;
434  bool_t res = B_FALSE;
435 
436  ASSERT(odb != NULL);
437  ASSERT(tile != NULL);
438  ASSERT(cc != NULL);
439 
440  latlon2path(tile->lat, tile->lon, subpath);
441  path = mkpathname(odb->cache_dir, cc, subpath, NULL);
442  dirpath = strdup(path);
443  VERIFY(dirpath != NULL);
444 
445  /* strip the last path component to create the directory */
446  p = strrchr(dirpath, DIRSEP);
447  ASSERT(p != NULL);
448  *p = 0;
449 
450  if (!create_directory_recursive(dirpath))
451  goto errout;
452  fp = fopen(path, "wb");
453  if (fp == NULL) {
454  logMsg("Error writing obstacle database tile %s: %s",
455  path, strerror (errno));
456  goto errout;
457  }
458  for (obst_t *obst = list_head(&tile->obst); obst != NULL;
459  obst = list_next(&tile->obst, obst)) {
460  fprintf(fp, ",,US,,,%f,%f,,,%s,%d,%.0f,%.0f,%c,1A,,,,\n",
461  obst->pos.lat, obst->pos.lon, type2dof(obst->type),
462  obst->quant, MET2FEET(obst->agl),
463  MET2FEET(obst->pos.elev + obst->agl),
464  light2dof(obst->light));
465  }
466 
467  res = B_TRUE;
468 errout:
469  if (fp != NULL)
470  fclose(fp);
471  lacf_free(path);
472  free(dirpath);
473 
474  return (res);
475 }

528 static void
529 odb_refresh_us_decompress(odb_t *odb, dl_info_t *dl_info)
530 {
531  size_t len;
532  void *buf;
533 
534  ASSERT(odb != NULL);
535  ASSERT(dl_info != NULL);
536 
537  if (dl_info->bufsz == 0) {
538  logMsg("Error updating obstacle database from %s: "
539  "server didn't send any data", FAA_DOF_URL);
540  return;
541  }
542  buf = decompress_zip(dl_info->buf, dl_info->bufsz, &len);
543  if (buf != NULL) {
544  char *subpath = mkpathname(odb->cache_dir, "US", NULL);
545  /*
546  * The downloaded DOF is HUUUGE, so DON'T lock here.
547  * add_obst_to_odb will lock the database when it's
548  * ready to add a single obstacle instead.
549  */
550  odb_proc_us_dof_impl(buf, len, add_obst_to_odb, odb);
551 
552  mutex_enter(&odb->tiles_lock);
553 
554  if (file_exists(subpath, NULL))
555  remove_directory(subpath);
556  create_directory_recursive(odb->cache_dir);
557  odb_write_tiles(odb, "US");
558  odb_flush_tiles(odb);
559  write_odb_refresh_date(odb, "US");
560 
561  lacf_free(subpath);
562  odb->refresh_times[ODB_REGION_US] = time(NULL);
563 
564  mutex_exit(&odb->tiles_lock);
565 
566  logMsg("Obstacle database update successful");
567  } else {
568  logMsg("Error updating obstacle database from %s: "
569  "failed to decompress downloaded ZIP file", FAA_DOF_URL);
570  odb->refresh_times[ODB_REGION_US] = -1u;
571  }
572  free(buf);
573 }

575 static void
576 odb_refresh_us(odb_t *odb)
577 {
578  CURL *curl;
579  CURLcode res;
580  dl_info_t dl_info = { .odb = odb };
581  time_t refresh_date;
582 
583  thread_set_name("odb-refresh-us");
584 
585  ASSERT(odb != NULL);
586 
587  curl = curl_easy_init();
588  VERIFY(curl != NULL);
589 
590  logMsg("Downloading new obstacle data from \"%s\" for region \"US\"",
591  FAA_DOF_URL);
592  curl_easy_setopt(curl, CURLOPT_URL, FAA_DOF_URL);
593  chart_setup_curl(curl, odb->cainfo);
594  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, dl_write);
595  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &dl_info);
596  curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME, LOW_SPD_TIME);
597  curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT, LOW_SPD_LIM);
598  curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);
599  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
600  curl_easy_setopt(curl, CURLOPT_TIMEOUT, DL_TIMEOUT);
601  mutex_enter(&odb->proxy_lock);
602  if (odb->proxy != NULL)
603  curl_easy_setopt(curl, CURLOPT_PROXY, odb->proxy);
604  mutex_exit(&odb->proxy_lock);
605 
606  refresh_date = odb_get_cc_refresh_date(odb, "US");
607  if (refresh_date != 0) {
608  /* If-Modified-Since the above time stamp */
609  curl_easy_setopt(curl, CURLOPT_TIMEVALUE, (long)refresh_date);
610  curl_easy_setopt(curl, CURLOPT_TIMECONDITION,
611  (long)CURL_TIMECOND_IFMODSINCE);
612  }
613  res = curl_easy_perform(curl);
614 
615  if (res == CURLE_OK) {
616  long code = 0;
617 
618  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
619  if (code == 200) {
620  odb_refresh_us_decompress(odb, &dl_info);
621  } else if (code == 304) {
622  logMsg("Obstacle database %s unchanged", FAA_DOF_URL);
623  mutex_enter(&odb->tiles_lock);
624  odb->refresh_times[ODB_REGION_US] = time(NULL);
625  mutex_exit(&odb->tiles_lock);
626  } else {
627  logMsg("Error updating obstacle database from %s: "
628  "server responded with code %ld", FAA_DOF_URL,
629  code);
630  }
631  } else {
632  logMsg("Error updating obstacle database from %s: %s",
633  FAA_DOF_URL, curl_easy_strerror(res));
634  odb->refresh_times[ODB_REGION_US] = -1u;
635  }
636 
637  curl_easy_cleanup(curl);
638  free(dl_info.buf);
639 
640  mutex_enter(&odb->refresh_lock);
641  odb->refresh_run = B_FALSE;
642  mutex_exit(&odb->refresh_lock);
643 }

645 bool_t

647 {
648  void (*refresh_op)(odb_t *odb) = NULL;
649 
650  ASSERT(odb != NULL);
651  ASSERT(cc != NULL);
