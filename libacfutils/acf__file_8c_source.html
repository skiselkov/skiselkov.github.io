57 /*
58  * Reads an X-Plane .acf file and returns a structure which can be used
59  * to access its properties.
60  *
61  * @param filename A full path to the .acf file to read.
62  *
63  * @return A constructed .acf file in an accessible structure, or NULL
64  * on error. The exact error is emitted via logMsg.
65  */
66 acf_file_t *
67 acf_file_read(const char *filename)
68 {
69  acf_file_t *acf;
70  FILE *fp = fopen(filename, "rb");
71  char *line = NULL;
72  size_t cap = 0;
73  bool_t parsing_props = B_FALSE;
74 
75  if (fp == NULL) {
76  logMsg("Error reading acf file %s: %s", filename,
77  strerror(errno));
78  return (NULL);
79  }
80  acf = safe_calloc(1, sizeof (*acf));
81  avl_create(&acf->props, acf_prop_compar, sizeof (acf_prop_t),
82  offsetof(acf_prop_t, node));
83 
84  for (int line_num = 1; getline(&line, &cap, fp) > 0; line_num++) {
85  char *name_end;
86  acf_prop_t *prop;
87  avl_index_t where;
88 
89  strip_space(line);
90  if (line_num <= 3) {
91  size_t n_comps;
92  char **comps = strsplit(line, " ", B_TRUE, &n_comps);
93 
94  if (n_comps >= 2 && strcmp(comps[1], "Version") == 0) {
95  acf->version = atoi(comps[0]);
96  } else if (line_num == 3 &&
97  strcmp(comps[0], "ACF") != 0) {
98  logMsg("Error reading acf file %s: missing "
99  "file header. Are you sure this is an "
100  "ACF file?", filename);
101  free_strlist(comps, n_comps);
102  goto errout;
103  }
104  free_strlist(comps, n_comps);
105  continue;
106  }
107  if (!parsing_props) {
108  if (strncmp(line, "PROPERTIES_BEGIN", 16) == 0)
109  parsing_props = B_TRUE;
110  continue;
111  }
112  if (strncmp(line, "PROPERTIES_END", 14) == 0)
113  break;
114  if (strncmp(line, "P ", 2) != 0)
115  continue;
116 
117  name_end = strchr(&line[2], ' ');
118  if (name_end == NULL) {
119  logMsg("Error reading acf file %s:%d: bad parameter "
120  "line.", filename, line_num);
121  goto errout;
122  }
123  prop = safe_calloc(1, sizeof (*prop));
124  prop->name = safe_malloc(name_end - line - 1);
125  prop->value = safe_malloc(strlen(&name_end[1]) + 1);
126  lacf_strlcpy(prop->name, &line[2], name_end - line - 1);
127  lacf_strlcpy(prop->value, &name_end[1], strlen(&name_end[1]) +
128  1);
129  if (avl_find(&acf->props, prop, &where) != NULL) {
130  logMsg("Error reading acf file %s:%d duplicate "
131  "property \"%s\" found.", filename, line_num,
132  prop->name);
133  free(prop->name);
134  free(prop->value);
135  free(prop);
136  goto errout;
137  }
138  avl_insert(&acf->props, prop, where);
139  }
140 
141  fclose(fp);
142  return (acf);
143 errout:
144  acf_file_free(acf);
145  fclose(fp);
146  return (NULL);
147 }

182  acf_prop_t *prop = avl_find(&acf->props, &srch, NULL);
