47const char *
48font_utils_ft_err2str(FT_Error err)
49{
50 for (int i = 0; ft_errors[i].err_msg != NULL; i++)
51 if (ft_errors[i].err_code == err)
52 return (ft_errors[i].err_msg);
53 return (NULL);

56bool_t
57font_utils_try_load_font(const char *fontdir, const char *fontfile,
58 FT_Library ft,
59 FT_Face *font, cairo_font_face_t **cr_font)
60{
61 char *fontpath = mkpathname(fontdir, fontfile, NULL);
62 FT_Error err;
63
64 if ((err = FT_New_Face(ft, fontpath, 0, font)) != 0) {
65 logMsg("Error loading font file %s: %s", fontpath,
66 ft_err2str(err));
67 free(fontpath);
68 return (B_FALSE);
69 }
70
71 *cr_font = cairo_ft_font_face_create_for_ft_face(*font, 0);
72
73 free(fontpath);
74
75 return (B_TRUE);
76}
