233 /*
234  * Provides a gradual method of integrating an old value until it approaches
235  * a new target value. This is used in iterative processes by calling the
236  * FILTER_IN macro repeatedly a certain time intervals (d_t = delta-time).
237  * As time progresses, old_val will gradually be made to approach new_val.
238  * The lag serves to make the approach slower or faster (e.g. a value of
239  * '2' and d_t in seconds makes old_val approach new_val with a ramp that
240  * is approximately 2 seconds long).
241  */
242 #define FILTER_IN(old_val, new_val, d_t, lag) \
243  do { \
244  FILTER_IN_TYPE(old_val) o = (old_val); \
245  FILTER_IN_TYPE(new_val) n = (new_val); \
246  ASSERT(!isnan(o)); \
247  (old_val) += (n - o) * ((d_t) / (lag)); \
248  /* Prevent an overshoot */ \
249  if ((o < n && (old_val) > n) || \
250  (o > n && (old_val) < n)) \
251  (old_val) = n; \
252  } while (0)
253 /*
254  * Same as FILTER_IN, but handles NAN values for old_val and new_val properly.
255  * If new_val is NAN, old_val is set to NAN. Otherwise if old_val is NAN,
256  * it is set to new_val directly (without gradual filtering). Otherwise this
257  * simply calls the FILTER_IN macro as normal.
258  */
259 #define FILTER_IN_NAN(old_val, new_val, d_t, lag) \
260  do { \
261  FILTER_IN_TYPE(old_val) o = (old_val); \
262  FILTER_IN_TYPE(new_val) n = (new_val); \
263  if (isnan(n)) \
264  (old_val) = NAN; \
265  else if (isnan(o)) \
266  (old_val) = (new_val); \
267  else \
268  FILTER_IN(old_val, new_val, d_t, lag); \
269  } while (0)

271 /*
272  * Linearly interpolates old_val until it is equal to tgt. The current
273  * time delta is d_t (in seconds). The interpolation speed is step/second.
274  */
275 #define FILTER_IN_LIN(old_val, tgt, d_t, step) \
276  do { \
277  double o = (old_val); \
278  double t = (tgt); \
279  double s; \
280  if (isnan(o)) \
281  o = t; \
282  if (o < t) \
283  s = (d_t) * (step); \
284  else \
285  s = (d_t) * (-(step)); \
286  if ((o <= t && o + s > t) || (o >= t && o + s < t)) \
287  (old_val) = t; \
288  else \
289  (old_val) += s; \
290  } while (0)
