65 static bool_t
66 load_image(lacf_gl_pic_t *pic)
67 {
68  uint8_t *buf;
69 
70  ASSERT(pic != NULL);
71  ASSERT0(pic->tex);
72  ASSERT(pic->path != NULL);
73 
74  buf = png_load_from_file_rgba(pic->path, &pic->w, &pic->h);
75  if (buf == NULL)
76  return (B_FALSE);
77 
78  glGenTextures(1, &pic->tex);
79  ASSERT(pic->tex != 0);
80  glBindTexture(GL_TEXTURE_2D, pic->tex);
81  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
82  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
83  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, pic->w, pic->h, 0, GL_RGBA,
84  GL_UNSIGNED_BYTE, buf);
85  lacf_free(buf);
86 
87  return (B_TRUE);
88 }

165 void
166 lacf_gl_pic_draw(lacf_gl_pic_t *pic, vect2_t pos, vect2_t size,
167  float alpha)
168 {
169  mat4 proj_matrix, mv_matrix, pvm;
170 
171  ASSERT(pic != NULL);
172 
173  VERIFY3F(dr_getvf32(&pic->proj_matrix, (float *)proj_matrix, 0, 16),
174  ==, 16);
175  VERIFY3F(dr_getvf32(&pic->mv_matrix, (float *)mv_matrix, 0, 16),
176  ==, 16);
177  glm_mat4_mul(proj_matrix, mv_matrix, pvm);
178 
179  if (pic->shader == 0) {
180  pic->shader = shader_prog_from_text("lacf_gl_pic_shader",
181  vert_shader, frag_shader,
182  "vtx_pos", VTX_ATTRIB_POS,
183  "vtx_tex0", VTX_ATTRIB_TEX0, NULL);
184  ASSERT(pic->shader != 0);
185  }
186  glUseProgram(pic->shader);
187  glUniform1f(glGetUniformLocation(pic->shader, "alpha"), alpha);
188  lacf_gl_pic_draw_custom(pic, pos, size, pic->shader, pvm);
189  glUseProgram(0);
190 }

192 void
193 lacf_gl_pic_draw_custom(lacf_gl_pic_t *pic, vect2_t pos, vect2_t size,
194  GLuint prog, const mat4 pvm)
195 {
196  glutils_quads_t *quads;
197  vect2_t p[4];
198  const vect2_t t[4] = {
199  VECT2(0, 1), VECT2(0, 0), VECT2(1, 0), VECT2(1, 1)
200  };
201 
202  ASSERT(pic != NULL);
203  ASSERT(pvm != NULL);
204 
205  if (pic->tex == 0 && !load_image(pic))
206  return;
207  if (IS_NULL_VECT(size))
208  size = VECT2(pic->w, pic->h);
209  if (pic->cache == NULL)
210  pic->cache = glutils_cache_new(LACF_GL_PIC_CACHE_SIZE);
211 
212  p[0] = VECT2(pos.x, pos.y);
213  p[1] = VECT2(pos.x, pos.y + size.y);
214  p[2] = VECT2(pos.x + size.x, pos.y + size.y);
215  p[3] = VECT2(pos.x + size.x, pos.y);
216  quads = glutils_cache_get_2D_quads(pic->cache, p, t, 4);
217 
218  XPLMBindTexture2d(pic->tex, 0);
219  glUniform1i(glGetUniformLocation(prog, "tex"), 0);
220  glUniformMatrix4fv(glGetUniformLocation(prog, "pvm"),
221  1, GL_FALSE, (const GLfloat *)pvm);
222  glutils_draw_quads(quads, prog);
223  XPLMBindTexture2d(0, 0);
224 }
