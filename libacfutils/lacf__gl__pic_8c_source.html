65 static bool_t
66 load_image(lacf_gl_pic_t *pic)
67 {
68  uint8_t *buf;
69 
70  ASSERT(pic != NULL);
71  ASSERT0(pic->tex);
72  ASSERT(pic->path != NULL);
73 
74  buf = png_load_from_file_rgba(pic->path, &pic->w, &pic->h);
75  if (buf == NULL)
76  return (B_FALSE);
77 
78  glGenTextures(1, &pic->tex);
79  ASSERT(pic->tex != 0);
80  glBindTexture(GL_TEXTURE_2D, pic->tex);
81  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
82  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
83  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, pic->w, pic->h, 0, GL_RGBA,
84  GL_UNSIGNED_BYTE, buf);
85  lacf_free(buf);
86 
87  return (B_TRUE);
88 }

235 void
236 lacf_gl_pic_draw(lacf_gl_pic_t *pic, vect2_t pos, vect2_t size,
237  float alpha)
238 {
239  mat4 proj_matrix, mv_matrix, pvm;
240 
241  ASSERT(pic != NULL);
242 
243  VERIFY3F(dr_getvf32(&pic->proj_matrix, (float *)proj_matrix, 0, 16),
244  ==, 16);
245  VERIFY3F(dr_getvf32(&pic->mv_matrix, (float *)mv_matrix, 0, 16),
246  ==, 16);
247  glm_mat4_mul(proj_matrix, mv_matrix, pvm);
248 
249  if (pic->shader == 0) {
250  pic->shader = shader_prog_from_text("lacf_gl_pic_shader",
251  vert_shader, frag_shader,
252  "vtx_pos", VTX_ATTRIB_POS,
253  "vtx_tex0", VTX_ATTRIB_TEX0, NULL);
254  ASSERT(pic->shader != 0);
255  }
256  glUseProgram(pic->shader);
257  glUniform1f(glGetUniformLocation(pic->shader, "alpha"), alpha);
258  glUniformMatrix4fv(glGetUniformLocation(pic->shader, "pvm"),
259  1, GL_FALSE, (const GLfloat *)pvm);
260  lacf_gl_pic_draw_custom(pic, pos, size, pic->shader);
261  glUseProgram(0);
262 }

277 void
278 lacf_gl_pic_draw_custom(lacf_gl_pic_t *pic, vect2_t pos, vect2_t size,
279  GLuint prog)
280 {
281  glutils_quads_t *quads;
282  vect2_t p[4];
283  const vect2_t t[4] = {
284  VECT2(0, 1), VECT2(0, 0), VECT2(1, 0), VECT2(1, 1)
285  };
286 
287  ASSERT(pic != NULL);
288 
289  if (pic->tex == 0 && !load_image(pic))
290  return;
291  if (IS_NULL_VECT(size))
292  size = VECT2(pic->w, pic->h);
293  if (pic->cache == NULL)
294  pic->cache = glutils_cache_new(LACF_GL_PIC_CACHE_SIZE);
295 
296  p[0] = VECT2(pos.x, pos.y);
297  p[1] = VECT2(pos.x, pos.y + size.y);
298  p[2] = VECT2(pos.x + size.x, pos.y + size.y);
299  p[3] = VECT2(pos.x + size.x, pos.y);
300  quads = glutils_cache_get_2D_quads(pic->cache, p, t, 4);
301 
302  XPLMBindTexture2d(pic->tex, 0);
303  glUniform1i(glGetUniformLocation(prog, "vtx_tex0"), 0);
304  glutils_draw_quads(quads, prog);
305  XPLMBindTexture2d(0, 0);
306 }
