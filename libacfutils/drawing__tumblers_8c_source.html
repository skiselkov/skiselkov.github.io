50 /*
51  * Example of a tumbler drawing function. You will ideally want to extend
52  * this with things like font size selection, variable column widths, etc.
53  *
54  * @param tumblers The tumbler set you defined ahead of time.
55  * @param n_tumblers The number of tumblers in the tumbler set.
56  * @param display_value The actual numerical value of the rolling display.
57  * @param display_x The X position of the right-most column of the display.
58  * @param display_y The Y position of the center of the display.
59  *
60  * Please note that COLUMN_WIDTH and LINE_HEIGHT below aren't defined.
61  * They are meant to represent the concept of moving between columns
62  * and line heights in the displays. You should adjust those to match
63  * your particular needs.
64  */
65 void
66 draw_tumblers(const tumbler_t *tumblers, size_t n_tumblers,
67  double display_value, double display_x, double display_y)
68 {
69  /*
70  * This is used for storing fractional line offsets between columns.
71  */
72  double fract[n_tumblers] = {};
73 
74  for (unsigned i = 0; i < n_tumblers; i++) {
75  /*
76  * The lines of text we're going to draw. The tumbler solver
77  * will put the text it wants us to draw in here.
78  */
79  char out_str[TUMBLER_LINES][TUMBLER_CAP];
80  int n = tumbler_solve(tumblers, i, display_value,
81  i > 0 ? fract[i - 1] : 0, out_str, &fract[i]);
82  /*
83  * If no digits were emitted, we can stop.
84  */
85  if (n == 0)
86  break;
87  /*
88  * The solver gave us one or more lines of text it wants us
89  * to show and their fractional line height offset.
90  */
91  for (int j = 0; j < n; j++) {
92  double text_x, text_y;
93 
94  cairo_text_extents(cr, out_str[j], &te);
95  /*
96  * Center horizontally on the column.
97  */
98  text_x = display_x - te.width / 2;
99  /*
100  * We apply a fractional line height offset (fract[i]),
101  * and then draw lines one-by-one (j) on top of each
102  * other.
103  * We round the line height, because cairo_show_text
104  * always aligns text on pixel boundaries. To avoid
105  * each line jumping around independently as they try
106  * to align to pixels, we need to make sure all the
107  * rounding for all lines is the same.
108  * We also center vertically on the line.
109  */
110  text_y = display_y +
111  (fract[i] - j) * round(1.5 * LINE_HEIGHT) -
112  te.height / 2 - te.y_bearing;
113  cairo_move_to(cr, text_x, text_y);
114  cairo_show_text(cr, out_str[j]);
115  }
116  /*
117  * Draw the next column to the left. If your columns are
118  * of unequal width, you will want to use a more complex
119  * algorithm.
120  */
121  display_x -= COLUMN_WIDTH;
122  }
123 }
