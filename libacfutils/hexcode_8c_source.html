50 hex_dec(const void *in_enc, size_t len, void *out_raw, size_t out_cap)
51 {

53  uint8_t *out = out_raw;
54 
55  /* Input length must be a multiple of two and fit into output */
56  if ((len % 2 != 0) || (out_cap < len / 2))
57  return (B_FALSE);
58 
59  for (size_t i = 0, j = 0; i < len && j < out_cap; i += 2, j++) {
60  int a = xdigit2int(in_str[i]);
61  int b = xdigit2int(in_str[i + 1]);
62

66  }
