83 hex_dec(const void *in_enc, size_t len, void *out_raw, size_t out_cap)
84 {
85  const char *in_str = in_enc;
86  uint8_t *out = out_raw;
87 
88  /* Input length must be a multiple of two and fit into output */
89  if ((len % 2 != 0) || (out_cap < len / 2))
90  return (B_FALSE);
91 
92  for (size_t i = 0, j = 0; i < len && j < out_cap; i += 2, j++) {
93  int a = xdigit2int(in_str[i]);
94  int b = xdigit2int(in_str[i + 1]);
95 
96  if (a == -1 || b == -1)
97  return (B_FALSE);
98  out[j] = (a << 4) | b;
99  }
100 
101  return (B_TRUE);
102 }
