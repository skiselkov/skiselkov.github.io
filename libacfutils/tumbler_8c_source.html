28 static double
29 wrap(const tumbler_t *tumbler, double v)
30 {
31  ASSERT(tumbler != NULL);
32  ASSERT3F(tumbler->quant, >, 0);
33 
34  while (v < 0)
35  v += tumbler->mod;
36  while (v >= tumbler->mod)
37  v -= tumbler->mod;
38 
39  return (ABS(v));
40 }

56 int
57 tumbler_solve(const tumbler_t *tumblers, int idx, double value,
58  double prev_fract, char out_str[TUMBLER_LINES][TUMBLER_CAP], double *fract)
59 {
60  const tumbler_t *tumbler, *prev_tumbler;
61 
62  ASSERT(tumblers != NULL);
63  ASSERT(out_str != NULL);
64  ASSERT(fract != NULL);
65 
66  tumbler = &tumblers[idx];
67  if (idx > 0)
68  prev_tumbler = &tumblers[idx - 1];
69  else
70  prev_tumbler = NULL;
71 
72  if (prev_tumbler == NULL) {
73  double v = fabs(fmod(value / tumbler->div, tumbler->mod));
74  double nearest = round(v / tumbler->quant) * tumbler->quant;
75  double lower2 = nearest - 2 * tumbler->quant;
76  double lower = nearest - tumbler->quant;
77  double upper = nearest + tumbler->quant;
78  double upper2 = nearest + 2 * tumbler->quant;
79 
80  *fract = iter_fract(v, lower, nearest, B_FALSE) + 1;
81 
82  lower2 = wrap(tumbler, lower2);
83  lower = wrap(tumbler, lower);
84  nearest = wrap(tumbler, nearest);
85  upper = wrap(tumbler, upper);
86  upper2 = wrap(tumbler, upper2);
87  if (value < 0) {
88  double tmp = lower;
89  lower = upper;
90  upper = tmp;
91  tmp = lower2;
92  lower2 = upper2;
93  upper2 = tmp;
94  *fract = 4 - (*fract);
95  }
96 
97  snprintf(out_str[0], TUMBLER_CAP, tumbler->fmt, fabs(lower2));
98  snprintf(out_str[1], TUMBLER_CAP, tumbler->fmt, fabs(lower));
99  snprintf(out_str[2], TUMBLER_CAP, tumbler->fmt, fabs(nearest));
100  snprintf(out_str[3], TUMBLER_CAP, tumbler->fmt, fabs(upper));
101  snprintf(out_str[4], TUMBLER_CAP, tumbler->fmt, fabs(upper2));
102 
103  return (5);
104  } else {
105  double prev_v, v, lower, lower2, upper, upper2;
106  double rollover_fract;
107 
108  prev_v = fmod(value / prev_tumbler->div, prev_tumbler->mod);
109  rollover_fract = 1 - prev_tumbler->quant / prev_tumbler->mod;
110  /*
111  * If the value of the previous tumbler is between the
112  * rollover begin value and the rollover end, then calculate
113  * the display
114  */
115  if (ABS(prev_v) > prev_tumbler->mod * rollover_fract &&
116  ABS(prev_v) <= prev_tumbler->mod) {
117  if (prev_fract > 2.0)
118  *fract = fmod(prev_fract, 1) + 1;
119  else
120  *fract = prev_fract;
121  } else if (value >= 0) {
122  *fract = 1;
123  } else {
124  *fract = 2;
125  }
126 
127  v = fmod(value / tumbler->div, tumbler->mod);
128  lower = floor(v);
129  upper = ceil(v);
130  lower2 = lower - (upper - lower);
131  upper2 = upper + (upper - lower);
132 
133  snprintf(out_str[0], TUMBLER_CAP, tumbler->fmt,
134  fmod(fabs(lower2), tumbler->mod));
135  snprintf(out_str[1], TUMBLER_CAP, tumbler->fmt,
136  fmod(fabs(lower), tumbler->mod));
137  snprintf(out_str[2], TUMBLER_CAP, tumbler->fmt,
138  fmod(fabs(upper), tumbler->mod));
139  snprintf(out_str[3], TUMBLER_CAP, tumbler->fmt,
140  fmod(fabs(upper2), tumbler->mod));
141  memset(out_str[4], 0, TUMBLER_CAP);
142 
143  handle_leading_chars(tumbler, prev_tumbler, value, out_str[0]);
144  handle_leading_chars(tumbler, prev_tumbler, value, out_str[1]);
145  handle_leading_chars(tumbler, prev_tumbler, value, out_str[2]);
146  handle_leading_chars(tumbler, prev_tumbler, value, out_str[3]);
147 
148  return (4);
149  }
