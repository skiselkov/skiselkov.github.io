56size_t
57chart_dl_info_write(char *ptr, size_t size, size_t nmemb, void *userdata)
58{
59 chart_dl_info_t *dl_info = userdata;
60 size_t bytes = size * nmemb;
61
62 ASSERT(dl_info != NULL);
63
64 /* Respond to an early termination request */
65 if (dl_info->cdb != NULL && !dl_info->cdb->loader.run)
66 return (0);
67
68 if (dl_info->bufcap < dl_info->bufsz + bytes) {
69 dl_info->bufcap += REALLOC_STEP;
70 if (dl_info->bufcap > MAX_DL_SIZE) {
71 logMsg("Error downloading %s: too much data received "
72 "(%ld bytes)", dl_info->url, (long)dl_info->bufcap);
73 return (0);
74 }
75 dl_info->buf = realloc(dl_info->buf, dl_info->bufcap);
76 }
77 memcpy(&dl_info->buf[dl_info->bufsz], ptr, bytes);
78 dl_info->bufsz += bytes;
79
80 return (bytes);
81}

83static struct curl_slist *
84append_if_mod_since_hdr(struct curl_slist *hdrs, const char *path)
85{
86 struct stat st;
87
88 if (file_exists(path, NULL) && stat(path, &st) == 0) {
89 char buf[64];
90 time_t t = st.st_mtime;
91 struct tm tm;
92
93 lacf_gmtime_r(&t, &tm);
94 strftime(buf, sizeof (buf),
95 "If-Modified-Since: %a, %d %b %Y %H:%M:%S GMT", &tm);
96 hdrs = curl_slist_append(hdrs, buf);
97 }
98
99 return (hdrs);
100}

102static bool_t
103write_dl(chart_dl_info_t *dl_info, const char *filepath, const char *url,
104 const char *error_prefix)
105{
106 char *dname = lacf_dirname(filepath);
107 FILE *fp;
108
109 if (!create_directory_recursive(dname)) {
110 free(dname);
111 return (B_FALSE);
112 }
113 free(dname);
114 fp = fopen(filepath, "wb");
115
116 if (fp == NULL) {
117 logMsg("%s %s: error writing disk file %s: %s",
118 error_prefix, url, filepath, strerror(errno));
119 return (B_FALSE);
120 }
121 fwrite(dl_info->buf, 1, dl_info->bufsz, fp);
122 fclose(fp);
123
124 return (B_TRUE);
125}

149bool_t
150chart_download_multi2(CURL **curl_p, const char *proxy, const char *url,
151 const char *filepath, const char *method,
152 const chart_prov_info_login_t *login, int timeout,
153 const char *error_prefix, chart_dl_info_t *raw_output)
154{
155 CURL *curl;
156 struct curl_slist *hdrs = NULL;
157 chart_dl_info_t dl_info = { .cdb = NULL };
158 CURLcode res;
159 long code = 0;
160 bool_t result = B_TRUE;
161
162 ASSERT(curl_p != NULL);
163 /* cdb can be NULL */
164 ASSERT(url != NULL);
165 /* filepath can be NULL */
166 /* method can be NULL */
167 /* login can be NULL */
168 ASSERT(error_prefix != NULL);
169 /* raw_output can be NULL */
170
171 dl_info.url = url;
172
173 if (*curl_p != NULL) {
174 curl = *curl_p;
175 } else {
176 const char *cainfo = (login != NULL ? login->cainfo : NULL);
177
178 curl = curl_easy_init();
179 VERIFY(curl != NULL);
180
181 curl_easy_setopt(curl, CURLOPT_TIMEOUT,
182 timeout < 0 ? DL_TIMEOUT : timeout);
183 chart_setup_curl(curl, cainfo);
184 if (login != NULL && login->username != NULL) {
185 curl_easy_setopt(curl, CURLOPT_USERNAME,
186 login->username);
187 }
188 if (login != NULL && login->password != NULL) {
189 curl_easy_setopt(curl, CURLOPT_PASSWORD,
190 login->password);
191 }
192 if (proxy != NULL)
193 curl_easy_setopt(curl, CURLOPT_PROXY, proxy);
194 *curl_p = curl;
195 }
196
197 if (method != NULL)
198 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, method);
199 else
200 curl_easy_setopt(curl, CURLOPT_HTTPGET, 1L);
201
202 if (filepath != NULL)
203 hdrs = append_if_mod_since_hdr(hdrs, filepath);
204
205 curl_easy_setopt(curl, CURLOPT_URL, url);
206 curl_easy_setopt(curl, CURLOPT_WRITEDATA, &dl_info);
207 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, hdrs);
208
209 res = curl_easy_perform(curl);
210 if (res == CURLE_OK)
211 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
212 /*
213 * HTTP code 207 is "Multi-Status" for WebDAV.
214 */
215 if (res == CURLE_OK && (code == 200 || code == 207) &&
216 dl_info.bufsz != 0) {
217 if (filepath != NULL) {
218 result = write_dl(&dl_info, filepath, url,
219 error_prefix);
220 }
221 } else {
222 if (res != CURLE_OK) {
223 logMsg("%s %s: %s", error_prefix, url,
224 curl_easy_strerror(res));
225 result = B_FALSE;
226 } else if (code != 304) {
227 /*
228 * Code `304' indicates we have a cached good copy.
229 */
230 logMsg("%s %s: HTTP error %ld", error_prefix, url,
231 code);
232 result = B_FALSE;
233 }
234 }
235 if (hdrs != NULL) {
236 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, NULL);
237 curl_slist_free_all(hdrs);
238 }
239 if (raw_output != NULL) {
240 if (code == 200 || code == 207) {
241 *raw_output = dl_info;
242 } else {
243 memset(raw_output, 0, sizeof (*raw_output));
244 free(dl_info.buf);
245 }
246 } else {
247 free(dl_info.buf);
248 }
249
250 return (result);
251}

278void
279word_subst(char *str, const char **subst)
280{
281 for (int i = 0; subst[i] != NULL; i += 2) {
282 char *p = strstr(str, subst[i]);
283
284 if (p != NULL &&
285 /*
286 * Make sure the word either ends the line,
287 * or it's followed by a space.
288 */
289 (strlen(p) == strlen(subst[i]) ||
290 !isalnum(p[strlen(subst[i])]))) {
291 unsigned l1 = strlen(subst[i]);
292 unsigned l2 = strlen(subst[i + 1]);
293 memcpy(p, subst[i + 1], l2);
294 memmove(&p[l2], &p[l1],
295 strlen(str) - (&p[l1] - str) + 1);
296 }
297 }
298}
