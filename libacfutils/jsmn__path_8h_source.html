68 {
69  int num = 1;

71  ASSERT(toks != NULL);

73 
74  switch (toks[cur_tok].type) {

77  cur_tok++;
78  for (int i = 0; i < obj->size; i++) {
79  int n;
80

82  cur_tok += n;
83  num += n;
84  if (cur_tok >= n_toks)
85  break;
86  n = jsmn_count_toks_r(toks, n_toks, cur_tok);
87  cur_tok += n;
88  num += n;
89  if (cur_tok >= n_toks)
90  break;
91  }
92  break;
93  }
94  case JSMN_ARRAY: {
95  const jsmntok_t *arr = &toks[cur_tok];
96  cur_tok++;
97  for (int i = 0; i < arr->size; i++) {
98  int n = jsmn_count_toks_r(toks, n_toks, cur_tok);
99  cur_tok += n;
100  num += n;
101  if (cur_tok >= n_toks)
102  break;
103  }
104  break;
105  }
106  default:
107  break;
108  }
109  return (num);

112 UNUSED_ATTR static const jsmntok_t *
113 jsmn_path_lookup(const char *json, const jsmntok_t *toks, int n_toks,
114  const char *path)
115 {
116  size_t n_comps;
117  char **comps;
118  int cur_tok = 0;
119 
120  ASSERT(json != NULL);
121  ASSERT(toks != NULL);
122  ASSERT(path != NULL);
123  comps = strsplit(path, "/", B_TRUE, &n_comps);
124  for (size_t i = 0; i < n_comps; i++) {
125  if (comps[i][0] == '[') {
126  const jsmntok_t *arr = &toks[cur_tok];
127  int idx = atoi(&comps[i][1]);
128 
129  if (arr->type != JSMN_ARRAY || idx >= arr->size)
130  return (NULL);
131  cur_tok++;
132  for (; idx > 0; idx--) {
133  cur_tok += jsmn_count_toks_r(toks, n_toks,
134  cur_tok);
135  }
136  } else {
137  bool_t found = B_FALSE;
138  const jsmntok_t *obj = &toks[cur_tok];
139 
140  if (obj->type != JSMN_OBJECT)
141  return (NULL);
142  cur_tok++;
143  for (int j = cur_tok, idx = 0; idx < obj->size &&
144  j + 1 < n_toks; idx++) {
145  const jsmntok_t *key = &toks[j];
146  unsigned len;
147 
148  if (key->type != JSMN_STRING)
149  return (NULL);
150  len = key->end - key->start;
151  if (strlen(comps[i]) == len && strncmp(
152  &json[key->start], comps[i], len) == 0) {
153  found = B_TRUE;
154  cur_tok = j + 1;
155  break;
156  }
157  j += 1 + jsmn_count_toks_r(toks, n_toks, j + 1);
158  }
159  if (!found)
160  return (NULL);
161  }
162  }
163  free_strlist(comps, n_comps);
164 
165  return (&toks[cur_tok]);
166 }
