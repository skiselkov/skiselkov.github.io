73 static bool_t
74 parse_proc_name(const char *faanfd18, chart_type_t type, char proc_name[8])
75 {
76  const char *period;
77 
78  ASSERT(faanfd18 != NULL);
79  ASSERT(proc_name != NULL);
80 
81  if (faanfd18[0] == '\0')
82  return (B_FALSE);
83  switch (type) {
84  case CHART_TYPE_DP:
85  period = strchr(faanfd18, '.');
86  if (period != NULL) {
87  strlcpy(proc_name, faanfd18, MIN(8,
88  (period - faanfd18) + 1));
89  } else {
90  strlcpy(proc_name, faanfd18, 8);
91  }
92  return (B_TRUE);
93  case CHART_TYPE_STAR:
94  case CHART_TYPE_IAP:
95  period = strchr(faanfd18, '.');
96  if (period != NULL)
97  strlcpy(proc_name, period + 1, 8);
98  else
99  strlcpy(proc_name, faanfd18, 8);
100  return (B_TRUE);
101  default:
102  return (B_FALSE);
103  }
104 }

106 static void
107 load_record(chart_arpt_t *arpt, const xmlNode *rec)
108 {
109  chart_t *chart = safe_calloc(1, sizeof (*chart));
110 
111  for (const xmlNode *node = rec->children; node != NULL;
112  node = node->next) {
113  const char *content;
114 
115  if (node->children == NULL || node->name == NULL)
116  continue;
117  content = (char *)node->children[0].content;
118  if (strcmp((char *)node->name, "chart_name") == 0) {
119  free(chart->name);
120  chart->name = safe_strdup(content);
121  } else if (strcmp((char *)node->name, "faanfd18") == 0) {
122  free(chart->codename);
123  chart->codename = safe_strdup(content);
124  } else if (strcmp((char *)node->name, "pdf_name") == 0) {
125  /*
126  * "DELETED_JOB.PDF" in the PDF filename means that
127  * the chart no longer exists, so get rid of it.
128  */
129  if (strcmp(content, "DELETED_JOB.PDF") == 0) {
130  chart->type = CHART_TYPE_UNKNOWN;
131  break;
132  }
133  free(chart->filename);
134  chart->filename = safe_strdup(content);
135  } else if (strcmp((char *)node->name, "chart_code") == 0) {
136  if (strcmp(content, "APD") == 0)
137  chart->type = CHART_TYPE_APD;
138  else if (strcmp(content, "IAP") == 0)
139  chart->type = CHART_TYPE_IAP;
140  else if (strcmp(content, "DP") == 0)
141  chart->type = CHART_TYPE_DP;
142  else if (strcmp(content, "ODP") == 0)
143  chart->type = CHART_TYPE_ODP;
144  else if (strcmp(content, "STAR") == 0)
145  chart->type = CHART_TYPE_STAR;
146  else if (strcmp(content, "MIN") == 0)
147  chart->type = CHART_TYPE_MIN;
148  else
149  chart->type = CHART_TYPE_UNKNOWN;
150  }
151  }
152  if (chart->name[0] == '\0' || chart->type == CHART_TYPE_UNKNOWN ||
153  !chartdb_add_chart(arpt, chart)) {
154  free(chart->name);
155  free(chart->codename);
156  free(chart->filename);
157  free(chart);
158  } else {
159  /*
160  * Extract the procedure name from the faanfd18 field.
161  */
162  if (chart->codename != NULL && parse_proc_name(chart->codename,
163  chart->type, chart->procs.procs[0])) {
164  chart->procs.n_procs++;
165  }
166  }
167 }

169 static void
170 load_airport(chartdb_t *cdb, const xmlNode *arpt_node, const char *city_id,
171  const char *state_id)
172 {
173  char *icao_ident = (char *)xmlGetProp(arpt_node,
174  (xmlChar *)"icao_ident");
175  char *apt_ident = (char *)xmlGetProp(arpt_node, (xmlChar *)"apt_ident");
176  char *apt_name = (char *)xmlGetProp(arpt_node, (xmlChar *)"ID");
177  chart_arpt_t *arpt;
178  char icao[8];
179 
180  if (icao_ident == NULL || apt_ident == NULL || apt_name == NULL) {
181  /* Malformed file */
182  goto out;
183  }
184  if (is_valid_icao_code(icao_ident)) {
185  /* Normal ICAO identifier present, use it */
186  lacf_strlcpy(icao, icao_ident, sizeof (icao));
187  } else if (strlen(apt_ident) > 0) {
188  /*
189  * Local non-ICAO identifier, convert it into a pseudo-ICAO
190  * identifier.
191  */
192  snprintf(icao, sizeof (icao), "K%s", apt_ident);
193  } else {
194  /* No valid ID present, skip the airport */
195  goto out;
196  }
197  arpt = chartdb_add_arpt(cdb, icao, apt_name, city_id, state_id);
198 
199  for (const xmlNode *rec = arpt_node->children; rec != NULL;
200  rec = rec->next) {
201  if (rec->name == NULL ||
202  strcmp((char *)rec->name, "record") != 0)
203  continue;
204  load_record(arpt, rec);
205  }
206 out:
207  if (icao_ident != NULL)
208  xmlFree(icao_ident);
209  if (apt_ident != NULL)
210  xmlFree(apt_ident);
211  if (apt_name != NULL)
212  xmlFree(apt_name);
213 }

251 static bool_t
252 load_index(chartdb_t *cdb)
253 {
254  char *index_path = mk_index_path(cdb);
255  xmlDoc *doc = NULL;
256  xmlXPathContext *xpath_ctx = NULL;
257  xmlXPathObject *states_obj = NULL;
258 
259  doc = xmlParseFile(index_path);
260  if (doc == NULL) {
261  logMsg("Error parsing chart index %s: XML parsing error",
262  index_path);
263  goto errout;
264  }
265  xpath_ctx = xmlXPathNewContext(doc);
266  if (xpath_ctx == NULL) {
267  logMsg("Error creating XPath context for document %s.",
268  index_path);
269  goto errout;
270  }
271  states_obj = xmlXPathEvalExpression((xmlChar *)
272  "/digital_tpp/state_code", xpath_ctx);
273  for (int i = 0; i < states_obj->nodesetval->nodeNr; i++)
274  load_state(cdb, states_obj->nodesetval->nodeTab[i]);
275  xmlXPathFreeObject(states_obj);
276  states_obj = NULL;
277 
278  xmlXPathFreeContext(xpath_ctx);
279  xmlFreeDoc(doc);
280  free(index_path);
281 
282  return (B_TRUE);
283 errout:
284  if (states_obj != NULL)
285  xmlXPathFreeObject(states_obj);
286  if (xpath_ctx != NULL)
287  xmlXPathFreeContext(xpath_ctx);
288  if (doc != NULL)
289  xmlFreeDoc(doc);
290  free(index_path);
291 
292  return (B_FALSE);
293 }
