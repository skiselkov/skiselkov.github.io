601 /*
602  * Given a runway threshold vector, direction vector, width, length and
603  * threshold longitudinal displacement, prepares a bounding box which
604  * encompasses that runway.
605  */
606 static vect2_t *
607 make_rwy_bbox(vect2_t thresh_v, vect2_t dir_v, double width, double len,
608  double long_displ)
609 {
610  vect2_t *bbox;
611  vect2_t len_displ_v;
612 
613  ASSERT(!IS_NULL_VECT(thresh_v));
614  ASSERT(!IS_NULL_VECT(dir_v));
615  ASSERT(!isnan(width));
616  ASSERT(!isnan(len));
617  ASSERT(!isnan(long_displ));
618 
619  bbox = safe_malloc(sizeof (*bbox) * 5);
620 
621  /*
622  * Displace the 'a' point from the runway threshold laterally
623  * by 1/2 width to the right.
624  */
625  bbox[0] = vect2_add(thresh_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
626  width / 2));
627  /* pull it back by `long_displ' */
628  bbox[0] = vect2_add(bbox[0], vect2_set_abs(vect2_neg(dir_v),
629  long_displ));
630 
631  /* do the same for the `d' point, but displace to the left */
632  bbox[3] = vect2_add(thresh_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
633  width / 2));
634  /* pull it back by `long_displ' */
635  bbox[3] = vect2_add(bbox[3], vect2_set_abs(vect2_neg(dir_v),
636  long_displ));
637 
638  /*
639  * points `b' and `c' are along the runway simply as runway len +
640  * long_displ
641  */
642  len_displ_v = vect2_set_abs(dir_v, len + long_displ);
643  bbox[1] = vect2_add(bbox[0], len_displ_v);
644  bbox[2] = vect2_add(bbox[3], len_displ_v);
645 
646  bbox[4] = NULL_VECT2;
647 
648  return (bbox);
649 }

917 /*
918  * Parses an airport line in apt.dat. The default apt.dat spec only supplies
919  * the identifier and field elevation on this line. Our extended format which
920  * we use in the data cache also adds the TA, TL and reference point LAT &
921  * LON to this. If the apt.dat being parsed is a standard (non-extended) one,
922  * the additional info is inferred later on from other sources during the
923  * airport data cache creation process.
924  */
925 static airport_t *
926 parse_apt_dat_1_line(airportdb_t *db, const char *line, iconv_t *cd_p,
927  airport_t **dup_arpt_p)
928 {
929  /*
930  * pre-allocate the buffer to be large enough that most names are
931  * already gonna fit in there without too much reallocation.
932  */
933  char *name = NULL;
934  const char *new_ident;
935  geo_pos3_t pos = NULL_GEO_POS3;
936  size_t ncomps;
937  char **comps = strsplit(line, " ", B_TRUE, &ncomps);
938  airport_t *arpt = NULL;
939 
940  ASSERT(db != NULL);
941  ASSERT(line != NULL);
942 
943  if (dup_arpt_p != NULL)
944  *dup_arpt_p = NULL;
945 
946  ASSERT(strcmp(comps[0], "1") == 0);
947  if (ncomps < 5)
948  goto out;
949 
950  new_ident = comps[4];
951  pos.elev = atof(comps[1]);
952  if (!is_valid_elev(pos.elev))
953  /* Small GA fields might not have valid identifiers. */
954  goto out;
955  name = concat_comps(&comps[5], ncomps - 5);
956  if (name == NULL)
957  name = safe_strdup("");

959  if (arpt != NULL) {
960  /*
961  * This airport was already known from a previously loaded
962  * apt.dat. Avoid overwriting its data.
963  */

965  *dup_arpt_p = arpt;
966  arpt = NULL;
967  goto out;
968  }

970  avl_create(&arpt->rwys, runway_compar, sizeof (runway_t),
971  offsetof(runway_t, node));
972  list_create(&arpt->freqs, sizeof (freq_info_t),
973  offsetof(freq_info_t, node));
974  lacf_strlcpy(arpt->ident, new_ident, sizeof (arpt->ident));
975  strtoupper(arpt->ident);
976  /*
977  * Legacy scenery doesn't include '1302' metainfo lines with
978  * the ICAO code listed separately, so for those we just assume
979  * that the code listed in the ident here is the ICAO code.
980  */
981  strlcpy(arpt->icao, arpt->ident, sizeof (arpt->icao));
982 
983  avl_create(&arpt->ramp_starts, ramp_start_compar,
984  sizeof (ramp_start_t), offsetof(ramp_start_t, node));
985 
986  /*
987  * Unfortunately, X-Plane's scenery authors put all kinds of
988  * weird chars into their airport names. So we employ libiconv
989  * to hopefully transliterate that junk away as much as possible.
990  */
991  if (cd_p != NULL) {
992  LACF_DESTROY(arpt->name_orig);
993  arpt->name_orig = safe_strdup(name);
994  normalize_name(cd_p, name, arpt->name, sizeof (arpt->name));
995  strtoupper(arpt->name);
996  } else {
997  /*
998  * iconv is NOT used when reading our own apt.dat cache.
999  * So for those cases, we can just verbatim copy the airport
1000  * name directly without charset issues.
1001  */
1002  lacf_strlcpy(arpt->name, name, sizeof (arpt->name));
1003  }
1004 
1005  arpt->refpt = pos;
1006  arpt->refpt_m = GEO3_FT2M(pos);
1007 out:
1008  free_strlist(comps, ncomps);
1009  free(name);
1010  return (arpt);
1011 }

1013 /*
1014  * This is the matching function that attempts to determine if a VGSI
1015  * (row code '21' in apt.dat) belongs to a specific runway. Returns the
1016  * lateral displacement (in meters) from the runway centerline if the
1017  * VGSI matches the runway or a huge number (1e10) otherwise.
1018  */
1019 static double
1020 runway_vgsi_fuzzy_match(runway_t *rwy, int end, vgsi_t type, vect2_t pos_v,
1021  double true_hdg)
1022 {
1023  runway_end_t *re = &rwy->ends[end], *ore = &rwy->ends[!end];
1024  vect2_t thr2light_v = vect2_sub(pos_v, re->thr_v);
1025  vect2_t thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
1026  vect2_t thr2thr_uv = vect2_unit(thr2thr_v, NULL);
1027  vect2_t thr2thr_norm_uv = vect2_norm(thr2thr_uv, B_TRUE);
1028  double lat_displ = vect2_dotprod(thr2light_v, thr2thr_norm_uv),
1029  lon_displ = vect2_dotprod(thr2light_v, thr2thr_uv);
1030 
1031  ASSERT(rwy != NULL);
1032 
1033  /*
1034  * The checks we perform are:
1035  * 1) the lateral displacement from the runway centerline must be
1036  * no more than 2x the runway width (VGSI_LAT_DISPL_FACT).
1037  * 2) the longitudinal displacement must be sit between the thresholds
1038  * 3) the true heading of the light fixture must be within 5 degrees
1039  * of true runway heading (VGSI_HDG_MATCH_THRESH).
1040  * 4) if the VGSI is a left PAPI, it must be on the left
1041  * 5) if the VGSI is a right PAPI, it must be on the right
1042  */
1043  if (fabs(lat_displ) > VGSI_LAT_DISPL_FACT * rwy->width ||
1044  lon_displ < 0 || lon_displ > rwy->length ||
1045  fabs(rel_hdg(re->hdg, true_hdg)) > VGSI_HDG_MATCH_THRESH ||
1046  (lat_displ > 0 && type == VGSI_PAPI_4L) ||
1047  (lat_displ < 0 && type == VGSI_PAPI_4R))
1048  return (1e10);
1049  return (lat_displ);
1050 }

1052 static void
1053 find_nearest_runway_to_vgsi(airport_t *arpt, vgsi_t type, vect2_t pos_v,
1054  double true_hdg, runway_t **rwy, runway_end_t **re, runway_end_t **ore)
1055 {
1056  double max_displ = 100000;
1057 
1058  ASSERT(arpt != NULL);
1059  ASSERT(rwy != NULL);
1060  ASSERT(re != NULL);
1061  ASSERT(ore != NULL);
1062  /*
1063  * Runway unknown. Let's try to do a more fuzzy search.
1064  * We will look for the closest runway from which we are
1065  * displaced no more than 2x the runway's width. We also
1066  * check that the sense of the displacement is kept (left
1067  * PAPI on the left side of the runway and vice versa).
1068  */
1069  for (runway_t *crwy = avl_first(&arpt->rwys); crwy != NULL;
1070  crwy = AVL_NEXT(&arpt->rwys, crwy)) {
1071  double displ;
1072  if ((displ = runway_vgsi_fuzzy_match(crwy, 0,
1073  type, pos_v, true_hdg)) < max_displ) {
1074  *rwy = crwy;
1075  *re = &crwy->ends[0];
1076  *ore = &crwy->ends[1];
1077  max_displ = displ;
1078  } else if ((displ = runway_vgsi_fuzzy_match(crwy, 1,
1079  type, pos_v, true_hdg)) < max_displ) {
1080  *rwy = crwy;
1081  *re = &crwy->ends[1];
1082  *ore = &crwy->ends[0];
1083  max_displ = displ;
1084  }
1085  }
1086 }

1088 /*
1089  * Row codes `21' denote lighting objects. We detect if the object is a
1090  * PAPI or VASI and use it to compute the GPA and TCH.
1091  */
1092 static void
1093 parse_apt_dat_21_line(airport_t *arpt, const char *line)
1094 {
1095  char **comps;
1096  size_t ncomps;
1097  vgsi_t type;
1098  geo_pos2_t pos;
1099  double gpa, tch, displ, true_hdg;
1100  const char *rwy_id;
1101  runway_t *rwy = NULL;
1102  runway_end_t *re = NULL, *ore = NULL;
1103  vect2_t pos_v, thr2light_v, thr2thr_v;
1104 
1105  ASSERT(arpt != NULL);
1106  ASSERT(line != NULL);
1107 
1108  /* Construct the airport fpp to compute the thresholds */
1109  if (!load_airport(arpt))
1110  return;
1111 
1112  comps = strsplit(line, " ", B_TRUE, &ncomps);
1113  ASSERT(strcmp(comps[0], "21") == 0);
1114  if (ncomps < 7)
1115  /* No need to report, sometimes the rwy_ID is missing. */
1116  goto out;
1117  type = atoi(comps[3]);
1118  if (type < VGSI_VASI || type > VGSI_PAPI_3C || type == VGSI_PAPI_20DEG)
1119  goto out;
1120  pos = GEO_POS2(atof(comps[1]), atof(comps[2]));
1121  pos_v = geo2fpp(pos, &arpt->fpp);
1122  true_hdg = atof(comps[4]);
1123  if (!is_valid_hdg(true_hdg))
1124  goto out;
1125  gpa = atof(comps[5]);
1126  if (isnan(gpa) || gpa <= 0.0 || gpa > RWY_GPA_LIMIT)
1127  goto out;
1128  rwy_id = comps[6];
1129 
1130  /*
1131  * Locate the associated runway. The VGSI line should denote which
1132  * runway it belongs to.
1133  */
1134  for (rwy = avl_first(&arpt->rwys); rwy != NULL;
1135  rwy = AVL_NEXT(&arpt->rwys, rwy)) {
1136  if (strcmp(rwy->ends[0].id, rwy_id) == 0) {
1137  re = &rwy->ends[0];
1138  ore = &rwy->ends[1];
1139  break;
1140  } else if (strcmp(rwy->ends[1].id, rwy_id) == 0) {
1141  ore = &rwy->ends[0];
1142  re = &rwy->ends[1];
1143  break;
1144  }
1145  }
1146  if (rwy == NULL) {
1147  find_nearest_runway_to_vgsi(arpt, type, pos_v, true_hdg,
1148  &rwy, &re, &ore);
1149  if (rwy == NULL)
1150  goto out;
1151  }
1152  /*
1153  * We can compute the longitudinal displacement along the associated
1154  * runway of the light from the runway threshold.
1155  */
1156  thr2light_v = vect2_sub(pos_v, re->thr_v);
1157  thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
1158  displ = vect2_dotprod(thr2light_v, vect2_unit(thr2thr_v, NULL));
1159  /*
1160  * Check that the VGSI sits somewhere between the two thresholds
1161  * and that it's aligned properly. Some scenery is broken like that!
1162  * This condition will only fail if we didn't use the matching in
1163  * find_nearest_runway_to_vgsi, because that function already
1164  * perform these checks.
1165  */
1166  if (displ < 0 || displ > rwy->length ||
1167  fabs(rel_hdg(true_hdg, re->hdg)) > VGSI_HDG_MATCH_THRESH) {
1168  rwy = NULL;
1169  re = NULL;
1170  ore = NULL;
1171  /* Fallback check - try to match it to ANY runway */
1172  find_nearest_runway_to_vgsi(arpt, type, pos_v, true_hdg,
1173  &rwy, &re, &ore);
1174  if (rwy == NULL)
1175  goto out;
1176  thr2light_v = vect2_sub(pos_v, re->thr_v);
1177  thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
1178  displ = vect2_dotprod(thr2light_v, vect2_unit(thr2thr_v,
1179  NULL));
1180  }
1181  /* Finally, given the displacement and GPA, compute the TCH. */
1182  tch = MET2FEET(sin(DEG2RAD(gpa)) * displ);
1183  ASSERT(tch >= 0.0);
1184  if (TCH_IS_VALID(tch)) {
1185  re->gpa = gpa;
1186  re->tch = tch;
1187  }
1188 out:
1189  free_strlist(comps, ncomps);
1190 }

1258  * not tens of thousands.

1260  freq->type = atoi(comps[0]) - (use833 ? 1050 : 50);

1268  * wastes space, so remove that. And some even more stupidly
1269  * contain the word "frequency" - DOH!

1277  sizeof (freq->name) - strlen(freq->name));

1292  * Our data cache features three additional special fields: GPA, TCH and

1303  avl_index_t where;

1307  ASSERT(line != NULL);

1312  (hard_surf_only && !rwy_is_hard(atoi(comps[2]))))
1313  goto out;

1316

1319  rwy->surf = atoi(comps[2]);
1320

1322  rwy->ends[0].thr = GEO_POS3(atof(comps[8 + 1]), atof(comps[8 + 2]),
1323  arpt->refpt.elev);
1324  rwy->ends[0].thr_m = GEO3_FT2M(rwy->ends[0].thr);
1325  rwy->ends[0].displ = atof(comps[8 + 3]);
1326  rwy->ends[0].blast = atof(comps[8 + 4]);
1327

1329  rwy->ends[1].thr = GEO_POS3(atof(comps[8 + 9 + 1]),
1330  atof(comps[8 + 9 + 2]), arpt->refpt.elev);
1331  rwy->ends[1].thr_m = GEO3_FT2M(rwy->ends[1].thr);
1332  rwy->ends[1].displ = atof(comps[8 + 9 + 3]);
1333  rwy->ends[1].blast = atof(comps[8 + 9 + 4]);
1334 
1335  /*
1336  * ARINC 424 says in field reference 5.67 that if no explicit TCH is
1337  * specified, 50 feet shall be assumed. The GPA cannot be assumed
1338  * this easily and unfortunately field 5.226 from ARINC 424 isn't in
1339  * X-Plane 11's navdata, so we instead parse it in a later step from
1340  * instrument approach procedures (X-Plane 11) or from an Airports.txt
1341  * (X-Plane 10), falling back to VGSI triangulation in the scenery if
1342  * those methods fail. We won't provide vertical approach monitoring
1343  * unless both GPA & TCH are non-zero.
1344  */
1345  rwy->ends[0].tch = 50;
1346  rwy->ends[1].tch = 50;

1349  rwy->ends[0].id, rwy->ends[1].id);

1351  rwy->ends[1].id, rwy->ends[0].id);

1355  strstr(comps[23], "GPA2:") == comps[23] &&
1356  strstr(comps[24], "TCH1:") == comps[24] &&
1357  strstr(comps[25], "TCH2:") == comps[25] &&
1358  strstr(comps[26], "TELEV1:") == comps[26] &&
1359  strstr(comps[27], "TELEV2:") == comps[27]) {
1360  rwy->ends[0].gpa = atof(&comps[22][5]);
1361  rwy->ends[1].gpa = atof(&comps[23][5]);
1362  rwy->ends[0].tch = atof(&comps[24][5]);
1363  rwy->ends[1].tch = atof(&comps[25][5]);
1364  rwy->ends[0].thr.elev = atof(&comps[26][7]);
1365  rwy->ends[1].thr.elev = atof(&comps[27][7]);
1366  rwy->ends[0].thr_m.elev = FEET2MET(rwy->ends[0].thr.elev);
1367  rwy->ends[1].thr_m.elev = FEET2MET(rwy->ends[1].thr.elev);
1368  }
1369 
1370  /* Validate the runway ends individually. */
1371  if (!validate_rwy_end(&rwy->ends[0], error_descr) ||
1372  !validate_rwy_end(&rwy->ends[1], error_descr)) {
1373  free(rwy);
1374  goto out;
1375  }
1376  /*
1377  * Are the runway ends sufficiently far apart? Protects against runways
1378  * with overlapping thresholds, which results in a NAN runway hdg.
1379  */

1381  geo2ecef_ft(rwy->ends[1].thr, &wgs84)) < MIN_RWY_LEN) {
1382  free(rwy);
1383  goto out;
1384  }
1385  /* Duplicate runway present? */
1386  if (avl_find(&arpt->rwys, rwy, &where) != NULL) {
1387  free(rwy);
1388  goto out;
1389  }
1390  avl_insert(&arpt->rwys, rwy, where);
1391  if (arpt->load_complete) {
1392  /* do a supplemental runway info load */
1393  load_rwy_info(rwy);
1394  } else if (isnan(arpt->refpt.lat) || isnan(arpt->refpt.lon)) {
1395  arpt->refpt.lat = NAN;
1396  arpt->refpt.lon = NAN;
1397  arpt->refpt_m.lat = NAN;
1398  arpt->refpt_m.lon = NAN;
1399  airport_auto_refpt(arpt);
1400  }
1401 out:
1402  free_strlist(comps, ncomps);
1403 }

1617

1627  }
1628

1631

1633  if (getline(&line, &linecap, apt_dat_f) <= 0)
1634  continue;

1638  continue;
1639  /* Read the version header */
1640  if (line_num == 2) {
1641  version = row_code;
1642  continue;
1643  }

1713  ncomps - 2);

2643 /*
2644  * The approach proximity bounding box is constructed as follows:
2645  *
2646  * 5500 meters
2647  * |<=======>|
2648  * | |
2649  * d +-_ (c1) |
2650  * | -._3 degrees
2651  * | -_ c
2652  * | +-------------------------------+
2653  * | | ==== ---- ---- ==== |
2654  * x + thr_v-+ ==== - ------> dir_v - - ==== |
2655  * | | ==== ---- ---- ==== |
2656  * | +-------------------------------+
2657  * | _- b
2658  * | _-.
2659  * a +-- (b1)
2660  *
2661  * If there is another parallel runway, we make sure our bounding boxes
2662  * don't overlap. We do this by introducing two additional points, b1 and
2663  * c1, in between a and b or c and d respectively. We essentially shear
2664  * the overlapping excess from the bounding polygon.
2665  */
2666 static vect2_t *
2667 make_apch_prox_bbox(const runway_t *rwy, int end_i)
2668 {
2669  const runway_end_t *end, *oend;
2670  const fpp_t *fpp;
2671  double limit_left = 1000000, limit_right = 1000000;
2672  vect2_t x, a, b, b1, c, c1, d, thr_v, othr_v, dir_v;
2673  vect2_t *bbox = safe_calloc(7, sizeof (vect2_t));
2674  size_t n_pts = 0;
2675 
2676  ASSERT(rwy != NULL);
2677  fpp = &rwy->arpt->fpp;
2678  ASSERT(end_i == 0 || end_i == 1);
2679 
2680  /*
2681  * By pre-initing the whole array to null vectors, we can make the
2682  * bbox either contain 4, 5 or 6 points, depending on whether
2683  * shearing due to a close parallel runway needs to be applied.
2684  */
2685  for (int i = 0; i < 7; i++)
2686  bbox[i] = NULL_VECT2;
2687 
2688  end = &rwy->ends[end_i];
2689  oend = &rwy->ends[!end_i];
2690  thr_v = end->thr_v;
2691  othr_v = oend->thr_v;
2692  dir_v = vect2_sub(othr_v, thr_v);
2693 
2694  x = vect2_add(thr_v, vect2_set_abs(vect2_neg(dir_v),
2695  RWY_APCH_PROXIMITY_LON_DISPL));
2696  a = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2697  rwy->width / 2 + RWY_APCH_PROXIMITY_LAT_DISPL));
2698  b = vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2699  rwy->width / 2));
2700  c = vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2701  rwy->width / 2));
2702  d = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2703  rwy->width / 2 + RWY_APCH_PROXIMITY_LAT_DISPL));
2704 
2705  b1 = NULL_VECT2;
2706  c1 = NULL_VECT2;
2707 
2708  /*
2709  * If our rwy_id designator contains a L/C/R, then we need to
2710  * look for another parallel runway.
2711  */
2712  if (strlen(end->id) >= 3) {
2713  int my_num_id = atoi(end->id);
2714 
2715  for (const runway_t *orwy = avl_first(&rwy->arpt->rwys);
2716  orwy != NULL; orwy = AVL_NEXT(&rwy->arpt->rwys, orwy)) {
2717  const runway_end_t *orwy_end;
2718  vect2_t othr_v, v;
2719  double a, dist;
2720 
2721  if (orwy == rwy)
2722  continue;
2723  if (atoi(orwy->ends[0].id) == my_num_id)
2724  orwy_end = &orwy->ends[0];
2725  else if (atoi(orwy->ends[1].id) == my_num_id)
2726  orwy_end = &orwy->ends[1];
2727  else
2728  continue;
2729 
2730  /*
2731  * This is a parallel runway, measure the
2732  * distance to it from us.
2733  */
2734  othr_v = geo2fpp(GEO3_TO_GEO2(orwy_end->thr), fpp);
2735  v = vect2_sub(othr_v, thr_v);
2736  if (IS_ZERO_VECT2(v)) {
2737  logMsg("CAUTION: your nav DB is looking very "
2738  "strange: runways %s and %s at %s are on "
2739  "top of each other (coords: %fx%f)",
2740  end->id, orwy_end->id, rwy->arpt->icao,
2741  orwy_end->thr.lat, orwy_end->thr.lon);
2742  continue;
2743  }
2744  a = rel_hdg(dir2hdg(dir_v), dir2hdg(v));
2745  dist = fabs(sin(DEG2RAD(a)) * vect2_abs(v));
2746 
2747  if (a < 0)
2748  limit_left = MIN(dist / 2, limit_left);
2749  else
2750  limit_right = MIN(dist / 2, limit_right);
2751  }
2752  }
2753 
2754  if (limit_left < RWY_APCH_PROXIMITY_LAT_DISPL) {
2755  c1 = vect2vect_isect(vect2_sub(d, c), c, vect2_neg(dir_v),
2756  vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2757  limit_left)), B_FALSE);
2758  d = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2759  limit_left));
2760  }
2761  if (limit_right < RWY_APCH_PROXIMITY_LAT_DISPL) {
2762  b1 = vect2vect_isect(vect2_sub(b, a), a, vect2_neg(dir_v),
2763  vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2764  limit_right)), B_FALSE);
2765  a = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2766  limit_right));
2767  }
2768 
2769  bbox[n_pts++] = a;
2770  if (!IS_NULL_VECT(b1))
2771  bbox[n_pts++] = b1;
2772  bbox[n_pts++] = b;
2773  bbox[n_pts++] = c;
2774  if (!IS_NULL_VECT(c1))
2775  bbox[n_pts++] = c1;
2776  bbox[n_pts++] = d;
2777 
2778  return (bbox);
2779 }

2781 /*
2782  * Prepares a runway's bounding box vector coordinates using the airport
2783  * coord fpp transform.
2784  */
2785 static void
2786 load_rwy_info(runway_t *rwy)
2787 {
2788  ASSERT(rwy != NULL);
2789  ASSERT(rwy->arpt->load_complete);
2790  /*
2791  * RAAS runway proximity entry bounding box is defined as:
2792  *
2793  * 1000ft 1000ft
2794  * |<======>| |<======>|
2795  * | | | |
2796  * ---- d +-------------------------------------------------+ c
2797  * 1.5x ^ | | | |
2798  * rwy | | | | |
2799  * width | | +-------------------------------+ |
2800  * v | | ==== ---- ---- ==== | |
2801  * -------|-thresh-x ==== - - - - - - - - - - ==== | |
2802  * ^ | | ==== ---- ---- ==== | |
2803  * 1.5x | | +-------------------------------+ |
2804  * rwy | | |
2805  * width v | |
2806  * ---- a +-------------------------------------------------+ b
2807  */
2808  vect2_t dt1v = geo2fpp(GEO3_TO_GEO2(rwy->ends[0].thr), &rwy->arpt->fpp);
2809  vect2_t dt2v = geo2fpp(GEO3_TO_GEO2(rwy->ends[1].thr), &rwy->arpt->fpp);
2810  double displ1 = rwy->ends[0].displ;
2811  double displ2 = rwy->ends[1].displ;
2812  double blast1 = rwy->ends[0].blast;
2813  double blast2 = rwy->ends[1].blast;
2814 
2815  vect2_t dir_v = vect2_sub(dt2v, dt1v);
2816  double dlen = vect2_abs(dir_v);
2817  double hdg1 = dir2hdg(dir_v);
2818  double hdg2 = dir2hdg(vect2_neg(dir_v));
2819 
2820  vect2_t t1v = vect2_add(dt1v, vect2_set_abs(dir_v, displ1));
2821  vect2_t t2v = vect2_add(dt2v, vect2_set_abs(vect2_neg(dir_v), displ2));
2822  double len = vect2_abs(vect2_sub(t2v, t1v));
2823 
2824  double prox_lon_bonus1 = MAX(displ1, RWY_PROXIMITY_LON_DISPL - displ1);
2825  double prox_lon_bonus2 = MAX(displ2, RWY_PROXIMITY_LON_DISPL - displ2);
2826 
2827  rwy->ends[0].thr_v = t1v;
2828  rwy->ends[1].thr_v = t2v;
2829  rwy->ends[0].dthr_v = dt1v;
2830  rwy->ends[1].dthr_v = dt2v;
2831  rwy->ends[0].hdg = hdg1;
2832  rwy->ends[1].hdg = hdg2;
2833  rwy->ends[0].land_len = vect2_abs(vect2_sub(dt2v, t1v));
2834  rwy->ends[1].land_len = vect2_abs(vect2_sub(dt1v, t2v));
2835  rwy->length = len;
2836 
2837  ASSERT(rwy->rwy_bbox == NULL);
2838 
2839  rwy->rwy_bbox = make_rwy_bbox(t1v, dir_v, rwy->width, len, 0);
2840  rwy->tora_bbox = make_rwy_bbox(dt1v, dir_v, rwy->width, dlen, 0);
2841  rwy->asda_bbox = make_rwy_bbox(dt1v, dir_v, rwy->width,
2842  dlen + blast2, blast1);
2843  rwy->prox_bbox = make_rwy_bbox(t1v, dir_v, RWY_PROXIMITY_LAT_FRACT *
2844  rwy->width, len + prox_lon_bonus2, prox_lon_bonus1);
2845 
2846  rwy->ends[0].apch_bbox = make_apch_prox_bbox(rwy, 0);
2847  rwy->ends[1].apch_bbox = make_apch_prox_bbox(rwy, 1);
2848 }
