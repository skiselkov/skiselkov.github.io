412 /*
413  * AVL tree comparator for tile_t's based on latitude and longitude.
414  */
415 static int
416 tile_compar(const void *a, const void *b)
417 {
418  const tile_t *ta = a, *tb = b;
419 
420  if (ta->pos.lat < tb->pos.lat) {
421  return (-1);
422  } else if (ta->pos.lat == tb->pos.lat) {
423  if (ta->pos.lon < tb->pos.lon)
424  return (-1);
425  else if (ta->pos.lon == tb->pos.lon)
426  return (0);
427  else
428  return (1);
429  } else {
430  return (1);
431  }
432 }

434 /*
435  * AVL tree comparator for runway_t's based on the joint runway ID.
436  */
437 static int

439 {
440  const runway_t *ra = a, *rb = b;

465 /*
466  * Retrieves the geo table tile which contains position `pos'. If create is
467  * B_TRUE, if the tile doesn't exit, it will be created.
468  * Returns the table tile (if it exists) and a boolean (in created_p if
469  * non-NULL) informing whether the table tile was created in this call
470  * (if create == B_TRUE).
471  */
472 static tile_t *
473 geo_table_get_tile(airportdb_t *db, geo_pos2_t pos, bool_t create,
474  bool_t *created_p)
475 {
476  pos.lat = floor(pos.lat);
477  pos.lon = floor(pos.lon);
478 
479  bool_t created = B_FALSE;
480  tile_t srch = { .pos = pos };
481  tile_t *tile;
482  avl_index_t where;
483 
484  ASSERT(db != NULL);
485  ASSERT(!IS_NULL_GEO_POS(pos));
486 
487  tile = avl_find(&db->geo_table, &srch, &where);
488  if (tile == NULL && create) {
489  tile = safe_malloc(sizeof (*tile));
490  tile->pos = pos;
491  avl_create(&tile->arpts, airport_compar, sizeof (airport_t),
492  offsetof(airport_t, tile_node));
493  avl_insert(&db->geo_table, tile, where);
494  created = B_TRUE;
495  }
496  if (created_p != NULL)
497  *created_p = created;
498 
499  return (tile);
500 }

502 /*
503  * Given a runway threshold vector, direction vector, width, length and
504  * threshold longitudinal displacement, prepares a bounding box which
505  * encompasses that runway.
506  */
507 static vect2_t *
508 make_rwy_bbox(vect2_t thresh_v, vect2_t dir_v, double width, double len,
509  double long_displ)
510 {
511  vect2_t *bbox;
512  vect2_t len_displ_v;
513 
514  ASSERT(!IS_NULL_VECT(thresh_v));
515  ASSERT(!IS_NULL_VECT(dir_v));
516  ASSERT(!isnan(width));
517  ASSERT(!isnan(len));
518  ASSERT(!isnan(long_displ));
519 
520  bbox = safe_malloc(sizeof (*bbox) * 5);
521 
522  /*
523  * Displace the 'a' point from the runway threshold laterally
524  * by 1/2 width to the right.
525  */
526  bbox[0] = vect2_add(thresh_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
527  width / 2));
528  /* pull it back by `long_displ' */
529  bbox[0] = vect2_add(bbox[0], vect2_set_abs(vect2_neg(dir_v),
530  long_displ));
531 
532  /* do the same for the `d' point, but displace to the left */
533  bbox[3] = vect2_add(thresh_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
534  width / 2));
535  /* pull it back by `long_displ' */
536  bbox[3] = vect2_add(bbox[3], vect2_set_abs(vect2_neg(dir_v),
537  long_displ));
538 
539  /*
540  * points `b' and `c' are along the runway simply as runway len +
541  * long_displ
542  */
543  len_displ_v = vect2_set_abs(dir_v, len + long_displ);
544  bbox[1] = vect2_add(bbox[0], len_displ_v);
545  bbox[2] = vect2_add(bbox[3], len_displ_v);
546 
547  bbox[4] = NULL_VECT2;
548 
549  return (bbox);
550 }

577 /*
578  * Performs a lookup for an airport based on ICAO code in an airportdb_t.
579  * The lookup is case-insensitive, because some data providers sometimes
580  * provided ICAO identifiers in lowercase.
581  */

583 apt_dat_lookup(airportdb_t *db, const char *ident)
584 {
585  airport_t search, *result;
586

588  ASSERT(ident != NULL);
589 
590  lacf_strlcpy(search.ident, ident, sizeof (search.ident));
591  strtoupper(search.ident);

818 /*
819  * Parses an airport line in apt.dat. The default apt.dat spec only supplies
820  * the identifier and field elevation on this line. Our extended format which
821  * we use in the data cache also adds the TA, TL and reference point LAT &
822  * LON to this. If the apt.dat being parsed is a standard (non-extended) one,
823  * the additional info is inferred later on from other sources during the
824  * airport data cache creation process.
825  */
826 static airport_t *
827 parse_apt_dat_1_line(airportdb_t *db, const char *line, iconv_t *cd_p,
828  airport_t **dup_arpt_p)
829 {
830  /*
831  * pre-allocate the buffer to be large enough that most names are
832  * already gonna fit in there without too much reallocation.
833  */
834  char *name = NULL;
835  const char *new_ident;
836  geo_pos3_t pos = NULL_GEO_POS3;
837  size_t ncomps;
838  char **comps = strsplit(line, " ", B_TRUE, &ncomps);
839  airport_t *arpt = NULL;
840 
841  ASSERT(db != NULL);
842  ASSERT(line != NULL);
843 
844  if (dup_arpt_p != NULL)
845  *dup_arpt_p = NULL;
846 
847  ASSERT(strcmp(comps[0], "1") == 0);
848  if (ncomps < 5)
849  goto out;
850 
851  new_ident = comps[4];
852  pos.elev = atof(comps[1]);
853  if (!is_valid_elev(pos.elev))
854  /* Small GA fields might not have valid identifiers. */
855  goto out;
856  name = concat_comps(&comps[5], ncomps - 5);
857  if (name == NULL)
858  name = safe_strdup("");

860  if (arpt != NULL) {
861  /*
862  * This airport was already known from a previously loaded
863  * apt.dat. Avoid overwriting its data.
864  */
865  if (dup_arpt_p != NULL)
866  *dup_arpt_p = arpt;
867  arpt = NULL;
868  goto out;
869  }
870  arpt = safe_calloc(1, sizeof (*arpt));
871  avl_create(&arpt->rwys, runway_compar, sizeof (runway_t),
872  offsetof(runway_t, node));
873  list_create(&arpt->freqs, sizeof (freq_info_t),
874  offsetof(freq_info_t, node));
875  lacf_strlcpy(arpt->ident, new_ident, sizeof (arpt->ident));
876  strtoupper(arpt->ident);
877  /*
878  * Legacy scenery doesn't include '1302' metainfo lines with
879  * the ICAO code listed separately, so for those we just assume
880  * that the code listed in the ident here is the ICAO code.
881  */
882  strlcpy(arpt->icao, arpt->ident, sizeof (arpt->icao));
883 
884  avl_create(&arpt->ramp_starts, ramp_start_compar,
885  sizeof (ramp_start_t), offsetof(ramp_start_t, node));
886 
887  /*
888  * Unfortunately, X-Plane's scenery authors put all kinds of
889  * weird chars into their airport names. So we employ libiconv
890  * to hopefully transliterate that junk away as much as possible.
891  */
892  if (cd_p != NULL) {
893  LACF_DESTROY(arpt->name_orig);
894  arpt->name_orig = safe_strdup(name);
895  normalize_name(cd_p, name, arpt->name, sizeof (arpt->name));
896  strtoupper(arpt->name);
897  } else {
898  /*
899  * iconv is NOT used when reading our own apt.dat cache.
900  * So for those cases, we can just verbatim copy the airport
901  * name directly without charset issues.
902  */
903  lacf_strlcpy(arpt->name, name, sizeof (arpt->name));
904  }
905 
906  arpt->refpt = pos;
907  arpt->refpt_m = GEO3_FT2M(pos);
908 out:
909  free_strlist(comps, ncomps);
910  free(name);
911  return (arpt);
912 }

914 /*
915  * This is the matching function that attempts to determine if a VGSI
916  * (row code '21' in apt.dat) belongs to a specific runway. Returns the
917  * lateral displacement (in meters) from the runway centerline if the
918  * VGSI matches the runway or a huge number (1e10) otherwise.
919  */
920 static double
921 runway_vgsi_fuzzy_match(runway_t *rwy, int end, vgsi_t type, vect2_t pos_v,
922  double true_hdg)
923 {
924  runway_end_t *re = &rwy->ends[end], *ore = &rwy->ends[!end];
925  vect2_t thr2light_v = vect2_sub(pos_v, re->thr_v);
926  vect2_t thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
927  vect2_t thr2thr_uv = vect2_unit(thr2thr_v, NULL);
928  vect2_t thr2thr_norm_uv = vect2_norm(thr2thr_uv, B_TRUE);
929  double lat_displ = vect2_dotprod(thr2light_v, thr2thr_norm_uv),
930  lon_displ = vect2_dotprod(thr2light_v, thr2thr_uv);
931 
932  ASSERT(rwy != NULL);
933 
934  /*
935  * The checks we perform are:
936  * 1) the lateral displacement from the runway centerline must be
937  * no more than 2x the runway width (VGSI_LAT_DISPL_FACT).
938  * 2) the longitudinal displacement must be sit between the thresholds
939  * 3) the true heading of the light fixture must be within 5 degrees
940  * of true runway heading (VGSI_HDG_MATCH_THRESH).
941  * 4) if the VGSI is a left PAPI, it must be on the left
942  * 5) if the VGSI is a right PAPI, it must be on the right
943  */
944  if (fabs(lat_displ) > VGSI_LAT_DISPL_FACT * rwy->width ||
945  lon_displ < 0 || lon_displ > rwy->length ||
946  fabs(rel_hdg(re->hdg, true_hdg)) > VGSI_HDG_MATCH_THRESH ||
947  (lat_displ > 0 && type == VGSI_PAPI_4L) ||
948  (lat_displ < 0 && type == VGSI_PAPI_4R))
949  return (1e10);
950  return (lat_displ);
951 }

953 static void
954 find_nearest_runway_to_vgsi(airport_t *arpt, vgsi_t type, vect2_t pos_v,
955  double true_hdg, runway_t **rwy, runway_end_t **re, runway_end_t **ore)
956 {
957  double max_displ = 100000;
958 
959  ASSERT(arpt != NULL);
960  ASSERT(rwy != NULL);
961  ASSERT(re != NULL);
962  ASSERT(ore != NULL);
963  /*
964  * Runway unknown. Let's try to do a more fuzzy search.
965  * We will look for the closest runway from which we are
966  * displaced no more than 2x the runway's width. We also
967  * check that the sense of the displacement is kept (left
968  * PAPI on the left side of the runway and vice versa).
969  */
970  for (runway_t *crwy = avl_first(&arpt->rwys); crwy != NULL;
971  crwy = AVL_NEXT(&arpt->rwys, crwy)) {
972  double displ;
973  if ((displ = runway_vgsi_fuzzy_match(crwy, 0,
974  type, pos_v, true_hdg)) < max_displ) {
975  *rwy = crwy;
976  *re = &crwy->ends[0];
977  *ore = &crwy->ends[1];
978  max_displ = displ;
979  } else if ((displ = runway_vgsi_fuzzy_match(crwy, 1,
980  type, pos_v, true_hdg)) < max_displ) {
981  *rwy = crwy;
982  *re = &crwy->ends[1];
983  *ore = &crwy->ends[0];
984  max_displ = displ;
985  }
986  }
987 }

989 /*
990  * Row codes `21' denote lighting objects. We detect if the object is a
991  * PAPI or VASI and use it to compute the GPA and TCH.
992  */
993 static void
994 parse_apt_dat_21_line(airport_t *arpt, const char *line)
995 {
996  char **comps;
997  size_t ncomps;
998  vgsi_t type;
999  geo_pos2_t pos;
1000  double gpa, tch, displ, true_hdg;
1001  const char *rwy_id;
1002  runway_t *rwy = NULL;
1003  runway_end_t *re = NULL, *ore = NULL;
1004  vect2_t pos_v, thr2light_v, thr2thr_v;
1005 
1006  ASSERT(arpt != NULL);
1007  ASSERT(line != NULL);
1008 
1009  /* Construct the airport fpp to compute the thresholds */
1010  if (!load_airport(arpt))
1011  return;
1012 
1013  comps = strsplit(line, " ", B_TRUE, &ncomps);
1014  ASSERT(strcmp(comps[0], "21") == 0);
1015  if (ncomps < 7)
1016  /* No need to report, sometimes the rwy_ID is missing. */
1017  goto out;
1018  type = atoi(comps[3]);
1019  if (type < VGSI_VASI || type > VGSI_PAPI_3C || type == VGSI_PAPI_20DEG)
1020  goto out;
1021  pos = GEO_POS2(atof(comps[1]), atof(comps[2]));
1022  pos_v = geo2fpp(pos, &arpt->fpp);
1023  true_hdg = atof(comps[4]);
1024  if (!is_valid_hdg(true_hdg))
1025  goto out;
1026  gpa = atof(comps[5]);
1027  if (isnan(gpa) || gpa <= 0.0 || gpa > RWY_GPA_LIMIT)
1028  goto out;
1029  rwy_id = comps[6];
1030 
1031  /*
1032  * Locate the associated runway. The VGSI line should denote which
1033  * runway it belongs to.
1034  */
1035  for (rwy = avl_first(&arpt->rwys); rwy != NULL;
1036  rwy = AVL_NEXT(&arpt->rwys, rwy)) {
1037  if (strcmp(rwy->ends[0].id, rwy_id) == 0) {
1038  re = &rwy->ends[0];
1039  ore = &rwy->ends[1];
1040  break;
1041  } else if (strcmp(rwy->ends[1].id, rwy_id) == 0) {
1042  ore = &rwy->ends[0];
1043  re = &rwy->ends[1];
1044  break;
1045  }
1046  }
1047  if (rwy == NULL) {
1048  find_nearest_runway_to_vgsi(arpt, type, pos_v, true_hdg,
1049  &rwy, &re, &ore);
1050  if (rwy == NULL)
1051  goto out;
1052  }
1053  /*
1054  * We can compute the longitudinal displacement along the associated
1055  * runway of the light from the runway threshold.
1056  */
1057  thr2light_v = vect2_sub(pos_v, re->thr_v);
1058  thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
1059  displ = vect2_dotprod(thr2light_v, vect2_unit(thr2thr_v, NULL));
1060  /*
1061  * Check that the VGSI sits somewhere between the two thresholds
1062  * and that it's aligned properly. Some scenery is broken like that!
1063  * This condition will only fail if we didn't use the matching in
1064  * find_nearest_runway_to_vgsi, because that function already
1065  * perform these checks.
1066  */
1067  if (displ < 0 || displ > rwy->length ||
1068  fabs(rel_hdg(true_hdg, re->hdg)) > VGSI_HDG_MATCH_THRESH) {
1069  rwy = NULL;
1070  re = NULL;
1071  ore = NULL;
1072  /* Fallback check - try to match it to ANY runway */
1073  find_nearest_runway_to_vgsi(arpt, type, pos_v, true_hdg,
1074  &rwy, &re, &ore);
1075  if (rwy == NULL)
1076  goto out;
1077  thr2light_v = vect2_sub(pos_v, re->thr_v);
1078  thr2thr_v = vect2_sub(ore->thr_v, re->thr_v);
1079  displ = vect2_dotprod(thr2light_v, vect2_unit(thr2thr_v,
1080  NULL));
1081  }
1082  /* Finally, given the displacement and GPA, compute the TCH. */
1083  tch = MET2FEET(sin(DEG2RAD(gpa)) * displ);
1084  ASSERT(tch >= 0.0);
1085  if (TCH_IS_VALID(tch)) {
1086  re->gpa = gpa;
1087  re->tch = tch;
1088  }
1089 out:
1090  free_strlist(comps, ncomps);
1091 }

1093 /*
1094  * Validates the data parsed from an apt.dat for a runway end:
1095  * 1) it has a valid runway identifier
1096  * 2) it has a valid threshold lat x lon
1097  * 3) its latitude is within our latitude limits
1098  * 4) it has a valid elevation (or no known elevation)
1099  * 5) it has a non-negative threshold displacement value
1100  * 6) it has a non-negative blastpath length value
1101  * 7) it has a valid (or zero) glidepath angle
1102  * 8) it has a valid (or zero) threshold clearing height

1105 validate_rwy_end(const runway_end_t *re, char error_descr[128])
1106 {

1108  ASSERT(error_descr != NULL);

1114  } \

1116  VALIDATE(is_valid_rwy_ID(re->id), "Runway ID \"%s\" invalid", re->id);

1125  VALIDATE(re->gpa >= 0.0 && re->gpa < RWY_GPA_LIMIT,
1126  "GPA \"%g\" is invalid", re->gpa);

1133 static void
1134 parse_apt_dat_freq_line(airport_t *arpt, char *line, bool_t use833)
1135 {

1137  size_t ncomps;
1138  freq_info_t *freq;
1139 
1140  ASSERT(arpt != NULL);
1141  ASSERT(line != NULL);
1142  /*
1143  * Remove spurious underscores and dashes that some sceneries insist
1144  * on using in frequency names. Do this before the strsplit pass, so
1145  * we subdivide at those boundaries.

1189 /*

1191  * denote the runway's surface type, width (in meters) the position of

1197  */

1203  runway_t *rwy;
1204  avl_index_t where;
1205  char error_descr[128];

1208  ASSERT(line != NULL);
1209

1211  ASSERT(strcmp(comps[0], "100") == 0);

1213  (hard_surf_only && !rwy_is_hard(atoi(comps[2]))))
1214  goto out;
1215 
1216  rwy = safe_calloc(1, sizeof (*rwy));
1217

1220  rwy->surf = atoi(comps[2]);
1221 
1222  copy_rwy_ID(comps[8 + 0], rwy->ends[0].id);
1223  rwy->ends[0].thr = GEO_POS3(atof(comps[8 + 1]), atof(comps[8 + 2]),
1224  arpt->refpt.elev);
1225  rwy->ends[0].thr_m = GEO3_FT2M(rwy->ends[0].thr);
1226  rwy->ends[0].displ = atof(comps[8 + 3]);
1227  rwy->ends[0].blast = atof(comps[8 + 4]);
1228 
1229  copy_rwy_ID(comps[8 + 9 + 0], rwy->ends[1].id);
1230  rwy->ends[1].thr = GEO_POS3(atof(comps[8 + 9 + 1]),
1231  atof(comps[8 + 9 + 2]), arpt->refpt.elev);
1232  rwy->ends[1].thr_m = GEO3_FT2M(rwy->ends[1].thr);
1233  rwy->ends[1].displ = atof(comps[8 + 9 + 3]);
1234  rwy->ends[1].blast = atof(comps[8 + 9 + 4]);
1235 
1236  /*
1237  * ARINC 424 says in field reference 5.67 that if no explicit TCH is
1238  * specified, 50 feet shall be assumed. The GPA cannot be assumed
1239  * this easily and unfortunately field 5.226 from ARINC 424 isn't in
1240  * X-Plane 11's navdata, so we instead parse it in a later step from
1241  * instrument approach procedures (X-Plane 11) or from an Airports.txt
1242  * (X-Plane 10), falling back to VGSI triangulation in the scenery if
1243  * those methods fail. We won't provide vertical approach monitoring
1244  * unless both GPA & TCH are non-zero.
1245  */
1246  rwy->ends[0].tch = 50;
1247  rwy->ends[1].tch = 50;

1250  rwy->ends[0].id, rwy->ends[1].id);
1251  snprintf(rwy->rev_joint_id, sizeof (rwy->rev_joint_id), "%s%s",
1252  rwy->ends[1].id, rwy->ends[0].id);
1253 
1254  /* Our extended data cache format */
1255  if (ncomps >= 28 && strstr(comps[22], "GPA1:") == comps[22] &&
1256  strstr(comps[23], "GPA2:") == comps[23] &&
1257  strstr(comps[24], "TCH1:") == comps[24] &&
1258  strstr(comps[25], "TCH2:") == comps[25] &&
1259  strstr(comps[26], "TELEV1:") == comps[26] &&
1260  strstr(comps[27], "TELEV2:") == comps[27]) {
1261  rwy->ends[0].gpa = atof(&comps[22][5]);
1262  rwy->ends[1].gpa = atof(&comps[23][5]);
1263  rwy->ends[0].tch = atof(&comps[24][5]);
1264  rwy->ends[1].tch = atof(&comps[25][5]);
1265  rwy->ends[0].thr.elev = atof(&comps[26][7]);
1266  rwy->ends[1].thr.elev = atof(&comps[27][7]);
1267  rwy->ends[0].thr_m.elev = FEET2MET(rwy->ends[0].thr.elev);
1268  rwy->ends[1].thr_m.elev = FEET2MET(rwy->ends[1].thr.elev);
1269  }
1270 
1271  /* Validate the runway ends individually. */
1272  if (!validate_rwy_end(&rwy->ends[0], error_descr) ||
1273  !validate_rwy_end(&rwy->ends[1], error_descr)) {
1274  free(rwy);
1275  goto out;
1276  }
1277  /*
1278  * Are the runway ends sufficiently far apart? Protects against runways
1279  * with overlapping thresholds, which results in a NAN runway hdg.
1280  */
1281  if (vect3_dist(geo2ecef_ft(rwy->ends[0].thr, &wgs84),
1282  geo2ecef_ft(rwy->ends[1].thr, &wgs84)) < MIN_RWY_LEN) {
1283  free(rwy);
1284  goto out;
1285  }
1286  /* Duplicate runway present? */
1287  if (avl_find(&arpt->rwys, rwy, &where) != NULL) {
1288  free(rwy);
1289  goto out;
1290  }
1291  avl_insert(&arpt->rwys, rwy, where);
1292  if (arpt->load_complete) {
1293  /* do a supplemental runway info load */
1294  load_rwy_info(rwy);
1295  } else if (isnan(arpt->refpt.lat) || isnan(arpt->refpt.lon)) {
1296  arpt->refpt.lat = NAN;
1297  arpt->refpt.lon = NAN;
1298  arpt->refpt_m.lat = NAN;
1299  arpt->refpt_m.lon = NAN;
1300  airport_auto_refpt(arpt);
1301  }
1302 out:
1303  free_strlist(comps, ncomps);
1304 }

1503 /*
1504  * Parses an apt.dat (either from regular scenery or from CACHE_DIR) to
1505  * cache the airports contained in it.
1506  */
1507 static void
1508 read_apt_dat(airportdb_t *db, const char *apt_dat_fname, bool_t fail_ok,
1509  iconv_t *cd_p, bool_t fill_in_dups)
1510 {
1511  FILE *apt_dat_f;
1512  airport_t *arpt = NULL, *dup_arpt = NULL;
1513  char *line = NULL;
1514  size_t linecap = 0;
1515  int line_num = 0, version = 0;
1516  char **comps;
1517  size_t ncomps;
1518 
1519  ASSERT(db != NULL);
1520  ASSERT(apt_dat_fname != NULL);
1521 
1522  apt_dat_f = fopen(apt_dat_fname, "r");
1523  if (apt_dat_f == NULL) {
1524  if (!fail_ok)
1525  logMsg("Can't open %s: %s", apt_dat_fname,
1526  strerror(errno));
1527  return;
1528  }
1529 
1530  while (!feof(apt_dat_f)) {
1531  int row_code;
1532 
1533  line_num++;
1534  if (getline(&line, &linecap, apt_dat_f) <= 0)
1535  continue;
1536  strip_space(line);
1537 
1538  if (sscanf(line, "%d", &row_code) != 1)
1539  continue;
1540  /* Read the version header */
1541  if (line_num == 2) {
1542  version = row_code;
1543  continue;
1544  }
1545  /*
1546  * Finish the current airport on an empty line or a new
1547  * airport line.
1548  */
1549  if (strlen(line) == 0 || row_code == 1 || row_code == 16 ||
1550  row_code == 17) {
1551  if (arpt != NULL)
1552  read_apt_dat_insert(db, arpt);
1553  arpt = NULL;
1554  dup_arpt = NULL;
1555  }
1556  if (row_code == 1) {
1557  arpt = parse_apt_dat_1_line(db, line, cd_p,
1558  fill_in_dups ? &dup_arpt : NULL);
1559  }
1560  if (arpt == NULL) {
1561  if (dup_arpt != NULL)
1562  fill_dup_arpt_info(dup_arpt, line, row_code);
1563  continue;
1564  }
1565 
1566  switch (row_code) {
1567  case 21:
1568  parse_apt_dat_21_line(arpt, line);
1569  break;
1570  case 50 ... 56:
1571  parse_apt_dat_freq_line(arpt, line, B_FALSE);

1573  case 100:

1576  case 1050 ... 1056:

1587  * X-Plane 11. This line can contain varying numbers
1588  * of components, but we only care when it's 3.

1592  continue;
1593  }

1595  ASSERT(!arpt->geo_linked);

1598  * for an Airports.txt.
1599  */
1600  if (strcmp(comps[1], "icao_code") == 0 &&
1601  is_valid_icao_code(comps[2])) {
1602  lacf_strlcpy(arpt->icao, comps[2],
1603  sizeof (arpt->icao));
1604  } else if (strcmp(comps[1], "iata_code") == 0 &&
1605  is_valid_iata_code(comps[2])) {
1606  lacf_strlcpy(arpt->iata, comps[2],
1607  sizeof (arpt->iata));
1608  } else if (strcmp(comps[1], "country") == 0) {
1609  parse_attr_country(&comps[2], ncomps - 2,
1610  version, arpt);
1611  } else if (strcmp(comps[1], "city") == 0) {
1612  LACF_DESTROY(arpt->city);
1613  arpt->city = concat_comps(&comps[2],
1614  ncomps - 2);
1615  } else if (strcmp(comps[1], "name_orig") == 0) {
1616  LACF_DESTROY(arpt->name_orig);
1617  arpt->name_orig = concat_comps(&comps[2],
1618  ncomps - 2);
1619  } else if (strcmp(comps[1], "transition_alt") == 0) {
1620  extract_TA(arpt, comps);
1621  } else if (strcmp(comps[1], "transition_level") == 0) {
1622  extract_TL(arpt, comps);
1623  } else if (strcmp(comps[1], "datum_lat") == 0) {
1624  double lat = atof(comps[2]);
1625  if (is_valid_lat(lat)) {
1626  arpt->refpt.lat = lat;
1627  arpt->refpt_m.lat = lat;
1628  } else {
1629  free_airport(arpt);
1630  arpt = NULL;
1631  }
1632  } else if (strcmp(comps[1], "datum_lon") == 0) {
1633  double lon = atof(comps[2]);
1634  if (is_valid_lon(lon)) {
1635  arpt->refpt.lon = lon;
1636  arpt->refpt_m.lon = lon;
1637  }
1638  } else if (strcmp(comps[1], "region_code") == 0 &&
1639  strcmp(comps[1], "-") != 0) {
1640  lacf_strlcpy(arpt->cc, comps[2],
1641  sizeof (arpt->cc));
1642  }
1643 
1644  free_strlist(comps, ncomps);
1645  break;
1646  }
1647  }

1650  read_apt_dat_insert(db, arpt);
1651 
1652  free(line);
1653  fclose(apt_dat_f);

2544 /*
2545  * The approach proximity bounding box is constructed as follows:
2546  *
2547  * 5500 meters
2548  * |<=======>|
2549  * | |
2550  * d +-_ (c1) |
2551  * | -._3 degrees
2552  * | -_ c
2553  * | +-------------------------------+
2554  * | | ==== ---- ---- ==== |
2555  * x + thr_v-+ ==== - ------> dir_v - - ==== |
2556  * | | ==== ---- ---- ==== |
2557  * | +-------------------------------+
2558  * | _- b
2559  * | _-.
2560  * a +-- (b1)
2561  *
2562  * If there is another parallel runway, we make sure our bounding boxes
2563  * don't overlap. We do this by introducing two additional points, b1 and
2564  * c1, in between a and b or c and d respectively. We essentially shear
2565  * the overlapping excess from the bounding polygon.
2566  */
2567 static vect2_t *
2568 make_apch_prox_bbox(const runway_t *rwy, int end_i)
2569 {
2570  const runway_end_t *end, *oend;
2571  const fpp_t *fpp;
2572  double limit_left = 1000000, limit_right = 1000000;
2573  vect2_t x, a, b, b1, c, c1, d, thr_v, othr_v, dir_v;
2574  vect2_t *bbox = safe_calloc(7, sizeof (vect2_t));
2575  size_t n_pts = 0;
2576 
2577  ASSERT(rwy != NULL);
2578  fpp = &rwy->arpt->fpp;
2579  ASSERT(end_i == 0 || end_i == 1);
2580 
2581  /*
2582  * By pre-initing the whole array to null vectors, we can make the
2583  * bbox either contain 4, 5 or 6 points, depending on whether
2584  * shearing due to a close parallel runway needs to be applied.
2585  */
2586  for (int i = 0; i < 7; i++)
2587  bbox[i] = NULL_VECT2;
2588 
2589  end = &rwy->ends[end_i];
2590  oend = &rwy->ends[!end_i];
2591  thr_v = end->thr_v;
2592  othr_v = oend->thr_v;
2593  dir_v = vect2_sub(othr_v, thr_v);
2594 
2595  x = vect2_add(thr_v, vect2_set_abs(vect2_neg(dir_v),
2596  RWY_APCH_PROXIMITY_LON_DISPL));
2597  a = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2598  rwy->width / 2 + RWY_APCH_PROXIMITY_LAT_DISPL));
2599  b = vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2600  rwy->width / 2));
2601  c = vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2602  rwy->width / 2));
2603  d = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2604  rwy->width / 2 + RWY_APCH_PROXIMITY_LAT_DISPL));
2605 
2606  b1 = NULL_VECT2;
2607  c1 = NULL_VECT2;
2608 
2609  /*
2610  * If our rwy_id designator contains a L/C/R, then we need to
2611  * look for another parallel runway.
2612  */
2613  if (strlen(end->id) >= 3) {
2614  int my_num_id = atoi(end->id);
2615 
2616  for (const runway_t *orwy = avl_first(&rwy->arpt->rwys);
2617  orwy != NULL; orwy = AVL_NEXT(&rwy->arpt->rwys, orwy)) {
2618  const runway_end_t *orwy_end;
2619  vect2_t othr_v, v;
2620  double a, dist;
2621 
2622  if (orwy == rwy)
2623  continue;
2624  if (atoi(orwy->ends[0].id) == my_num_id)
2625  orwy_end = &orwy->ends[0];
2626  else if (atoi(orwy->ends[1].id) == my_num_id)
2627  orwy_end = &orwy->ends[1];
2628  else
2629  continue;
2630 
2631  /*
2632  * This is a parallel runway, measure the
2633  * distance to it from us.
2634  */
2635  othr_v = geo2fpp(GEO3_TO_GEO2(orwy_end->thr), fpp);
2636  v = vect2_sub(othr_v, thr_v);
2637  if (IS_ZERO_VECT2(v)) {
2638  logMsg("CAUTION: your nav DB is looking very "
2639  "strange: runways %s and %s at %s are on "
2640  "top of each other (coords: %fx%f)",
2641  end->id, orwy_end->id, rwy->arpt->icao,
2642  orwy_end->thr.lat, orwy_end->thr.lon);
2643  continue;
2644  }
2645  a = rel_hdg(dir2hdg(dir_v), dir2hdg(v));
2646  dist = fabs(sin(DEG2RAD(a)) * vect2_abs(v));
2647 
2648  if (a < 0)
2649  limit_left = MIN(dist / 2, limit_left);
2650  else
2651  limit_right = MIN(dist / 2, limit_right);
2652  }
2653  }
2654 
2655  if (limit_left < RWY_APCH_PROXIMITY_LAT_DISPL) {
2656  c1 = vect2vect_isect(vect2_sub(d, c), c, vect2_neg(dir_v),
2657  vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2658  limit_left)), B_FALSE);
2659  d = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_FALSE),
2660  limit_left));
2661  }
2662  if (limit_right < RWY_APCH_PROXIMITY_LAT_DISPL) {
2663  b1 = vect2vect_isect(vect2_sub(b, a), a, vect2_neg(dir_v),
2664  vect2_add(thr_v, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2665  limit_right)), B_FALSE);
2666  a = vect2_add(x, vect2_set_abs(vect2_norm(dir_v, B_TRUE),
2667  limit_right));
2668  }
2669 
2670  bbox[n_pts++] = a;
2671  if (!IS_NULL_VECT(b1))
2672  bbox[n_pts++] = b1;
2673  bbox[n_pts++] = b;
2674  bbox[n_pts++] = c;
2675  if (!IS_NULL_VECT(c1))
2676  bbox[n_pts++] = c1;
2677  bbox[n_pts++] = d;
2678 
2679  return (bbox);
2680 }

2682 /*
2683  * Prepares a runway's bounding box vector coordinates using the airport
2684  * coord fpp transform.
2685  */
2686 static void
2687 load_rwy_info(runway_t *rwy)
2688 {
2689  ASSERT(rwy != NULL);
2690  ASSERT(rwy->arpt->load_complete);
2691  /*
2692  * RAAS runway proximity entry bounding box is defined as:
2693  *
2694  * 1000ft 1000ft
2695  * |<======>| |<======>|
2696  * | | | |
2697  * ---- d +-------------------------------------------------+ c
2698  * 1.5x ^ | | | |
2699  * rwy | | | | |
2700  * width | | +-------------------------------+ |
2701  * v | | ==== ---- ---- ==== | |
2702  * -------|-thresh-x ==== - - - - - - - - - - ==== | |
2703  * ^ | | ==== ---- ---- ==== | |
2704  * 1.5x | | +-------------------------------+ |
2705  * rwy | | |
2706  * width v | |
2707  * ---- a +-------------------------------------------------+ b
2708  */
2709  vect2_t dt1v = geo2fpp(GEO3_TO_GEO2(rwy->ends[0].thr), &rwy->arpt->fpp);
2710  vect2_t dt2v = geo2fpp(GEO3_TO_GEO2(rwy->ends[1].thr), &rwy->arpt->fpp);
2711  double displ1 = rwy->ends[0].displ;
2712  double displ2 = rwy->ends[1].displ;
2713  double blast1 = rwy->ends[0].blast;
2714  double blast2 = rwy->ends[1].blast;
2715 
2716  vect2_t dir_v = vect2_sub(dt2v, dt1v);
2717  double dlen = vect2_abs(dir_v);
2718  double hdg1 = dir2hdg(dir_v);
2719  double hdg2 = dir2hdg(vect2_neg(dir_v));
2720 
2721  vect2_t t1v = vect2_add(dt1v, vect2_set_abs(dir_v, displ1));
2722  vect2_t t2v = vect2_add(dt2v, vect2_set_abs(vect2_neg(dir_v), displ2));
2723  double len = vect2_abs(vect2_sub(t2v, t1v));
2724 
2725  double prox_lon_bonus1 = MAX(displ1, RWY_PROXIMITY_LON_DISPL - displ1);
2726  double prox_lon_bonus2 = MAX(displ2, RWY_PROXIMITY_LON_DISPL - displ2);
2727 
2728  rwy->ends[0].thr_v = t1v;
2729  rwy->ends[1].thr_v = t2v;
2730  rwy->ends[0].dthr_v = dt1v;
2731  rwy->ends[1].dthr_v = dt2v;
2732  rwy->ends[0].hdg = hdg1;
2733  rwy->ends[1].hdg = hdg2;
2734  rwy->ends[0].land_len = vect2_abs(vect2_sub(dt2v, t1v));
2735  rwy->ends[1].land_len = vect2_abs(vect2_sub(dt1v, t2v));
2736  rwy->length = len;
2737 
2738  ASSERT(rwy->rwy_bbox == NULL);
2739 
2740  rwy->rwy_bbox = make_rwy_bbox(t1v, dir_v, rwy->width, len, 0);
2741  rwy->tora_bbox = make_rwy_bbox(dt1v, dir_v, rwy->width, dlen, 0);
2742  rwy->asda_bbox = make_rwy_bbox(dt1v, dir_v, rwy->width,
2743  dlen + blast2, blast1);
2744  rwy->prox_bbox = make_rwy_bbox(t1v, dir_v, RWY_PROXIMITY_LAT_FRACT *
2745  rwy->width, len + prox_lon_bonus2, prox_lon_bonus1);
2746 
2747  rwy->ends[0].apch_bbox = make_apch_prox_bbox(rwy, 0);
2748  rwy->ends[1].apch_bbox = make_apch_prox_bbox(rwy, 1);
2749 }
