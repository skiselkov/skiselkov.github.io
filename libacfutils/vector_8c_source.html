24static inline void
25vector_grow(vector_t REQ_PTR(v), size_t new_size)
26{
27 if (new_size <= v->cap) {
28 return;
29 }
30 v->cap = P2ROUNDUP(new_size);
31 v->buf = safe_realloc(v->buf, v->cap * sizeof (*v->buf));

100void
101vector_insert(vector_t REQ_PTR(v), void *elem, size_t index)
102{
103 VERIFY(elem != NULL);
104 VERIFY3U(index, <=, v->fill);
105 vector_grow(v, v->fill + 1);
106 memmove(&v->buf[index + 1], &v->buf[index],
107 (v->fill - index) * sizeof (*v->buf));
108 v->buf[index] = elem;
109 v->fill++;
110}

163size_t
164vector_shrink(vector_t REQ_PTR(v))
165{
166 if (v->fill == 0) {
167 if (v->cap != 0) {
168 free(v->buf);
169 v->buf = NULL;
170 v->cap = 0;
171 }
172 } else {
173 size_t new_cap = P2ROUNDUP(v->fill);
174 if (new_cap != v->cap) {
175 v->cap = new_cap;
176 v->buf = safe_realloc(v->buf,
177 v->cap * sizeof (*v->buf));
178 }
179 }
180 return (v->cap);
181}
