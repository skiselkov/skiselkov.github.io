35 /*
36  * Generates high-quality random numbers using the OS'es own PRNG. Returns
37  * B_TRUE iff `buf' has been successfully populated with `len' bytes.
38  * Use this to generate high-quality encryption key material.
39  *
40  * Be sparing with this function. High-quality random numbers are expensive
41  * and/or slow to generate in large quantities. For a low-quality but cheap
42  * PRNG, see crc64.c.
43  */
44 bool_t
45 osrand(void *buf, size_t len)
46 {
47 #if IBM
48  HCRYPTPROV prov;
49  /*
50  * First try the New-New RNG API. MS, will you ever stop making new
51  * APIs for old functionality?
52  */
53  if (BCryptGenRandom(NULL, buf, len, BCRYPT_USE_SYSTEM_PREFERRED_RNG) ==
54  STATUS_SUCCESS) {
55  return (B_TRUE);
56  }
57  /*
58  * If that fails, try the crypto interface.
59  */
60  if (!CryptAcquireContext(&prov, NULL, NULL, PROV_RSA_FULL, 0) &&
61  !CryptAcquireContext(&prov, NULL, NULL, PROV_RSA_FULL,
62  CRYPT_NEWKEYSET) &&
63  !CryptAcquireContext(&prov, NULL, NULL, PROV_RSA_FULL,
64  CRYPT_NEWKEYSET | CRYPT_MACHINE_KEYSET)) {
65  win_perror(GetLastError(), "Error generating random data: "
66  "error during CryptAcquireContext");
67  return (B_FALSE);
68  }
69  if (!CryptGenRandom(prov, len, buf)) {
70  win_perror(GetLastError(), "Error generating random data: "
71  "error during CryptGenRandom");
72  CryptReleaseContext(prov, 0);
73  return (B_FALSE);
74  }
75  CryptReleaseContext(prov, 0);
76  return (B_TRUE);
77 #else /* !IBM */
78  FILE *fp = fopen("/dev/random", "rb");
79 
80  if (fp == NULL) {
81  logMsg("Error reading /dev/random to generate random data: %s",
82  strerror(errno));
83  return (B_FALSE);
84  }
85  if (fread(buf, 1, len, fp) != len) {
86  logMsg("Error reading /dev/random to generate random data: %s",
87  strerror(errno));
88  fclose(fp);
89  return (B_FALSE);
90  }
91  fclose(fp);
92  return (B_TRUE);
93 #endif /* !IBM */
94 }
