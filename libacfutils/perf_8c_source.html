230  unsigned last_data_col, size_t offset)
231 {

235  ASSERT(table != NULL);

237  ASSERT3U(last_data_col, <, col);
238  ASSERT3U(offset, <, sizeof (perf_table_cell_t));

241  /*
242  * Single data element in table, can't extrapolate.
243  * So just copy the value.
244  */

246  return (*(double *)((void *)(cell1) + offset));

248

250  cell2 = &table->rows[table->num_alts - 1][last_data_col];

252  m1 = table->wts[last_data_col - 1];
253  v2 = *(double *)((void *)(cell2) + offset);
254  m2 = table->wts[last_data_col];
255  m = table->wts[col];
256 
257  return (fx_lin(m, m1, v1, m2, v2));

285 static bool_t
286 perf_table_parse(FILE *fp, perf_table_set_t *ts, unsigned num_eng,
287  double ff_corr, unsigned *line_num)
288 {
289  perf_table_isa_t *isa, srch_isa;
290  perf_table_t *table = safe_calloc(1, sizeof (*table));
291  char *line = NULL;
292  size_t line_cap = 0;
293  avl_index_t where;
294 
295  ASSERT(fp != NULL);
296  ASSERT(ts != NULL);
297  ASSERT(line_num != NULL);
298 
299  table->isa = 0;
300  table->ias = NAN;
301  table->mach = NAN;
302 
303  for (;;) {
304  ssize_t line_len =
305  parser_get_next_line(fp, &line, &line_cap, line_num);
306  size_t n_comps;
307  char **comps;
308  double alt;
309 
310  if (line_len == 0)
311  break;
312  comps = strsplit(line, " ", B_TRUE, &n_comps);
313  if (strcmp(comps[0], "ISA") == 0 && n_comps == 2) {
314  table->isa = atof(comps[1]);
315  } else if (strcmp(comps[0], "IAS") == 0 && n_comps == 2) {
316  table->ias = atof(comps[1]);
317  } else if (strcmp(comps[0], "KIAS") == 0 && n_comps == 2) {
318  table->ias = KT2MPS(atof(comps[1]));
319  } else if (strcmp(comps[0], "MACH") == 0 && n_comps == 2) {
320  table->mach = atof(comps[1]);
321  } else if (strcmp(comps[0], "GWLBK") == 0 && n_comps >= 2) {
322  table->num_wts = n_comps - 1;
323  table->wts = safe_calloc(table->num_wts,
324  sizeof (*table->wts));
325  for (size_t i = 1; i < n_comps; i++) {
326  table->wts[i - 1] =
327  LBS2KG(1000 * atof(comps[i]));
328  }
329  } else if (!isnan(alt = parse_table_alt(comps[0]))) {
330  table->num_alts++;
331  table->alts = realloc(table->alts, table->num_alts *
332  sizeof (*table->alts));
333  table->alts[table->num_alts - 1] = alt;
334  table->rows = realloc(table->rows, table->num_alts *
335  sizeof (*table->rows));
336  table->rows[table->num_alts - 1] = safe_calloc(
337  table->num_wts, sizeof (perf_table_cell_t));
338  } else if (strcmp(comps[0], "FPM") == 0) {
339  perf_table_cells_populate(comps, n_comps, table,
340  offsetof(perf_table_cell_t, vs),
341  FPM2MPS(1));
342  } else if (strcmp(comps[0], "TIMM") == 0) {
343  perf_table_cells_populate(comps, n_comps, table,
344  offsetof(perf_table_cell_t, fused_t), 60);
345  } else if (strcmp(comps[0], "FULB") == 0) {
346  perf_table_cells_populate(comps, n_comps, table,
347  offsetof(perf_table_cell_t, fused),
348  LBS2KG(1) * ff_corr);
349  } else if (strcmp(comps[0], "FFLB/ENG") == 0) {
350  perf_table_cells_populate(comps, n_comps, table,
351  offsetof(perf_table_cell_t, ff),
352  (LBS2KG(1) / SECS_PER_HR) * num_eng * ff_corr);
353  } else if (strcmp(comps[0], "ENDTABLE") == 0) {
354  free_strlist(comps, n_comps);
355  break;
356  } else {
357  free_strlist(comps, n_comps);
358  goto errout;
359  }
360  free_strlist(comps, n_comps);
361  }
362 
363  if ((isnan(table->ias) && isnan(table->mach)) || table->num_wts == 0 ||
364  table->num_alts == 0)
365  goto errout;
366 
367  /*
368  * For climb/descent tables, we need to calculate the immediate
369  * fuel flow. The table contains aggregate climb time & fuel use
370  * figures. So to compute local fuel flow, we first subtract the
371  * fuel use and time-to-reach from one altitude lower. This then
372  * gives the fuel use & time delta to go from the lower altitude
373  * bracket to the altitude being examined. It isn't super-duper
374  * accurate, but should be reasonably close to immediate FF.
375  */
376  for (unsigned i_alt = 0; i_alt < table->num_alts; i_alt++) {
377  for (unsigned i_wt = 0; i_wt < table->num_wts; i_wt++) {
378  perf_table_cell_t *cell = &table->rows[i_alt][i_wt];
379 
380  if (cell->fused_t == 0)
381  continue;
382  if (i_alt == 0) {
383  cell->ff = cell->fused / cell->fused_t;
384  } else {
385  perf_table_cell_t *subcell =
386  &table->rows[i_alt - 1][i_wt];
387  double fused, fused_t;
388 
389  ASSERT3F(subcell->fused, >, 0);
390  ASSERT3F(subcell->fused_t, >, 0);
391  fused = cell->fused - subcell->fused;
392  fused_t = cell->fused_t - subcell->fused_t;
393  cell->ff = fused / fused_t;
394  }
395  ASSERT_MSG(cell->ff >= 0, "Malformed table with "
396  "negative fuel flow: ISA=%.0f KIAS=%.0f Mach=%.2f "
397  "ALT=%.0fft WT=%.0flbs", table->isa,
398  MPS2KT(table->ias), table->mach,
399  MET2FEET(table->alts[i_alt]),
400  KG2LBS(table->wts[i_wt]));
401  }
402  }
403 
404  srch_isa.isa = table->isa;
405  isa = avl_find(&ts->by_isa, &srch_isa, &where);
406  if (isa == NULL) {
407  isa = safe_calloc(1, sizeof (*isa));
408  isa->isa = table->isa;
409  avl_create(&isa->by_ias, perf_tables_ias_compar,
410  sizeof (perf_table_t), offsetof(perf_table_t, ias_node));
411  avl_create(&isa->by_mach, perf_tables_mach_compar,
412  sizeof (perf_table_t), offsetof(perf_table_t, mach_node));
413  list_create(&isa->tables, sizeof (perf_table_t),
414  offsetof(perf_table_t, isa_node));
415  avl_insert(&ts->by_isa, isa, where);
416  }
417  if (!isnan(table->ias)) {
418  if (avl_find(&isa->by_ias, table, &where) != NULL) {
419  logMsg("Duplicate table for ISA %.1f/IAS %.1f",
420  table->isa, MPS2KT(table->ias));
421  goto errout;
422  }
423  avl_insert(&isa->by_ias, table, where);
424  }
425  if (!isnan(table->mach)) {
426  if (avl_find(&isa->by_mach, table, &where) != NULL) {
427  logMsg("Duplicate table for ISA %.1f/Mach %.3f",
428  table->isa, MPS2KT(table->mach));
429  goto errout;
430  }
431  avl_insert(&isa->by_mach, table, where);
432  }
433 
434  return (B_TRUE);
435 errout:
436  perf_table_free(table);
437  return (B_FALSE);
438 }

491 perf_tables_find_spd(perf_table_isa_t *isa, double spd, bool_t is_mach,
492  perf_table_t **t_min_p, perf_table_t **t_max_p)
493 {
494  avl_index_t where;
495  perf_table_t srch;
496  perf_table_t *t_min, *t_max;
497  avl_tree_t *tree;
498 
499  ASSERT(isa != NULL);
500  ASSERT(t_min_p != NULL);
501  ASSERT(t_max_p != NULL);
502 
503  if (is_mach) {
504  srch.mach = spd;
505  tree = &isa->by_mach;
506  } else {
507  srch.ias = spd;
508  tree = &isa->by_ias;
509  }
510  ASSERT(avl_numnodes(tree) != 0);
511  t_min = avl_find(tree, &srch, &where);
512 
513  if (t_min != NULL) {
514  t_max = t_min;
515  } else {
516  t_min = avl_nearest(tree, where, AVL_BEFORE);
517  t_max = avl_nearest(tree, where, AVL_AFTER);
518  ASSERT(t_min != NULL || t_max != NULL);
519  if (t_min == NULL) {
520  if (AVL_NEXT(tree, t_max) != NULL) {
521  t_min = t_max;
522  t_max = AVL_NEXT(tree, t_max);
523  } else {
524  t_min = t_max;
525  }
526  }
527  if (t_max == NULL) {
528  if (AVL_PREV(tree, t_min) != NULL) {
529  t_max = t_min;
530  t_min = AVL_PREV(tree, t_min);
531  } else {
532  t_max = t_min;
533  }
534  }
535  ASSERT(t_min != NULL);
536  ASSERT(t_max != NULL);
537  }
538 
539  *t_min_p = t_min;
540  *t_max_p = t_max;
541 }

543 static bool_t
544 perf_tables_find(perf_table_set_t *ts,
545  double isadev, double spd, bool_t is_mach,
546  perf_table_t **isa0_min, perf_table_t **isa0_max,
547  perf_table_t **isa1_min, perf_table_t **isa1_max)
548 {
549  avl_index_t where;
550  perf_table_isa_t srch = { .isa = isadev };
551  perf_table_isa_t *isa0, *isa1;
552 
553  ASSERT(ts != NULL);
554  ASSERT(isa0_min != NULL);
555  ASSERT(isa0_max != NULL);
556  ASSERT(isa1_min != NULL);
557  ASSERT(isa1_max != NULL);
558 
559  if (avl_numnodes(&ts->by_isa) == 0)
560  return (B_FALSE);
561 
562  isa0 = avl_find(&ts->by_isa, &srch, &where);
563  if (isa0 != NULL) {
564  /* Exact hit */
565  isa1 = isa0;
566  } else {
567  /*
568  * Try to find nearest two data points and interpolate, or
569  * even extrapolate from the nearest two data points.
570  */
571  isa0 = avl_nearest(&ts->by_isa, where, AVL_BEFORE);
572  isa1 = avl_nearest(&ts->by_isa, where, AVL_AFTER);
573  ASSERT(isa0 != NULL || isa1 != NULL);
574  /*
575  * We are at the edge of the data range. If we have more than
576  * one adjacent table to either side, we can try extrapolating.
577  */
578  if (isa0 == NULL) {
579  if (AVL_NEXT(&ts->by_isa, isa1) != NULL) {
580  isa0 = isa1;
581  isa1 = AVL_NEXT(&ts->by_isa, isa1);
582  } else {
583  isa0 = isa1;
584  }
585  }
586  if (isa1 == NULL) {
587  if (AVL_PREV(&ts->by_isa, isa0) != NULL) {
588  isa1 = isa0;
589  isa0 = AVL_PREV(&ts->by_isa, isa0);
590  } else {
591  isa1 = isa0;
592  }
593  }
594  ASSERT(isa0 != NULL);
595  ASSERT(isa1 != NULL);
596  }
597 
598  perf_tables_find_spd(isa0, spd, is_mach, isa0_min, isa0_max);
599  perf_tables_find_spd(isa1, spd, is_mach, isa1_min, isa1_max);
600 
601  return (B_FALSE);
602 }

604 static double
605 perf_table_lookup_row(perf_table_t *table, perf_table_cell_t *row,
606  double mass, size_t field_offset)
607 {
608  unsigned col1 = 0, col2 = 1;
609  double v1, v2, v;
610 
611  ASSERT(table != NULL);
612  ASSERT(table->num_wts != 0);
613  ASSERT(row != NULL);
614  ASSERT3U(field_offset, <, sizeof (perf_table_cell_t));
615 
616  /* clamp the mass to our tabulated range */
617  mass = clamp(mass, table->wts[0], table->wts[table->num_wts - 1] - 1);
618 
619  for (unsigned i = 0; i + 1 < table->num_wts; i++) {
620  if (mass >= table->wts[i] && mass <= table->wts[i + 1]) {
621  col1 = i;
622  col2 = i + 1;
623  break;
624  }
625  }
626 
627  v1 = *(double *)((void *)(&row[col1]) + field_offset);
628  v2 = *(double *)((void *)(&row[col2]) + field_offset);
629 
630  if (col1 != col2)
631  v = fx_lin(mass, table->wts[col1], v1, table->wts[col2], v2);

639 static double
640 perf_table_lookup(perf_table_t *table, double mass, double alt,
641  size_t field_offset)
642 {
643  unsigned row1 = UINT32_MAX, row2 = UINT32_MAX;
644  double row1_val, row2_val, value;
645 
646  ASSERT(table != NULL);
647  ASSERT3U(table->num_alts, >, 1);
648  ASSERT3U(field_offset, <, sizeof (perf_table_cell_t));
649 
650  /*
651  * If the requested altitude lies outside of our tabulated range,
652  * extrapolate to it from the nearest pair of rows.
653  */
654  if (alt > table->alts[0]) {
655  row1 = 0;
656  row2 = 1;
657  } else if (alt < table->alts[table->num_alts - 1]) {
658  row1 = table->num_alts - 2;
659  row2 = table->num_alts - 1;
660  } else {
661  for (unsigned i = 0; i + 1 < table->num_alts; i++) {
662  if (alt <= table->alts[i] &&
663  alt >= table->alts[i + 1]) {
664  row1 = i;
665  row2 = i + 1;
666  break;
667  }
668  }
669  }
670  ASSERT3U(row1 + 1, ==, row2);
671  ASSERT3U(row2, <, table->num_alts);
672 
673  row1_val = perf_table_lookup_row(table, table->rows[row1], mass,
674  field_offset);
675  row2_val = perf_table_lookup_row(table, table->rows[row2], mass,
676  field_offset);
677  value = fx_lin(alt, table->alts[row1], row1_val, table->alts[row2],
678  row2_val);
679  ASSERT(!isnan(value));
680 
681  return (value);
682 }

684 static double
685 table_lookup_common(perf_table_set_t *ts, double isadev, double mass,
686  double spd_mps_or_mach, bool_t is_mach, double alt, size_t offset)
687 {
688  perf_table_t *isa0_min = NULL, *isa0_max = NULL;
689  perf_table_t *isa1_min = NULL, *isa1_max = NULL;
690  double isa0_min_param, isa0_max_param, isa1_min_param, isa1_max_param;
691  double isa0_param, isa1_param, param;
692 
693  ASSERT(ts != NULL);
694  ASSERT3U(offset + sizeof (double), <=, sizeof (perf_table_cell_t));
695 
696  perf_tables_find(ts, isadev, spd_mps_or_mach, is_mach,
697  &isa0_min, &isa0_max, &isa1_min, &isa1_max);
698 
699  if (isa0_min != isa0_max) {
700  double x0 = (is_mach ? isa0_min->mach : isa0_min->ias);
701  double x1 = (is_mach ? isa0_max->mach : isa0_max->ias);
702  double rat = iter_fract(spd_mps_or_mach, x0, x1, B_FALSE);
703  isa0_min_param = perf_table_lookup(isa0_min, mass, alt, offset);
704  isa0_max_param = perf_table_lookup(isa0_max, mass, alt, offset);
705  /*
706  * We need to be careful about extrapolating speed estimates
707  * too much. There are drag-nonlinearities inherent in this,
708  * so we limit the estimator to reasonable ranges only.
709  */
710  isa0_param = wavg2(isa0_min_param, isa0_max_param,
711  clamp(rat, -0.25, 2));
712  ASSERT(!isnan(isa0_param));
713  } else {
714  isa0_param = perf_table_lookup(isa0_min, mass, alt, offset);
715  }
716  if (isa1_min != isa1_max) {
717  double x0 = (is_mach ? isa1_min->mach : isa1_min->ias);
718  double x1 = (is_mach ? isa1_max->mach : isa1_max->ias);
719  double rat = iter_fract(spd_mps_or_mach, x0, x1, B_FALSE);
720  isa1_min_param = perf_table_lookup(isa1_min, mass, alt, offset);
721  isa1_max_param = perf_table_lookup(isa1_max, mass, alt, offset);
722  isa1_param = wavg2(isa1_min_param, isa1_max_param,
723  clamp(rat, -0.25, 2));
724  ASSERT(!isnan(isa1_param));
725  } else {
726  isa1_param = perf_table_lookup(isa1_min, mass, alt, offset);
727  }
728 
729  if (isa0_param != isa1_param) {
730  double rat = clamp(iter_fract(isadev, isa0_min->isa,
731  isa1_min->isa, B_FALSE), -0.5, 1.5);
732  param = wavg2(isa0_param, isa1_param, rat);
733  } else {
734  param = isa0_param;
735  }
736 
737  return (param);

1023  * This takes into account atmospheric conditions as well as any currently
1024  * effective engine derates. Number of engines running is configured via
1025  * flt_perf->num_eng.
1026  *

1028  * @param acft Aircraft performance tables.
1029  * @param throttle Relative linear throttle position (0.0 to 1.0).
1030  * @param alt Altitude in feet.
1031  * @param ktas True air speed in knots.
1032  * @param qnh Barometric altimeter setting in hPa.
1033  * @param isadev ISA temperature deviation in degrees C.
1034  * @param tp_alt Altitude of the tropopause in feet.
1035  *
1036  * @return Maximum available engine thrust in Newtons.
1037  */
1038 double
1039 eng_get_thrust(const flt_perf_t *flt, const acft_perf_t *acft, double throttle,
1040  double alt, double ktas, double qnh, double isadev, double tp_alt)
1041 {

1043  unsigned num_eng;
1044  double min_thr, max_thr;
1045 
1046  ASSERT(flt != NULL);
1047  ASSERT(acft != NULL);
1048  ASSERT3F(throttle, >=, 0);
1049  ASSERT3F(throttle, <=, 1);

1056  dmod = D / ISA_SL_DENS;

1063  min_thr = num_eng * acft->eng_min_thr * dmod * mmod * flt->thr_derate;
1064

1250  *

1262  * mach_lim, it will be down-adjusted, otherwise it remains unchanged.

1264  * or target airspeeds exceed this value at the provided altitude,
1265  * we down-adjust both. This allows for continuously accelerating while
1266  * climbing until reaching the Mach limit and then slowly bleeding off

1273  * @param flt Flight performance settings.

1275  * covered during the acceleration phase. As such, it must be
1276  * pre-initialized.

1278  * seconds that the acceleration step can be allowed to happen. This is
1279  * then adjusted with the actual amount of time that the acceleration
1280  * was happening. If the target airspeed is greater than what can be
1281  * achieved during the acceleration step, timep is left unmodified

1283  * target speed is achievable, timep is modified to the proportion of
1284  * the time taken to accelerate to that speed. If the target speed is
1285  * less than the current speed, timep is set proportionally negative

1289  * fuel consumed during the acceleration phase. If no acceleration was
1290  * performed or speed was even given up for reuse for climbing, no
1291  * adjustment to this value will be made.
1292  *

1294  * to overcome drag and accelerate).

1296 static bool_t
1297 spd_chg_step(bool_t accel, double isadev, double tp_alt, double qnh, bool_t gnd,
1298  double alt, double *kcasp, double kcas_targ, double wind_mps, double mass,
1299  double flap_ratio, const acft_perf_t *acft, const flt_perf_t *flt,
1300  double *distp, double *timep, double *burnp)
1301 {

1303  double fl = alt2fl(alt, qnh);
1304  double Ps = alt2press(alt, qnh);
1305  double oat = isadev2sat(fl, isadev);
1306  double ktas_now = kcas2ktas(*kcasp, Ps, oat);
1307  double tas_now = KT2MPS(ktas_now);
1308  double tas_targ = KT2MPS(kcas2ktas(kcas_targ, Ps, oat));
1309  double Pd = dyn_press(ktas_now, Ps, oat);
1310  double throttle = accel ? 1.0 : 0.0;
1311  double thr = eng_get_thrust(flt, acft, throttle, alt, ktas_now, qnh,
1312  isadev, tp_alt);
1313  double burn = *burnp;
1314  double t = *timep;
1315  double altm = FEET2MET(alt);
1316 
1317  if (gnd) {
1318  aoa = 0;
1319  } else {
1320  aoa = get_aoa(Pd, mass, flap_ratio, acft);
1321  if (isnan(aoa))
1322  return (B_FALSE);
1323  }
1324  drag = get_drag(Pd, aoa, flap_ratio, acft);
1325  delta_v = (thr - drag) / mass;
1326  if (accel && delta_v < MIN_ACCEL)
1327  return (B_FALSE);
1328 
1329  tas_lim = tas_now + delta_v * t;

1332  E_targ = calc_total_E(mass, altm, tas_targ);
1333

1335  *kcasp = ktas2kcas(MPS2KT(tas_lim), Ps, oat);
1336  } else {
1337  t *= ((E_targ - E_now) / (E_lim - E_now));
1338  *kcasp = ktas2kcas(MPS2KT(tas_targ), Ps, oat);
1339  *timep = t;
1340  }
1341 
1342  if (t > 0) {
1343  burn += acft_get_sfc(flt, acft, thr, alt, ktas_now, qnh,
1344  isadev, tp_alt) * (t / SECS_PER_HR);
1345  }
1346 
1347  *burnp = burn;
1348  (*distp) += MET2NM(tas_now * t + 0.5 * delta_v * POW2(t) +
1349  wind_mps * t);
1350 
1351  return (B_TRUE);
1352 }

1652  * performance estimates, especially when constructing altitude-terminated
1653  * procedure legs. This function assumes the engines will be running at
1654  * maximum thrust during the climb/acceleration phase (subject to
1655  * environmental limitations and configured performance derates).
1656  *
1657  * @param flt Flight performance settings.
1658  * @param acft Aircraft performance tables.
1659  * @param isadev Estimated average ISA deviation during climb phase.
1660  * @param qnh Estimated average QNH during climb phase.
1661  * @param tp_alt Altitude of the tropopause in feet AMSL.
1662  * @param fuel Current fuel state in kg.
1663  * @param dir Unit vector pointing in the flight direction of the aircraft.
1664  * @param alt1 Climb/acceleration starting altitude in feet AMSL.
1665  * @param kcas1 Climb/acceleration starting calibrated airspeed in knots.
1666  * @param wind1 Wind vector at start of climb/acceleration phase. The
1667  * vector's direction expresses the wind direction and its magnitude
1668  * the wind's true speed in knots.
1669  * @param alt2 Climb/acceleration target altitude AMSL in feet. Must be
1670  * greater than or equal to alt1.
1671  * @param kcas2 Climb/acceleration target calibrated airspeed in knots.
1672  * @param wind2 Wind vector at end of climb/acceleration phase. The wind is
1673  * assumed to vary smoothly between alt1/wind1 and alt2/wind2.
1674  * @param flap_ratio Average flap setting between 0 and 1 (inclusive) during
1675  * climb/acceleration phase. This expresses how muct lift and drag is
1676  * produced by the wings as a fraction between CL/CLFLAP and CD/CDFLAP.
1677  * A setting of 0 means flaps up, whereas 1 is flaps fully extended.
1678  * @param type Type of acceleration/climb procedure to execute.
1679  * @param burnp Return parameter which if not NULL will be filled with the

1687  double qnh, double tp_alt, double accel_alt, double fuel, vect2_t dir,
1688  double alt1_ft, double kcas1, vect2_t wind1,

1703 
1704  /* Iterate in steps of `step'. */
1705  while (alt2_ft - alt > ALT_THRESH && kcas2 - kcas > KCAS_THRESH) {

1736  * factored into the distance traveled estimation below.

1756  * conditions are satisfied:

1759  * 3) our speed is within the airspeed target (acceleration

1768  spd, is_mach, flt->zfw + fuel - burn, wind_mps,
1769  step, &nalt, &nburn, &ndist)) {
1770  return (NAN);
1771  }
1772  alt = MET2FEET(nalt);
1773  burn += nburn;
1774  dist += MET2NM(ndist);
1775  clb_t = step - accel_t;
1776  if (is_mach)
1777  kcas = kcas_lim_mach;
1778  if (step_debug)
1779  table = B_TRUE;
1780  } else {
1781  if (accel_t > 0 && !spd_chg_step(B_TRUE, isadev, tp_alt,
1782  qnh, type == ACCEL_TAKEOFF && alt == alt1_ft, alt,
1783  &kcas, kcas_lim, wind_mps, flt->zfw + fuel - burn,
1784  flap_ratio_act, acft, flt, &dist, &accel_t,
1785  &burn)) {
1786  return (NAN);
1787  }
1788 
1789  clb_t = step - accel_t;
1790  if (clb_t > 0 && alt2_ft - alt > ALT_THRESH &&
1791  !alt_chg_step(B_TRUE, isadev, tp_alt, qnh, &alt,
1792  &vs, &kcas, alt2_ft, wind_mps,
1793  flt->zfw + fuel - burn, flap_ratio_act, acft, flt,
1794  &dist, &clb_t, &burn)) {
1795  return (NAN);
1796  }
1797  }
1798 
1799  if (step_debug) {
1800  double total_t;
1801 
1802  total_t = accel_t + clb_t;
1803  oat = isadev2sat(alt2fl(alt, qnh), isadev);
1804 
1805  printf("V:%3.0f KT +V:%5.02lf H:%5.0lf fpm:%4.0lf "
1806  "s:%6.0lf M:%5.03lf tab:%d\n", kcas,
1807  (kcas - old_kcas) / total_t, alt,
1808  ((alt - old_alt) / total_t) * 60, NM2MET(dist),
1809  ktas2mach(kcas2ktas(kcas, alt2press(alt, qnh), oat),
1810  oat), table);
1811  }
1812 
1813  iter_counter++;
1814  }
1815  if (burnp != NULL)
1816  *burnp = burn;
1817  if (kcas_out != NULL)
1818  *kcas_out = kcas;
1819 
1820  return (dist);
1821 }

2143  }
