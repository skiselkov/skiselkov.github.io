44 static void
45 worker(void *ui)
46 {
47  worker_t *wk = ui;
48  uint64_t now = microclock();
49 #if IBM
50  TIMECAPS tc;
51 #endif
52  thread_set_name(wk->name);
53  /*
54  * SIGPIPE is almost never desired on worker threads (typically due
55  * to writing to broken network sockets).
56  */
57  lacf_mask_sigpipe();
58 
59  if (wk->init_func != NULL) {
60  if (!wk->init_func(wk->userinfo))
61  return;
62  }
63 #if IBM
64  /*
65  * On Windows, the sleep interval is constrained to a multiple of
66  * the default clock tick. This is often something really crude,
67  * like 20ms or so. That is way too crude for our needs, so we'll
68  * drop the time tick to the minimum the hardware can support
69  * (usually 1ms).
70  */
71  timeGetDevCaps(&tc, sizeof (tc));
72  timeBeginPeriod(tc.wPeriodMin);
73 #endif /* IBM */
74 
75  mutex_enter(&wk->lock);
76  while (wk->run) {
77  bool_t result;
78  uint64_t intval_us = wk->intval_us;
79 
80  if (intval_us == 0 && !wk->dontstop) {
81  cv_wait(&wk->cv, &wk->lock);
82  if (!wk->run)
83  break;
84  now = microclock();
85  }
86  /*
87  * Avoid holding the worker lock in the user callback, as
88  * that can result in locking inversions if the worker needs
89  * to grab locks, while external threads might be holding
90  * those locks and trying to wake us up.
91  */
92  wk->inside_cb = B_TRUE;
93  mutex_exit(&wk->lock);
94  result = wk->worker_func(wk->userinfo);
95  mutex_enter(&wk->lock);
96  wk->inside_cb = B_FALSE;
97  if (!result)
98  break;
99  /*
100  * If another thread is waiting on us to finish executing,
101  * signal it.
102  */
103  cv_broadcast(&wk->cv);
104 
105  if (intval_us != 0 && !wk->dontstop) {
106  uint64_t new_now;
107 
108  cv_timedwait(&wk->cv, &wk->lock, now + intval_us);
109  /*
110  * If the timeout expired, we have waited for the
111  * full duration. So jump our idea of current time
112  * forward in increments of the interval. This
113  * maintains a fixed execution schedule, but allows
114  * for skipped intervals in case the callback took
115  * too long to execute.
116  */
117  new_now = microclock();
118  if (new_now >= now + intval_us) {
119  uint64_t d_t = new_now - now;
120  now += (d_t / intval_us) * intval_us;
121  }
122  }
123  wk->dontstop = B_FALSE;
124  }
125  mutex_exit(&wk->lock);
126 
127  if (wk->fini_func != NULL)
128  wk->fini_func(wk->userinfo);
129 
130 #if IBM
131  timeEndPeriod(tc.wPeriodMin);
132 #endif /* IBM */
133 }

167 void
168 worker_fini(worker_t *wk)
169 {
170  if (!wk->run)
171  return;
172 
173  /*
174  * Stop the thread before grabbing the lock, to shut it down
175  * while it is executing its callback.
176  */
177  wk->run = B_FALSE;
178 
179  mutex_enter(&wk->lock);
180  cv_broadcast(&wk->cv);
181  mutex_exit(&wk->lock);
182 
183  thread_join(&wk->thread);
184 
185  mutex_destroy(&wk->lock);
186  cv_destroy(&wk->cv);
187 }

190 worker_set_interval(worker_t *wk, uint64_t intval_us)
191 {
192  mutex_enter(&wk->lock);
193  /* If the worker is in the callback, wait for it to exit first */
194  while (wk->inside_cb)
195  cv_wait(&wk->cv, &wk->lock);
196  if (wk->intval_us != intval_us) {
197  wk->intval_us = intval_us;
198  cv_broadcast(&wk->cv);
199  }
