214 /*
215  * Given a module path in `filename' and a relative module address in `addr',
216  * attempts to resolve the symbol name and relative symbol address. This is
217  * done by looking for a syms.txt file in the same directory as the module's
218  * filename.
219  * If found, the symbol name + symbol relative address is placed into
220  * `symname' in the "symbol+offset" format.
221  * This function is deliberately designed to be simple and use as little
222  * memory as possible, because when called from an exception handler, the
223  * process' memory state can be assumed to be quite broken already.
224  */
225 static void
226 find_symbol(const char *filename, void *addr, char *symname,
227  size_t symname_cap)
228 {
229  /*
230  * Note the `static' here is deliberate to cause these to become
231  * BSS-allocated variables instead of stack-allocated. When parsing
232  * through a stack trace we are in a pretty precarious state, so we
233  * can't rely on having much stack space available.
234  */
235  static char symstxtname[MAX_PATH];
236  static char prevsym[SYMNAME_MAXLEN + 1];
237  static const char *sep;
238  static FILE *fp;
239  static void *prevptr = NULL;
240  static void *image_base = NULL;
241 
242  *symname = 0;
243  *prevsym = 0;
244 
245  sep = strrchr(filename, DIRSEP);
246  if (sep == NULL)
247  return;
248  lacf_strlcpy(symstxtname, filename, MIN((uintptr_t)(sep - filename) + 1,
249  sizeof (symstxtname)));
250  strncat(symstxtname, DIRSEP_S "syms.txt", sizeof (symstxtname));
251  fp = fopen(symstxtname, "rb");
252  if (fp == NULL)
253  return;
254 
255  while (!feof(fp)) {
256  static char unused_c;
257  static void *ptr;
258  static char sym[SYMNAME_MAXLEN + 1];
259 
260  if (fscanf(fp, "%p %c %" SCANF_STR_AUTOLEN(SYMNAME_MAXLEN) "s",
261  &ptr, &unused_c, sym) != 3) {
262  break;
263  }
264  if (strcmp(sym, "__image_base__") == 0) {
265  image_base = ptr;
266  continue;
267  }
268  ptr = (void *)(ptr - image_base);
269  if (addr >= prevptr && addr < ptr) {
270  snprintf(symname, symname_cap, "%s+%x", prevsym,
271  (unsigned)(addr - prevptr));
272  break;
273  }
274  prevptr = ptr;
275  lacf_strlcpy(prevsym, sym, sizeof (prevsym));
276  }
277  fclose(fp);
278 }

280 static HMODULE
281 find_module(LPVOID pc, DWORD64 *module_base)
282 {
283  static DWORD i;
284  for (i = 0; i < num_modules; i++) {
285  static LPVOID start, end;
286  start = mi[i].lpBaseOfDll;
287  end = start + mi[i].SizeOfImage;
288  if (start <= pc && end > pc) {
289  *module_base = (DWORD64)start;
290  return (modules[i]);
291  }
292  }
293  *module_base = 0;
294  return (NULL);
295 }

309 void
310 log_backtrace(int skip_frames)
311 {
312  static unsigned frames;
313  static void *stack[MAX_STACK_FRAMES];
314  static SYMBOL_INFO *symbol;
315  static HANDLE process;
316  static DWORD displacement;
317  static IMAGEHLP_LINE64 *line;
318  static char filename[MAX_PATH];
319 
320  frames = RtlCaptureStackBackTrace(skip_frames + 1, MAX_STACK_FRAMES,
321  stack, NULL);
322 
323  process = GetCurrentProcess();
324  mutex_enter(&backtrace_lock);
325 
326  SymInitialize(process, NULL, TRUE);
327  SymSetOptions(SYMOPT_LOAD_LINES);
328 
329  gather_module_info();
330 
331  memset(symbol_buf, 0, sizeof (symbol_buf));
332  memset(line_buf, 0, sizeof (line_buf));
333 
334  symbol = (SYMBOL_INFO *)symbol_buf;
335  symbol->MaxNameLen = MAX_SYM_NAME_LEN - 1;
336  symbol->SizeOfStruct = sizeof (SYMBOL_INFO);
337 
338  line = (IMAGEHLP_LINE64 *)line_buf;
339  line->SizeOfStruct = sizeof (*line);
340 
341  backtrace_buf[0] = '\0';
342  lacf_strlcpy(backtrace_buf, BACKTRACE_STR, sizeof (backtrace_buf));
343 
344  for (unsigned frame_nr = 0; frame_nr < frames; frame_nr++) {
345  static DWORD64 address;
346  static int fill;
347 
348  address = (DWORD64)(uintptr_t)stack[frame_nr];
349  fill = strlen(backtrace_buf);
350 
351  memset(symbol_buf, 0, sizeof (symbol_buf));
352  /*
353  * Try to grab the symbol name from the stored %rip data.
354  */
355  if (!SymFromAddr(process, address, 0, symbol)) {
356  static DWORD64 start;
357  static HMODULE module;
358 
359  module = find_module((void *)address, &start);
360  if (module != NULL) {
361  static char symname[SYMNAME_MAXLEN + 1];
362 
363  GetModuleFileNameA(module, filename,
364  sizeof (filename));
365  find_symbol(filename, stack[frame_nr] - start,
366  symname, sizeof (symname));
367  fill += snprintf(&backtrace_buf[fill],
368  sizeof (backtrace_buf) - fill,
369  "%d %p %s+%p (%s)\n", frame_nr,
370  stack[frame_nr], filename,
371  stack[frame_nr] - start, symname);
372  } else {
373  fill += snprintf(&backtrace_buf[fill],
374  sizeof (backtrace_buf) - fill,
375  "%d %p <unknown module>\n", frame_nr,
376  stack[frame_nr]);
377  }
378  continue;
379  }
380  /*
381  * See if we have debug info available with file names and
382  * line numbers.
383  */
384  if (SymGetLineFromAddr64(process, address, &displacement,
385  line)) {
386  snprintf(&backtrace_buf[fill], sizeof (backtrace_buf) -
387  fill, "%d: %s (0x%lx) [%s:%d]\n", frame_nr,
388  symbol->Name, (unsigned long)symbol->Address,
389  line->FileName, (int)line->LineNumber);
390  } else {
391  snprintf(&backtrace_buf[fill], sizeof (backtrace_buf) -
392  fill, "%d: %s - 0x%lx\n", frame_nr, symbol->Name,
393  (unsigned long)symbol->Address);
394  }
395  }
396 
397  if (log_func == NULL)
398  abort();
399  log_func(backtrace_buf);
400  fputs(backtrace_buf, stderr);
401  fflush(stderr);
402  SymCleanup(process);
403 
404  mutex_exit(&backtrace_lock);
405 }

407 void
408 log_backtrace_sw64(PCONTEXT ctx)
409 {
410  static char filename[MAX_PATH];
411  static DWORD64 pcs[MAX_STACK_FRAMES];
412  static unsigned num_stack_frames;
413  static STACKFRAME64 sf;
414  static HANDLE process, thread;
415  static DWORD machine;
416 
417  process = GetCurrentProcess();
418  thread = GetCurrentThread();
419 
420  mutex_enter(&backtrace_lock);
421 
422  SymInitialize(process, NULL, TRUE);
423  SymSetOptions(SYMOPT_LOAD_LINES);
424 
425  gather_module_info();
426 
427  memset(&sf, 0, sizeof (sf));
428  sf.AddrPC.Mode = AddrModeFlat;
429  sf.AddrStack.Mode = AddrModeFlat;
430  sf.AddrFrame.Mode = AddrModeFlat;
431 #if defined(_M_IX86)
432  machine = IMAGE_FILE_MACHINE_I386;
433  sf.AddrPC.Offset = ctx->Eip;
434  sf.AddrStack.Offset = ctx->Esp;
435  sf.AddrFrame.Offset = ctx->Ebp;
436 #elif defined(_M_X64)
437  machine = IMAGE_FILE_MACHINE_AMD64;
438  sf.AddrPC.Offset = ctx->Rip;
439  sf.AddrStack.Offset = ctx->Rsp;
440  sf.AddrFrame.Offset = ctx->Rbp;
441 #elif defined(_M_IA64)
442  machine = IMAGE_FILE_MACHINE_IA64;
443  sf.AddrPC.Offset = ctx->StIIP;
444  sf.AddrFrame.Offset = ctx->IntSp;
445  sf.AddrBStore.Offset = ctx->RsBSP;
446  sf.AddrBStore.Mode = AddrModeFlat;
447  sf.AddrStack.Offset = ctx->IntSp;
448 #else
449 #error "Unsupported architecture"
450 #endif /* _M_X64 */
451 
452  for (num_stack_frames = 0; num_stack_frames < MAX_STACK_FRAMES;
453  num_stack_frames++) {
454  if (!StackWalk64(machine, process, thread, &sf, ctx, NULL,
455  SymFunctionTableAccess64, SymGetModuleBase64, NULL)) {
456  break;
457  }
458  pcs[num_stack_frames] = sf.AddrPC.Offset;
459  }
460 
461  backtrace_buf[0] = '\0';
462  lacf_strlcpy(backtrace_buf, BACKTRACE_STR, sizeof (backtrace_buf));
463 
464  for (unsigned i = 0; i < num_stack_frames; i++) {
465  static int fill;
466  static DWORD64 pc;
467  static char symname[SYMNAME_MAXLEN + 1];
468  static HMODULE module;
469  static DWORD64 mbase;
470 
471  fill = strlen(backtrace_buf);
472  pc = pcs[i];
473 
474  module = find_module((LPVOID)pc, &mbase);
475  GetModuleFileNameA(module, filename, sizeof (filename));
476  find_symbol(filename, (void *)(pc - mbase),
477  symname, sizeof (symname));
478  fill += snprintf(&backtrace_buf[fill],
479  sizeof (backtrace_buf) - fill,
480  "%d %p %s+%p (%s)\n", i, (void *)pc, filename,
481  (void *)(pc - mbase), symname);
482  }
483 
484  if (log_func == NULL)
485  abort();
486  log_func(backtrace_buf);
487  fputs(backtrace_buf, stderr);
488  fflush(stderr);
489  SymCleanup(process);
490 
491  mutex_exit(&backtrace_lock);
492 }
