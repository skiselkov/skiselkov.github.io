138 /*
139  * Given a module path in `filename' and a relative module address in `addr',
140  * attempts to resolve the symbol name and relative symbol address. This is
141  * done by looking for a syms.txt file in the same directory as the module's
142  * filename.
143  * If found, the symbol name + symbol relative address is placed into
144  * `symname' in the "symbol+offset" format.
145  * This function is deliberately designed to be simple and use as little
146  * memory as possible, because when called from an exception handler, the
147  * process' memory state can be assumed to be quite broken already.
148  */
149 static void
150 find_symbol(const char *filename, void *addr, char *symname,
151  size_t symname_cap)
152 {
153  /*
154  * Note the `static' here is deliberate to cause these to become
155  * BSS-allocated variables instead of stack-allocated. When parsing
156  * through a stack trace we are in a pretty precarious state, so we
157  * can't rely on having much stack space available.
158  */
159  static char symstxtname[MAX_PATH];
160  static char prevsym[SYMNAME_MAXLEN + 1];
161  static const char *sep;
162  static FILE *fp;
163  static void *prevptr = NULL;
164  static void *image_base = NULL;
165 
166  *symname = 0;
167  *prevsym = 0;
168 
169  sep = strrchr(filename, DIRSEP);
170  if (sep == NULL)
171  return;
172  lacf_strlcpy(symstxtname, filename, MIN((uintptr_t)(sep - filename) + 1,
173  sizeof (symstxtname)));
174  strncat(symstxtname, DIRSEP_S "syms.txt", sizeof (symstxtname));
175  fp = fopen(symstxtname, "rb");
176  if (fp == NULL)
177  return;
178 
179  while (!feof(fp)) {
180  static char unused_c;
181  static void *ptr;
182  static char sym[SYMNAME_MAXLEN + 1];
183 
184  if (fscanf(fp, "%p %c %" SCANF_STR_AUTOLEN(SYMNAME_MAXLEN) "s",
185  &ptr, &unused_c, sym) != 3) {
186  break;
187  }
188  if (strcmp(sym, "__image_base__") == 0) {
189  image_base = ptr;
190  continue;
191  }
192  ptr = (void *)(ptr - image_base);
193  if (addr >= prevptr && addr < ptr) {
194  snprintf(symname, symname_cap, "%s+%x", prevsym,
195  (unsigned)(addr - prevptr));
196  break;
197  }
198  prevptr = ptr;
199  lacf_strlcpy(prevsym, sym, sizeof (prevsym));
200  }
201  fclose(fp);
202 }

204 static HMODULE
205 find_module(LPVOID pc, DWORD64 *module_base)
206 {
207  static DWORD i;
208  for (i = 0; i < num_modules; i++) {
209  static LPVOID start, end;
210  start = mi[i].lpBaseOfDll;
211  end = start + mi[i].SizeOfImage;
212  if (start <= pc && end > pc) {
213  *module_base = (DWORD64)start;
214  return (modules[i]);
215  }
216  }
217  *module_base = 0;
218  return (NULL);
219 }

233 void
234 log_backtrace(int skip_frames)
235 {
236  static unsigned frames;
237  static void *stack[MAX_STACK_FRAMES];
238  static SYMBOL_INFO *symbol;
239  static HANDLE process;
240  static DWORD displacement;
241  static IMAGEHLP_LINE64 *line;
242  static char filename[MAX_PATH];
243 
244  frames = RtlCaptureStackBackTrace(skip_frames + 1, MAX_STACK_FRAMES,
245  stack, NULL);
246 
247  process = GetCurrentProcess();
248  mutex_enter(&backtrace_lock);
249 
250  SymInitialize(process, NULL, TRUE);
251  SymSetOptions(SYMOPT_LOAD_LINES);
252 
253  gather_module_info();
254 
255  memset(symbol_buf, 0, sizeof (symbol_buf));
256  memset(line_buf, 0, sizeof (line_buf));
257 
258  symbol = (SYMBOL_INFO *)symbol_buf;
259  symbol->MaxNameLen = MAX_SYM_NAME_LEN - 1;
260  symbol->SizeOfStruct = sizeof (SYMBOL_INFO);
261 
262  line = (IMAGEHLP_LINE64 *)line_buf;
263  line->SizeOfStruct = sizeof (*line);
264 
265  backtrace_buf[0] = '\0';
266  lacf_strlcpy(backtrace_buf, BACKTRACE_STR, sizeof (backtrace_buf));
267 
268  for (unsigned frame_nr = 0; frame_nr < frames; frame_nr++) {
269  static DWORD64 address;
270  static int fill;
271 
272  address = (DWORD64)(uintptr_t)stack[frame_nr];
273  fill = strlen(backtrace_buf);
274 
275  memset(symbol_buf, 0, sizeof (symbol_buf));
276  /*
277  * Try to grab the symbol name from the stored %rip data.
278  */
279  if (!SymFromAddr(process, address, 0, symbol)) {
280  static DWORD64 start;
281  static HMODULE module;
282 
283  module = find_module((void *)address, &start);
284  if (module != NULL) {
285  static char symname[SYMNAME_MAXLEN + 1];
286 
287  GetModuleFileNameA(module, filename,
288  sizeof (filename));
289  find_symbol(filename, stack[frame_nr] - start,
290  symname, sizeof (symname));
291  fill += snprintf(&backtrace_buf[fill],
292  sizeof (backtrace_buf) - fill,
293  "%d %p %s+%p (%s)\n", frame_nr,
294  stack[frame_nr], filename,
295  stack[frame_nr] - start, symname);
296  } else {
297  fill += snprintf(&backtrace_buf[fill],
298  sizeof (backtrace_buf) - fill,
299  "%d %p <unknown module>\n", frame_nr,
300  stack[frame_nr]);
301  }
302  continue;
303  }
304  /*
305  * See if we have debug info available with file names and
306  * line numbers.
307  */
308  if (SymGetLineFromAddr64(process, address, &displacement,
309  line)) {
310  snprintf(&backtrace_buf[fill], sizeof (backtrace_buf) -
311  fill, "%d: %s (0x%lx) [%s:%d]\n", frame_nr,
312  symbol->Name, (unsigned long)symbol->Address,
313  line->FileName, (int)line->LineNumber);
314  } else {
315  snprintf(&backtrace_buf[fill], sizeof (backtrace_buf) -
316  fill, "%d: %s - 0x%lx\n", frame_nr, symbol->Name,
317  (unsigned long)symbol->Address);
318  }
319  }
320 
321  if (log_func == NULL)
322  abort();
323  log_func(backtrace_buf);
324  fputs(backtrace_buf, stderr);
325  fflush(stderr);
326  SymCleanup(process);
327 
328  mutex_exit(&backtrace_lock);
329 }

332 log_backtrace_sw64(PCONTEXT ctx)
333 {
334  static char filename[MAX_PATH];
335  static DWORD64 pcs[MAX_STACK_FRAMES];
336  static unsigned num_stack_frames;
337  static STACKFRAME64 sf;
338  static HANDLE process, thread;
339  static DWORD machine;
340 
341  process = GetCurrentProcess();
342  thread = GetCurrentThread();
343 
344  mutex_enter(&backtrace_lock);
345 
346  SymInitialize(process, NULL, TRUE);
347  SymSetOptions(SYMOPT_LOAD_LINES);
348 
349  gather_module_info();
350 
351  memset(&sf, 0, sizeof (sf));
352  sf.AddrPC.Mode = AddrModeFlat;
353  sf.AddrStack.Mode = AddrModeFlat;
354  sf.AddrFrame.Mode = AddrModeFlat;
355 #if defined(_M_IX86)
356  machine = IMAGE_FILE_MACHINE_I386;
357  sf.AddrPC.Offset = ctx->Eip;
358  sf.AddrStack.Offset = ctx->Esp;
359  sf.AddrFrame.Offset = ctx->Ebp;
360 #elif defined(_M_X64)
361  machine = IMAGE_FILE_MACHINE_AMD64;
362  sf.AddrPC.Offset = ctx->Rip;
363  sf.AddrStack.Offset = ctx->Rsp;
364  sf.AddrFrame.Offset = ctx->Rbp;
365 #elif defined(_M_IA64)
366  machine = IMAGE_FILE_MACHINE_IA64;
367  sf.AddrPC.Offset = ctx->StIIP;
368  sf.AddrFrame.Offset = ctx->IntSp;
369  sf.AddrBStore.Offset = ctx->RsBSP;
370  sf.AddrBStore.Mode = AddrModeFlat;
371  sf.AddrStack.Offset = ctx->IntSp;
372 #else
373 #error "Unsupported architecture"
374 #endif /* _M_X64 */
375 
376  for (num_stack_frames = 0; num_stack_frames < MAX_STACK_FRAMES;
377  num_stack_frames++) {
378  if (!StackWalk64(machine, process, thread, &sf, ctx, NULL,
379  SymFunctionTableAccess64, SymGetModuleBase64, NULL)) {
380  break;
381  }
382  pcs[num_stack_frames] = sf.AddrPC.Offset;
383  }
384 
385  backtrace_buf[0] = '\0';
386  lacf_strlcpy(backtrace_buf, BACKTRACE_STR, sizeof (backtrace_buf));
387 
388  for (unsigned i = 0; i < num_stack_frames; i++) {
389  static int fill;
390  static DWORD64 pc;
391  static char symname[SYMNAME_MAXLEN + 1];
392  static HMODULE module;
393  static DWORD64 mbase;
394 
395  fill = strlen(backtrace_buf);
396  pc = pcs[i];
397 
398  module = find_module((LPVOID)pc, &mbase);
399  GetModuleFileNameA(module, filename, sizeof (filename));
400  find_symbol(filename, (void *)(pc - mbase),
401  symname, sizeof (symname));
402  fill += snprintf(&backtrace_buf[fill],
403  sizeof (backtrace_buf) - fill,
404  "%d %p %s+%p (%s)\n", i, (void *)pc, filename,
405  (void *)(pc - mbase), symname);
406  }
407 
408  if (log_func == NULL)
409  abort();
410  log_func(backtrace_buf);
411  fputs(backtrace_buf, stderr);
412  fflush(stderr);
413  SymCleanup(process);
414 
415  mutex_exit(&backtrace_lock);
416 }
