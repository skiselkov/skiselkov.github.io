57 /*
58  * Basic portable multi-threading API. We have 3 kinds of objects and
59  * associated manipulation functions here:
60  * 1) threat_t - A generic thread identification structure.
61  * 2) mutex_t - A generic mutual exclusion lock.
62  * 3) condvar_t - A generic condition variable.
63  *
64  * The actual implementation is all contained in this header and simply
65  * works as a set of macro expansions on top of the OS-specific threading
66  * API (pthreads on Unix/Linux, winthreads on Win32).
67  *
68  * Example of how to create a thread:
69  * thread_t my_thread;
70  * if (!thread_create(&my_thread, thread_main_func, thread_arg))
71  * fprintf(stderr, "thread create failed!\n");
72  *
73  * Example of how to wait for a thread to exit:
74  * thread_t my_thread;
75  * thread_join(my_thread);
76  * ... thread disposed of, no need for further cleanup ...
77  *
78  * Example of how to use a mutex_t:
79  * mutex_t my_lock; -- the lock object itself
80  * mutex_init(&my_lock); -- create a lock
81  * mutex_enter(&my_lock); -- grab a lock
82  * ... do some critical, exclusiony-type stuff ...
83  * mutex_exit(&my_lock); -- release a lock
84  * mutex_destroy(&my_lock); -- free a lock
85  *
86  * Example of how to use a condvar_t:
87  * mutex_t my_lock;
88  * condvar_t my_cv;
89  *
90  * mutex_init(&my_lock); -- create a lock to control the CV
91  * cv_init(&my_cv); -- create the condition variable
92  *
93  * -- thread that's going to signal the condition:
94  * mutex_enter(&my_lock); -- grab the lock
95  * ... set up some resource that others might be waiting on ...
96  * cv_broadcast(&my_lock); -- wake up all waiters
97  * mutex_exit(&my_lock); -- release the lock
98  *
99  * -- thread that's going to wait on the condition:
100  * mutex_enter(&my_lock); -- grab the lock
101  * while (!condition_met())
102  * cv_wait(&my_cv, &my_lock); -- wait for the CV
103  * -- to be signalled
104  * ... condition fulfilled, use the resource ...
105  * mutex_exit(&my_lock); -- release the lock
106  *
107  * You can also performed a "timed" wait on a CV using cv_timedwait. The
108  * function will exit when either the condition has been signalled, or the
109  * timer has expired. The return value of the function indicates whether
110  * the condition was signalled before the timer expired (returns zero),
111  * or if the wait timed out (returns ETIMEDOUT) or another error occurred
112  * (returns -1).
113  *
114  * mutex_enter(&my_lock); -- grab the lock
115  * -- Wait for the CV to be signalled. Time argument is an
116  * -- absolute time as returned by the 'microclock' function +
117  * -- whatever extra time delay you want to apply.
118  * uint64_t deadline = microclock() + timeout_usecs;
119  * while (!condition_met()) {
120  * if (cv_timedwait(&my_cv, &my_lock, deadline) ==
121  * ETIMEDOUT) {
122  * -- timed out waiting for CV to signal
123  * break;
124  * }
125  * }
126  * mutex_exit(&my_lock); -- release the lock
127  */

288 static inline void
289 _lacf_thread_list_fini(void)
290 {
291  if (lacf_thread_list_inited) {
292  mutex_enter(&lacf_thread_list_lock);
293  if (list_count(&lacf_thread_list) == 0) {
294  list_destroy(&lacf_thread_list);
295  mutex_exit(&lacf_thread_list_lock);
296  mutex_destroy(&lacf_thread_list_lock);
297  lacf_thread_list_inited = B_FALSE;
298  } else {
299  mutex_exit(&lacf_thread_list_lock);
300  }
301  }

324 /*
325  * Checks to see if all threads that were created using thread_create
326  * have been properly disposed of. If not, this trips an assertion
327  * failure and lists all threads (including filenames and line numbers
328  * where they have been spawned) that weren't properly stopped. You
329  * should call this just as your plugin is exiting.
330  */
331 static inline void
332 lacf_threads_fini(void)
333 {
334  if (!lacf_thread_list_inited)
335  return;
336 
337  mutex_enter(&lacf_thread_list_lock);
338  for (lacf_thread_info_t *ti =
339  (lacf_thread_info_t *)list_head(&lacf_thread_list); ti != NULL;
340  ti = (lacf_thread_info_t *)list_next(&lacf_thread_list, ti)) {
341  logMsg("Leaked thread, created here %s:%d", ti->filename,
342  ti->linenum);
343  }
344  VERIFY0(list_count(&lacf_thread_list));
345  mutex_exit(&lacf_thread_list_lock);
346 
347  _lacf_thread_list_fini();
348 }

502 static inline int
503 cv_timedwait(condvar_t *cv, mutex_t *mtx, uint64_t limit)
504 {
505  uint64_t now = microclock();
506  if (now < limit) {
507  /*
508  * The only way to guarantee that when we return due to a
509  * timeout the full microsecond-accurate quantum has elapsed
510  * is to round-up to the nearest millisecond.
511  */
512  if (SleepConditionVariableCS(cv, &mtx->cs,
513  ceil((limit - now) / 1000.0)) != 0) {
514  return (0);
515  }
516  if (GetLastError() == ERROR_TIMEOUT)
517  return (ETIMEDOUT);
518  return (-1);
519  }
520  return (ETIMEDOUT);
521 }

570 static void
571 rwmutex_enter(rwmutex_t *rw, bool_t write)
572 {
573  mutex_enter(&rw->lock);
574 
575  if (rw->write_locked && rw->writer != curthread_id) {
576  rw->waiters++;
577  while (rw->write_locked)
578  cv_wait(&rw->cv, &rw->lock);
579  rw->waiters--;
580  }
581  if (write) {
582  if (rw->refcount > 0 && rw->writer != curthread_id) {
583  rw->waiters++;
584  while (rw->refcount > 0)
585  cv_wait(&rw->cv, &rw->lock);
586  rw->waiters--;
587  }
588  rw->write_locked = B_TRUE;
589  rw->writer = curthread_id;
590  }
591  rw->refcount++;
592 
593  mutex_exit(&rw->lock);
594 }
