883 static inline int
884 cv_timedwait(condvar_t *cv, mutex_t *mtx, uint64_t limit)
885 {
886  uint64_t now = microclock();
887  if (now < limit) {
888  /*

895  return (0);

898  return (ETIMEDOUT);
899  return (-1);
900  }

902 }

1128 UNUSED_ATTR static void
1129 rwmutex_enter(rwmutex_t *rw, bool_t write)
1130 {
1131  rwlock_waiter_t wt = {0};
1132 
1133  wt.write = write;
1134  /*
1135  * No recursion allowed! We can't check for recursive read attempts,
1136  * only write (since readers don't retain any ownership information),
1137  * so it's best to avoid recursion altogether.
1138  */
1139  ASSERT_MSG(!rwmutex_held_write(rw), "%s", "Attempted to recursively "
1140  "acquire an rwmutex_t. This is NOT supported!");
1141 
1142  mutex_enter(&rw->lock);
1143  /*
1144  * Enter the queue of threads waiting to acquire the mutex.
1145  */
1146  list_insert_tail(&rw->waiters, &wt);
1147 
1148  if (write) {
1149  /*
1150  * Wait until everybody else is out of the mutex
1151  * and we're next to enter.
1152  */
1153  while (rw->refcount != 0 ||
1154  list_head(&rw->waiters) != (void *)&wt) {
1155  cv_wait(&rw->cv, &rw->lock);
1156  }
1157  /*
1158  * We're clear to proceed, mark the mutex as
1159  * write-locked by us.
1160  */
1161  rw->writer = curthread_id;
1162  rw->write_locked = B_TRUE;
1163  } else {
1164  /*
1165  * If the mutex is currently held by a writer, or
1166  * there's another writer ahead of us, wait.
1167  */
1168  while (rw->write_locked ||
1169  !rwmutex_can_enter_impl(rw, &wt)) {
1170  cv_wait(&rw->cv, &rw->lock);
1171  }
1172  }
1173  /*
1174  * Exit the wait queue. We've now acquired the mutex.
1175  */
1176  list_remove(&rw->waiters, &wt);
1177  rw->refcount++;
1178 
1179  mutex_exit(&rw->lock);
1180 }

1187 {
