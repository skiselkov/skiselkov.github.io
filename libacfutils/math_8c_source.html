26 /*
27  * Solves quadratic equation ax^2 + bx + c = 0. Solutions are placed in 'x'.
28  * Returns the number of solutions (0, 1 or 2).
29  */
30 unsigned
31 quadratic_solve(double a, double b, double c, double x[2])
32 {
33  double tmp;
34 
35  /* Actually just a linear equation. */
36  if (a == 0) {
37  if (b == 0)
38  return (0);
39  x[0] = -c / b;
40  return (1);
41  }
42 
43  tmp = POW2(b) - 4 * a * c;
44  if (tmp > ROUND_ERROR) {
45  double tmp_sqrt = sqrt(tmp);
46  x[0] = (-b + tmp_sqrt) / (2 * a);
47  x[1] = (-b - tmp_sqrt) / (2 * a);
48  return (2);
49  } else if (tmp > -ROUND_ERROR) {
50  x[0] = -b / (2 * a);
51  return (1);
52  } else {
53  return (0);
54  }
55 }

57 /*
58  * Interpolates a linear function defined by two points.
59  *
60  * @param x Point who's 'y' value we're looking for on the function.
61  * @param x1 First reference point's x coordinate.
62  * @param y1 First reference point's y coordinate.
63  * @param x2 Second reference point's x coordinate.
64  * @param y2 Second reference point's y coordinate.
65  */
66 double
67 fx_lin(double x, double x1, double y1, double x2, double y2)
68 {
69  ASSERT3F(x1, !=, x2);
70  return (((x - x1) / (x2 - x1)) * (y2 - y1) + y1);
71 }

73 /*
74  * Multi-segment version of fx_lin. The segments are defined as a series of
75  * x-y coordinate points (list terminated with a NULL_VECT2). The list must
76  * contain AT LEAST 2 points. The value of 'x' is then computed using the
77  * fx_lin function from the appropriate segment. If 'x' falls outside of the
78  * curve range, the `extrapolate' argument controls behavior. If extrapolate
79  * is B_TRUE, the nearest segment is extrapolated to the value of 'x'.
80  * Otherwise the function returns NAN.
81  */
82 double
83 fx_lin_multi(double x, const struct vect2_s *points, bool_t extrapolate)
84 {
85  ASSERT(!IS_NULL_VECT(points[0]));
86  ASSERT(!IS_NULL_VECT(points[1]));
87 
88  for (;;) {
89  vect2_t p1 = points[0], p2 = points[1];
90 
91  ASSERT3F(p1.x, <, p2.x);
92 
93  if (x < p1.x) {
94  /* X outside of range to the left */
95  if (extrapolate)
96  return (fx_lin(x, p1.x, p1.y, p2.x, p2.y));
97  break;
98  }
99  /* X in range of current segment */
100  if (x <= p2.x)
101  return (fx_lin(x, p1.x, p1.y, p2.x, p2.y));
102  /* X outside of range to the right */
103  if (IS_NULL_VECT(points[2])) {
104  if (extrapolate)
105  return (fx_lin(x, p1.x, p1.y, p2.x, p2.y));
106  break;
107  }
108 
109  points++;
110  }
111 
112  return (NAN);
113 }

115 double *
116 fx_lin_multi_inv2(double y, const struct vect2_s *points, bool_t extrapolate,
117  size_t *num_out)
118 {
119  double *out = NULL;
120  size_t cap = 0, num = 0;
121 
122  ASSERT(points != NULL);
123  ASSERT(num_out != NULL);
124  ASSERT(!IS_NULL_VECT(points[0]));
125  ASSERT(!IS_NULL_VECT(points[1]));
126 
127  for (int i = 0; !IS_NULL_VECT(points[i + 1]); i++) {
128  vect2_t p1 = points[i], p2 = points[i + 1];
129  double min_val = MIN(p1.y, p2.y);
130  double max_val = MAX(p1.y, p2.y);
131 
132  if (min_val <= y && y <= max_val)
133  cap++;
134  }
135  if (extrapolate)
136  cap += 2;
137  if (cap == 0) {
138  *num_out = 0;
139  return (NULL);
140  }
141 
142  out = safe_calloc(cap, sizeof (*out));
143  for (int i = 0; !IS_NULL_VECT(points[i + 1]); i++) {
144  vect2_t p1 = points[i], p2 = points[i + 1];
145  double min_val = MIN(p1.y, p2.y);
146  double max_val = MAX(p1.y, p2.y);
147 
148  if (extrapolate) {
149  bool_t first = (i == 0);
150  bool_t up_slope = (p1.y <= p2.y);
151  if (first && ((up_slope && y < p1.y) ||
152  (!up_slope && y > p1.y)) && p1.y != p2.y) {
153  out[num++] = fx_lin(y, p1.y, p1.x, p2.y, p2.x);
154  }
155  }
156  if (min_val <= y && y <= max_val)
157  out[num++] = fx_lin(y, p1.y, p1.x, p2.y, p2.x);
158  if (extrapolate) {
159  bool_t last = IS_NULL_VECT(points[i + 2]);
160  bool_t up_slope = (p1.y <= p2.y);
161  if (last && ((up_slope && y > p2.y) ||
162  (!up_slope && y < p2.y)) && p1.y != p2.y) {
163  out[num++] = fx_lin(y, p1.y, p1.x, p2.y, p2.x);
164  }
165  }
166  }
167  /* We might have slightly over-accounted here if extrapolate was set */
168  *num_out = num;
169 
170  return (out);
171 }

179 /*
180  * Algorithm credit: https://en.wikibooks.org/wiki/\
181  * Algorithm_Implementation/Mathematics/Polynomial_interpolation
182  */
183 void
184 pn_interp_init(pn_interp_t *interp, const vect2_t *points, unsigned numpts)
185 {
186  ASSERT(interp != NULL);
187  ASSERT(points != 0);
188  ASSERT(numpts != 0);
189  ASSERT3U(numpts, <=, MAX_PN_INTERP_ORDER);
190 
191  memset(interp, 0, sizeof (*interp));
192  interp->order = numpts;
193 
194  for (unsigned i = 0; i < numpts; i++) {
195  double terms[MAX_PN_INTERP_ORDER] = { 0 };
196  double product = 1.0;
197 
198  /* Compute Prod_{j != i} (x_i - x_j) */
199  for (unsigned j = 0; j < numpts; j++) {
200  if (i == j)
201  continue;
202  product *= points[i].x - points[j].x;
203  }
204  /* Compute y_i/Prod_{j != i} (x_i - x_j) */
205  product = points[i].y / product;
206  terms[0] = product;
207  /* Compute theterm := product * Prod_{j != i} (x - x_j) */
208  for (unsigned j = 0; j < numpts; j++) {
209  if (i == j)
210  continue;
211  for (int k = numpts - 1; k > 0; k--) {
212  terms[k] += terms[k-1];
213  terms[k-1] *= -points[j].x;
214  }
215  }
216  /* coeff += terms (as coeff vectors) */
217  for (unsigned j = 0; j < numpts; j++)
218  interp->coeff[j] += terms[j];
219  }
220 }
