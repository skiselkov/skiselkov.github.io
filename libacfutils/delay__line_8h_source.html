43 /*
44  * Implements a generic variable that changes after a short delay.
45  * You need to initialize the variable using delay_line_init. Subsequently,
46  * you can push changes to it and read the current value. When a change is
47  * made, it is propagated into the variable after the delay with which the
48  * variable was initialized.
49  */
50 typedef struct {
51  union {
52  int64_t i64;
53  uint64_t u64;
54  double f64;
55  };
56  union {
57  int64_t i64_new;
58  uint64_t u64_new;
59  double f64_new;
60  };
61  uint64_t changed_t;
62  uint64_t delay_us;
63  uint64_t delay_base_us;
64  double delay_rand_fract;
65 #ifndef _MSC_VER
66  int __serialize_marker[0];
67 #else
68  /* MSVC really hates a zero-length array */
69  int __serialize_marker[1];
70 #endif
71  delay_line_time_func_t time_func;
72  void *time_func_userinfo;
73 } delay_line_t;

104 static inline void
105 delay_line_refresh_delay(delay_line_t *line)
106 {
107  ASSERT(line != NULL);
108  if (line->delay_rand_fract == 0) {
109  line->delay_us = line->delay_base_us;
110  } else {
111  uint64_t rand_us = line->delay_rand_fract * line->delay_base_us;
112  line->delay_us = line->delay_base_us +
113  (uint64_t)((crc64_rand_fract() - 0.5) * rand_us);
114  }
115 }

155 /*
156  * Functions to pull the current value from a delay line:
157  * delay_line_pull_i64 - reads the delay line as an int64_t
158  * delay_line_pull_u64 - reads the delay line as a uint64_t
159  * delay_line_pull_f64 - reads the delay line as a double
160  * If a new value has been pushed to the delay line, these functions
161  * keep returning the old value until `delay_us' microseconds have
162  * elapsed, after which they start returning the new value.
163  */
164 #define DEF_DELAY_LINE_PULL(typename, abbrev_type) \
165 static inline typename \
166 delay_line_pull_ ## abbrev_type(delay_line_t *line) \
167 { \
168  uint64_t now; \
169  ASSERT(line != NULL); \
170  now = (line->time_func != NULL ? \
171  line->time_func(line->time_func_userinfo) : microclock()); \
172  if (line->abbrev_type ## _new != line->abbrev_type && \
173  now - line->changed_t >= line->delay_us) { \
174  line->abbrev_type = line->abbrev_type ## _new; \
175  delay_line_refresh_delay(line); \
176  } \
177  return (line->abbrev_type); \
178 }

216 /*
217  * Functions that push a new value to a delay line:
218  * delay_line_push_i64 - pushes an int64_t to the delay line
219  * delay_line_push_u64 - pushes a uint64_t to the delay line
220  * delay_line_push_f64 - pushes a double to the delay line
221  * In each case, the current value of the delay line is returned
222  * (equivalent to calling `delay_line_pull_*'). If the new value is
223  * different from the current value of the delay line, the new value
224  * will become the delay line's current value after `delay_us' microsecs.
225  */
226 #define DEF_DELAY_LINE_PUSH(typename, abbrev_type) \
227 static inline typename \
228 delay_line_push_ ## abbrev_type(delay_line_t *line, typename value) \
229 { \
230  uint64_t now; \
231  ASSERT(line != NULL); \
232  now = (line->time_func != NULL ? \
233  line->time_func(line->time_func_userinfo) : microclock()); \
234  if (line->abbrev_type == line->abbrev_type ## _new && \
235  value != line->abbrev_type ## _new) { \
236  line->changed_t = now; \
237  delay_line_refresh_delay(line); \
238  } \
239  line->abbrev_type ## _new = value; \
240  return (delay_line_pull_ ## abbrev_type(line)); \
241 }
242 DEF_DELAY_LINE_PUSH(int64_t, i64)
243 DEF_DELAY_LINE_PUSH(uint64_t, u64)
244 DEF_DELAY_LINE_PUSH(double, f64)

301 /*
302  * CAUTION: do NOT use this for precise interval timing. Delay lines can be
303  * used as simple timed triggers, but they don't keep time accurately, or
304  * account for triggering-overshoot. If you try this, your clock will end
305  * up slipping and running too slow!
306  */
307 static inline uint64_t
308 delay_line_get_time_since_change(const delay_line_t *line)
309 {
310  uint64_t now;
311  ASSERT(line != NULL);
312  now = (line->time_func != NULL ?
313  line->time_func(line->time_func_userinfo) : microclock());
314  return (now - line->changed_t);
315 }
