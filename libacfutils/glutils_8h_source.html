143 /*
144  * The infrastructure below is for debugging GPU VRAM memory leaks.
145  *
146  * At plugin load time (in XPluginStart and before doing any libacfutils
147  * calls that might generate OpenGL calls), you must first initialize the
148  * system using a call to glutils_texsz_init(). At plugin exit time, and
149  * after having torn down all resources, call glutils_texsz_fini(). This
150  * collects all garbage and crashes the app with diagnostic information
151  * in case any leaks have been detected.
152  *
153  * Each allocation can be tracked in a two-level hierarchy:
154  * - using a symbolic token name
155  * - each token can track allocations to a particular anonymous pointer
156  * (plus filename:line where it occurred)
157  *
158  * The tokens are used to identify large blocks of functionality. You'd
159  * use a token for, for example, "efis_textures" or "custom_drawing_pbo",
160  * etc. These must be declared ahead at the top of each module file using
161  * the TEXSZ_MK_TOKEN macro, for example:
162  *
163  * TEXSZ_MK_TOKEN(efis_textures);
164  *
165  * Don't put spaces into the token name, the name must be a valid C
166  * identifier. You can subsequently track allocations to this token using
167  * the TEXSZ_ALLOC and TEXSZ_FREE macros. These macros take 5 arguments:
168  * 1) the token name
169  * 2) the texture format (e.g. GL_RGBA)
170  * 3) the texture data type (e.g. GL_UNSIGNED_BYTE)
171  * 4) texture width
172  * 5) texture height
173  * They then either increment or decrement the token byte counter. If a
174  * token is non-zero at the time glutils_texsz_fini() is called, the
175  * offending token name(s) are printed in sequence, with the amount of
176  * bytes leaked in them. In this mode, no filenames or line numbers are
177  * printed.
178  *
179  * Please note that TEXSZ_MK_TOKEN creates a static (single-module) token
180  * that cannot be shared between multiple C files. If you plan on using a
181  * TEXSZ token from more than one C/C++ file, you must declare the token
182  * in a header file using the TEXSZ_DECL_TOKEN_GLOB macro instead. Then
183  * define the token in a single C file using TEXSZ_DEF_TOKEN_GLOB:
184  *
185  * TEXSZ_DECL_TOKEN_GLOB(efis_textures); <--- goes in a header file
186  *
187  * TEXSZ_DEF_TOKEN_GLOB(efis_textures); <--- goes in a single C file
188  *
189  * In case you are using a generic facility (e.g. a picture loader) to
190  * provide texturing service to other parts of the code, a simple token
191  * name might not be specific enough to pinpoint the offending allocation.
192  * In that case, you can use the TEXSZ_ALLOC_INSTANCE and
193  * TEXSZ_FREE_INSTANCE macros to generate per-pointer statistics. The
194  * TEX_ALLOC_INSTANCE macro takes two additional parameters over TEXSZ_ALLOC:
195  * 1) An instance pointer - this is used to discriminate individual
196  * allocations. Usually you'd want to pass a containing structure
197  * pointer or something similar in here.
198  * 2) An allocation point filename.
199  * 3) An allocation point line number.
200  * The filename should be shortened at build time using log_backtrace()
201  * to only contain the last portion of the filename of the call site.
202  * You should wrap your functions which use TEXSZ_ALLOC_INSTANCE into a
203  * macro and extract the call site information automatically using the
204  * __FILE__ and __LINE__ built-in pre-processor variables (see the
205  * glutils_init_*_quads functions above for an example on how to do that).
206  * With instancing, glutils_texsz_fini() will print a list of leaked
207  * instance pointers and call sites in your code where that resource was
208  * leaked.
209  *
210  * In addition to the TEXSZ_ALLOC_* and TEXSZ_FREE_* macros, there are
211  * variations of these macros with the "_BYTES" suffix. These let you
212  * pass the raw byte size of the allocation directly, instead of having
213  * to pass texture formats and sizes. Use these when the data isn't a
214  * texture, but instead something more generic, like a vertex buffer.
215  */
216 
217 #define TEXSZ_MK_TOKEN(name) \
218  static const char *__texsz_token_ ## name = #name
219 #define TEXSZ_DECL_TOKEN_GLOB(name) \
220  extern const char *__texsz_token_ ## name
221 #define TEXSZ_DEF_TOKEN_GLOB(name) \
222  const char *__texsz_token_ ## name = #name
223 #define TEXSZ_ALLOC(__token_id, __format, __type, __w, __h) \
224  TEXSZ_ALLOC_INSTANCE(__token_id, NULL, NULL, -1, (__format), \
225  (__type), (__w), (__h))
226 #define TEXSZ_FREE(__token_id, __format, __type, __w, __h) \
227  TEXSZ_FREE_INSTANCE(__token_id, NULL, (__format), \
228  (__type), (__w), (__h))
229 #define TEXSZ_ALLOC_INSTANCE(__token_id, __instance, __filename, __line, \
230  __format, __type, __w, __h) \
231  glutils_texsz_alloc(__texsz_token_ ## __token_id, (__instance), \
232  (__filename), (__line), (__format), (__type), (__w), (__h))
233 #define TEXSZ_FREE_INSTANCE(__token_id, __instance, __format, __type, \
234  __w, __h) \
235  glutils_texsz_free(__texsz_token_ ## __token_id, (__instance), \
236  (__format), (__type), (__w), (__h))
237 #define TEXSZ_ALLOC_BYTES(__token_id, __bytes) \
238  TEXSZ_ALLOC_BYTES_INSTANCE(__token_id, NULL, NULL, -1, (__bytes))
239 #define TEXSZ_FREE_BYTES(__token_id, __bytes) \
240  TEXSZ_FREE_BYTES_INSTANCE(__token_id, NULL, (__bytes))
241 #define TEXSZ_ALLOC_BYTES_INSTANCE(__token_id, __instance, __filename, \
242  __line, __bytes) \
243  glutils_texsz_alloc_bytes(__texsz_token_ ## __token_id, (__instance), \
244  (__filename), (__line), (__bytes))
245 #define TEXSZ_FREE_BYTES_INSTANCE(__token_id, __instance, __bytes) \

314 static inline void
315 glutils_debug_push(GLuint msgid, const char *format, ...)
316 {
317  char buf_stack[128];
318  int len;
319  va_list ap;
320 
321  va_start(ap, format);
322  len = vsnprintf(buf_stack, sizeof (buf_stack), format, ap);
323  va_end(ap);
324 
325  if (len >= (int)sizeof (buf_stack)) {
326  char *buf_heap = (char *)safe_malloc(len + 1);
327 
328  va_start(ap, format);
329  vsnprintf(buf_heap, len + 1, format, ap);
330  va_end(ap);
331 
332  glPushDebugGroup(GL_DEBUG_SOURCE_APPLICATION, msgid, len,
333  buf_heap);
334 
335  free(buf_heap);
336  } else {
337  glPushDebugGroup(GL_DEBUG_SOURCE_APPLICATION, msgid, len,
338  buf_stack);
339  }
340 }
