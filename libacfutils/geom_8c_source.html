45 /*
46  * Naive implementation of matrix multiplication. We don't use this very
47  * heavily and can thus afford to rely on compiler auto-vectorization to
48  * get this optimized.
49  * Multiplies `x' and `y' and places result in 'z'. `xrows', `ycols' and `sz'
50  * have meanings as explained below:
51  * sz ycols ycols
52  * | | | | | |
53  * | | | | | |
54  *
55  * --- ++= =++ .-++= =++ ++= =++ --
56  * xrows ^ || x00 x01 || \/ / || y00 y01 || --- || z00 z01 || ^ xrows
57  * v || x10 x11 || /\ / || y10 y11 || --- || z10 z11 || v
58  * --- ++= =++ / .-++= =++ ++= =++ --
59  * --' /
60  * sz ^ /
61  * v /
62  * --'
63  */
64 static void
65 matrix_mul(const double *x, const double *y, double *z,
66  size_t xrows, size_t ycols, size_t sz)
67 {
68  memset(z, 0, sz * ycols * sizeof (double));
69  for (size_t row = 0; row < xrows; row++) {
70  for (size_t col = 0; col < ycols; col++) {
71  for (size_t i = 0; i < sz; i++) {
72  z[row * ycols + col] += x[row * sz + i] *
73  y[i * ycols + col];
74  }
75  }
76  }
77 }

660 vect3_t
661 sph2ecef(geo_pos3_t pos)
662 {
663  vect3_t result;
664  double lat_rad, lon_rad, R, R0;
665 
666  lat_rad = DEG2RAD(pos.lat);
667  lon_rad = DEG2RAD(pos.lon);
668 
669  /* R is total distance from center at alt_msl */
670  R = pos.elev + EARTH_MSL;
671  /*
672  * R0 is the radius of a circular cut parallel to the equator at the
673  * given latitude of a sphere with radius R.
674  */
675  R0 = R * cos(lat_rad);
676  /* Given R and R0, we can transform the geo coords into 3-space: */
677  result.x = R0 * cos(lon_rad);
678  result.y = R0 * sin(lon_rad);
679  result.z = R * sin(lat_rad);
680 
681  return (result);
682 }

828 geo_pos3_t
829 geo2sph(geo_pos3_t pos, const ellip_t *ellip)
830 {
831  double lat_r = DEG2RAD(pos.lat);
832  double sin_lat = sin(lat_r);
833  double p, z;
834  double Rc; /* curvature of the prime vertical */
835  geo_pos3_t res;
836 
837  Rc = ellip->a / sqrt(1 - ellip->ecc2 * POW2(sin_lat));
838  p = (Rc + pos.elev) * cos(lat_r);
839  z = ((Rc * (1 - ellip->ecc2)) + pos.elev) * sin_lat;
840 
841  res.elev = sqrt(POW2(p) + POW2(z));
842  res.lat = RAD2DEG(asin(z / res.elev));
843  res.lon = pos.lon;
844 
845  return (res);
846 }

858 vect3_t
859 geo2ecef_mtr(geo_pos3_t pos, const ellip_t *ellip)
860 {
861  double lat_r = DEG2RAD(pos.lat);
862  double lon_r = DEG2RAD(pos.lon);
863  double Rc; /* curvature of the prime vertical */
864  vect3_t res;
865  double sin_lat = sin(lat_r), cos_lat = cos(lat_r);
866  double sin_lon = sin(lon_r), cos_lon = cos(lon_r);
867 
868  Rc = ellip->a / sqrt(1 - ellip->ecc2 * POW2(sin_lat));
869 
870  res.x = (Rc + pos.elev) * cos_lat * cos_lon;
871  res.y = (Rc + pos.elev) * cos_lat * sin_lon;
872  res.z = (Rc * (1 - ellip->ecc2) + pos.elev) * sin_lat;
873 
874  return (res);
875 }

897 geo_pos3_t
898 ecef2geo(vect3_t pos, const ellip_t *ellip)
899 {
900  geo_pos3_t res;
901  double B;
902  double d;
903  double e;
904  double f;
905  double g;
906  double p;
907  double q;
908  double r;
909  double t;
910  double v;
911  double zlong;
912 
913  /*
914  * 1.0 compute semi-minor axis and set sign to that of z in order
915  * to get sign of Phi correct.
916  */
917  B = (pos.z >= 0 ? ellip->b : -ellip->b);
918 
919  /*
920  * 2.0 compute intermediate values for latitude
921  */
922  r = sqrt(POW2(pos.x) + POW2(pos.y));
923  e = (B * pos.z - (POW2(ellip->a) - POW2(B))) / (ellip->a * r);
924  f = (B * pos.z + (POW2(ellip->a) - POW2(B))) / (ellip->a * r);
925 
926  /*
927  * 3.0 find solution to:
928  * t^4 + 2*E*t^3 + 2*F*t - 1 = 0
929  */
930  p = (4.0 / 3.0) * (e * f + 1.0);
931  q = 2.0 * (POW2(e) - POW2(f));
932  d = POW3(p) + POW2(q);
933 
934  if (d >= 0.0) {
935  v = pow((sqrt(d) - q), (1.0 / 3.0)) - pow((sqrt(d) + q),
936  (1.0 / 3.0));
937  } else {
938  v = 2.0 * sqrt(-p) * cos(acos(q / (p * sqrt(-p))) / 3.0);
939  }
940 
941  /*
942  * 4.0 improve v
943  * NOTE: not really necessary unless point is near pole
944  */
945  if (POW2(v) < fabs(p))
946  v = -(POW3(v) + 2.0 * q) / (3.0 * p);
947  g = (sqrt(POW2(e) + v) + e) / 2.0;
948  t = sqrt(POW2(g) + (f - v * g) / (2.0 * g - e)) - g;
949 
950  res.lat = atan((ellip->a * (1.0 - POW2(t))) / (2.0 * B * t));
951 
952  /*
953  * 5.0 compute height above ellipsoid
954  */
955  res.elev = (r - ellip->a * t) * cos(res.lat) +
956  (pos.z - B) * sin(res.lat);
957 
958  /*
959  * 6.0 compute longitude east of Greenwich
960  */
961  zlong = atan2(pos.y, pos.x);
962  if (zlong < 0.0)
963  zlong = zlong + (2 * M_PI);
964 
965  res.lon = zlong;
966 
967  /*
968  * 7.0 convert latitude and longitude to degrees (elev is in meters).
969  */
970  res.lat = RAD2DEG(res.lat);
971  res.lon = RAD2DEG(res.lon);
972  if (res.lon >= 180.0)
973  res.lon -= 360.0;
974 
975  return (res);
976 }

1032 unsigned
1033 vect2sph_isect(vect3_t v, vect3_t o, vect3_t c, double r, bool_t confined,
1034  vect3_t i[2])
1035 {
1036  vect3_t l, o_min_c;
1037  double d, l_dot_o_min_c, sqrt_tmp, o_min_c_abs;
1038 
1039  /* convert v into a unit vector 'l' and scalar distance 'd' */
1040  l = vect3_unit(v, &d);
1041 
1042  /* compute (o - c) and the dot product of l.(o - c) */
1043  o_min_c = vect3_sub(o, c);
1044  l_dot_o_min_c = vect3_dotprod(l, o_min_c);
1045 
1046  /*
1047  * The full formula for the distance along L for the intersects is:
1048  * -(l.(o - c)) +- sqrt((l.(o - c))^2 - abs(o - c)^2 + r^2)
1049  * The part in the sqrt may be negative, zero or positive, indicating
1050  * respectively no intersection, one touching point or two points, so
1051  * before we start sqrt()ing away, we check which it is. Also, this
1052  * checks for a solution on an infinite line between extending along
1053  * v. Before we declare victory, we check that the computed
1054  * points lie on the vector.
1055  */
1056  o_min_c_abs = vect3_abs(o_min_c);
1057  sqrt_tmp = POW2(l_dot_o_min_c) - POW2(o_min_c_abs) + POW2(r);
1058  if (sqrt_tmp > 0) {
1059  /* Two solutions */
1060  double i1_d, i2_d;
1061  unsigned intersects = 0;
1062 
1063  sqrt_tmp = sqrt(sqrt_tmp);
1064 
1065  i1_d = -l_dot_o_min_c - sqrt_tmp;
1066  if ((i1_d >= 0 && i1_d <= d) || !confined) {
1067  /*
1068  * Solution lies on vector, store a vector to it
1069  * if the caller requested it.
1070  */
1071  if (i != NULL)
1072  i[intersects] = vect3_add(vect3_scmul(l, i1_d),
1073  o);
1074  intersects++;
1075  } else {
1076  /* Solution lies outside of line between o1 & o2 */
1077  i1_d = NAN;
1078  if (i != NULL)
1079  i[intersects] = NULL_VECT3;
1080  }
1081 
1082  /* ditto for the second intersect */
1083  i2_d = -l_dot_o_min_c + sqrt_tmp;
1084  if ((i2_d >= 0 && i2_d <= d) || !confined) {
1085  if (i != NULL)
1086  i[intersects] = vect3_add(vect3_scmul(l, i2_d),
1087  o);
1088  intersects++;
1089  } else {
1090  i2_d = NAN;
1091  if (i != NULL)
1092  i[intersects] = NULL_VECT3;
1093  }
1094 
1095  return (intersects);
1096  } else if (sqrt_tmp == 0) {
1097  /* One solution */
1098  double i1_d;
1099 
1100  if (i != NULL)
1101  i[1] = NULL_VECT3;
1102 
1103  i1_d = -l_dot_o_min_c;
1104  if ((i1_d >= 0 && i1_d <= d) || !confined) {
1105  if (i != NULL)
1106  i[0] = vect3_add(vect3_scmul(l, i1_d), o);
1107  return (1);
1108  } else {
1109  if (i != NULL)
1110  i[0] = NULL_VECT3;
1111  return (0);
1112  }
1113  } else {
1114  /* No solution, no intersections, NaN i1 & i2 */
1115  if (i != NULL) {
1116  i[0] = NULL_VECT3;
1117  i[1] = NULL_VECT3;
1118  }
1119  return (0);
1120  }
1121 }

1179  p3 = ob;

1516  * +- -+
1517  * | cos(a) 0 sin(a) |

1520  * +- -+
1521  */

1960 double
1961 quad_bezier_func(double x, const bezier_t *func)
1962 {
1963  ASSERT(func->n_pts >= 3 || func->n_pts % 3 == 2);
1964  /* Check boundary conditions */
1965  if (x < func->pts[0].x)
1966  return (func->pts[0].y);
1967  else if (x > func->pts[func->n_pts - 1].x)
1968  return (func->pts[func->n_pts - 1].y);
1969 
1970  /* Point lies on a curve segment */
1971  for (size_t i = 0; i + 2 < func->n_pts; i += 2) {
1972  if (x <= func->pts[i + 2].x) {
1973  vect2_t p0 = func->pts[i], p1 = func->pts[i + 1];
1974  vect2_t p2 = func->pts[i + 2];
1975  double y, t, ts[2];
1976  unsigned n;
1977 
1978  /*
1979  * Quadratic Bezier curves are defined as follows:
1980  *
1981  * B(t) = (1-t)^2.P0 + 2(1-t)t.P1 + t^2.P2
1982  *
1983  * Where 't' is a number 0.0 - 1.0, increasing along
1984  * curve as it progresses from P0 to P2 (P{0,1,2} are
1985  * 2D vectors). However, since we are using the curves
1986  * as functions, we first need to find which 't' along
1987  * the curve corresponds to our 'x' input.
1988  * Geometrically, quadratic Bezier curves are defined
1989  * as a set of points `P' which satisfy:
1990  *
1991  * 1) let `A' be a point that is on the line between
1992  * P0 and P1. Its distance from P0 is `t' times
1993  * the distance of P1 from P0.
1994  * 2) let `B' be a point that is on the line between
1995  * P1 and P2. Its distance from P1 is `t' times
1996  * the distance of P2 from P1.
1997  * 3) `P' is a point that is on a line between A and B.
1998  * Its distance from A is `t' times the distance
1999  * of B from A.
2000  *
2001  * The x coordinate of `P' corresponds to the input
2002  * `x' argument. Transforming this into an analytical
2003  * representation and simplifying to only consider
2004  * the X axis, we get (here 'a', 'b' represent the
2005  * respective 2D vector's X coordinate only):
2006  *
2007  * a = t(p1 - p0) + p0
2008  * b = t(p2 - p1) + p1
2009  * x = t(b - a) + a
2010  * x = t(t(p2 - p1) + p1 - t(p1 - p0) + p0) +
2011  * + t(p1 - p0) + p0
2012  *
2013  * Rearranging, we obtain this quadratic equation:
2014  *
2015  * 0 = (p2 - 2.p1 + p0)t^2 + 2(p1 - p0)t + p0 - x
2016  *
2017  * The solution we're interested in is only the one
2018  * between 0.0 - 1.0 (inclusive). We then take that
2019  * value and use it in the bezier curve equation at
2020  * the top to obtain the function value at point 'x'.
2021  */
2022  ASSERT3F(p0.x, <, p2.x);
2023  ASSERT3F(p0.x, <=, p1.x);
2024  ASSERT3F(p1.x, <=, p2.x);
2025  n = quadratic_solve(p2.x - 2 * p1.x + p0.x,
2026  2 * (p1.x - p0.x), p0.x - x, ts);
2027  ASSERT(n != 0);
2028  if (ts[0] >= 0 && ts[0] <= 1) {
2029  t = ts[0];
2030  } else if (n == 2 && ts[1] >= 0 && ts[1] <= 1) {
2031  t = ts[1];
2032  } else {
2033  continue;
2034  }
2035  ASSERT3F(t, >=, 0.0);
2036  ASSERT3F(t, <=, 1.0);
2037  y = POW2(1 - t) * p0.y + 2 * (1 - t) * t * p1.y +
2038  POW2(t) * p2.y;
2039 
2040  return (y);
2041  }
2042  }
2043  VERIFY(0);
2044 }
