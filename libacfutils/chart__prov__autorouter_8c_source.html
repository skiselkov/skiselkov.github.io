78 static bool_t
79 webdav_foreach_dirlist(chartdb_t *cdb, CURL *curl, const char *path,
80  const char *cachefile,
81  bool_t (*cb)(chartdb_t *cdb, CURL *curl, const char *href))
82 {
83  chart_dl_info_t dl = { .cdb = NULL };
84  const chart_prov_info_login_t *login = cdb->prov_login;
85  xmlDoc *doc = NULL;
86  xmlXPathContext *xpath_ctx = NULL;
87  xmlXPathObject *xpath_obj = NULL;
88  char url[256];
89 
90  snprintf(url, sizeof(url), "%s%s", BASE_URL, path);
91 
92  if (!file_exists(cachefile, NULL) ||
93  (doc = xmlParseFile(cachefile)) == NULL) {
94  bool_t result = chart_download_multi(&curl, cdb, url,
95  cachefile, "PROPFIND", login, -1,
96  "Error downloading chart index", &dl);
97  if (result == B_FALSE || dl.buf == NULL)
98  goto errout;
99  doc = xmlParseMemory((char *)dl.buf, dl.bufsz);
100  if (doc == NULL) {
101  logMsg("Error parsing chart index: "
102  "malformed WebDAV XML response from server");
103  goto errout;
104  }
105  }
106  xpath_ctx = xmlXPathNewContext(doc);
107  if (xpath_ctx == NULL) {
108  logMsg("Error parsing chart index: error creating XPath "
109  "context for XML");
110  goto errout;
111  };
112  VERIFY0(xmlXPathRegisterNs(xpath_ctx, (xmlChar *)"d",
113  (xmlChar *)"DAV:"));
114  xpath_obj = xmlXPathEvalExpression(
115  (xmlChar *)"/d:multistatus/d:response/d:href", xpath_ctx);
116  if (xpath_obj == NULL || xpath_obj->nodesetval->nodeNr < 1) {
117  logMsg("Error parsing chart index: malformed WebDAV dirlist "
118  "%s %p.", path, xpath_obj);
119  goto errout;
120  }
121  for (int i = 0; i < xpath_obj->nodesetval->nodeNr; i++) {
122  const xmlNode *href = xpath_obj->nodesetval->nodeTab[i];
123 
124  if (href->children == NULL || href->children->content == NULL) {
125  logMsg("Error parsing chart index: directory list "
126  "entry was missing HREF content.");
127  goto errout;
128  }
129  /* Skip references back to the index */
130  if (strcmp((char *)href->children->content, path) == 0)
131  continue;
132  if (!cb(cdb, curl, (const char *)href->children->content)) {
133  goto errout;
134  }
135  }
136 
137  xmlXPathFreeObject(xpath_obj);
138  xmlXPathFreeContext(xpath_ctx);
139  xmlFreeDoc(doc);
140  free(dl.buf);
141 
142  return (B_TRUE);
143 errout:
144  if (xpath_obj != NULL)
145  xmlXPathFreeObject(xpath_obj);
146  if (xpath_ctx != NULL)
147  xmlXPathFreeContext(xpath_ctx);
148  if (doc != NULL)
149  xmlFreeDoc(doc);
150  free(dl.buf);
151  return (B_FALSE);
152 }

172  "OBSTACLE", "OBST",
173  "TRANSITION", "TRANS",
174  "NOISE ABATEMENT", "NOISE ABTMT",
175  "PRECISION APPROACH", "PRECISION APP",
176  "STANDARD DEPARTURE ROUTES - INSTRUMENT", "",
177  "STANDARD ARRIVAL ROUTES - INSTRUMENT", "",
178  "STANDARD DEPARTURE ROUTES INSTRUMENT", "",
179  "STANDARD ARRIVAL ROUTES INSTRUMENT", "",
180  "FINAL APCH", "FINAL APP",
181  "FINAL APPROACH", "FINAL APP",
182  "(SID)", "",
183  "(STAR)", "",
184  NULL /* list terminator */
185  };
186 
187  strtoupper(name);
188  word_subst(name, subst);
189  /* Kill any leading or trailing whitespace */
190  strip_space(name);
191  /*
192  * Kill double spaces that might have been generated as a result
193  * of the word removal.
194  */
195  for (unsigned i = 0, n = strlen(name); i + 1 < n;) {
196  if (isspace(name[i]) && isspace(name[i + 1])) {
197  memmove(&name[i], &name[i + 1], strlen(&name[i]));
198  n--;
199  } else {
200  i++;
201  }
202  }
203 }

205 static bool_t
206 parse_chart(chartdb_t *cdb, CURL *curl, const char *path)
207 {
208  char **comps;
209  size_t n_comps;
210  char icao[8];
211  char *filename, *country, *arpt_name, *typename, *chartname;
212  chart_type_t chart_type;
213  char common_prefix[16];
214  chart_arpt_t *arpt;
215  chart_t *chart;
216 
217  LACF_UNUSED(curl);
218 
219  comps = strsplit(path, "/", B_TRUE, &n_comps);
220  if (n_comps < 5)
221  goto out;
222 
223  unescape_percent(comps[1]);
224  country = comps[1];
225 
226  unescape_percent(comps[2]);
227  if (strlen(comps[2]) < 8)
228  goto out;
229  lacf_strlcpy(icao, comps[2], 5); /* first 4 chars is ICAO */
230  arpt_name = &(comps[2][7]);
231 
232  unescape_percent(comps[3]);
233  typename = comps[3];
234  if (strcmp(typename, "Airport") == 0 || strcmp(typename, "VFR") == 0)
235  chart_type = CHART_TYPE_APD;
236  else if (strcmp(typename, "Arrival") == 0)
237  chart_type = CHART_TYPE_STAR;
238  else if (strcmp(typename, "Approach") == 0)
239  chart_type = CHART_TYPE_IAP;
240  else if (strcmp(typename, "Departure") == 0)
241  chart_type = CHART_TYPE_DP;
242  else
243  goto out;
244 
245  /* Use the unescaped version as the on-disk filename */
246  filename = strdup(comps[4]);
247 
248  snprintf(common_prefix, sizeof (common_prefix), "AD 2 %s ", icao);
249  unescape_percent(comps[4]);
250  chartname = comps[4];
251  if (strncmp(chartname, common_prefix, strlen(common_prefix)) == 0) {
252  int l = strlen(common_prefix);
253  memmove(chartname, &chartname[l], strlen(chartname) - l + 1);
254  }
255  chart_name_process(chartname);
256 
257  arpt = chartdb_add_arpt(cdb, icao, arpt_name, country, "");
258  chart = safe_calloc(1, sizeof (*chart));
259  chart->name = strdup(chartname);
260  chart->codename = strdup(path);
261  chart->type = chart_type;
262  chart->filename = filename;
263  if (!chartdb_add_chart(arpt, chart)) {
264  free(chart->name);
265  free(chart->codename);
266  free(chart->filename);
267  free(chart);
268  }
269 out:
270  free_strlist(comps, n_comps);
271  return (B_TRUE);
272 }
