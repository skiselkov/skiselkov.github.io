163 dsf_t *
164 dsf_init(const char *filename)
165 {
166  dsf_t *dsf = NULL;
167  uint8_t *buf = NULL;
168  ssize_t bufsz = filesz(filename);
169  ssize_t readsz = 0;
170  FILE *fp = fopen(filename, "rb");
171  char reason[DSF_REASON_SZ];
172  static const uint8_t magic[8] = {
173  'X', 'P', 'L', 'N', 'E', 'D', 'S', 'F'
174  };
175 
176  if (bufsz < 12 + 16 || fp == NULL)
177  goto errout;
178  buf = safe_malloc(bufsz);
179 
180  if (fread(buf, 1, sizeof (magic), fp) != sizeof (magic))
181  goto errout;
182  readsz = sizeof (magic);
183 
184  if (memcmp(buf, magic, sizeof (magic)) == 0) {
185  /* raw DSF, just read its contents */
186  while (readsz < bufsz) {
187  size_t n = fread(&buf[readsz], 1, bufsz - readsz, fp);
188  if (n == 0)
189  break;
190  readsz += n;
191  }
192  if (!feof(fp) && readsz < bufsz) {
193  logMsg("Error reading DSF %s: %s", filename,
194  strerror(errno));
195  goto errout;
196  }
197  fclose(fp);
198  fp = NULL;
199  } else if (test_7z(buf, sizeof (magic))) {
200  fclose(fp);
201  fp = NULL;
202  free(buf);
203  buf = decompress_7z(filename, (size_t *)&bufsz);
204  if (buf == NULL)
205  goto errout;
206  } else {
207  goto errout;
208  }
209 
210  dsf = dsf_parse(buf, bufsz, reason);
211  if (dsf == NULL) {
212  logMsg("Error parsing DSF %s: %s", filename, reason);
213  goto errout;
214  }
215  /* No need to release `buf', the dsf_t has taken ownership of it. */
216 
217  return (dsf);
218 errout:
219  free(buf);
220  if (fp != NULL)
221  fclose(fp);
222  return (NULL);
223 }

225 static bool_t
226 parse_prop_atom(dsf_atom_t *atom, char reason[DSF_REASON_SZ])
227 {
228  const char *payload_end =
229  (const char *)(atom->payload + atom->payload_sz);
230 
231  ASSERT3U(atom->id, ==, DSF_ATOM_PROP);
232 
233  list_create(&atom->prop_atom.props, sizeof (dsf_prop_t),
234  offsetof(dsf_prop_t, prop_node));
235  atom->subtype_inited = B_TRUE;
236 
237  if (atom->payload_sz < 2) {
238  snprintf(reason, DSF_REASON_SZ, "PROP atom too short");
239  return (B_FALSE);
240  }
241  for (const char *name = (const char *)atom->payload;
242  name < payload_end;) {
243  size_t name_len = strnlen(name, payload_end - name);
244  size_t value_len;
245  const char *value;
246  dsf_prop_t *prop;
247 
248  if (name_len == (size_t)(payload_end - name)) {
249  snprintf(reason, DSF_REASON_SZ, "PROP atom contains "
250  "an unterminated name string");
251  return (B_FALSE);
252  }
253  value = name + name_len + 1;
254  if (value >= payload_end) {
255  snprintf(reason, DSF_REASON_SZ, "Last name-value pair "
256  "in PROP atom is missing the value");
257  return (B_FALSE);
258  }
259  value_len = strnlen(value, payload_end - value);
260  if (value_len == (size_t)(payload_end - value)) {
261  snprintf(reason, DSF_REASON_SZ, "PROP atom contains "
262  "an unterminated value string");
263  return (B_FALSE);
264  }
265  prop = safe_calloc(1, sizeof (*prop));
266  prop->name = name;
267  prop->value = value;
268  list_insert_tail(&atom->prop_atom.props, prop);
269 
270  name = value + value_len + 1;
271  }
272 
273  return (B_TRUE);
274 }

315 static void *
316 rle_decode(dsf_atom_t *atom, unsigned plane, const uint8_t *start,
317  const uint8_t *end, dsf_data_type_t data_type, uint32_t num_values,
318  size_t *bytes_consumed, char reason[DSF_REASON_SZ])
319 {
320  unsigned datalen = type2datalen(data_type);
321  uint8_t *out = NULL;
322  size_t out_cap = 0;
323  size_t i = 0;
324 
325  for (uint32_t num = 0; num < num_values;) {
326  unsigned repeat, cnt;
327 
328  CHECK_LEN(1);
329  repeat = start[i];
330  cnt = repeat & 0x7f;
331  i++;
332 
333  RESIZE(out, num, out_cap, datalen, cnt);
334  if (repeat & 0x80) {
335  CHECK_LEN(datalen);
336  /* repeating case */
337  for (unsigned k = 0; k < cnt; k++) {
338  memcpy(&out[num * datalen], &start[i], datalen);
339  num++;
340  }
341  i += datalen;
342  } else {
343  CHECK_LEN(cnt * datalen);
344  for (unsigned k = 0; k < cnt; k++) {
345  memcpy(&out[num * datalen], &start[i], datalen);
346  i += datalen;
347  num++;
348  }
349  }
350  }
351 
352  if (bytes_consumed != NULL)
353  *bytes_consumed = i;
354  return (out);
355 errout:
356  free(out);
357  return (NULL);
358 }

361 diff_decode(dsf_atom_t *atom, unsigned plane, const uint8_t *start,
362  const uint8_t *end, dsf_data_type_t data_type, uint32_t num_values,
363  size_t *bytes_consumed, char reason[DSF_REASON_SZ])
364 {
365  unsigned datalen = type2datalen(data_type);
366  void *out;
367 
368  CHECK_LEN(num_values * datalen);
369 
370  out = safe_calloc(datalen, num_values);
371 
372 #if __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
373 #error "TODO: implement big-endian"
374 #endif

376 #define DIFF_DEC(ctype) \
377  do { \
378  ctype prev = 0; \
379  ctype *out_type = out; \
380  for (size_t i = 0; start + i * sizeof (ctype) < end; i++) { \
381  ctype val = *(ctype *)(start + i * sizeof (ctype)); \
382  prev += val; \
383  out_type[i] = prev; \
384  } \
385  } while (0)
386  switch (data_type) {
387  case DSF_DATA_SINT16:
388  DIFF_DEC(int16_t);
389  break;
390  case DSF_DATA_UINT16:
391  DIFF_DEC(uint16_t);
392  break;
393  case DSF_DATA_SINT32:
394  DIFF_DEC(int32_t);
395  break;
396  case DSF_DATA_UINT32:
397  DIFF_DEC(uint32_t);
398  break;
399  case DSF_DATA_SINT64:
400  DIFF_DEC(int64_t);
401  break;
402  case DSF_DATA_UINT64:
403  DIFF_DEC(uint64_t);
404  break;
405  case DSF_DATA_FP32:
406  DIFF_DEC(float);
407  break;
408  case DSF_DATA_FP64:
409  DIFF_DEC(double);
410  break;
411  default:
412  VERIFY(0);
413  }
414 #undef DIFF_DEC
415 
416  if (bytes_consumed != NULL)
417  *bytes_consumed = num_values * datalen;
418  return (out);
419 errout:
420  return (NULL);
421 }

447 static ssize_t
448 parse_plane(dsf_atom_t *atom, unsigned plane, dsf_data_type_t data_type,
449  const uint8_t *start, const uint8_t *end, char reason[DSF_REASON_SZ])
450 {
451  void *out = NULL;
452  size_t consumed = 0;
453  uint32_t datacnt = atom->planar_atom.data_count;
454  unsigned datalen = type2datalen(data_type);
455  unsigned enc;
456 
457  CHECK_LEN(1);
458  enc = start[0];
459  start++;
460  if (enc > 3) {
461  snprintf(reason, DSF_REASON_SZ, "invalid encoding type %u for "
462  "planar numeric atom %c%c%c%c at %lx", enc,
463  DSF_ATOM_ID_PRINTF(atom), (unsigned long)atom->file_off);
464  return (-1);
465  }
466 
467  if (datacnt == 0)
468  return (1);
469 
470  switch (enc) {
471  case DSF_ENC_RAW:
472  out = raw_decode(atom, plane, start, end, data_type, datacnt,
473  &consumed, reason);
474  break;
475  case DSF_ENC_DIFF:
476  out = diff_decode(atom, plane, start, end, data_type,
477  datacnt, &consumed, reason);
478  break;
479  case DSF_ENC_DIFF | DSF_ENC_RLE: {
480  void *tmp = rle_decode(atom, plane, start, end, data_type,
481  datacnt, &consumed, reason);
482  if (tmp == NULL)
483  return (-1);
484  out = diff_decode(atom, plane, tmp, tmp + datacnt * datalen,
485  data_type, datacnt, NULL, reason);
486  free(tmp);
487  break;
488  }
489  default: {
490  void *tmp;
491  ASSERT3U(enc, ==, DSF_ENC_RLE);
492  tmp = rle_decode(atom, plane, start, end, data_type, datacnt,
493  &consumed, reason);
494  if (tmp == NULL)
495  return (-1);
496  out = raw_decode(atom, plane, start, end, data_type, datacnt,
497  &consumed, reason);
498  free(tmp);
499  break;
500  }
501  }
502 
503  if (out == NULL)
504  return (-1);
505 
506  atom->planar_atom.data[plane] = out;
507 
508  return (consumed + 1);
509 errout:
510  return (-1);
511 }

517 static bool_t
518 parse_planar_numeric_atom(dsf_atom_t *atom, dsf_data_type_t data_type,
519  char reason[DSF_REASON_SZ])
520 {
521  dsf_planar_atom_t *pa = &atom->planar_atom;
522  const uint8_t *plane_p = &atom->payload[5];
523  const uint8_t *end = atom->payload + atom->payload_sz;
524 
525  if (atom->payload_sz < 5) {
526  snprintf(reason, DSF_REASON_SZ, "invalid planar numeric atom "
527  "%c%c%c%c at %lx: not enough payload",
528  DSF_ATOM_ID_PRINTF(atom), (unsigned long)atom->file_off);
529  return (B_FALSE);
530  }
531 
532  pa->data_type = data_type;
533  pa->data_count = read_u32(atom->payload);
534  pa->plane_count = atom->payload[4];
535  pa->data = safe_calloc(pa->plane_count, sizeof (*pa->data));
536  atom->subtype_inited = B_TRUE;
537 
538  for (unsigned i = 0; i < pa->plane_count; i++) {
539  ssize_t len = parse_plane(atom, i, data_type,
540  plane_p, end, reason);
541  if (len < 0) {
542  printf("plane parse error\n");
543  return (B_FALSE);
544  }
545  plane_p += len;
546  }
547 
548  if (plane_p != end) {
549  snprintf(reason, DSF_REASON_SZ, "planar numeric atom %c%c%c%c "
550  "at %lx contained trailing garbage",
551  DSF_ATOM_ID_PRINTF(atom), (unsigned long)atom->file_off);
552  return (B_FALSE);
553  }
554 
555  return (B_TRUE);
556 }

613 static dsf_atom_t *
614 parse_atom(const uint8_t *buf, size_t bufsz, char reason[DSF_REASON_SZ],
615  uint64_t abs_off)
616 {
617  dsf_atom_t *atom = safe_calloc(1, sizeof (*atom));
618 
619  ASSERT(reason != NULL);
620  list_create(&atom->subatoms, sizeof (dsf_atom_t),
621  offsetof(dsf_atom_t, atom_list));
622 
623  if (bufsz < 8)
624  goto errout;
625 #if __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
626  atom->id = BSWAP32(read_u32(buf));
627 #else
628  atom->id = read_u32(buf);
629 #endif
630  atom->payload_sz = read_u32(&buf[4]) - 8;
631  atom->payload = &buf[8];
632  if (atom->payload_sz + 8 > bufsz) {
633  snprintf(reason, DSF_REASON_SZ, "invalid atom at %lx, "
634  "size (%lx) is too large",
635  (unsigned long)abs_off, (unsigned long)atom->payload_sz);
636  goto errout;
637  }
638  atom->file_off = abs_off;
639 
640  if (atom->id == DSF_ATOM_HEAD || atom->id == DSF_ATOM_DEFN ||
641  atom->id == DSF_ATOM_GEOD || atom->id == DSF_ATOM_DEMS) {
642  if (!parse_atom_list(atom->payload, atom->payload_sz,
643  &atom->subatoms, reason, abs_off + 8))
644  goto errout;
645  } else if (atom->id == DSF_ATOM_PROP) {
646  if (!parse_prop_atom(atom, reason))
647  goto errout;
648  } else if (atom->id == DSF_ATOM_POOL) {
649  if (!parse_planar_numeric_atom(atom, DSF_DATA_UINT16, reason))
650  goto errout;
651  } else if (atom->id == DSF_ATOM_PO32) {
652  if (!parse_planar_numeric_atom(atom, DSF_DATA_UINT32, reason))
653  goto errout;
654  } else if (atom->id == DSF_ATOM_DEMI) {
655  if (!parse_demi_atom(atom, reason))
656  goto errout;
657  }
658 
659  return (atom);
660 errout:
661  free_atom(atom);
662  return (NULL);
663 }

665 static bool_t
666 parse_atom_list(const uint8_t *buf, uint64_t bufsz, list_t *atoms,
667  char reason[DSF_REASON_SZ], uint64_t abs_off)
668 {
669  ASSERT(reason != NULL);
670 
671  for (const uint8_t *atom_buf = buf; atom_buf < buf + bufsz;) {
672  dsf_atom_t *atom = parse_atom(atom_buf,
673  (buf + bufsz) - atom_buf, reason,
674  abs_off + (atom_buf - buf));
675 
676  if (atom == NULL)
677  return (B_FALSE);
678  list_insert_tail(atoms, atom);
679  atom_buf += atom->payload_sz + 8;
680  }
681 
682  return (B_TRUE);

685 dsf_t *
686 dsf_parse(uint8_t *buf, size_t bufsz, char reason[DSF_REASON_SZ])
687 {
688  dsf_t *dsf = safe_calloc(1, sizeof (*dsf));
689  static const uint8_t magic[8] = {
690  'X', 'P', 'L', 'N', 'E', 'D', 'S', 'F'
691  };
692 
693  VERIFY(reason != NULL);
694 
695  list_create(&dsf->atoms, sizeof (dsf_atom_t),
696  offsetof(dsf_atom_t, atom_list));
697 
698  if (bufsz < 12 + 16) {
699  snprintf(reason, DSF_REASON_SZ,
700  "file is too short (%d) to be a valid DSF", (int)bufsz);
701  goto errout;
702  }
703  if (memcmp(buf, magic, sizeof (magic)) != 0) {
704  snprintf(reason, DSF_REASON_SZ,
705  "file premable missing DSF magic number");
706  goto errout;
707  }
708  dsf->version = read_u32(&buf[8]);
709  if (dsf->version > DSF_MAX_VERSION) {
710  snprintf(reason, DSF_REASON_SZ,
711  "file version (%d) exceeds our max supported version (%d)",
712  dsf->version, DSF_MAX_VERSION);
713  goto errout;
714  }
715  memcpy(dsf->md5sum, &buf[bufsz - 16], 16);
716  if (!parse_atom_list(&buf[12], bufsz - (12 + 16), &dsf->atoms, reason,
717  12))
718  goto errout;
719 
720  /* Set this last, this confirms our ownership of the data buffer. */
721  dsf->data = buf;
722  dsf->size = bufsz;
723 
724  return (dsf);
725 errout:
726  dsf_fini(dsf);
727  return (NULL);
728 }
