56static double
57demd_read(const dsf_atom_t *demi, const dsf_atom_t *demd,
58 unsigned row, unsigned col)
59{
60 double v;
61
62#define DEMD_READ(data_type, val) \
63 do { \
64 (val) = ((data_type *)demd->payload)[row * \
65 demi->demi_atom.width + col] * \
66 demi->demi_atom.scale + demi->demi_atom.offset; \
67 } while (0)
68
69 switch (demi->demi_atom.flags & DEMI_DATA_MASK) {
70 case DEMI_DATA_FP32:
71 DEMD_READ(float, v);
72 break;
73 case DEMI_DATA_SINT:
74 switch (demi->demi_atom.bpp) {
75 case 1:
76 DEMD_READ(int8_t, v);
77 break;
78 case 2:
79 DEMD_READ(int16_t, v);
80 break;
81 case 4:
82 DEMD_READ(int32_t, v);
83 break;
84 default:
85 VERIFY_FAIL();
86 }
87 break;
88 case DEMI_DATA_UINT:
89 switch (demi->demi_atom.bpp) {
90 case 1:
91 DEMD_READ(uint8_t, v);
92 break;
93 case 2:
94 DEMD_READ(uint16_t, v);
95 break;
96 case 4:
97 DEMD_READ(uint32_t, v);
98 break;
99 default:
100 VERIFY_FAIL();
101 }
102 break;
103 default:
104 VERIFY_FAIL();
105 }
106
107#undef DEMD_READ
108
109 return (v);
110}

112static void
113dump_dem(const dsf_atom_t *demi, const dsf_atom_t *demd, int seq)
114{
115 uint8_t *buf;
116 char filename[32];
117 double max_val = -10000, min_val = 10000;
118
119 ASSERT3U(demd->payload_sz, ==, demi->demi_atom.width *
120 demi->demi_atom.height * demi->demi_atom.bpp);
121
122 /* we write 8-bit grey-scale */
123 buf = malloc(demi->demi_atom.width * demi->demi_atom.height);
124
125 for (unsigned row = 0; row < demi->demi_atom.height; row++) {
126 for (unsigned col = 0; col < demi->demi_atom.width; col++) {
127 double v = demd_read(demi, demd, row, col);
128 max_val = MAX(v, max_val);
129 min_val = MIN(v, min_val);
130 }
131 }
132
133 for (unsigned row = 0; row < demi->demi_atom.height; row++) {
134 for (unsigned col = 0; col < demi->demi_atom.width; col++) {
135 double v = demd_read(demi, demd, row, col);
136 unsigned pixel_pos =
137 ((demi->demi_atom.height - row - 1) *
138 demi->demi_atom.width + col);
139
140 buf[pixel_pos] =
141 ((v - min_val) / (max_val - min_val)) * 255;
142 }
143 }
144
145 snprintf(filename, sizeof (filename), "DEM_%d.png", seq);
146 png_write_to_file_grey8(filename, demi->demi_atom.width,
147 demi->demi_atom.height, buf);
148
149 printf("min: %f max: %f\n", min_val, max_val);
150
151 free(buf);
152}

154static void
155water_mask(dsf_t *dsf)
156{
157 const dsf_atom_t *demi_land, *demd_land;
158 const dsf_atom_t *demi_water, *demd_water;
159 uint8_t *buf;
160 double rat_x, rat_y;
161
162 demi_land = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMI, 0, 0);
163 demd_land = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMD, 0, 0);
164 demi_water = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMI, 1, 0);
165 demd_water = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMD, 1, 0);
166
167 if (demi_land == NULL || demd_land == NULL ||
168 demi_water == NULL || demd_water == NULL) {
169 fprintf(stderr, "Error producing water mask: required "
170 "atoms are missing in DSF.\n");
171 return;
172 }
173
174 ASSERT3U(demd_land->payload_sz, ==, demi_land->demi_atom.width *
175 demi_land->demi_atom.height * demi_land->demi_atom.bpp);
176 ASSERT3U(demd_water->payload_sz, ==, demi_water->demi_atom.width *
177 demi_water->demi_atom.height * demi_water->demi_atom.bpp);
178
179 rat_x = (double)demi_water->demi_atom.width /
180 demi_land->demi_atom.width;
181 rat_y = (double)demi_water->demi_atom.height /
182 demi_land->demi_atom.height;
183
184 /* we write 8-bit grey-scale */
185 buf = malloc(demi_land->demi_atom.width * demi_land->demi_atom.height);
186 for (unsigned y = 0; y < demi_land->demi_atom.height; y++) {
187 for (unsigned x = 0; x < demi_land->demi_atom.width; x++) {
188 unsigned wx = MIN(x * rat_x,
189 demi_water->demi_atom.width - 1);
190 unsigned wy = MIN(y * rat_y,
191 demi_water->demi_atom.height - 1);
192 double h_land = demd_read(demi_land, demd_land, y, x);
193 double h_wat = demd_read(demi_water, demd_water, wy,
194 wx);
195 unsigned pixel = x + (demi_land->demi_atom.height * y);
196
197 buf[pixel] = (h_land > h_wat) * 255;
198 }
199 }
200
201 png_write_to_file_grey8("water_mask.png", demi_land->demi_atom.width,
202 demi_land->demi_atom.height, buf);
203
204 free(buf);
205}

207int
208main(int argc, char *argv[])
209{
210 dsf_t *dsf;
211 char *dump;
212 char reason[DSF_REASON_SZ];
213 int opt;
214 bool_t quiet = B_FALSE;
215 bool_t dump_cmds = B_FALSE;
216 bool_t do_dump_dem = B_FALSE;
217 bool_t do_water_mask = B_FALSE;
218
219 memset(cmd_cbs, 0, sizeof (cmd_cbs));
220 for (int i = 0; i < NUM_DSF_CMDS; i++)
221 cmd_cbs[i] = cmd_cb;
222
223 log_init(logfunc, "dsfdump");
224
225 while ((opt = getopt(argc, argv, "hqcdw")) != -1) {
226 switch (opt) {
227 case 'q':
228 quiet = B_TRUE;
229 break;
230 case 'c':
231 dump_cmds = B_TRUE;
232 break;
233 case 'd':
234 do_dump_dem = B_TRUE;
235 break;
236 case 'h':
237 printf("Usage: %s [-q] <dsf-file>\n", argv[0]);
238 exit(EXIT_SUCCESS);
239 case 'w':
240 do_water_mask = B_TRUE;
241 break;
242 default:
243 fprintf(stderr, "Usage: %s [-q] <dsf-file>\n", argv[0]);
244 exit(EXIT_FAILURE);
245 }
246 }
247
248 if (optind >= argc) {
249 fprintf(stderr, "Missing filename argument. "
250 "Try -h for help.\n");
251 exit(EXIT_FAILURE);
252 }
253
254 dsf = dsf_init(argv[optind]);
255 if (dsf == NULL)
256 return (1);
257
258 dump = dsf_dump(dsf);
259 if (!quiet)
260 puts(dump);
261 free(dump);
262
263 if (dump_cmds &&
264 !dsf_parse_cmds(dsf, cmd_cbs, (void *)(uintptr_t)quiet, reason)) {
265 fprintf(stderr, "Error parsing DSF commands: %s\n", reason);
266 }
267 if (do_dump_dem) {
268 const dsf_atom_t *demi;
269 const dsf_atom_t *demd;
270
271 for (int i = 0;
272 (demi = dsf_lookup(dsf, DSF_ATOM_DEMS, 0,
273 DSF_ATOM_DEMI, i, 0)) != NULL &&
274 (demd = dsf_lookup(dsf, DSF_ATOM_DEMS, 0,
275 DSF_ATOM_DEMD, i, 0)) != NULL && i < 2; i++) {
276 dump_dem(demi, demd, i);
277 }
278 }
279 if (do_water_mask)
280 water_mask(dsf);
281
282 dsf_fini(dsf);
283
284 return (0);
285}
