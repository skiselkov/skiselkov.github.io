55 static double
56 demd_read(const dsf_atom_t *demi, const dsf_atom_t *demd,
57  unsigned row, unsigned col)
58 {
59  double v;
60 
61 #define DEMD_READ(data_type, val) \
62  do { \
63  (val) = ((data_type *)demd->payload)[row * \
64  demi->demi_atom.width + col] * \
65  demi->demi_atom.scale + demi->demi_atom.offset; \
66  } while (0)
67 
68  switch (demi->demi_atom.flags & DEMI_DATA_MASK) {
69  case DEMI_DATA_FP32:
70  DEMD_READ(float, v);
71  break;
72  case DEMI_DATA_SINT:
73  switch (demi->demi_atom.bpp) {
74  case 1:
75  DEMD_READ(int8_t, v);
76  break;
77  case 2:
78  DEMD_READ(int16_t, v);
79  break;
80  case 4:
81  DEMD_READ(int32_t, v);
82  break;
83  }
84  break;
85  case DEMI_DATA_UINT:
86  switch (demi->demi_atom.bpp) {
87  case 1:
88  DEMD_READ(uint8_t, v);
89  break;
90  case 2:
91  DEMD_READ(uint16_t, v);
92  break;
93  case 4:
94  DEMD_READ(uint32_t, v);
95  break;
96  }
97  break;
98  default:
99  VERIFY(0);
100  }
101 
102 #undef DEMD_READ
103 
104  return (v);
105 }

107 static void
108 dump_dem(const dsf_atom_t *demi, const dsf_atom_t *demd, int seq)
109 {
110  uint8_t *buf;
111  char filename[32];
112  double max_val = -10000, min_val = 10000;
113 
114  ASSERT3U(demd->payload_sz, ==, demi->demi_atom.width *
115  demi->demi_atom.height * demi->demi_atom.bpp);
116 
117  /* we write 8-bit grey-scale */
118  buf = malloc(demi->demi_atom.width * demi->demi_atom.height);
119 
120  for (unsigned row = 0; row < demi->demi_atom.height; row++) {
121  for (unsigned col = 0; col < demi->demi_atom.width; col++) {
122  double v = demd_read(demi, demd, row, col);
123  max_val = MAX(v, max_val);
124  min_val = MIN(v, min_val);
125  }
126  }
127 
128  for (unsigned row = 0; row < demi->demi_atom.height; row++) {
129  for (unsigned col = 0; col < demi->demi_atom.width; col++) {
130  double v = demd_read(demi, demd, row, col);
131  unsigned pixel_pos =
132  ((demi->demi_atom.height - row - 1) *
133  demi->demi_atom.width + col);
134 
135  buf[pixel_pos] =
136  ((v - min_val) / (max_val - min_val)) * 255;
137  }
138  }
139 
140  snprintf(filename, sizeof (filename), "DEM_%d.png", seq);
141  png_write_to_file_grey8(filename, demi->demi_atom.width,
142  demi->demi_atom.height, buf);
143 
144  printf("min: %f max: %f\n", min_val, max_val);
145 
146  free(buf);
147 }

149 static void
150 water_mask(dsf_t *dsf)
151 {
152  const dsf_atom_t *demi_land, *demd_land;
153  const dsf_atom_t *demi_water, *demd_water;
154  uint8_t *buf;
155  double rat_x, rat_y;
156 
157  demi_land = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMI, 0, 0);
158  demd_land = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMD, 0, 0);
159  demi_water = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMI, 1, 0);
160  demd_water = dsf_lookup(dsf, DSF_ATOM_DEMS, 0, DSF_ATOM_DEMD, 1, 0);
161 
162  if (demi_land == NULL || demd_land == NULL ||
163  demi_water == NULL || demd_water == NULL) {
164  fprintf(stderr, "Error producing water mask: required "
165  "atoms are missing in DSF.\n");
166  return;
167  }
168 
169  ASSERT3U(demd_land->payload_sz, ==, demi_land->demi_atom.width *
170  demi_land->demi_atom.height * demi_land->demi_atom.bpp);
171  ASSERT3U(demd_water->payload_sz, ==, demi_water->demi_atom.width *
172  demi_water->demi_atom.height * demi_water->demi_atom.bpp);
173 
174  rat_x = (double)demi_water->demi_atom.width /
175  demi_land->demi_atom.width;
176  rat_y = (double)demi_water->demi_atom.height /
177  demi_land->demi_atom.height;
178 
179  /* we write 8-bit grey-scale */
180  buf = malloc(demi_land->demi_atom.width * demi_land->demi_atom.height);
181  for (unsigned y = 0; y < demi_land->demi_atom.height; y++) {
182  for (unsigned x = 0; x < demi_land->demi_atom.width; x++) {
183  unsigned wx = MIN(x * rat_x,
184  demi_water->demi_atom.width - 1);
185  unsigned wy = MIN(y * rat_y,
186  demi_water->demi_atom.height - 1);
187  double h_land = demd_read(demi_land, demd_land, y, x);
188  double h_wat = demd_read(demi_water, demd_water, wy,
189  wx);
190  unsigned pixel = x + (demi_land->demi_atom.height * y);
191 
192  buf[pixel] = (h_land > h_wat) * 255;
193  }
194  }
195 
196  png_write_to_file_grey8("water_mask.png", demi_land->demi_atom.width,
197  demi_land->demi_atom.height, buf);
198 
199  free(buf);
200 }

233  exit(EXIT_SUCCESS);

239  exit(EXIT_FAILURE);
240  }
241  }
242

246  exit(EXIT_FAILURE);

269  (demd = dsf_lookup(dsf, DSF_ATOM_DEMS, 0,
270  DSF_ATOM_DEMD, i, 0)) != NULL && i < 2; i++) {
271  dump_dem(demi, demd, i);
272  }
273  }
