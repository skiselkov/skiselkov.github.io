214 static inline double
215 normalize_hdg(double hdg)
216 {
217  if (isnan(hdg))
218  return (hdg);
219  hdg = fmod(hdg, 360);
220  /* Flip negative into positive */
221  if (hdg < 0.0)
222  hdg += 360.0;
223  /* Necessary to avoid FP rounding errors */
224  if (hdg <= 0.0 || hdg > 360.0) {
225  hdg = clamp(hdg, 0, 360);
226  /* Avoid negative zero */
227  if (hdg == -0.0)
228  hdg = 0.0;
229  }
230  return (hdg);
231 }

559 {

564  ASSERT(s2 != NULL);
565 
566  l1 = strlen(s1);
567  l2 = strlen(s2);
568

571 
572  lacf_strlcpy(s1_lower, s1, sizeof (s1_lower));
573  lacf_strlcpy(s2_lower, s2, sizeof (s2_lower));
574  strtolower(s1_lower);
575  strtolower(s2_lower);
576  res = strncmp(s1_lower, s2_lower, n);
577  } else {

579  char *s2_lower = (char *)safe_malloc(l2 + 1);

581  lacf_strlcpy(s1_lower, s1, l1 + 1);
582  lacf_strlcpy(s2_lower, s2, l2 + 1);
583  strtolower(s1_lower);
584  strtolower(s2_lower);
585  res = strncmp(s1_lower, s2_lower, n);
586  free(s1_lower);

588  }

682 static inline int
683 fixed_decimals(double x, int digits)
684 {
685  if (x > -1e-10 && x < 1e-10)
686  return (MAX(digits - 1, 0));
687  if (x < 0)
688  x *= -1;
689  /* This avoids the leading "0." not counting to the digit number */
690  if (x < 1)

692  return (clampi(digits - ceil(log10(x)), 0, digits));
