124 static inline double
125 normalize_hdg(double hdg)
126 {
127  hdg = fmod(hdg, 360);
128  /* Flip negative into positive */
129  if (hdg < 0.0)
130  hdg += 360.0;
131  /* Necessary to avoid FP rounding errors */
132  if (hdg <= 0.0 || hdg > 360.0) {
133  hdg = clamp(hdg, 0, 360);
134  /* Avoid negative zero */
135  if (hdg == -0.0)
136  hdg = 0.0;
137  }
138  return (hdg);
139 }

418  * Portable version of BSD & POSIX strcasecmp.
419  */
420 static inline int
421 lacf_strcasecmp(const char *s1, const char *s2)
422 {
423  int l1, l2, res;
424 
425  ASSERT(s1 != NULL);
426  ASSERT(s2 != NULL);
427 
428  l1 = strlen(s1);
429  l2 = strlen(s2);
430 
431  if (l1 < 4096 && l2 < 4096) {
432  char s1_lower[l1 + 1], s2_lower[l2 + 1];
433 
434  lacf_strlcpy(s1_lower, s1, l1 + 1);
435  lacf_strlcpy(s2_lower, s2, l2 + 1);
436  strtolower(s1_lower);
437  strtolower(s2_lower);
438  res = strcmp(s1_lower, s2_lower);
439  } else {
440  char *s1_lower = (char *)safe_malloc(l1 + 1);
441  char *s2_lower = (char *)safe_malloc(l2 + 1);
442 
443  lacf_strlcpy(s1_lower, s1, l1 + 1);
444  lacf_strlcpy(s2_lower, s2, l2 + 1);
445  strtolower(s1_lower);
446  strtolower(s2_lower);
447  res = strcmp(s1_lower, s2_lower);
448  free(s1_lower);
449  free(s2_lower);
450  }
451  return (res);
452 }

496 static inline int
497 fixed_decimals(double x, int digits)
498 {
499  if (x > -1e-10 && x < 1e-10)
500  return (MAX(digits - 1, 0));
501  if (x < 0)
502  x *= -1;
503  /* This avoids the leading "0." not counting to the digit number */
504  if (x < 1)
505  digits = MAX(digits - 1, 0);
506  return (clampi(digits - ceil(log10(x)), 0, digits));
