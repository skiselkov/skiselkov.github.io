188 /*
189  * How to turn to get from hdg1 to hdg2 with positive being right and negative
190  * being left. Always turns the shortest way around (<= 180 degrees).
191  */
192 double
193 rel_hdg_impl(double hdg1, double hdg2, const char *file, int line)
194 {
195  ASSERT_MSG(is_valid_hdg(hdg1) && is_valid_hdg(hdg2),
196  "from: %s:%d %f -> %f", file, line, hdg1, hdg2);
197  if (hdg1 > hdg2) {
198  if (hdg1 > hdg2 + 180)
199  return (360 - hdg1 + hdg2);
200  else
201  return (-(hdg1 - hdg2));
202  } else {
203  if (hdg2 > hdg1 + 180)
204  return (-(360 - hdg2 + hdg1));
205  else
206  return (hdg2 - hdg1);
207  }
208 }

220 bool_t
221 is_valid_vor_freq(double freq_mhz)
222 {
223  unsigned freq_khz = freq_mhz * 1000;
224 
225  /* Check correct frequency band */
226  if (freq_khz < 108000 || freq_khz > 117950)
227  return (B_FALSE);
228  /*
229  * Check the LOC band - freq must be multiple of 200 kHz or
230  * remainder must be 50 kHz.
231  */
232  if (freq_khz >= 108000 && freq_khz <= 112000 &&
233  freq_khz % 200 != 0 && freq_khz % 200 != 50)
234  return (B_FALSE);
235  /* Above 112 MHz, frequency must be multiple of 50 kHz */
236  if (freq_khz % 50 != 0)
237  return (B_FALSE);
238 
239  return (B_TRUE);
240 }

242 bool_t
243 is_valid_loc_freq(double freq_mhz)
244 {
245  unsigned freq_khz = freq_mhz * 1000;
246 
247  /* Check correct frequency band */
248  if (freq_khz < 108100 || freq_khz > 111950)
249  return (B_FALSE);
250  /* Check 200 kHz spacing with 100 kHz or 150 kHz remainder. */
251  if (freq_khz % 200 != 100 && freq_khz % 200 != 150)
252  return (B_FALSE);
253 
254  return (B_TRUE);
255 }

277 /*
278  * Checks if a string is a valid ICAO airport code. ICAO airport codes always:
279  * 1) are 4 characters long
280  * 2) are all upper case
281  * 3) contain only the letters A-Z
282  * 4) may not start with I, J, Q or X
283  */
284 bool_t
285 is_valid_icao_code(const char *icao)
286 {
287  if (strlen(icao) != 4)
288  return (B_FALSE);
289  for (int i = 0; i < 4; i++) {
290  if ((icao[i] < 'A' || icao[i] > 'Z') &&
291  (icao[i] < '0' || icao[i] > '9')) {
292  return (B_FALSE);
293  }
294  }
295  if (icao[0] == 'I' || icao[0] == 'J' || icao[0] == 'Q' ||
296  icao[0] == 'X')
297  return (B_FALSE);
298  return (B_TRUE);
299 }

301 /*
302  * Checks if a string is a valid IATA airport code. ICAO airport codes always:
303  * 1) are 3 characters long
304  * 2) are all upper case
305  * 3) contain only the letters A-Z
306  * 4) may not start with Q
307  */
308 bool_t
309 is_valid_iata_code(const char *iata)
310 {
311  if (strlen(iata) != 3)
312  return (B_FALSE);
313  for (int i = 0; i < 3; i++) {
314  if (iata[i] < 'A' || iata[i] > 'Z')
315  return (B_FALSE);
316  }
317  if (iata[0] == 'Q')
318  return (B_FALSE);
319  return (B_TRUE);

322 /*
323  * Extracts the country code portion from an ICAO airport code. Because the
324  * returned string pointer for any given country code is always the same, it
325  * is possible to simply compare country code correspondence using a simple
326  * pointer value comparison. If the ICAO country code is not known, returns
327  * NULL instead.
328  * If the passed string isn't a valid ICAO country code, returns NULL as well.
329  */
330 const char *
331 extract_icao_country_code(const char *icao)

333  if (!is_valid_icao_code(icao))
334  return (NULL);

338  strlen(icao_country_codes[i])) == 0)
339  return (icao_country_codes[i]);
340  }
341  return (NULL);
342 }

344 /*
345  * Checks a runway ID for correct formatting. Runway IDs are
346  * 2- or 3-character strings conforming to the following format:
347  * *) Two-digit runway heading between 01 and 36. Headings less
348  * than 10 are always prefixed by a '0'. "00" is NOT valid.
349  * *) An optional parallel runway discriminator, one of 'L', 'R', 'C'
350  * or 'T'.
351  */
352 bool_t
353 is_valid_rwy_ID(const char *rwy_ID)
354 {
355  int hdg;
356  int len = strlen(rwy_ID);
357 
358  if (len < 2 || len > 3 || !isdigit(rwy_ID[0]) || !isdigit(rwy_ID[1]))
359  return (B_FALSE);
360  hdg = atoi(rwy_ID);
361  if (hdg == 0 || hdg > 36)
362  return (B_FALSE);
363  if (len == 3) {
364  if (rwy_ID[2] != 'R' && rwy_ID[2] != 'L' && rwy_ID[2] != 'C' &&
365  rwy_ID[2] != 'T') {
366  return (B_FALSE);
367  }
368  }
369 
370  return (B_TRUE);
371 }

373 /*
374  * Copies a runway identifier from an unknown source to a 4-character runway
375  * identifier buffer. This also performs the following transformations on the
376  * runway identifier:
377  * 1) if the runway identifier had a trailing 'T', it is stripped
378  * 2) if the runway was a US single-digit runway number, a '0' is prepended
379  * The function performs no validity checking other than making sure that the
380  * output buffer is not overflown. The caller is responsible for validating
381  * the result after copy_rwy_ID() returns using is_valid_rwy_ID().
382  */
383 void
384 copy_rwy_ID(const char *src, char dst[4])
385 {
386  int len;
387 
388  /* make sure dst is populated with *something* */
389  memset(dst, 0, 4);
390  lacf_strlcpy(dst, src, 4);
391 
392  /*
393  * Strip an optional trailing true hdg indicator. If the runway had
394  * a suffix (e.g. '08LT'), the previous copy step will have already
395  * stripped it.
396  */
397  len = strlen(dst);
398  if (len > 0 && dst[len - 1] == 'T') {
399  dst[len - 1] = '\0';
400  len--;
401  }
402 
403  /* check if the identifier is a US runway number and translate it */
404  if (dst[0] >= '1' && dst[0] <= '9' &&
405  (len == 1 || (len == 2 && !isdigit(dst[1])))) {
406  memmove(dst + 1, dst, len + 1);
407  dst[0] = '0';
408  }
409 }

428 static time_t
429 cycle2start(int i)
430 {
431  struct tm cyc_tm = { .tm_year = 0 };
432  char day[4], month[4];
433  int year = (airac_eff_dates[i].cycle / 100) + 100;
434  const char *start = airac_eff_dates[i].start;
435 
436  snprintf(day, sizeof (day), "%c%c", start[0], start[1]);
437  snprintf(month, sizeof (month), "%c%c%c", start[3], start[4], start[5]);
438 
439  cyc_tm.tm_year = year;
440  cyc_tm.tm_mday = atoi(day);
441  cyc_tm.tm_mon = month2nr(month);
442  cyc_tm.tm_wday = -1;
443  cyc_tm.tm_yday = -1;
444  cyc_tm.tm_isdst = -1;
445 
446  return (lacf_timegm(&cyc_tm));
447 }

468 bool_t
469 airac_cycle2exp_date(int cycle, char buf[16], time_t *cycle_end_p)
470 {
471  for (int i = 0; airac_eff_dates[i].cycle != -1; i++) {
472  if (airac_eff_dates[i].cycle == cycle) {
473  /*
474  * Grab the effective date of the next cycle and
475  * subtract one day second. The end time will be
476  * the previous day at 23:59:59.
477  */
478  time_t cycle_end_t = cycle2start(i) + 28 * 86400;
479  struct tm end_tm;
480 
481  lacf_gmtime_r(&cycle_end_t, &end_tm);
482  if (buf != NULL) {
483  snprintf(buf, 16, "%02d-%s", end_tm.tm_mday,
484  nr2month(end_tm.tm_mon));
485  }
486  if (cycle_end_p != NULL)
487  *cycle_end_p = cycle_end_t;
488 
489  return (B_TRUE);
490  }
491  }
492  return (B_FALSE);
493 }

546 /*
547  * Splits up an input string by separator into individual components.
548  * The input string is `input' and the separator string is `sep'. The
549  * skip_empty flag indicates whether to skip empty components (i.e.
550  * two occurences of the separator string are next to each other).
551  * Returns an array of pointers to the component strings and the number
552  * of components in the `num' return parameter. The array of pointers
553  * as well as the strings themselves are safe_malloc'd and should be freed
554  * by the caller. Use free_strlist for that.
555  */
556 char **
557 strsplit(const char *input, const char *sep, bool_t skip_empty, size_t *num)
558 {
559  char **result;
560  size_t i = 0, n = 0;
561  size_t seplen = strlen(sep);
562 
563  for (const char *a = input, *b = strstr(a, sep);
564  a != NULL; a = b + seplen, b = strstr(a, sep)) {
565  if (b == NULL) {
566  b = input + strlen(input);
567  if (a != b || !skip_empty)
568  n++;
569  break;
570  }
571  if (a == b && skip_empty)
572  continue;
573  n++;
574  }
575 
576  result = safe_calloc(n, sizeof (char *));
577 
578  for (const char *a = input, *b = strstr(a, sep);
579  a != NULL; a = b + seplen, b = strstr(a, sep)) {
580  if (b == NULL) {
581  b = input + strlen(input);
582  if (a != b || !skip_empty) {
583  result[i] = safe_calloc(b - a + 1,
584  sizeof (char));
585  memcpy(result[i], a, b - a);
586  }
587  break;
588  }
589  if (a == b && skip_empty)
590  continue;
591  ASSERT(i < n);
592  result[i] = safe_calloc(b - a + 1, sizeof (char));
593  memcpy(result[i], a, b - a);
594  i++;
595  }
596 
597  if (num != NULL)
598  *num = n;
599 
600  return (result);
601 }

860  * the path, resolving directory returns as necessary. This also first flips
861  * all path separators to the platform-appropriate type using fix_pathsep.

865 {

872  for (elem = path; *elem != '\0';) {
