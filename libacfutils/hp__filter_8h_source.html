87 static inline double
88 hp_filter_update(hp_filter_t *filt, double m, double d_t)
89 {
90  ASSERT(filt != NULL);
91  ASSERT3F(d_t, >, 0);
92  if (isnan(filt->state)) {
93  filt->state = m;
94  filt->prev = m;
95  } else {
96  double alpha = filt->RC / (filt->RC + d_t);
97  ASSERT(!isnan(filt->prev));
98  filt->state = alpha * filt->state + alpha * (m - filt->prev);
99  filt->prev = m;
100  }
101  return (filt->state);
102 }
