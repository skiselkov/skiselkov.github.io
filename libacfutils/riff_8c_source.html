37#define RIFF_ID FOURCC("RIFF")
38#define LIST_ID FOURCC("LIST")
39#define IS_LIST(chunk) (chunk->fourcc == RIFF_ID || chunk->fourcc == LIST_ID)

41/*
42 * Frees a RIFF chunk and all its subchunks. Call this function on the
43 * top-level chunk returned by riff_parse to free the whole structure.
44 */
45void
46riff_free_chunk(riff_chunk_t *c)
47{
48 if (c->fourcc == RIFF_ID || c->fourcc == LIST_ID) {
49 for (riff_chunk_t *sc = list_head(&c->subchunks); sc != NULL;
50 sc = list_head(&c->subchunks)) {
51 list_remove(&c->subchunks, sc);
52 riff_free_chunk(sc);
53 }
54 list_destroy(&c->subchunks);
55 }
56 free(c);
57}

59/*
60 * Recursive chunk parser. This is the actual worker of riff_parse. It is
61 * invoked for every chunk found in the RIFF file.
62 */
63static riff_chunk_t *
64riff_parse_chunk(uint8_t *buf, size_t bufsz, bool_t bswap)
65{
66 riff_chunk_t *c = safe_calloc(1, sizeof (*c));
67
68 memcpy(&c->fourcc, buf, sizeof (c->fourcc));
69 memcpy(&c->datasz, buf + 4, sizeof (c->datasz));
70 if (bswap) {
71 c->fourcc = BSWAP32(c->fourcc);
72 c->datasz = BSWAP32(c->datasz);
73 c->bswap = B_TRUE;
74 }
75 if (c->datasz > bufsz - 8) {
76 free(c);
77 return (NULL);
78 }
79 if (c->fourcc == RIFF_ID || c->fourcc == LIST_ID) {
80 size_t consumed = 0;
81 uint8_t *subbuf;
82
83 /* check there's enough data for a listcc field */
84 if (c->datasz < 4) {
85 free(c);
86 return (NULL);
87 }
88 memcpy(&c->listcc, buf + 8, sizeof (c->listcc));
89 if (bswap)
90 c->listcc = BSWAP32(c->listcc);
91
92 /* we exclude the listcc field from our data pointer */
93 c->data = buf + 12;
94 c->datasz -= 4;
95
96 list_create(&c->subchunks, sizeof (riff_chunk_t),
97 offsetof(riff_chunk_t, node));
98
99 subbuf = c->data;
100 while (consumed != c->datasz) {
101 riff_chunk_t *sc = riff_parse_chunk(subbuf,
102 c->datasz - consumed, bswap);
103
104 if (sc == NULL) {
105 riff_free_chunk(c);
106 return (NULL);
107 }
108 list_insert_tail(&c->subchunks, sc);
109 consumed += sc->datasz + 8;
110 subbuf += sc->datasz + 8;
111 if (IS_LIST(sc)) {
112 /* because we exclude the listcc from datasz */
113 consumed += 4;
114 subbuf += 4;
115 }
116 if (consumed & 1) {
117 /* realign to two-byte boundary */
118 consumed++;
119 subbuf++;
120 }
121 ASSERT(consumed <= c->datasz);
122 ASSERT(subbuf <= buf + bufsz);
123 }
124 } else {
125 /* plain data chunk */
126 c->data = buf + 8;
127 }
128
129 return (c);
130}

132/*
133 * Parses a RIFF file and returns its top-level chunk. The parsed structure
134 * contains references to the input buffer, so it must not be freed by the
135 * caller. Chunk's fourcc's are automatically endian-swapped if the file's
136 * endianness doesn't match ours. Chunk contents AREN'T byteswapped. Instead,
137 * the caller should use the `bswap' field in the riff_chunk_t to determine
138 * if the file was found to be reverse endian and thus if it needs to
139 * perform byteswapping of chunk contents.
140 *
141 * @param filetype The 32-bit little-endian fourcc code identifying the
142 * RIFF file type (e.g. 'WAVE').
143 * @param buf Input buffer to parse which contains the entire RIFF file.
144 * @param bufsz Input buffer size.
145 *
146 * If the file isn't of the appropriate type or is malformed, NULL is
147 * returned instead.
148 */
149riff_chunk_t *
150riff_parse(uint32_t filetype, uint8_t *buf, size_t bufsz)
151{
152 bool_t bswap;
153 const uint32_t *buf32 = (uint32_t *)buf;
154 uint32_t ftype;
155
156 /* check the buffer isn't too small to be a valid RIFF file */
157 if (bufsz < 3 * sizeof (uint32_t))
158 return (NULL);
159
160 /* make sure the header signature matches & determine endianness */
161 if (buf32[0] == RIFF_ID)
162 bswap = B_FALSE;
163 else if (buf32[0] == BSWAP32(RIFF_ID))
164 bswap = B_TRUE;
165 else
166 return (NULL);
167
168 /* check the file size fits in our buffer */
169 if ((bswap ? BSWAP32(buf32[1]) : buf32[1]) > bufsz - 8)
170 return (NULL);
171 /* check the file type requested by the caller */
172 ftype = (bswap ? BSWAP32(buf32[2]) : buf32[2]);
173 if (ftype != filetype)
174 return (NULL);
175
176 /* now we can be reasonably sure that this is a somewhat valid file */
177 return (riff_parse_chunk(buf, bufsz, bswap));
178}

180/*
181 * Locates a specific chunk in a RIFF file. In `topchunk' pass the top-level
182 * RIFF file chunk. The remaining arguments must be a 0-terminated list of
183 * uint32_t FourCCs of the nested chunks. Don't include the top-level chunk ID.
184 * Returns a pointer to a riff_chunk_t (if found), or NULL if not found.
185 */
186riff_chunk_t *
187riff_find_chunk(riff_chunk_t *topchunk, ...)
188{
189 va_list ap;
190 uint32_t fourcc;
191
192 ASSERT(topchunk != NULL);
193 ASSERT(topchunk->fourcc == RIFF_ID);
194
195 va_start(ap, topchunk);
196 while ((fourcc = va_arg(ap, uint32_t)) != 0) {
197 riff_chunk_t *sc;
198
199 ASSERT(fourcc != LIST_ID && fourcc != RIFF_ID);
200 if (topchunk->fourcc != LIST_ID &&
201 topchunk->fourcc != RIFF_ID)
202 return (NULL);
203 for (sc = list_head(&topchunk->subchunks); sc != NULL;
204 sc = list_next(&topchunk->subchunks, sc)) {
205 if (sc->fourcc == fourcc || (sc->listcc == fourcc &&
206 (sc->fourcc == LIST_ID || sc->fourcc == RIFF_ID)))
207 break;
208 }
209 if (sc == NULL)
210 return (NULL);
211 topchunk = sc;
212 }
213 va_end(ap);
214
215 ASSERT(topchunk != NULL);
216
217 return (topchunk);
218}

220/*
221 * The actual recursive implementation of riff_dump with variable indent.
222 */
223static char *
224riff_dump_impl(const riff_chunk_t *chunk, int indent)
225{
226 char fourcc[5] = {
227 chunk->fourcc & 0xff, (chunk->fourcc >> 8) & 0xff,
228 (chunk->fourcc >> 16) & 0xff, (chunk->fourcc >> 24) & 0xff, 0
229 };
230 char listcc[5] = {
231 chunk->listcc & 0xff, (chunk->listcc >> 8) & 0xff,
232 (chunk->listcc >> 16) & 0xff, (chunk->listcc >> 24) & 0xff, 0
233 };
234 char *indent_str = safe_calloc(indent + 1, sizeof (*indent_str));
235 int len;
236 char *buf;
237
238 memset(indent_str, ' ', indent);
239 indent_str[indent] = 0;
240
241#define DUMP_FMT_ARGS "%s%s%s%s%s%08x\n", indent_str, fourcc, \
242 IS_LIST(chunk) ? " " : " ", IS_LIST(chunk) ? listcc : "", \
243 IS_LIST(chunk) ? " " : "", chunk->datasz
244 len = snprintf(NULL, 0, DUMP_FMT_ARGS);
245 buf = safe_malloc(len + 1);
246 snprintf(buf, len + 1, DUMP_FMT_ARGS);
247#undef DUMP_FMT_ARGS
248
249 if (!IS_LIST(chunk)) {
250 free(indent_str);
251 return (buf);
252 }
253 for (const riff_chunk_t *sc = list_head(&chunk->subchunks); sc != NULL;
254 sc = list_next(&chunk->subchunks, sc)) {
255 char *subbuf = riff_dump_impl(sc, indent + 2);
256
257 len += strlen(subbuf);
258 buf = safe_realloc(buf, len + 1);
259 strcat(buf, subbuf);
260 free(subbuf);
261 }
262 free(indent_str);
263 return (buf);
264}

266/*
267 * Returns a malloc'd string containing the format dump of a RIFF file
268 * (for debugging). The string is formatted into individual lines:
269 *
270 * RIFF FTYP ChunkSz
271 * CHNK ChunkSz
272 * LIST CHNK ChunkSz
273 * CHNK ChunkSz
274 * CHNK ChunkSz
275 * CHNK ChunkSz
