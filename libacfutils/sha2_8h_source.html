308static inline void
309sha256_update(sha256_ctx *ctx, const void *message_in, unsigned len)
310{
311 unsigned block_nb;
312 unsigned new_len, rem_len, tmp_len;
313 const unsigned char *shifted_message;
314 const unsigned char *message = message_in;
315
316 tmp_len = SHA256_BLOCK_SIZE - ctx->len;
317 rem_len = len < tmp_len ? len : tmp_len;
318
319 memcpy(&ctx->block[ctx->len], message, rem_len);
320
321 if (ctx->len + len < SHA256_BLOCK_SIZE) {
322 ctx->len += len;
323 return;
324 }
325
326 new_len = len - rem_len;
327 block_nb = new_len / SHA256_BLOCK_SIZE;
328
329 shifted_message = message + rem_len;
330
331 sha256_transf(ctx, ctx->block, 1);
332 sha256_transf(ctx, shifted_message, block_nb);
333
334 rem_len = new_len % SHA256_BLOCK_SIZE;
335
336 memcpy(ctx->block, &shifted_message[block_nb << 6],
337 rem_len);
338
339 ctx->len = rem_len;
340 ctx->tot_len += (block_nb + 1) << 6;
341}

343static inline void
344sha256_final(sha256_ctx *ctx, void *digest_out)
345{
346 unsigned block_nb;
347 unsigned pm_len;
348 unsigned len_b;
349 unsigned char *digest = digest_out;
350
351#ifndef SHA2_UNROLL_LOOPS
352 int i;
353#endif
354
355 block_nb = (1 + ((SHA256_BLOCK_SIZE - 9)
356 < (ctx->len % SHA256_BLOCK_SIZE)));
357
358 len_b = (ctx->tot_len + ctx->len) << 3;
359 pm_len = block_nb << 6;
360
361 memset(ctx->block + ctx->len, 0, pm_len - ctx->len);
362 ctx->block[ctx->len] = 0x80;
363 UNPACK32(len_b, ctx->block + pm_len - 4);
364
365 sha256_transf(ctx, ctx->block, block_nb);
366
367#ifndef SHA2_UNROLL_LOOPS
368 for (i = 0 ; i < 8; i++) {
369 UNPACK32(ctx->h[i], &digest[i << 2]);
370 }
371#else
372 UNPACK32(ctx->h[0], &digest[ 0]);
373 UNPACK32(ctx->h[1], &digest[ 4]);
374 UNPACK32(ctx->h[2], &digest[ 8]);
375 UNPACK32(ctx->h[3], &digest[12]);
376 UNPACK32(ctx->h[4], &digest[16]);
377 UNPACK32(ctx->h[5], &digest[20]);
378 UNPACK32(ctx->h[6], &digest[24]);
379 UNPACK32(ctx->h[7], &digest[28]);
380#endif /* !SHA2_UNROLL_LOOPS */
381}

383static inline void
384hmac_sha256_init(hmac_sha256_ctx *ctx, const unsigned char *key,
385 unsigned key_size)
386{
387 unsigned fill;
388 unsigned num;
389
390 const unsigned char *key_used;
391 unsigned char key_temp[SHA256_DIGEST_SIZE];
392 int i;
393
394 if (key_size == SHA256_BLOCK_SIZE) {
395 key_used = key;
396 num = SHA256_BLOCK_SIZE;
397 } else {
398 if (key_size > SHA256_BLOCK_SIZE){
399 key_used = key_temp;
400 num = SHA256_DIGEST_SIZE;
401 sha256(key, key_size, key_temp);
402 } else { /* key_size > SHA256_BLOCK_SIZE */
403 key_used = key;
404 num = key_size;
405 }
406 fill = SHA256_BLOCK_SIZE - num;
407
408 memset(ctx->block_ipad + num, 0x36, fill);
409 memset(ctx->block_opad + num, 0x5c, fill);
410 }
411
412 for (i = 0; (unsigned) i < num; i++) {
413 ctx->block_ipad[i] = key_used[i] ^ 0x36;
414 ctx->block_opad[i] = key_used[i] ^ 0x5c;
415 }
416
417 sha256_init(&ctx->ctx_inside);
418 sha256_update(&ctx->ctx_inside, ctx->block_ipad, SHA256_BLOCK_SIZE);
419
420 sha256_init(&ctx->ctx_outside);
421 sha256_update(&ctx->ctx_outside, ctx->block_opad, SHA256_BLOCK_SIZE);
422
423 /* for hmac_reinit */
424 memcpy(&ctx->ctx_inside_reinit, &ctx->ctx_inside,
425 sizeof(sha256_ctx));
426 memcpy(&ctx->ctx_outside_reinit, &ctx->ctx_outside,
427 sizeof(sha256_ctx));
428}

437static inline void hmac_sha256_final(hmac_sha256_ctx *ctx, void *mac,

439{
440 unsigned char digest_inside[SHA256_DIGEST_SIZE];
441 unsigned char mac_temp[SHA256_DIGEST_SIZE];
442
443 sha256_final(&ctx->ctx_inside, digest_inside);
444 sha256_update(&ctx->ctx_outside, digest_inside, SHA256_DIGEST_SIZE);

446 assert(mac_size <= SHA256_DIGEST_SIZE);
447 memcpy(mac, mac_temp, mac_size);
