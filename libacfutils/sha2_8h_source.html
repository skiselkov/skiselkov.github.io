310 {

312  unsigned new_len, rem_len, tmp_len;
313  const unsigned char *shifted_message;
314  const unsigned char *message = message_in;
315 
316  tmp_len = SHA256_BLOCK_SIZE - ctx->len;
317  rem_len = len < tmp_len ? len : tmp_len;
318 
319  memcpy(&ctx->block[ctx->len], message, rem_len);
320 
321  if (ctx->len + len < SHA256_BLOCK_SIZE) {
322  ctx->len += len;
323  return;
324  }
325

327  block_nb = new_len / SHA256_BLOCK_SIZE;
328 
329  shifted_message = message + rem_len;
330

332  sha256_transf(ctx, shifted_message, block_nb);
333 
334  rem_len = new_len % SHA256_BLOCK_SIZE;
335 
336  memcpy(ctx->block, &shifted_message[block_nb << 6],
337  rem_len);
338 
339  ctx->len = rem_len;
