26 /*
27  * A generic CRC64 implementation from OpenSolaris. Be sure to call
28  * crc64_init before using it. Then just call crc64() and pass it the
29  * data you want checksummed. Also includes a fast & light-weight
30  * portable pseudo random number generator.
31  */
32 
33 #define CRC64_POLY 0xC96C5795D7870F42ULL /* ECMA-182, reflected form */
34 
35 static uint64_t crc64_table[256];
36 static uint64_t rand_seed = 0;
37 
38 void
39 crc64_init(void)
40 {
41  for (int i = 0; i < 256; i++) {
42  uint64_t *ct;
43  int j;
44  for (ct = crc64_table + i, *ct = i, j = 8; j > 0; j--)
45  *ct = (*ct >> 1) ^ (-(*ct & 1) & CRC64_POLY);
46  }
47 }

49 /*
50  * Computes the CRC64 checksum of a block of input data.
51  */
52 uint64_t
53 crc64(const void *input, size_t sz)
54 {
55  uint64_t crc;
56 
57  crc64_state_init(&crc);
58  return (crc64_append(crc, input, sz));
59 }
60 
61 uint64_t
62 crc64_append(uint64_t crc, const void *input, size_t sz)
63 {
64  const uint8_t *in_bytes = input;
65 
66  ASSERT3U(crc64_table[128], ==, CRC64_POLY);
67  for (size_t i = 0; i < sz; i++)
68  crc = (crc >> 8) ^ crc64_table[(crc ^ in_bytes[i]) & 0xFF];
69 
70  return (crc);
71 }

84 /*
85  * Grabs a random 64-bit number from the PRNG. This function isn't
86  * thread-safe, so take care not to rely on its output being super-duper
87  * unpredictable in multi-threade apps. You shouldn't be relying on it
88  * for anything more than lightweight randomness duties which need to be
89  * fast above everything else.
90  *
91  * DO NOT use this for cryptographically secure randomness operations
92  * (e.g. generating encryption key material). See osrand.c for a
93  * high-quality PRNG.
94  */
95 uint64_t
96 crc64_rand(void)
97 {
98  rand_seed = crc64(&rand_seed, sizeof (rand_seed));
99  return (rand_seed);
100 }

102 /*
103  * Returns a random number from the PRNG, but instead of returning a 64-bit
104  * integer between 0 and UINT64_MAX, this function returns a double value
105  * from 0.0 to 1.0 (inclusive).
106  */
107 double
108 crc64_rand_fract(void)
109 {
110  return (crc64_rand() / (double)UINT64_MAX);
111 }

113 /*
114  * Returns a random number using a normal distribution with standard
115  * deviation equal to sigma.
116  */
117 double
118 crc64_rand_normal(double sigma)
119 {
120  double x = crc64_rand_fract();
121  double y = crc64_rand_fract();
122  double z = sqrt(-2 * log(x)) * cos(2 * M_PI * y);
123  return (sigma * z);
124 }
