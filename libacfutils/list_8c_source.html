230 list_move_tail(list_t *dst, list_t *src)
231 {
232  list_node_t *dstnode = &dst->list_head;
233  list_node_t *srcnode = &src->list_head;
234 
235  ASSERT(dst->list_size == src->list_size);
236  ASSERT(dst->list_offset == src->list_offset);
237 
238  if (list_empty(src))
239  return;
240 
241  dstnode->list_prev->list_next = srcnode->list_next;
242  srcnode->list_next->list_prev = dstnode->list_prev;
243  dstnode->list_prev = srcnode->list_prev;
244  srcnode->list_prev->list_next = dstnode;
245  dst->list_count += src->list_count;
246 
247  /* empty src list */
248  srcnode->list_next = srcnode->list_prev = srcnode;
249  src->list_count = 0;
250 }
