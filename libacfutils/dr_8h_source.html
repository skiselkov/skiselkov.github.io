61 /*
62  * Attempts to find a dataref in the X-Plane dataref system and store
63  * a handle to it in `dr'. The dataref must exist (have been created)
64  * at the time of this call. There is no support for late (lazy)
65  * resolving of datarefs.
66  * Returns B_TRUE if the dataref was found, B_FALSE otherwise.
67  */

70 /*
71  * Same as `dr_find', but does a "forcible" lookup. That means, if
72  * the dataref doesn't exist, instead of returning B_FALSE, this macro
73  * causes a hard assertion failure. Use this to look up datarefs that
74  * you absolutely require to exist (such as those that are part of
75  * X-Plane itself).
76  */
77 #define fdr_find(dr, ...) \
78  do { \
79  if (!dr_find(dr, __VA_ARGS__)) { \
80  char drname[DR_MAX_NAME_LEN]; \
81  snprintf(drname, sizeof (drname), __VA_ARGS__); \
82  VERIFY_MSG(0, "dataref \"%s\" not found", drname); \
83  } \
84  } while (0)

95  * Reads and writes an integer dataref. If the dataref is not an integer,
96  * these functions automatically convert the types as necessary, so you
97  * can read a float dataref as an int without having to worry about
98  * dataref types. These functions can also work with array datarefs. If
99  * the dataref is an array, these functions simply read/write the array's

119 /*
120  * Extra error-checking version of dr_getf. If the read value is
121  * a NAN, this trips an assertion failure. This helps protect
122  * critical sections of code from being fed unexpected garbage.
123  */
124 #define dr_getf_prot(__dr) dr_getf_prot_impl((__dr), DR_DEBUG(#__dr))
125 static inline double dr_getf_prot_impl(const dr_t *dr, DR_DEBUG_VARS) HOT_ATTR;
126 static inline double
127 dr_getf_prot_impl(const dr_t *dr, DR_DEBUG_VARS)
128 {
129  double x = dr_getf_impl(dr, filename, line, varname);
130  ASSERT_MSG(!isnan(x) && isfinite(x), "%s:%d: Dataref %s (varname %s) "
131  "contains a garbage value (%f). We didn't write that, somebody "
132  "else did! Remove extraneous plugins and try to isolate the cause.",
133  filename, line, dr->name, varname, x);
134  return (x);
135 }
