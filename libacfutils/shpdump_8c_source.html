41 static void
42 logfunc(const char *str)
43 {
44  puts(str);
45 }

47 int
48 main(int argc, char *argv[])
49 {
50  int opt;
51  SHPHandle shp;
52  int n_ent, shp_type;
53  cairo_t *cr;
54  cairo_surface_t *surf;
55  int lat, lon;
56  char lat_buf[8], lon_buf[8];
57  char *last_part;
58  char *out_filename = "shp.png";
59  bool_t verbose = B_FALSE;
60 
61  log_init(logfunc, "dsfdump");
62 
63  while ((opt = getopt(argc, argv, "hvo:")) != -1) {
64  switch (opt) {
65  case 'h':
66  printf("Usage: %s [-hv] [-o outfile.png] <shp-file>\n",
67  argv[0]);
68  exit(EXIT_SUCCESS);
69  case 'o':
70  out_filename = optarg;
71  break;
72  case 'v':
73  verbose = B_TRUE;
74  break;
75  default:
76  fprintf(stderr, "Usage: %s [-hv] [-o outfile.png] "
77  "<shp-file>\n", argv[0]);
78  exit(EXIT_FAILURE);
79  }
80  }
81 
82  if (optind >= argc) {
83  fprintf(stderr, "Missing filename argument. "
84  "Try -h for help.\n");
85  exit(EXIT_FAILURE);
86  }
87 
88  shp = SHPOpen(argv[optind], "rb");
89  if (shp == NULL) {
90  fprintf(stderr, "Error opening shp file: %s\n",
91  strerror(errno));
92  return (1);
93  }
94 
95  last_part = strrchr(argv[optind], '/');
96  if (last_part == NULL)
97  last_part = argv[optind];
98  else
99  last_part++;
100  if (strlen(last_part) < 7) {
101  fprintf(stderr, "Bad filename: %s\n", argv[optind]);
102  return (1);
103  }
104 
105  memset(lat_buf, 0, sizeof (lat_buf));
106  memset(lon_buf, 0, sizeof (lon_buf));
107  memcpy(lat_buf, last_part, 3);
108  memcpy(lon_buf, &last_part[3], 4);
109  lat = atoi(lat_buf);
110  lon = atoi(lon_buf);
111  if (verbose)
112  printf("lat: %d lon: %d\n", lat, lon);
113 
114  surf = cairo_image_surface_create(CAIRO_FORMAT_A8, WIDTH, HEIGHT);
115  cr = cairo_create(surf);
116  cairo_set_antialias(cr, CAIRO_ANTIALIAS_NONE);
117 
118  cairo_scale(cr, WIDTH, HEIGHT);
119 
120  cairo_set_source_rgb(cr, 1, 1, 1);
121 
122  SHPGetInfo(shp, &n_ent, &shp_type, NULL, NULL);
123  if (verbose)
124  printf("n_ent: %d shp_type: %d\n", n_ent, shp_type);
125  for (int i = 0; i < n_ent; i++) {
126  SHPObject *obj = SHPReadObject(shp, i);
127 
128  if (obj == NULL) {
129  fprintf(stderr, "Error reading shape %d\n", i);
130  continue;
131  }
132  if (verbose) {
133  printf(" nVertices: %d nParts: %d\n",
134  obj->nVertices, obj->nParts);
135  }
136  cairo_new_path(cr);
137  for (int j = 0; j < obj->nParts; j++) {
138  int start_k, end_k;
139 
140  start_k = obj->panPartStart[j];
141  if (j + 1 < obj->nParts)
142  end_k = obj->panPartStart[j + 1];
143  else
144  end_k = obj->nVertices;
145  if (verbose) {
146  printf(" part: %d (%d - %d)\n", j,
147  start_k, end_k);
148  }
149 
150  cairo_new_sub_path(cr);
151  /*
152  * Note that since `lat' is always the floor of the
153  * latitude, but cairo & PNG address the image from
154  * the top left, to get the PNG to look right, we
155  * need to flip the Y coordinates to make the image
156  * look right when drawn. This is not done in OpenGWPS,
157  * since it instead wants to address the image from the
158  * bottom left corner.
159  */
160  cairo_move_to(cr, obj->padfX[start_k] - lon,
161  (lat + 1) - obj->padfY[start_k]);
162  for (int k = start_k + 1; k < end_k; k++) {
163  if (verbose) {
164  printf(" %d: %f x %f\n", k,
165  obj->padfX[k], obj->padfY[k]);
166  }
167  cairo_line_to(cr, obj->padfX[k] - lon,
168  (lat + 1) - obj->padfY[k]);
169  }
170  }
171  cairo_fill(cr);
172  SHPDestroyObject(obj);
173  }
174  SHPClose(shp);
175 
176  cairo_surface_write_to_png(surf, out_filename);
177 
178  cairo_destroy(cr);
179  cairo_surface_destroy(surf);
180 
181  return (0);
182 }
