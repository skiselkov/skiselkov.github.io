76 /*
77  * Given two values min_val and max_val, returns how far between min_val
78  * and max_val a third value 'x' lies. If `clamp_output' is true, 'x' is
79  * clamped such that it always lies between min_val and max_val. In essence,
80  * this function computes:
81  *
82  * ^
83  * |
84  * 1 -------------------+
85  * | / |
86  * | / |
87  * f(x) ------------+ |
88  * | / | |
89  * | / | |
90  * 0 ------+ | |
91  * | | | |
92  * +-----|-----|------|----->
93  * min_val x max_val
94  */
95 static inline double
96 iter_fract(double x, double min_val, double max_val, bool_t clamp_output)
97 {
98  ASSERT3F(min_val, !=, max_val);
99  x = (x - min_val) / (max_val - min_val);
100  if (clamp_output)
101  x = clamp(x, 0, 1);
102  return (x);
103 }

132 /*
133  * Given an initialized pn_interp_t (see above), calculates the Y value
134  * at a given point.
135  * @param x The X point for which to calculate the interpolated Y value.
136  * @param interp An initialized interpolator (as initialized by pn_interp_init).
137  */
138 static inline double
139 pn_interp_run(double x, const pn_interp_t *interp)
140 {
141  double y = 0, power = 1;
142 
143  ASSERT(interp != NULL);
144  ASSERT(interp->order != 0);
145  for (unsigned i = 0; i < interp->order; i++) {
146  y += interp->coeff[i] * power;
147  power *= x;
148  }
149 
150  return (y);
151 }

153 /*
154  * Implements the smoothstep function from GLSL. See Wikipedia
155  * for more info: https://en.wikipedia.org/wiki/Smoothstep
156  */
157 static inline double
158 smoothstep(double x, double edge0, double edge1)
159 {
160  ASSERT(!isnan(x));
161  ASSERT3F(edge1, >, edge0);
162  x = clamp((x - edge0) / (edge1 - edge0), 0, 1);
163  return (POW2(x) * (3 - 2 * x));
164 }
