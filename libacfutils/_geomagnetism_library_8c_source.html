201 int
202 MAG_Geomag(MAGtype_Ellipsoid Ellip, MAGtype_CoordSpherical CoordSpherical,
203  MAGtype_CoordGeodetic CoordGeodetic,
204  MAGtype_MagneticModel * TimedMagneticModel,
205  MAGtype_GeoMagneticElements * GeoMagneticElements)
206 /*
207 The main subroutine that calls a sequence of WMM sub-functions to calculate the magnetic field elements for a single point.
208 The function expects the model coefficients and point coordinates as input and returns the magnetic field elements and
209 their rate of change. Though, this subroutine can be called successively to calculate a time series, profile or grid
210 of magnetic field, these are better achieved by the subroutine MAG_Grid.
211 
212 INPUT: Ellip
213  CoordSpherical
214  CoordGeodetic
215  TimedMagneticModel
216 
217 OUTPUT : GeoMagneticElements
218 
219 CALLS: MAG_AllocateLegendreFunctionMemory(NumTerms); ( For storing the ALF functions )
220  MAG_ComputeSphericalHarmonicVariables( Ellip, CoordSpherical, TimedMagneticModel->nMax, &SphVariables); (Compute Spherical Harmonic variables )
221  MAG_AssociatedLegendreFunction(CoordSpherical, TimedMagneticModel->nMax, LegendreFunction); Compute ALF
222  MAG_Summation(LegendreFunction, TimedMagneticModel, SphVariables, CoordSpherical, &MagneticResultsSph); Accumulate the spherical harmonic coefficients
223  MAG_SecVarSummation(LegendreFunction, TimedMagneticModel, SphVariables, CoordSpherical, &MagneticResultsSphVar); Sum the Secular Variation Coefficients
224  MAG_RotateMagneticVector(CoordSpherical, CoordGeodetic, MagneticResultsSph, &MagneticResultsGeo); Map the computed Magnetic fields to Geodetic coordinates
225  MAG_CalculateGeoMagneticElements(&MagneticResultsGeo, GeoMagneticElements); Calculate the Geomagnetic elements
226  MAG_CalculateSecularVariationElements(MagneticResultsGeoVar, GeoMagneticElements); Calculate the secular variation of each of the Geomagnetic elements
227 
228  */
229 {
230  MAGtype_LegendreFunction *LegendreFunction;
231  MAGtype_SphericalHarmonicVariables *SphVariables;
232  int NumTerms;
233  MAGtype_MagneticResults MagneticResultsSph, MagneticResultsGeo,
234  MagneticResultsSphVar, MagneticResultsGeoVar;
235 
236  NumTerms =
237  ((TimedMagneticModel->nMax + 1) * (TimedMagneticModel->nMax +
238  2) / 2);
239  LegendreFunction = MAG_AllocateLegendreFunctionMemory(NumTerms); /* For storing the ALF functions */
240  SphVariables = MAG_AllocateSphVarMemory(TimedMagneticModel->nMax);
241  MAG_ComputeSphericalHarmonicVariables(Ellip, CoordSpherical, TimedMagneticModel->nMax, SphVariables); /* Compute Spherical Harmonic variables */
242  MAG_AssociatedLegendreFunction(CoordSpherical, TimedMagneticModel->nMax, LegendreFunction); /* Compute ALF */
243  MAG_Summation(LegendreFunction, TimedMagneticModel, *SphVariables, CoordSpherical, &MagneticResultsSph); /* Accumulate the spherical harmonic coefficients */
244  MAG_SecVarSummation(LegendreFunction, TimedMagneticModel, *SphVariables, CoordSpherical, &MagneticResultsSphVar); /*Sum the Secular Variation Coefficients */
245  MAG_RotateMagneticVector(CoordSpherical, CoordGeodetic, MagneticResultsSph, &MagneticResultsGeo); /* Map the computed Magnetic fields to Geodeitic coordinates */
246  MAG_RotateMagneticVector(CoordSpherical, CoordGeodetic, MagneticResultsSphVar, &MagneticResultsGeoVar); /* Map the secular variation field components to Geodetic coordinates */
247  MAG_CalculateGeoMagneticElements(&MagneticResultsGeo, GeoMagneticElements); /* Calculate the Geomagnetic elements, Equation 19 , WMM Technical report */
248  MAG_CalculateSecularVariationElements(MagneticResultsGeoVar, GeoMagneticElements); /*Calculate the secular variation of each of the Geomagnetic elements */
249 
250  MAG_FreeLegendreMemory(LegendreFunction);
251  MAG_FreeSphVarMemory(SphVariables);
252 
253  return (TRUE);
254 } /*MAG_Geomag */

259 {
260  char line[MAXLINELENGTH];
261  int n, nMax = 0, num_terms, a;
262  FILE *MODELFILE;
263  MODELFILE = fopen(filename, "r");
264  if (MODELFILE == 0) {
265  return (0);
266  }
267  fgets(line, MAXLINELENGTH, MODELFILE);
268  if (line[0] == '%')
269  MAG_readMagneticModel_SHDF(filename, magneticmodel);
270  else {
271  do {
272  if (NULL == fgets(line, MAXLINELENGTH, MODELFILE))
273  break;
274  a = sscanf(line, "%d", &n);
275  if (n > nMax && (n < 99999 && a == 1 && n > 0))
276  nMax = n;
277  } while (n < 99999 && a == 1);
278  num_terms = CALCULATE_NUMTERMS(nMax);
279  (*magneticmodel) = MAG_AllocateModelMemory(num_terms);
280  (*magneticmodel)->nMax = nMax;
281  (*magneticmodel)->nMaxSecVar = nMax;
282  MAG_readMagneticModel(filename, *magneticmodel);
283  (*magneticmodel)->CoefficientFileEndDate =
284  (*magneticmodel)->epoch + 5;
285 
286  }
287  fclose(MODELFILE);
288  return (1);
289 } /*MAG_robustReadMagModels */

1071 void
1072 MAG_CartesianToGeodetic(MAGtype_Ellipsoid Ellip, double x, double y, double z,
1073  MAGtype_CoordGeodetic * CoordGeodetic)
1074 {
1075  /*This converts the Cartesian x, y, and z coordinates to Geodetic Coordinates
1076  x is defined as the direction pointing out of the core toward the point defined
1077  * by 0 degrees latitude and longitude.
1078  y is defined as the direction from the core toward 90 degrees east longitude along
1079  * the equator
1080  z is defined as the direction from the core out the geographic north pole */
1081 
1082  double modified_b, r, e, f, p, q, d, v, g, t, zlong, rlat;
1083 
1084 /*
1085  * 1.0 compute semi-minor axis and set sign to that of z in order
1086  * to get sign of Phi correct
1087  */
1088 
1089  if (z < 0.0)
1090  modified_b = -Ellip.b;
1091  else
1092  modified_b = Ellip.b;
1093 
1094 /*
1095  * 2.0 compute intermediate values for latitude
1096  */
1097  r = sqrt(x * x + y * y);
1098  e = (modified_b * z - (Ellip.a * Ellip.a -
1099  modified_b * modified_b)) / (Ellip.a * r);
1100  f = (modified_b * z + (Ellip.a * Ellip.a -
1101  modified_b * modified_b)) / (Ellip.a * r);
1102 /*
1103  * 3.0 find solution to:
1104  * t^4 + 2*E*t^3 + 2*F*t - 1 = 0
1105  */
1106  p = (4.0 / 3.0) * (e * f + 1.0);
1107  q = 2.0 * (e * e - f * f);
1108  d = p * p * p + q * q;
1109 
1110  if (d >= 0.0) {
1111  v = pow((sqrt(d) - q), (1.0 / 3.0))
1112  - pow((sqrt(d) + q), (1.0 / 3.0));
1113  } else {
1114  v = 2.0 * sqrt(-p)
1115  * cos(acos(q / (p * sqrt(-p))) / 3.0);
1116  }
1117 /*
1118  * 4.0 improve v
1119  * NOTE: not really necessary unless point is near pole
1120  */
1121  if (v * v < fabs(p)) {
1122  v = -(v * v * v + 2.0 * q) / (3.0 * p);
1123  }
1124  g = (sqrt(e * e + v) + e) / 2.0;
1125  t = sqrt(g * g + (f - v * g) / (2.0 * g - e)) - g;
1126 
1127  rlat = atan((Ellip.a * (1.0 - t * t)) / (2.0 * modified_b * t));
1128  CoordGeodetic->phi = RAD2DEG(rlat);
1129 
1130 /*
1131  * 5.0 compute height above ellipsoid
1132  */
1133  CoordGeodetic->HeightAboveEllipsoid =
1134  (r - Ellip.a * t) * cos(rlat) + (z - modified_b) * sin(rlat);
1135 /*
1136  * 6.0 compute longitude east of Greenwich
1137  */
1138  zlong = atan2(y, x);
1139  if (zlong < 0.0)
1140  zlong = zlong + 2 * M_PI;
1141 
1142  CoordGeodetic->lambda = RAD2DEG(zlong);
1143  while (CoordGeodetic->lambda > 180) {
1144  CoordGeodetic->lambda -= 360;
1145  }
1146 
1147 }

1149 int
1150 MAG_GeodeticToSpherical(MAGtype_Ellipsoid Ellip,
1151  MAGtype_CoordGeodetic CoordGeodetic,
1152  MAGtype_CoordSpherical * CoordSpherical)
1153 
1154 /* Converts Geodetic coordinates to Spherical coordinates
1155 
1156  INPUT Ellip data structure with the following elements
1157  double a; semi-major axis of the ellipsoid
1158  double b; semi-minor axis of the ellipsoid
1159  double fla; flattening
1160  double epssq; first eccentricity squared
1161  double eps; first eccentricity
1162  double re; mean radius of ellipsoid
1163 
1164  CoordGeodetic Pointer to the data structure with the following elements updates
1165  double lambda; ( longitude )
1166  double phi; ( geodetic latitude )
1167  double HeightAboveEllipsoid; ( height above the WGS84 ellipsoid (HaE) )
1168  double HeightAboveGeoid; (height above the EGM96 Geoid model )
1169 
1170  OUTPUT CoordSpherical Pointer to the data structure with the following elements
1171  double lambda; ( longitude)
1172  double phig; ( geocentric latitude )
1173  double r; ( distance from the center of the ellipsoid)
1174 
1175 CALLS : none
1176 
1177  */
1178 {
1179  double CosLat, SinLat, rc, xp, zp; /*all local variables */
1180 
1181  /*
1182  ** Convert geodetic coordinates, (defined by the WGS-84
1183  ** reference ellipsoid), to Earth Centered Earth Fixed Cartesian
1184  ** coordinates, and then to spherical coordinates.
1185  */
1186 
1187  CosLat = cos(DEG2RAD(CoordGeodetic.phi));
1188  SinLat = sin(DEG2RAD(CoordGeodetic.phi));
1189 
1190  /* compute the local radius of curvature on the WGS-84 reference ellipsoid */
1191 
1192  rc = Ellip.a / sqrt(1.0 - Ellip.epssq * SinLat * SinLat);
1193 
1194  /* compute ECEF Cartesian coordinates of specified point (for longitude=0) */
1195 
1196  xp = (rc + CoordGeodetic.HeightAboveEllipsoid) * CosLat;
1197  zp = (rc * (1.0 - Ellip.epssq) +
1198  CoordGeodetic.HeightAboveEllipsoid) * SinLat;
1199 
1200  /* compute spherical radius and angle lambda and phi of specified point */
1201 
1202  CoordSpherical->r = sqrt(xp * xp + zp * zp);
1203  CoordSpherical->phig = RAD2DEG(asin(zp / CoordSpherical->r)); /* geocentric latitude */
1204  CoordSpherical->lambda = CoordGeodetic.lambda; /* longitude */
1205 
1206  return (TRUE);
1207 } /*MAG_GeodeticToSpherical */

1330 int
1331 MAG_ComputeSphericalHarmonicVariables(MAGtype_Ellipsoid Ellip,
1332  MAGtype_CoordSpherical CoordSpherical, int nMax,
1333  MAGtype_SphericalHarmonicVariables * SphVariables)
1334 
1335 /* Computes Spherical variables
1336  Variables computed are (a/r)^(n+2), cos_m(lamda) and sin_m(lambda) for spherical harmonic
1337  summations. (Equations 10-12 in the WMM Technical Report)
1338  INPUT Ellip data structure with the following elements
1339  double a; semi-major axis of the ellipsoid
1340  double b; semi-minor axis of the ellipsoid
1341  double fla; flattening
1342  double epssq; first eccentricity squared
1343  double eps; first eccentricity
1344  double re; mean radius of ellipsoid
1345  CoordSpherical A data structure with the following elements
1346  double lambda; ( longitude)
1347  double phig; ( geocentric latitude )
1348  double r; ( distance from the center of the ellipsoid)
1349  nMax integer ( Maxumum degree of spherical harmonic secular model)\
1350 
1351  OUTPUT SphVariables Pointer to the data structure with the following elements
1352  double RelativeRadiusPower[MAG_MAX_MODEL_DEGREES+1]; [earth_reference_radius_km sph. radius ]^n
1353  double cos_mlambda[MAG_MAX_MODEL_DEGREES+1]; cp(m) - cosine of (mspherical coord. longitude)
1354  double sin_mlambda[MAG_MAX_MODEL_DEGREES+1]; sp(m) - sine of (mspherical coord. longitude)
1355  CALLS : none
1356  */
1357 {
1358  double cos_lambda, sin_lambda;
1359  int m, n;
1360  cos_lambda = cos(DEG2RAD(CoordSpherical.lambda));
1361  sin_lambda = sin(DEG2RAD(CoordSpherical.lambda));
1362  /* for n = 0 ... model_order, compute (Radius of Earth / Spherical radius r)^(n+2)
1363  for n 1..nMax-1 (this is much faster than calling pow MAX_N+1 times). */
1364  SphVariables->RelativeRadiusPower[0] =
1365  (Ellip.re / CoordSpherical.r) * (Ellip.re / CoordSpherical.r);
1366  for (n = 1; n <= nMax; n++) {
1367  SphVariables->RelativeRadiusPower[n] =
1368  SphVariables->RelativeRadiusPower[n -
1369  1] * (Ellip.re / CoordSpherical.r);
1370  }
1371 
1372  /*
1373  Compute cos(m*lambda), sin(m*lambda) for m = 0 ... nMax
1374  cos(a + b) = cos(a)*cos(b) - sin(a)*sin(b)
1375  sin(a + b) = cos(a)*sin(b) + sin(a)*cos(b)
1376  */
1377  SphVariables->cos_mlambda[0] = 1.0;
1378  SphVariables->sin_mlambda[0] = 0.0;
1379 
1380  SphVariables->cos_mlambda[1] = cos_lambda;
1381  SphVariables->sin_mlambda[1] = sin_lambda;
1382  for (m = 2; m <= nMax; m++) {
1383  SphVariables->cos_mlambda[m] =
1384  SphVariables->cos_mlambda[m - 1] * cos_lambda -
1385  SphVariables->sin_mlambda[m - 1] * sin_lambda;
1386  SphVariables->sin_mlambda[m] =
1387  SphVariables->cos_mlambda[m - 1] * sin_lambda +
1388  SphVariables->sin_mlambda[m - 1] * cos_lambda;
1389  }
1390  return (TRUE);
1391 } /*MAG_ComputeSphericalHarmonicVariables */

1393 int
1394 MAG_PcupHigh(double *Pcup, double *dPcup, double x, int nMax)
1395 
1396 /* This function evaluates all of the Schmidt-semi normalized associated Legendre
1397  functions up to degree nMax. The functions are initially scaled by
1398  10^280 sin^m in order to minimize the effects of underflow at large m
1399  near the poles (see Holmes and Featherstone 2002, J. Geodesy, 76, 279-299).
1400  Note that this function performs the same operation as MAG_PcupLow.
1401  However this function also can be used for high degree (large nMax) models.
1402 
1403  Calling Parameters:
1404  INPUT
1405  nMax: Maximum spherical harmonic degree to compute.
1406  x: cos(colatitude) or sin(latitude).
1407 
1408  OUTPUT
1409  Pcup: A vector of all associated Legendgre polynomials evaluated at
1410  x up to nMax. The lenght must by greater or equal to (nMax+1)*(nMax+2)/2.
1411  dPcup: Derivative of Pcup(x) with respect to latitude
1412 
1413  CALLS : none
1414  Notes:
1415 
1416  Adopted from the FORTRAN code written by Mark Wieczorek September 25, 2005.
1417 
1418  Manoj Nair, Nov, 2009 Manoj.C.Nair@Noaa.Gov
1419 
1420  Change from the previous version
1421  The prevous version computes the derivatives as
1422  dP(n,m)(x)/dx, where x = sin(latitude) (or cos(colatitude) ).
1423  However, the WMM Geomagnetic routines requires dP(n,m)(x)/dlatitude.
1424  Hence the derivatives are multiplied by sin(latitude).
1425  Removed the options for CS phase and normalizations.
1426 
1427  Note: In geomagnetism, the derivatives of ALF are usually found with
1428  respect to the colatitudes. Here the derivatives are found with respect
1429  to the latitude. The difference is a sign reversal for the derivative of
1430  the Associated Legendre Functions.
1431 
1432  The derivatives can't be computed for latitude = |90| degrees.
1433  */
1434 {
1435  double pm2, pm1, pmm, plm, rescalem, z, scalef;
1436  double *f1, *f2, *PreSqr;
1437  int k, kstart, m, n, NumTerms;
1438 
1439  NumTerms = ((nMax + 1) * (nMax + 2) / 2);
1440 
1441  if (fabs(x) == 1.0) {
1442  printf
1443  ("Error in PcupHigh: derivative cannot be calculated at poles\n");
1444  return (FALSE);
1445  }
1446 
1447  f1 = (double *)malloc((NumTerms + 1) * sizeof (double));
1448  if (f1 == NULL) {
1449  MAG_Error(18);
1450  return (FALSE);
1451  }
1452 
1453  PreSqr = (double *)malloc((NumTerms + 1) * sizeof (double));
1454 
1455  if (PreSqr == NULL) {
1456  MAG_Error(18);
1457  return (FALSE);
1458  }
1459 
1460  f2 = (double *)malloc((NumTerms + 1) * sizeof (double));
1461 
1462  if (f2 == NULL) {
1463  MAG_Error(18);
1464  return (FALSE);
1465  }
1466 
1467  scalef = 1.0e-280;
1468 
1469  for (n = 0; n <= 2 * nMax + 1; ++n) {
1470  PreSqr[n] = sqrt((double)(n));
1471  }
1472 
1473  k = 2;
1474 
1475  for (n = 2; n <= nMax; n++) {
1476  k = k + 1;
1477  f1[k] = (double)(2 * n - 1) / (double)(n);
1478  f2[k] = (double)(n - 1) / (double)(n);
1479  for (m = 1; m <= n - 2; m++) {
1480  k = k + 1;
1481  f1[k] =
1482  (double)(2 * n - 1) / PreSqr[n + m] / PreSqr[n - m];
1483  f2[k] =
1484  PreSqr[n - m - 1] * PreSqr[n + m - 1] / PreSqr[n +
1485  m] / PreSqr[n - m];
1486  }
1487  k = k + 2;
1488  }
1489 
1490  /*z = sin (geocentric latitude) */
1491  z = sqrt((1.0 - x) * (1.0 + x));
1492  pm2 = 1.0;
1493  Pcup[0] = 1.0;
1494  dPcup[0] = 0.0;
1495  if (nMax == 0)
1496  return (FALSE);
1497  pm1 = x;
1498  Pcup[1] = pm1;
1499  dPcup[1] = z;
1500  k = 1;
1501 
1502  for (n = 2; n <= nMax; n++) {
1503  k = k + n;
1504  plm = f1[k] * x * pm1 - f2[k] * pm2;
1505  Pcup[k] = plm;
1506  dPcup[k] = (double)(n) * (pm1 - x * plm) / z;
1507  pm2 = pm1;
1508  pm1 = plm;
1509  }
1510 
1511  pmm = PreSqr[2] * scalef;
1512  rescalem = 1.0 / scalef;
1513  kstart = 0;
1514 
1515  for (m = 1; m <= nMax - 1; ++m) {
1516  rescalem = rescalem * z;
1517 
1518  /* Calculate Pcup(m,m) */
1519  kstart = kstart + m + 1;
1520  pmm = pmm * PreSqr[2 * m + 1] / PreSqr[2 * m];
1521  Pcup[kstart] = pmm * rescalem / PreSqr[2 * m + 1];
1522  dPcup[kstart] = -((double)(m) * x * Pcup[kstart] / z);
1523  pm2 = pmm / PreSqr[2 * m + 1];
1524  /* Calculate Pcup(m+1,m) */
1525  k = kstart + m + 1;
1526  pm1 = x * PreSqr[2 * m + 1] * pm2;
1527  Pcup[k] = pm1 * rescalem;
1528  dPcup[k] =
1529  ((pm2 * rescalem) * PreSqr[2 * m + 1] - x * (double)(m +
1530  1) * Pcup[k]) / z;
1531  /* Calculate Pcup(n,m) */
1532  for (n = m + 2; n <= nMax; ++n) {
1533  k = k + n;
1534  plm = x * f1[k] * pm1 - f2[k] * pm2;
1535  Pcup[k] = plm * rescalem;
1536  dPcup[k] =
1537  (PreSqr[n + m] * PreSqr[n - m] * (pm1 * rescalem) -
1538  (double)(n) * x * Pcup[k]) / z;
1539  pm2 = pm1;
1540  pm1 = plm;
1541  }
1542  }
1543 
1544  /* Calculate Pcup(nMax,nMax) */
1545  rescalem = rescalem * z;
1546  kstart = kstart + m + 1;
1547  pmm = pmm / PreSqr[2 * nMax];
1548  Pcup[kstart] = pmm * rescalem;
1549  dPcup[kstart] = -(double)(nMax) * x * Pcup[kstart] / z;
1550  free(f1);
1551  free(PreSqr);
1552  free(f2);
1553 
1554  return (TRUE);
1555 } /* MAG_PcupHigh */

1557 int
1558 MAG_PcupLow(double *Pcup, double *dPcup, double x, int nMax)
1559 
1560 /* This function evaluates all of the Schmidt-semi normalized associated Legendre
1561  functions up to degree nMax.
1562 
1563  Calling Parameters:
1564  INPUT
1565  nMax: Maximum spherical harmonic degree to compute.
1566  x: cos(colatitude) or sin(latitude).
1567 
1568  OUTPUT
1569  Pcup: A vector of all associated Legendgre polynomials evaluated at
1570  x up to nMax.
1571  dPcup: Derivative of Pcup(x) with respect to latitude
1572 
1573  Notes: Overflow may occur if nMax > 20 , especially for high-latitudes.
1574  Use MAG_PcupHigh for large nMax.
1575 
1576  Written by Manoj Nair, June, 2009 . Manoj.C.Nair@Noaa.Gov.
1577 
1578  Note: In geomagnetism, the derivatives of ALF are usually found with
1579  respect to the colatitudes. Here the derivatives are found with respect
1580  to the latitude. The difference is a sign reversal for the derivative of
1581  the Associated Legendre Functions.
1582  */
1583 {
1584  int n, m, index, index1, index2, NumTerms;
1585  double k, z, *schmidtQuasiNorm;
1586  Pcup[0] = 1.0;
1587  dPcup[0] = 0.0;
1588  /*sin (geocentric latitude) - sin_phi */
1589  z = sqrt((1.0 - x) * (1.0 + x));
1590 
1591  NumTerms = ((nMax + 1) * (nMax + 2) / 2);
1592  schmidtQuasiNorm = (double *)malloc((NumTerms + 1) * sizeof (double));
1593 
1594  if (schmidtQuasiNorm == NULL) {
1595  MAG_Error(19);
1596  return (FALSE);
1597  }
1598 
1599  /* First, Compute the Gauss-normalized associated Legendre functions */
1600  for (n = 1; n <= nMax; n++) {
1601  for (m = 0; m <= n; m++) {
1602  index = (n * (n + 1) / 2 + m);
1603  if (n == m) {
1604  index1 = (n - 1) * n / 2 + m - 1;
1605  Pcup[index] = z * Pcup[index1];
1606  dPcup[index] =
1607  z * dPcup[index1] + x * Pcup[index1];
1608  } else if (n == 1 && m == 0) {
1609  index1 = (n - 1) * n / 2 + m;
1610  Pcup[index] = x * Pcup[index1];
1611  dPcup[index] =
1612  x * dPcup[index1] - z * Pcup[index1];
1613  } else if (n > 1 && n != m) {
1614  index1 = (n - 2) * (n - 1) / 2 + m;
1615  index2 = (n - 1) * n / 2 + m;
1616  if (m > n - 2) {
1617  Pcup[index] = x * Pcup[index2];
1618  dPcup[index] =
1619  x * dPcup[index2] -
1620  z * Pcup[index2];
1621  } else {
1622  k = (double)(((n - 1) * (n - 1)) -
1623  (m * m)) / (double)((2 * n -
1624  1) * (2 * n - 3));
1625  Pcup[index] =
1626  x * Pcup[index2] - k * Pcup[index1];
1627  dPcup[index] =
1628  x * dPcup[index2] -
1629  z * Pcup[index2] -
1630  k * dPcup[index1];
1631  }
1632  }
1633  }
1634  }
1635  /* Compute the ration between the the Schmidt quasi-normalized associated Legendre
1636  * functions and the Gauss-normalized version. */
1637 
1638  schmidtQuasiNorm[0] = 1.0;
1639  for (n = 1; n <= nMax; n++) {
1640  index = (n * (n + 1) / 2);
1641  index1 = (n - 1) * n / 2;
1642  /* for m = 0 */
1643  schmidtQuasiNorm[index] =
1644  schmidtQuasiNorm[index1] * (double)(2 * n - 1) / (double)n;
1645 
1646  for (m = 1; m <= n; m++) {
1647  index = (n * (n + 1) / 2 + m);
1648  index1 = (n * (n + 1) / 2 + m - 1);
1649  schmidtQuasiNorm[index] =
1650  schmidtQuasiNorm[index1] * sqrt((double)((n - m +
1651  1) * (m == 1 ? 2 : 1)) / (double)(n + m));
1652  }
1653 
1654  }
1655 
1656  /* Converts the Gauss-normalized associated Legendre
1657  functions to the Schmidt quasi-normalized version using pre-computed
1658  relation stored in the variable schmidtQuasiNorm */
1659 
1660  for (n = 1; n <= nMax; n++) {
1661  for (m = 0; m <= n; m++) {
1662  index = (n * (n + 1) / 2 + m);
1663  Pcup[index] = Pcup[index] * schmidtQuasiNorm[index];
1664  dPcup[index] = -dPcup[index] * schmidtQuasiNorm[index];
1665  /* The sign is changed since the new WMM routines use derivative with respect to latitude
1666  insted of co-latitude */
1667  }
1668  }
1669 
1670  if (schmidtQuasiNorm)
1671  free(schmidtQuasiNorm);
1672  return (TRUE);
1673 } /*MAG_PcupLow */
