81 static void
82 conv_base64_to_url(char *str)
83 {
84  ASSERT(str != NULL);
85  for (int i = 0, n = strlen(str); i < n; i++) {
86  if (str[i] == '+') {
87  str[i] = '-';
88  } else if (str[i] == '/') {
89  str[i] = '_';
90  } else if (str[i] == '=') {
91  str[i] = '\0';
92  break;
93  }
94  }
95 }

116 /*
117  * Sort the charts based on the index number.
118  */
119 static int
120 chart_sort_func_navigraph(const void *a, const void *b, void *c)
121 {
122  const char *na = *(char **)a, *nb = *(char **)b;
123  const char *idx_a = strrchr(na, '#'), *idx_b = strrchr(nb, '#');
124  const char *dash_a, *dash_b;
125  char sect_a_str[8] = {}, sect_b_str[8] = {};
126  char page_a_str[8] = {}, page_b_str[8] = {};
127  int sect_a, sect_b, page_a, page_b, cmp;
128 
129  LACF_UNUSED(c);
130 
131  if (idx_a == NULL || idx_a < &na[1] || idx_a[-1] != '#' ||
132  idx_b == NULL || idx_b < &nb[1] || idx_b[-1] != '#') {
133  return (0);
134  }
135  idx_a++;
136  idx_b++;
137  dash_a = strchr(idx_a, '-');
138  dash_b = strchr(idx_b, '-');
139  if (dash_a == NULL || dash_b == NULL)
140  return (0);
141  strlcpy(sect_a_str, idx_a,
142  MIN(sizeof (sect_a_str), (unsigned)(dash_a - idx_a) + 1));
143  strlcpy(page_a_str, &dash_a[1], sizeof (page_a_str));
144  strlcpy(sect_b_str, idx_b,
145  MIN(sizeof (sect_b_str), (unsigned)(dash_b - idx_b) + 1));
146  strlcpy(page_b_str, &dash_b[1], sizeof (page_b_str));
147 
148  sect_a = atoi(sect_a_str);
149  page_a = atoi(page_a_str);
150  sect_b = atoi(sect_b_str);
151  page_b = atoi(page_b_str);
152 
153  if (sect_a < sect_b)
154  return (-1);
155  if (sect_a > sect_b)
156  return (1);
157  if (page_a < page_b)
158  return (-1);
159  if (page_a > page_b)
160  return (1);
161  cmp = strcmp(idx_a, idx_b);
162  if (cmp < 0)
163  return (-1);
164  if (cmp > 0)
165  return (1);
166  return (0);
167 }

169 static bool_t
170 handle_dev_auth(const chart_dl_info_t *dl_info, navigraph_t *nav)
171 {
172  jsmn_parser parser;
173  jsmntok_t toks[128];
174  int n_toks;
175  const jsmntok_t *dev_code, *verif_uri, *expires, *intval;
176  const char *str;
177  char *uri;
178  char tmpbuf[32];
179 
180  ASSERT(dl_info != NULL);
181  str = (const char *)dl_info->buf;
182  ASSERT(nav != NULL);
183 
184  jsmn_init(&parser);
185  n_toks = jsmn_parse(&parser, str, dl_info->bufsz,
186  toks, ARRAY_NUM_ELEM(toks));
187  if (n_toks < 0) {
188  logMsg("Can't authorize device: server responded with "
189  "what looks like invalid JSON data");
190  return (B_FALSE);
191  }
192  if (jsmn_get_tok_data_path(str, toks, n_toks, "error",
193  tmpbuf, sizeof (tmpbuf))) {
194  logMsg("Can't authorize device: server responded with "
195  "error \"%s\"", tmpbuf);
196  return (B_FALSE);
197  }
198  dev_code = jsmn_path_lookup_format(str, toks, n_toks, "device_code");
199  verif_uri = jsmn_path_lookup_format(str, toks, n_toks,
200  "verification_uri_complete");
201  expires = jsmn_path_lookup_format(str, toks, n_toks, "expires_in");
202  intval = jsmn_path_lookup_format(str, toks, n_toks, "interval");
203  if (dev_code == NULL || verif_uri == NULL || expires == NULL ||
204  intval == NULL) {
205  logMsg("Can't authorize device: server responded with "
206  "invalid JSON structure");
207  return (B_FALSE);
208  }
209  uri = jsmn_strdup_tok_data(str, verif_uri);
210  NAV_DBG_LOG("Opening verification URI: %s", uri);
211  if (!lacf_open_URL(uri)) {
212  free(uri);
213  logMsg("Can't authorize device: can't launch default browser "
214  "for verification URI");
215  return (B_FALSE);
216  }
217  free(uri);
218 
219  if (nav->dev_code != NULL)
220  free(nav->dev_code);
221  nav->dev_code = jsmn_strdup_tok_data(str, dev_code);
222  jsmn_get_tok_data(str, expires, tmpbuf, sizeof (tmpbuf));
223  nav->expire_t = time(NULL) + atol(tmpbuf);
224  jsmn_get_tok_data(str, intval, tmpbuf, sizeof (tmpbuf));
225  nav->intval = MAX(atoi(tmpbuf), 1);
226  nav->next_check_t = time(NULL) + nav->intval;
227 
228  NAV_DBG_LOG("devauth started, dev code \"%s\", intval: %d",
229  nav->dev_code, nav->intval);
230 
231  return (B_TRUE);
232 }

234 static bool_t
235 start_auth(chartdb_t *cdb, navigraph_t *nav)
236 {
237  uint8_t code_verifier[32];
238  uint8_t chlg_raw[SHA256_DIGEST_SIZE];
239  char chlg_enc[64];
240  size_t len;
241  char post_body[512];
242  const chart_prov_info_login_t *login;
243  CURLcode res;
244  chart_dl_info_t dl_info;
245 
246  ASSERT(cdb != NULL);
247  ASSERT(cdb->prov_login != NULL);
248  login = cdb->prov_login;
249  ASSERT(nav != NULL);
250 
251  chart_dl_info_init(&dl_info, cdb, DEV_AUTH_ENDPT);
252  /*
253  * We generate the code verifier and Base64-URL encode it.
254  * This must be kept after successful device authorization.
255  */
256  if (!osrand(code_verifier, sizeof (code_verifier))) {
257  logMsg("Cannot generate auth request: osrand() failed");
258  goto errout;
259  }
260  len = lacf_base64_encode(code_verifier, sizeof (code_verifier),
261  (uint8_t *)nav->code_verifier);
262  ASSERT3U(len, ==, 44);
263  nav->code_verifier[len] = '\0';
264  conv_base64_to_url(nav->code_verifier);
265  /*
266  * Now SHA-256 hash the encoded code verifier, Base64-URL-encode
267  * it again and send it over the wire.
268  */
269  sha256(nav->code_verifier, 43, chlg_raw);
270  len = lacf_base64_encode(chlg_raw, sizeof (chlg_raw),
271  (uint8_t *)chlg_enc);
272  chlg_enc[len] = '\0';
273  conv_base64_to_url(chlg_enc);
274 
275  snprintf(post_body, sizeof (post_body), "client_id=%s&"
276  "client_secret=%s&code_challenge=%s&code_challenge_method=S256",
277  login->username, login->password, chlg_enc);
278 
279  curl_easy_setopt(nav->curl, CURLOPT_URL, DEV_AUTH_ENDPT);
280  curl_easy_setopt(nav->curl, CURLOPT_POST, 1L);
281  curl_easy_setopt(nav->curl, CURLOPT_POSTFIELDS, post_body);
282  curl_easy_setopt(nav->curl, CURLOPT_WRITEDATA, &dl_info);
283 
284  NAV_DBG_LOG("devauth starting, code_challenge \"%s\"", chlg_enc);
285  res = curl_easy_perform(nav->curl);
286  if (res == CURLE_OK) {
287  long code;
288 
289  curl_easy_getinfo(nav->curl, CURLINFO_RESPONSE_CODE, &code);
290  if (code == 200) {
291  NAV_DBG_LOG("devauth server responded with 200");
292  if (!handle_dev_auth(&dl_info, nav))
293  goto errout;
294  } else {
295  logMsg("Cannot generate auth request: server "
296  "responded with error code %ld", code);
297  goto errout;
298  }
299  } else {
300  logMsg("Cannot generate auth request: %s",
301  curl_easy_strerror(res));
302  goto errout;
303  }
304  curl_easy_setopt(nav->curl, CURLOPT_POST, 0L);
305  nav->pending_ext_setup = B_TRUE;
306  chart_dl_info_fini(&dl_info);
307  return (B_TRUE);
308 errout:
309  curl_easy_setopt(nav->curl, CURLOPT_POST, 0L);
310  chart_dl_info_fini(&dl_info);
311  return (B_FALSE);
312 }

334 static bool_t
335 handle_token_response(chartdb_t *cdb, const chart_dl_info_t *dl_info,
336  navigraph_t *nav)
337 {
338  jsmn_parser parser;
339  jsmntok_t toks[128];
340  int n_toks;
341  const char *json;
342  const jsmntok_t *access_tok, *expires_tok, *refresh_tok;
343  char tmpbuf[32];
344 
345  ASSERT(dl_info != NULL);
346  json = (const char *)dl_info->buf;
347  ASSERT(nav != NULL);
348 
349  jsmn_init(&parser);
350  n_toks = jsmn_parse(&parser, json, dl_info->bufsz, toks,
351  ARRAY_NUM_ELEM(toks));
352  if (n_toks < 0) {
353  logMsg("Cannot fetch access token: server responded with "
354  "what looks like invalid JSON data");
355  return (B_FALSE);
356  }
357  access_tok = jsmn_path_lookup_format(json, toks, n_toks, "access_token");
358  expires_tok = jsmn_path_lookup_format(json, toks, n_toks, "expires_in");
359  refresh_tok = jsmn_path_lookup_format(json, toks, n_toks,
360  "refresh_token");
361  /* We must at least have an access token */
362  if (access_tok == NULL || expires_tok == NULL ||
363  (nav->refresh_token == NULL && refresh_tok == NULL)) {
364  logMsg("Cannot fetch access token: server responded with "
365  "invalid JSON structure");
366  return (B_FALSE);
367  }
368  if (refresh_tok != NULL) {
369  LACF_DESTROY(nav->refresh_token);
370  nav->refresh_token = jsmn_strdup_tok_data(json, refresh_tok);
371  save_refresh_token(cdb, nav);
372  NAV_DBG_LOG("got new refresh_token \"%s\"",
373  nav->refresh_token);
374  }
375  LACF_DESTROY(nav->access_token);
376  nav->access_token = jsmn_strdup_tok_data(json, access_tok);
377  jsmn_get_tok_data(json, expires_tok, tmpbuf, sizeof (tmpbuf));
378  nav->access_expire_t = time(NULL) + atoi(tmpbuf);
379 
380  NAV_DBG_LOG("got new access_token \"%s\" expires in %d seconds",
381  nav->access_token, atoi(tmpbuf));
382 
383  return (B_TRUE);
384 }

386 static bool_t
387 get_tokens(chartdb_t *cdb, navigraph_t *nav)
388 {
389  CURLcode res;
390  chart_dl_info_t dl_info;
391  char post_body[1024];
392  const chart_prov_info_login_t *login;
393 
394  ASSERT(cdb != NULL);
395  ASSERT(nav != NULL);
396  ASSERT(cdb->prov_login != NULL);
397  login = cdb->prov_login;
398 
399  chart_dl_info_init(&dl_info, cdb, TOKEN_ENDPT);
400 
401  while (time(NULL) <= nav->next_check_t) {
402  NAV_DBG_LOG("waiting for next token check (in %ld secs)",
403  (long)(nav->next_check_t - time(NULL)));
404  usleep(1000000);
405  /* Early termination request from outside? */
406  if (!cdb->loader.run) {
407  NAV_DBG_LOG("wait aborted due to worker thread "
408  "shutdown request");
409  goto errout;
410  }
411  }
412  if (nav->refresh_token != NULL) {
413  /*
414  * Do an access token refresh.
415  */
416  NAV_DBG_LOG("have refresh_token, doing an access_token "
417  "refresh");
418  snprintf(post_body, sizeof (post_body),
419  "grant_type=refresh_token&client_id=%s&client_secret=%s&"
420  "refresh_token=%s", login->username, login->password,
421  nav->refresh_token);
422  } else {
423  NAV_DBG_LOG("no tokens, expecting both a refresh_token "
424  "and access_token");
425  snprintf(post_body, sizeof (post_body),
426  "grant_type=urn:ietf:params:oauth:grant-type:device_code&"
427  "device_code=%s&code_verifier=%s&client_id=%s&"
428  "client_secret=%s&scope=openid charts offline_access",
429  nav->dev_code, nav->code_verifier, login->username,
430  login->password);
431  }
432  curl_easy_setopt(nav->curl, CURLOPT_URL, TOKEN_ENDPT);
433  curl_easy_setopt(nav->curl, CURLOPT_POST, 1L);
434  curl_easy_setopt(nav->curl, CURLOPT_POSTFIELDS, post_body);
435  curl_easy_setopt(nav->curl, CURLOPT_WRITEDATA, &dl_info);
436 
437  res = curl_easy_perform(nav->curl);
438  if (res == CURLE_OK) {
439  long code;
440  char error[32];
441 
442  curl_easy_getinfo(nav->curl, CURLINFO_RESPONSE_CODE, &code);
443  if (code == 200) {
444  NAV_DBG_LOG("server responded with 200");
445  if (get_json_error(&dl_info, error, sizeof (error))) {
446  logMsg("Cannot fetch access token: server "
447  "responded with error \"%s\"", error);
448  goto errout;
449  } else if (!handle_token_response(cdb, &dl_info, nav)) {
450  goto errout;
451  }
452  } else if (code == 400 && get_json_error(&dl_info,
453  error, sizeof (error))) {
454  NAV_DBG_LOG("server responded with: 400 %s", error);
455  if (strcmp(error, "authorization_pending") == 0) {
456  /* Just keep on waiting */
457  nav->next_check_t = time(NULL) + nav->intval;
458  chart_dl_info_fini(&dl_info);
459  return (get_tokens(cdb, nav));
460  } else if (strcmp(error, "slow_down") == 0) {
461  /* Just keep on waiting - increase intval +5s */
462  nav->intval += 5;
463  nav->next_check_t = time(NULL) + nav->intval;
464  chart_dl_info_fini(&dl_info);
465  return (get_tokens(cdb, nav));
466  } else if (strcmp(error, "invalid_grant") == 0 &&
467  nav->refresh_token != NULL) {
468  /*
469  * Refresh token is invalid, we need to drop
470  * it and restart a new device authorization
471  */
472  LACF_DESTROY(nav->refresh_token);
473  logMsg("Refresh token has become stale: "
474  "restarting a new device authorization");
475  } else {
476  /* All other errors are fatal */
477  logMsg("Cannot fetch access token: server "
478  "responded with error \"%s\"", error);
479  goto errout;
480  }
481  } else {
482  logMsg("Cannot fetch access token: server "
483  "responded with error code %ld", code);
484  goto errout;
485  }
486  } else {
487  logMsg("Cannot generate auth request: %s",
488  curl_easy_strerror(res));
489  goto errout;
490  }
491  curl_easy_setopt(nav->curl, CURLOPT_POST, 0L);
492  nav->pending_ext_setup = B_FALSE;
493  chart_dl_info_fini(&dl_info);
494  return (B_TRUE);
495 errout:
496  curl_easy_setopt(nav->curl, CURLOPT_POST, 0L);
497  /* Even if we fail completely, we exit the pending auth state */
498  nav->pending_ext_setup = B_FALSE;
499  chart_dl_info_fini(&dl_info);
500  return (B_FALSE);
501 }

503 static void *
504 do_download(chartdb_t *cdb, navigraph_t *nav, const char *url, size_t *len,
505  bool_t *retry)
506 {
507  CURLcode res;
508  chart_dl_info_t dl_info;
509  struct curl_slist *slist = NULL;
510  char *auth_hdr;
511 
512  ASSERT(cdb != NULL);
513  ASSERT(nav != NULL);
514  ASSERT(url != NULL);
515  ASSERT(len != NULL);
516  ASSERT(retry != NULL);
517  *retry = B_FALSE;
518 
519  chart_dl_info_init(&dl_info, cdb, TOKEN_ENDPT);
520  ASSERT(nav->access_token != NULL);
521  auth_hdr = sprintf_alloc("Authorization: Bearer %s", nav->access_token);
522  slist = curl_slist_append(slist, auth_hdr);
523  LACF_DESTROY(auth_hdr);
524 
525  curl_easy_setopt(nav->curl, CURLOPT_URL, url);
526  curl_easy_setopt(nav->curl, CURLOPT_HTTPHEADER, slist);
527  curl_easy_setopt(nav->curl, CURLOPT_WRITEDATA, &dl_info);
528 
529  NAV_DBG_LOG("%s", url);
530  res = curl_easy_perform(nav->curl);
531  if (res == CURLE_OK) {
532  long code;
533  char error[32];
534 
535  curl_easy_getinfo(nav->curl, CURLINFO_RESPONSE_CODE, &code);
536  if (code == 200) {
537  /* no-op */
538  NAV_DBG_LOG("success");
539  } else if (code == 400 && get_json_error(&dl_info,
540  error, sizeof (error))) {
541  if (strcmp(error, "expired_token") == 0) {
542  NAV_DBG_LOG("server says token has expired, "
543  "will try to refresh it");
544  /*
545  * Token expired, even though we didn't expect
546  * it. Could just be a timing issue. Just drop
547  * the access token and retry to fetch anew.
548  */
549  LACF_DESTROY(nav->access_token);
550  *retry = B_TRUE;
551  goto errout;
552  } else {
553  logMsg("Cannot download %s: server responded "
554  "with error %s", url, error);
555  goto errout;
556  }
557  } else if (code == 401) {
558  NAV_DBG_LOG("401 error, will try to refresh access "
559  "token");
560  /*
561  * Server rejected our authorization completely.
562  * Let's try to start a completely new device auth.
563  */
564  LACF_DESTROY(nav->access_token);
565  LACF_DESTROY(nav->refresh_token);
566  LACF_DESTROY(nav->dev_code);
567  *retry = B_TRUE;
568  goto errout;
569  } else {
570  logMsg("Cannot download %s: server responded with "
571  "error code %ld", url, code);
572  }
573  } else {
574  logMsg("Cannot download %s: %s", url, curl_easy_strerror(res));
575  goto errout;
576  }
577  /* Don't destroy the dl_info, we want to return the buffer from it */
578  *len = dl_info.bufsz;
579  curl_slist_free_all(slist);
580  curl_easy_setopt(nav->curl, CURLOPT_HTTPHEADER, NULL);
581  return (dl_info.buf);
582 errout:
583  chart_dl_info_fini(&dl_info);
584  curl_slist_free_all(slist);
585  curl_easy_setopt(nav->curl, CURLOPT_HTTPHEADER, NULL);
586  return (B_FALSE);
587 }

589 static void *
590 navigraph_dl(chartdb_t *cdb, navigraph_t *nav, const char *url,
591  bool_t allow_blocking, size_t *len)
592 {
593  ASSERT(cdb != NULL);
594  ASSERT(nav != NULL);
595  ASSERT_MUTEX_HELD(&nav->lock);
596  ASSERT(url != NULL);
597  ASSERT(len != NULL);
598  *len = 0;
599  /*
600  * Pre-emptively try to fetch a new access token if we're nearing
601  * the end of the validity period.
602  */
603  if (nav->access_token != NULL &&
604  time(NULL) + 60 >= nav->access_expire_t) {
605  LACF_DESTROY(nav->access_token);
606  }
607  NAV_DBG_LOG("%s", url);
608  if (nav->access_token != NULL) {
609  bool_t retry;
610  uint8_t *result;
611  /*
612  * If we have an access token, we can try to just grab
613  * the actual resource we're interested in.
614  */
615  NAV_DBG_LOG("have access token, fetching target");
616  result = do_download(cdb, nav, url, len, &retry);
617  if (result != NULL) {
618  return (result);
619  } else if (retry) {
620  NAV_DBG_LOG("soft failure, retrying");
621  return (navigraph_dl(cdb, nav, url,
622  allow_blocking, len));
623  } else {
624  return (NULL);
625  }
626  } else if (nav->refresh_token != NULL ||
627  (nav->dev_code != NULL && nav->code_verifier[0] != '\0')) {
628  /*
629  * We either already have a refresh token, or we've started
630  * the device authorization and we're awaiting a new refresh
631  * (+access) token.
632  */
633  NAV_DBG_LOG("retrieving tokens");
634  if (get_tokens(cdb, nav)) {
635  if (nav->refresh_token != NULL)
636  NAV_DBG_LOG("tokens acquired, retrying dl");
637  else
638  NAV_DBG_LOG("tokens stale, dropping devauth");
639  return (navigraph_dl(cdb, nav, url,
640  allow_blocking, len));
641  } else {
642  return (NULL);
643  }
644  } else {
645  /*
646  * We don't have anything. Start a new device authorization.
647  * If we can't block device authorization, fail immediately.
648  */
649  NAV_DBG_LOG("device not authorized, starting devauth");
650  if (!allow_blocking) {
651  NAV_DBG_LOG("caller says we can't block, so failing");
652  return (NULL);
653  } else if (start_auth(cdb, nav)) {
654  NAV_DBG_LOG("devauth succeeded, retrying dl");
655  return (navigraph_dl(cdb, nav, url,
656  allow_blocking, len));
657  } else {
658  return (NULL);
659  }
660  }
661 }

1045 static bool_t
1046 parse_chart_json(const void *json, size_t len, chart_arpt_t *arpt)
1047 {
1048  jsmn_parser parser;
1049  size_t max_toks = MAX(len / 4, 10000);
1050  jsmntok_t *toks = safe_calloc(max_toks, sizeof (*toks));
1051  int n_toks;
1052  const jsmntok_t *charts_tok;
1053 
1054  ASSERT(json != NULL);
1055  ASSERT(arpt != NULL);
1056 
1057  jsmn_init(&parser);
1058  n_toks = jsmn_parse(&parser, json, len, toks, max_toks);
1059  if (n_toks < 0) {
1060  logMsg("Error parsing airport %s: chart data doesn't "
1061  "look like well-formed JSON", arpt->icao);
1062  goto errout;
1063  }
1064  charts_tok = jsmn_path_lookup(json, toks, n_toks, "charts");
1065  if (charts_tok == NULL) {
1066  logMsg("Error parsing airport %s: chart data JSON "
1067  "has invalid structure", arpt->icao);
1068  goto errout;
1069  }
1070  for (int i = 0; i < charts_tok->size; i++) {
1071  char idx_nr[32], name[128], cat[8];
1072  char image_day[128], image_night[128];
1073  const jsmntok_t *procs_tok;
1074  chart_t *chart;
1075 
1076  if (!jsmn_get_tok_data_path_format(json, toks, n_toks,
1077  "charts/[%d]/index_number", idx_nr, sizeof (idx_nr), i) ||
1078  !jsmn_get_tok_data_path_format(json, toks, n_toks,
1079  "charts/[%d]/name", name, sizeof (name), i) ||
1080  !jsmn_get_tok_data_path_format(json, toks, n_toks,
1081  "charts/[%d]/category", cat, sizeof (cat), i) ||
1082  !jsmn_get_tok_data_path_format(json, toks, n_toks,
1083  "charts/[%d]/image_day", image_day,
1084  sizeof (image_day), i)) {
1085  continue;
1086  }
1087  jsmn_get_tok_data_path_format(json, toks, n_toks,
1088  "charts/[%d]/image_night", image_night,
1089  sizeof (image_night), i);
1090 
1091  chart = safe_calloc(1, sizeof (*chart));
1092  chart->arpt = arpt;
1093  /*
1094  * Navigraph charts don't always contain a unique name,
1095  * so to avoid name conflicts, we suffix the readable
1096  * chart name with "##<index>". The apps using the
1097  * interface must handle this case specially to avoid
1098  * showing the suffix.
1099  */
1100  chart->name = sprintf_alloc("%s##%s", name, idx_nr);
1101  if (strcmp(cat, "ARR") == 0) {
1102  chart->type = CHART_TYPE_STAR;
1103  } else if (strcmp(cat, "DEP") == 0) {
1104  chart->type = CHART_TYPE_DP;
1105  } else if (strstr(idx_nr, "0-9") ==
1106  idx_nr + strlen(idx_nr) - 3) {
1107  /* Anything ending in "0-9" is an airport diagram */
1108  chart->type = CHART_TYPE_APD;
1109  } else if (strcmp(cat, "REF") == 0 || strcmp(cat, "APT") == 0) {
1110  chart->type = CHART_TYPE_INFO;
1111  } else if (strcmp(cat, "APP") == 0) {
1112  chart->type = CHART_TYPE_IAP;
1113  } else {
1114  chart->type = CHART_TYPE_UNKNOWN;
1115  }
1116  chart->codename = safe_strdup(idx_nr);
1117  chart->filename = safe_strdup(image_day);
1118  if (image_night[0] != '\0')
1119  chart->filename_night = safe_strdup(image_night);
1120 
1121  procs_tok = jsmn_path_lookup_format(json, toks, n_toks,
1122  "charts/[%d]/procedures", i);
1123  for (int j = 0; procs_tok != NULL && j < procs_tok->size &&
1124  chart->procs.n_procs < MAX_CHART_PROCS; j++) {
1125  if (jsmn_get_tok_data_path_format(json, toks, n_toks,
1126  "charts/[%d]/procedures/[%d]",
1127  chart->procs.procs[chart->procs.n_procs],
1128  sizeof (*chart->procs.procs), i, j)) {
1129  chart->procs.n_procs++;
1130  }
1131  }
1132  parse_chart_georef_data(json, toks, n_toks, chart, i);
1133  for (chart_view_t view = 0; view < NUM_CHART_VIEWS; view++) {
1134  parse_chart_view_data(json, toks, n_toks, chart, i,
1135  view);
1136  }
1137  if (!chartdb_add_chart(arpt, chart)) {
1138  /*
1139  * Duplicate chart - unfortunately, Navigraph is
1140  * prone to sending duplicates.
1141  */
1142  logMsg("Chart error: airport %s contains duplicate "
1143  "chart %s", arpt->icao, chart->name);
1144  chartdb_chart_destroy(chart);
