56 static shader_attr_bind_t *
57 attr_binds_from_args(va_list ap)
58 {
59  va_list ap2;
60  int n_attrs = 0;
61  shader_attr_bind_t *binds;
62 
63  va_copy(ap2, ap);
64  for (;;) {
65  const char *name = va_arg(ap2, const char *);
66 
67  if (name == NULL)
68  break;
69  va_arg(ap2, GLuint);
70  n_attrs++;
71  }
72  va_end(ap2);
73  /* +1 to terminate the list with an empty bind */
74  binds = safe_calloc(n_attrs + 1, sizeof (*binds));
75  for (int i = 0; i < n_attrs; i++) {
76  binds[i].name = va_arg(ap, const char *);
77  binds[i].idx = va_arg(ap, GLuint);
78  }
79 
80  return (binds);
81 }

83 /*
84  * Attempts to load a fallback shader to a SPIR-V shader, in case SPIR-V
85  * isn't supported. This attempts to locate a shader by replacing the filename
86  * extension of the original shader with:
87  * - .vert or .frag
88  * - .glsl
89  * If found, the shader is compiled and returned. Otherwise returns 0.
90  */
91 static GLuint
92 shader_from_spirv_fallback(GLenum shader_type, const char *filename,
93  const shader_spec_const_t *spec_const)
94 {
95  char *alt_filename, *alt_ext, *new_ext;
96  bool_t is_dir;
97  bool_t loaded = B_FALSE;
98  GLuint shader = 0;
99 
100  switch (shader_type) {
101  case GL_VERTEX_SHADER:
102  alt_ext = "vert";
103  break;
104  case GL_FRAGMENT_SHADER:
105  alt_ext = "frag";
106  break;
107  case GL_COMPUTE_SHADER:
108  alt_ext = "comp";
109  break;
110  default:
111  VERIFY_MSG(0, "Unknown shader type %d", shader_type);
112  }
113 
114  alt_filename = safe_calloc(strlen(filename) + 16, 1);
115  strcpy(alt_filename, filename);
116  new_ext = strrchr(alt_filename, '.');
117  ASSERT(new_ext != NULL);
118  new_ext++;
119  strcpy(new_ext, alt_ext);
120  if (!loaded && file_exists(alt_filename, &is_dir) && !is_dir) {
121  shader = shader_from_file(shader_type, alt_filename,
122  NULL, spec_const);
123  loaded = (shader != 0);
124  }
125 #define TEST_GLSL_VERSION(version, extension) \
126  do { \
127  if (!loaded && (version)) { \
128  strcpy(new_ext, (extension)); \
129  if (file_exists(alt_filename, &is_dir) && !is_dir) { \
130  shader = shader_from_file(shader_type, \
131  alt_filename, NULL, spec_const); \
132  loaded = (shader != 0); \
133  } \
134  } \
135  } while (0)
136  TEST_GLSL_VERSION(GLEW_VERSION_4_6, "glsl460");
137  TEST_GLSL_VERSION(GLEW_VERSION_4_5, "glsl450");
138  TEST_GLSL_VERSION(GLEW_VERSION_4_4, "glsl440");
139  TEST_GLSL_VERSION(GLEW_VERSION_4_3, "glsl430");
140  TEST_GLSL_VERSION(GLEW_VERSION_4_2, "glsl420");
141  TEST_GLSL_VERSION(GLEW_VERSION_4_1, "glsl410");
142  TEST_GLSL_VERSION(GLEW_VERSION_4_0, "glsl400");
143  if (!loaded) {
144  strcpy(new_ext, "glsl");
145  if (file_exists(alt_filename, &is_dir) && !is_dir) {
146  shader = shader_from_file(shader_type,
147  alt_filename, NULL, spec_const);
148  loaded = (shader != 0);
149  } else {
150  logMsg("Error loading shader %s: SPIR-V shaders "
151  "not supported and no fallback shader found.",
152  filename);
153  }
154  }
155  free(alt_filename);
156 
157  return (shader);
158 }

167 static bool_t
168 have_shader_binary_format(GLint fmt)
169 {
170  GLint num;
171  GLint *formats;
172  bool_t found = B_FALSE;
173 
174  glGetIntegerv(GL_NUM_SHADER_BINARY_FORMATS, &num);
175  if (glGetError() != GL_NO_ERROR || num > 1024 * 1024)
176  return (B_FALSE);
177 
178  formats = safe_calloc(num, sizeof (*formats));
179  glGetIntegerv(GL_SHADER_BINARY_FORMATS, formats);
180  if (glGetError() != GL_NO_ERROR) {
181  free(formats);
182  return (B_FALSE);
183  }
184  for (int i = 0; i < num; i++) {
185  if (formats[i] == fmt) {
186  found = B_TRUE;
187  break;
188  }
189  }
190  free(formats);
191 
192  return (found);
193 }

209 /*
210  * Attemps to load a SPIR-V shader. If SPIR-V is not supported, calls
211  * shader_from_spirv_fallback to attempt to load a backup alternate shader.
212  * The entry point must be provided in `entry_pt'. If present, set spec_const
213  * to an array of specialization constants. This array MUST be terminated by
214  * an element that has is_last set to B_TRUE. If no specialization constants
215  * are needed, set spec_const = NULL.
216  */
217 static GLuint
218 shader_from_spirv(GLenum shader_type, const char *filename,
219  const char *entry_pt, const shader_spec_const_t *spec_const)
220 {
221  GLuint shader = 0;
222  GLint compile_result;
223  void *buf = NULL;
224  size_t bufsz, n_spec;
225  GLuint *spec_indices = NULL, *spec_values = NULL;
226  GLenum err;
227  /* clear error state */
228  glGetError();
229 
230  if (entry_pt == NULL)
231  entry_pt = "main";
232 
233  /*
234  * AMD's SPIR-V loader is b0rked AF (crashes in glSpecializeShader).
235  * Zink declares SPIR-V support, but the resultant shader doesn't
236  * actually work.
237  * When running under Nvidia Nsight, avoid loading binary shaders.
238  * That way, we can see the source in the source explorer.
239  */
240  if (!GLEW_ARB_gl_spirv ||
241  !have_shader_binary_format(GL_SHADER_BINARY_FORMAT_SPIR_V) ||
242  is_amd() || glutils_in_zink_mode() ||
243  (glutils_nsight_debugger_present() && !force_spv())) {
244  /* SPIR-V shaders not supported. Try fallback shader. */
245  return (shader_from_spirv_fallback(shader_type, filename,
246  spec_const));
247  }
248  for (n_spec = 0; spec_const != NULL && !spec_const[n_spec].is_last;
249  n_spec++)
250  ;
251  ASSERT(spec_const == NULL ||
252  (spec_const[n_spec].idx == 0 && spec_const[n_spec].val == 0));
253  if (n_spec > 0) {
254  /*
255  * Windows such a pile of fuck that it returns non-NULL
256  * for zero-length allocations. And that fucks over the
257  * AMD driver.
258  */
259  spec_indices = safe_calloc(n_spec, sizeof (*spec_indices));
260  spec_values = safe_calloc(n_spec, sizeof (*spec_values));
261  }
262  for (size_t i = 0; i < n_spec; i++) {
263  spec_indices[i] = spec_const[i].idx;
264  spec_values[i] = spec_const[i].val;
265  }
266 
267  buf = file2buf(filename, &bufsz);
268  if (buf == NULL) {
269  logMsg("Cannot load shader %s: %s", filename, strerror(errno));
270  goto errout;
271  }
272 
273  shader = glCreateShader(shader_type);
274  if (shader == 0) {
275  logMsg("Cannot load shader %s: glCreateShader failed with "
276  "error 0x%x", filename, glGetError());
277  goto errout;
278  }
279  glShaderBinary(1, &shader, GL_SHADER_BINARY_FORMAT_SPIR_V, buf, bufsz);
280  if ((err = glGetError()) != GL_NO_ERROR) {
281  logMsg("Cannot load SPIR-V %s: error %x", filename, err);
282  goto errout;
283  }
284  glSpecializeShader(shader, entry_pt, n_spec, spec_indices, spec_values);
285 
286  glGetShaderiv(shader, GL_COMPILE_STATUS, &compile_result);
287  if (compile_result == GL_FALSE) {
288  GLint len;
289  GLchar *buf;
290 
291  glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &len);
292  buf = safe_calloc(len + 1, sizeof (*buf));
293  glGetShaderInfoLog(shader, len, NULL, buf);
294  logMsg("Cannot load shader %s: specialization error: %s",
295  filename, buf);
296 
297  goto errout;
298  }
299 
300  free(buf);
301  free(spec_indices);
302  free(spec_values);
303 
304  return (shader);
305 errout:
306  free(buf);
307  free(spec_indices);
308  free(spec_values);
309  if (shader != 0)
310  glDeleteShader(shader);
311 
312  return (0);
313 }

315 /*
316  * Loads and compiles GLSL shader from a file and returns the shader object
317  * ID. The shader type is passed in `shader_type'. Returns 0 if the shader
318  * failed to load (an error description is printed to the X-Plane log).
319  */
320 static GLuint
321 shader_from_file(GLenum shader_type, const char *filename,
322  const char *entry_pt, const shader_spec_const_t *spec_const)
323 {
324  const char *ext;
325  GLchar *shader_text = NULL;
326  GLuint shader;
327 
328  ASSERT(shader_type == GL_VERTEX_SHADER ||
329  shader_type == GL_FRAGMENT_SHADER ||
330  shader_type == GL_COMPUTE_SHADER);
331 
332  ext = strrchr(filename, '.');
333  if (ext == NULL) {
334  logMsg("Cannot load shader %s: filename missing required "
335  "extension", filename);
336  return (0);
337  }
338  ext++;
339 
340  if (strcmp(ext, "spv") == 0) {
341  return (shader_from_spirv(shader_type, filename,
342  entry_pt, spec_const));
343  }
344  shader_text = file2str(filename, NULL);
345  if (shader_text == NULL) {
346  logMsg("Cannot load shader %s: %s", filename, strerror(errno));
347  return (0);
348  }
349  shader = shader_from_text(shader_type, shader_text, filename,
350  spec_const);
351  lacf_free(shader_text);
352 
353  return (shader);
354 }

356 static void
357 construct_defines(const GLchar *shader_text, GLchar **defines,
358  GLchar **shader_proc, const shader_spec_const_t *spec_const)
359 {
360  /* just a default guess */
361  int version = 120;
362  GLchar *mod_text = strdup(shader_text);
363  GLchar *ver_start = strstr(mod_text, "#version");
364  size_t len;
365 
366  /*
367  * Grab the version number from the '#version' directive and slap it
368  * at the start of our preamble. Then delete the old one.
369  */
370  if (ver_start != NULL && sscanf(&ver_start[9], "%d", &version) == 1) {
371  for (; *ver_start != '\0' && *ver_start != '\n'; ver_start++)
372  *ver_start = ' ';
373  }
374 
375  *shader_proc = mod_text;
376  *defines = sprintf_alloc(
377  "#version %d\n"
378  "#define IBM=%d\n"
379  "#define APL=%d\n"
380  "#define LIN=%d\n"
381  "%s",
382  version, IBM, APL, LIN,
383  version == 420 ? EXTRA_2D_DEFINES : "");
384  len = strlen(*defines);
385  for (const shader_spec_const_t *sc = spec_const;
386  sc != NULL && !sc->is_last; sc++) {
387  if (sc->is_float) {
388  float *value = (float *)&sc->val;
389  append_format(defines, &len,
390  "#define SPIRV_CROSS_CONSTANT_ID_%d %f\n",
391  sc->idx, *value);
392  } else {
393  append_format(defines, &len,
394  "#define SPIRV_CROSS_CONSTANT_ID_%d %d\n",
395  sc->idx, sc->val);
396  }
397  }
398  append_format(defines, &len, "#line 0\n");
399 }

401 /*
402  * Loads and compiles GLSL shader from a string of text and returns the
403  * shader object ID. The shader type is passed in `shader_type'. Returns
404  * 0 if the shader failed to load (an error description is printed to the
405  * X-Plane log).
406  */
407 static GLuint
408 shader_from_text(GLenum shader_type, const GLchar *shader_text,
409  const char *filename, const shader_spec_const_t *spec_const)
410 {
411  GLint shader = 0;
412  GLint compile_result;
413  GLchar *defines;
414  const GLchar *text[2];
415  GLchar *shader_proc;
416 
417  ASSERT(shader_text != NULL);
418  if (filename == NULL)
419  filename = "<cstring>";
420 
421  shader = glCreateShader(shader_type);
422  if (shader == 0) {
423  logMsg("Cannot load shader %s: glCreateShader failed with "
424  "error 0x%x", filename, glGetError());
425  goto errout;
426  }
427 
428  construct_defines(shader_text, &defines, &shader_proc, spec_const);
429  text[0] = defines;
430  text[1] = shader_proc;
431  glShaderSource(shader, 2, text, NULL);
432  free(defines);
433  free(shader_proc);
434 
435  glCompileShader(shader);
436 
437  glGetShaderiv(shader, GL_COMPILE_STATUS, &compile_result);
438  if (compile_result == GL_FALSE) {
439  GLint len;
440  GLchar *buf;
441 
442  glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &len);
443  buf = safe_calloc(len + 1, sizeof (*buf));
444  glGetShaderInfoLog(shader, len, NULL, buf);
445  logMsg("Cannot load shader %s: compile error: %s", filename,
446  buf);
447  free(buf);
448 
449  goto errout;
450  }
451 
452  return (shader);
453 errout:
454  if (shader != 0)
455  glDeleteShader(shader);
456  return (0);
457 }

460  * Loads, compiles and links a GLSL shader program composed of a vertex
461  * shader and fragment shader object.
462  *
463  * @param prog_name Shader program name. This is used for diagnostic
464  * purposes only.
465  * @param vert_file Full file path to the file containing the vertex shader.

467  * @param frag_file Full file path to the file containing the fragment shader.

469  * @param ... Additional parameters contain vertex attribute array index
470  * bindings. These must be passed in pairs of "char *" for the name
471  * of the vertex shader attribute, and GLuint for the array index. To
472  * terminate the list, pass NULL. Example:
473  * GLuint program = shader_prog_from_file("my_test_prog",
474  * "/file/path/to/shader.vert", "/file/path/to/shader.frag",
475  * "vertex_pos", 0, "tex_coord", 1, NULL);
476  *
477  * @return The compiled and linked shader program, ready for use in
478  * glUseProgram. Returns 0 if compiling or linking of the shader
479  * program failed (an error description is printed to the X-Plane log).

483  const char *frag_file, ...)

485  GLuint res;
486  va_list ap;
487  shader_attr_bind_t *binds;

490  binds = attr_binds_from_args(ap);
491  va_end(ap);
492  res = shader_prog_from_file_v(progname, vert_file, frag_file, binds);
493  free(binds);
494 
495  return (res);

598 /*
599  * Handles the loading of a shader from the components of a shader_prog_info_t
600  * structure. The shader type must be passed in `shader_type' (currently must
601  * be GL_VERTEX_SHADER or GL_FRAGMENT_SHADER). The program name is only used
602  * for referring to text-based GLSL fallback shaders in case compiling them
603  * fails and to give the user an idea of the error that occurred. `dirpath'
604  * is the name of the directory from which to load the shaders. The rest of
605  * the arguments are the respective vert_ or frag_ fields from a
606  * shader_prog_info_t structure.
607  */
608 static bool_t
609 shader_from_file_or_text(GLenum shader_type, const char *dirpath,
610  const shader_prog_info_t *prog_info, const shader_info_t *shader_info,
611  GLuint *shader)
612 {
613  ASSERT(shader_info->filename != NULL || shader_info->glsl != NULL);
614 
615  if (shader_info->filename != NULL) {
616  char *path = mkpathname(dirpath, shader_info->filename, NULL);
617  ASSERT3P(shader_info->glsl, ==, NULL);
618  *shader = shader_from_file(shader_type, path,
619  shader_info->entry_pt, shader_info->spec_const);
620  free(path);
621  if ((*shader) == 0)
622  return (B_FALSE);
623  } else if (shader_info->glsl != NULL) {
624  *shader = shader_from_text(shader_type, shader_info->glsl,
625  prog_info->progname, NULL);
626  if ((*shader) == 0)
627  return (B_FALSE);
628  }
629 
630  return (B_TRUE);
631 }

634  * Loads, specializes/compiles and links a shader program a shader_prog_info_t
635  * structure. The info structure is a structure designed to allow loading a
636  * range of shader types with automatic fallback in case support for the
637  * given shader type is limited.
638  *
639  * @param dirpath The absolute directory path to which all paths in `info'
640  * are relative.
641  * @param info The shader program construction structure. See shader.h for
642  * a description of its fields.
643  *
644  * @return The shader, ready for use in rendering, or 0 in case an errors
645  * occurs during the loading process. An error message is appended to
646  * the X-Plane log explaining the error.
647  */
648 API_EXPORT GLuint
649 shader_prog_from_info(const char *dirpath, const shader_prog_info_t *info)
650 {
651  GLuint vert_shader = 0, frag_shader = 0, comp_shader = 0;
652  bool_t debugger = glutils_nsight_debugger_present();
653  GLuint prog;
654 
655  /* Caller must have provided at least one! */
656  ASSERT(info->vert != NULL || info->frag != NULL || info->comp != NULL);
657  /* Vertex & fragment shaders aren't allowed in compute shaders. */
658  ASSERT((info->vert == NULL && info->frag == NULL) || info->comp == NULL);
659 
660  if (info->vert != NULL && !shader_from_file_or_text(GL_VERTEX_SHADER,
661  dirpath, info, info->vert, &vert_shader))
662  goto errout;
663  if (info->frag != NULL && !shader_from_file_or_text(GL_FRAGMENT_SHADER,
664  dirpath, info, info->frag, &frag_shader))
665  goto errout;
666  if (info->comp != NULL && !shader_from_file_or_text(GL_COMPUTE_SHADER,
667  dirpath, info, info->comp, &comp_shader))
668  goto errout;
669 
670  if (debugger) {
671  logMsg("loading %s vert: %d frag: %d comp: %d",
672  info->progname, vert_shader, frag_shader, comp_shader);
673  }
674 
675  prog = shaders2prog(info->progname, vert_shader, frag_shader,
676  comp_shader, info->attr_binds);
677  if (debugger && prog != 0)
678  logMsg("loaded %s progID: %d", info->progname, prog);
679 
680  return (prog);
681 errout:
682  if (vert_shader != 0)
683  glDeleteShader(vert_shader);
684  if (frag_shader != 0)
685  glDeleteShader(frag_shader);
686  if (comp_shader != 0)
687  glDeleteShader(comp_shader);
688  return (0);
689 }

691 /*
692  * Takes a vertex and fragment shader object and links them together, applying
693  * vertex attribute array bindings as specified in the `ap' variadic list.
694  * Returns the compiled and linked shader program, or 0 on error (error
695  * description is appended to X-Plane log file). The passed vertex and fragment
696  * shader objects are *always* consumed and released at the end of the
697  * function (regardless if an error occurred or not), so the caller needn't
698  * dispose of them on its own.
699  */
700 static GLuint
701 shaders2prog(const char *progname, GLuint vert_shader, GLuint frag_shader,
702  GLuint comp_shader, const shader_attr_bind_t *attr_binds)
703 {
704  GLuint prog = 0;
705  GLint linked;
706 
707  ASSERT(progname != NULL);
708  ASSERT((vert_shader == 0 && frag_shader == 0) || comp_shader == 0);
709 
710  prog = glCreateProgram();
711  if (vert_shader != 0)
712  glAttachShader(prog, vert_shader);
713  if (frag_shader != 0)
714  glAttachShader(prog, frag_shader);
715  if (comp_shader != 0)
716  glAttachShader(prog, comp_shader);
717 
718  while (attr_binds != NULL && attr_binds->name != NULL) {
719  ASSERT(vert_shader != 0);
720  glBindAttribLocation(prog, attr_binds->idx, attr_binds->name);
721  attr_binds++;
722  }
723 
724  glLinkProgram(prog);
725  glGetProgramiv(prog, GL_LINK_STATUS, &linked);
726  if (linked == GL_FALSE) {
727  GLint len;
728  GLchar *buf;
729 
730  glGetProgramiv(prog, GL_INFO_LOG_LENGTH, &len);
731  buf = safe_calloc(len + 1, sizeof (*buf));
732  glGetProgramInfoLog(prog, len, NULL, buf);
733  logMsg("Failed to link GLSL program %s: %s", progname, buf);
734  free(buf);
735 
736  glDeleteProgram(prog);
737  if (vert_shader != 0)
738  glDeleteShader(vert_shader);
739  if (frag_shader != 0)
740  glDeleteShader(frag_shader);
741  return (0);
742  }
743  if (vert_shader != 0) {
744  glDetachShader(prog, vert_shader);
745  glDeleteShader(vert_shader);
746  }
747  if (frag_shader != 0) {
748  glDetachShader(prog, frag_shader);
749  glDeleteShader(frag_shader);
750  }
751  if (comp_shader != 0) {
752  glDetachShader(prog, comp_shader);
753  glDeleteShader(comp_shader);
754  }
755 
756  return (prog);
757 }
