79 void
80 htbl_create(htbl_t *htbl, size_t tbl_sz, size_t key_sz, bool_t multi_value)
81 {
82  ASSERT(htbl != NULL);
83  ASSERT(key_sz != 0);
84  ASSERT(tbl_sz != 0);
85  ASSERT(tbl_sz <= ((size_t)1 << ((sizeof(size_t) * 8) - 1)));
86 
87  memset(htbl, 0, sizeof (*htbl));
88  /* round table size up to nearest multiple of 2 */
89  htbl->tbl_sz = P2ROUNDUP(tbl_sz);
90  htbl->buckets = malloc(sizeof (*htbl->buckets) * htbl->tbl_sz);
91  ASSERT(htbl->buckets != NULL);
92  for (size_t i = 0; i < htbl->tbl_sz; i++)
93  list_create(&htbl->buckets[i], sizeof (htbl_bucket_item_t),
94  offsetof(htbl_bucket_item_t, bucket_node));
95  htbl->key_sz = key_sz;
96  htbl->multi_value = multi_value;
97 }

154 void
155 htbl_empty(htbl_t *htbl, void (*func)(void *value, void *userinfo),
156  void *userinfo)
157 {
158  ASSERT(htbl != NULL);
159 
160  if (htbl->num_values == 0)
161  return;
162  for (size_t i = 0; i < htbl->tbl_sz; i++) {
163  for (htbl_bucket_item_t *item = list_head(&htbl->buckets[i]);
164  item; item = list_head(&htbl->buckets[i])) {
165  if (htbl->multi_value)
166  htbl_empty_multi_item(item, func, userinfo);
167  else if (func != NULL)
168  func(item->value, userinfo);
169  list_remove_head(&htbl->buckets[i]);
170  free(item);
171  }
172  }
173  htbl->num_values = 0;

217 void
218 htbl_set(htbl_t *htbl, const void *key, void *value)
219 {
220  ASSERT(htbl != NULL);
221  ASSERT(key != NULL);
222 
223  list_t *bucket = &htbl->buckets[H(key, htbl->key_sz) &
224  (htbl->tbl_sz - 1)];
225  htbl_bucket_item_t *item;
226 
227  ASSERT(key != NULL);
228  ASSERT(value != NULL);
229  for (item = list_head(bucket); item; item = list_next(bucket, item)) {
230  if (memcmp(item->key, key, htbl->key_sz) == 0) {
231  if (htbl->multi_value)
232  htbl_multi_value_add(htbl, item, value);
233  else
234  item->value = value;
235  return;
236  }
237  }
238  item = calloc(sizeof (*item) + htbl->key_sz - 1, 1);
239  memcpy(item->key, key, htbl->key_sz);
240  if (htbl->multi_value) {
241  list_create(&item->multi.list, sizeof (htbl_multi_value_t),
242  offsetof(htbl_multi_value_t, node));
243  htbl_multi_value_add(htbl, item, value);
244  } else {
245  item->value = value;
246  }
247  list_insert_head(bucket, item);
248  htbl->num_values++;
249 }

265 void
266 htbl_remove(htbl_t *htbl, const void *key, bool_t nil_ok)
267 {
268  ASSERT(htbl != NULL);
269  ASSERT(key != NULL);
270 
271  list_t *bucket = &htbl->buckets[H(key, htbl->key_sz) &
272  (htbl->tbl_sz - 1)];
273  htbl_bucket_item_t *item;
274 
275  for (item = list_head(bucket); item; item = list_next(bucket, item)) {
276  if (memcmp(item->key, key, htbl->key_sz) == 0) {
277  list_remove(bucket, item);
278  if (htbl->multi_value) {
279  htbl_empty_multi_item(item, NULL, NULL);
280  ASSERT(htbl->num_values >= item->multi.num);
281  htbl->num_values -= item->multi.num;
282  } else {
283  free(item);
284  ASSERT(htbl->num_values != 0);
285  htbl->num_values--;
286  }
287  return;
288  }
289  }
290  ASSERT(nil_ok);
291 }

413 void
414 htbl_foreach(const htbl_t *htbl,
415  void (*func)(const void *key, void *value, void *userinfo), void *userinfo)
416 {
417  ASSERT(htbl != NULL);
418  ASSERT(func != NULL);
419  for (size_t i = 0; i < htbl->tbl_sz; i++) {
420  list_t *bucket = &htbl->buckets[i];
421  for (const htbl_bucket_item_t *item = list_head(bucket),
422  *next_item = NULL; item; item = next_item) {
423  /*
424  * To support current value removal in the callback,
425  * retrieve the next pointer right now.
426  */
427  next_item = list_next(bucket, item);
428  if (htbl->multi_value) {
429  const list_t *ml = &item->multi.list;
430  for (htbl_multi_value_t *mv = list_head(ml),
431  *mv_next = NULL; mv != NULL; mv = mv_next) {
432  mv_next = list_next(ml, mv);
433  func(item->key, mv->value, userinfo);
434  }
435  } else {
436  func(item->key, item->value, userinfo);
437  }
438  }
439  }
440 }
