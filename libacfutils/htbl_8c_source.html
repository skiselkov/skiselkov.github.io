40 void
41 htbl_create(htbl_t *htbl, size_t tbl_sz, size_t key_sz, int multi_value)
42 {
43  ASSERT(key_sz != 0);
44  ASSERT(tbl_sz != 0);
45  ASSERT(tbl_sz <= ((size_t)1 << ((sizeof(size_t) * 8) - 1)));
46 
47  memset(htbl, 0, sizeof (*htbl));
48  /* round table size up to nearest multiple of 2 */
49  htbl->tbl_sz = P2ROUNDUP(tbl_sz);
50  htbl->buckets = malloc(sizeof (*htbl->buckets) * htbl->tbl_sz);
51  ASSERT(htbl->buckets != NULL);
52  for (size_t i = 0; i < htbl->tbl_sz; i++)
53  list_create(&htbl->buckets[i], sizeof (htbl_bucket_item_t),
54  offsetof(htbl_bucket_item_t, bucket_node));
55  htbl->key_sz = key_sz;
56  htbl->multi_value = multi_value;
57 }

83 void
84 htbl_empty(htbl_t *htbl, void (*func)(void *, void *), void *arg)
85 {
86  if (htbl->num_values == 0)
87  return;
88 
89  for (size_t i = 0; i < htbl->tbl_sz; i++) {
90  for (htbl_bucket_item_t *item = list_head(&htbl->buckets[i]);
91  item; item = list_head(&htbl->buckets[i])) {
92  if (htbl->multi_value)
93  htbl_empty_multi_item(item, func, arg);
94  else if (func)
95  func(item->value, arg);
96  list_remove_head(&htbl->buckets[i]);
97  free(item);
98  }
99  }
100  htbl->num_values = 0;
101 }

121 void
122 htbl_set(htbl_t *htbl, const void *key, void *value)
123 {
124  list_t *bucket = &htbl->buckets[H(key, htbl->key_sz) &
125  (htbl->tbl_sz - 1)];
126  htbl_bucket_item_t *item;
127 
128  ASSERT(key != NULL);
129  ASSERT(value != NULL);
130  for (item = list_head(bucket); item; item = list_next(bucket, item)) {
131  if (memcmp(item->key, key, htbl->key_sz) == 0) {
132  if (htbl->multi_value)
133  htbl_multi_value_add(htbl, item, value);
134  else
135  item->value = value;
136  return;
137  }
138  }
139  item = calloc(sizeof (*item) + htbl->key_sz - 1, 1);
140  memcpy(item->key, key, htbl->key_sz);
141  if (htbl->multi_value) {
142  list_create(&item->multi.list, sizeof (htbl_multi_value_t),
143  offsetof(htbl_multi_value_t, node));
144  htbl_multi_value_add(htbl, item, value);
145  } else {
146  item->value = value;
147  }
148  list_insert_head(bucket, item);
149  htbl->num_values++;
150 }

152 void
153 htbl_remove(htbl_t *htbl, const void *key, int nil_ok)
154 {
155  list_t *bucket = &htbl->buckets[H(key, htbl->key_sz) &
156  (htbl->tbl_sz - 1)];
157  htbl_bucket_item_t *item;
158 
159  for (item = list_head(bucket); item; item = list_next(bucket, item)) {
160  if (memcmp(item->key, key, htbl->key_sz) == 0) {
161  list_remove(bucket, item);
162  if (htbl->multi_value) {
163  htbl_empty_multi_item(item, NULL, NULL);
164  ASSERT(htbl->num_values >= item->multi.num);
165  htbl->num_values -= item->multi.num;
166  } else {
167  free(item);
168  ASSERT(htbl->num_values != 0);
169  htbl->num_values--;
170  }
171  return;
172  }
173  }
174  ASSERT(nil_ok != 0);
175 }

177 void htbl_remove_multi(htbl_t *htbl, const void *key, void *list_item)
178 {
179  htbl_multi_value_t *mv = list_item;
180  htbl_bucket_item_t *item = mv->item;
181 
182  ASSERT(htbl->multi_value != 0);
183  ASSERT(key != NULL);
184  ASSERT(item != NULL);
185  ASSERT(item->multi.num != 0);
186  ASSERT(htbl->num_values != 0);
187 
188  list_remove(&item->multi.list, mv);
189  item->multi.num--;
190  htbl->num_values--;
191  free(mv);
192  if (item->multi.num == 0) {
193  list_t *bucket =
194  &htbl->buckets[H(key, htbl->key_sz) & (htbl->tbl_sz - 1)];
195  list_remove(bucket, item);
196  list_destroy(&item->multi.list);
197  free(item);
198  }

233 void
234 htbl_foreach(const htbl_t *htbl, void (*func)(const void *, void *, void *),
235  void *arg)
236 {
237  for (size_t i = 0; i < htbl->tbl_sz; i++) {
238  list_t *bucket = &htbl->buckets[i];
239  for (const htbl_bucket_item_t *item = list_head(bucket),
240  *next_item = NULL; item; item = next_item) {
241  /*
242  * To support current value removal in the callback,
243  * retrieve the next pointer right now.
244  */
245  next_item = list_next(bucket, item);
246  if (htbl->multi_value) {
247  const list_t *ml = &item->multi.list;
248  for (htbl_multi_value_t *mv = list_head(ml),
249  *mv_next = NULL; mv != NULL; mv = mv_next) {
250  mv_next = list_next(ml, mv);
251  func(item->key, mv->value, arg);
252  }
253  } else {
254  func(item->key, item->value, arg);
255  }
256  }
257  }
258 }
