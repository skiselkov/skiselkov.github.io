265 API_EXPORT void
266 glutils_update_3D_quads_impl(glutils_quads_t *quads, const char *filename,
267  int line, const vect3_t *p, const vect2_t *t, size_t num_pts)
268 {
269  vtx_t *vtx_data = safe_calloc(num_pts, sizeof (*vtx_data));
270 
271  ASSERT(quads != NULL);
272  ASSERT(p != NULL);
273  ASSERT(glutils_quads_inited(quads));
274 
275  for (size_t i = 0; i < num_pts; i++) {
276  vtx_data[i].pos[0] = p[i].x;
277  vtx_data[i].pos[1] = p[i].y;
278  vtx_data[i].pos[2] = p[i].z;
279  if (t != NULL) {
280  vtx_data[i].tex0[0] = t[i].x;
281  vtx_data[i].tex0[1] = t[i].y;
282  }
283  }
284  glBindBuffer(GL_ARRAY_BUFFER, quads->vbo);
285  glBufferData(GL_ARRAY_BUFFER, num_pts * sizeof (vtx_t), vtx_data,
286  GL_STATIC_DRAW);
287  if (quads->num_vtx != num_pts || quads->ibo == 0) {
288  if (quads->ibo != 0)
289  glDeleteBuffers(1, &quads->ibo);
290  quads->ibo = glutils_make_quads_IBO(num_pts);
291  }
292  if (quads->num_vtx != num_pts) {
293  IF_TEXSZ(TEXSZ_FREE_BYTES_INSTANCE(glutils_quads_vbo, quads,
294  quads->num_vtx * sizeof (vtx_t)));
295  IF_TEXSZ(TEXSZ_ALLOC_BYTES_INSTANCE(glutils_quads_vbo, quads,
296  filename, line, num_pts * sizeof (vtx_t)));
297  }
298  quads->num_vtx = num_pts;
299 
300  free(vtx_data);
301 }

323 static void
324 glutils_draw_common(GLenum mode, GLuint vao, GLuint vbo, GLuint ibo,
325  bool_t *setup, size_t num_vtx, GLint prog)
326 {
327  GLint pos_loc = -1, tex0_loc = -1;
328 
329  ASSERT(vbo != 0);
330  ASSERT(prog != 0);
331 
332  if (vao != 0) {
333  /* Vertex arrays supported */
334  glBindVertexArray(vao);
335  }
336 
337  if (vao == 0 || !(*setup)) {
338  glBindBuffer(GL_ARRAY_BUFFER, vbo);
339  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo);
340 
341  pos_loc = glGetAttribLocation(prog, "vtx_pos");
342  tex0_loc = glGetAttribLocation(prog, "vtx_tex0");
343 
344  glutils_enable_vtx_attr_ptr(pos_loc, 3, GL_FLOAT, GL_FALSE,
345  sizeof (vtx_t), offsetof(vtx_t, pos));
346  glutils_enable_vtx_attr_ptr(tex0_loc, 2, GL_FLOAT, GL_FALSE,
347  sizeof (vtx_t), offsetof(vtx_t, tex0));
348  *setup = B_TRUE;
349  }
350 
351  if (ibo != 0)
352  glDrawElements(mode, num_vtx, GL_UNSIGNED_INT, NULL);
353  else
354  glDrawArrays(mode, 0, num_vtx);
355 
356  if (vao != 0) {
357  glBindVertexArray(0);
358  } else {
359  glutils_disable_vtx_attr_ptr(pos_loc);
360  glutils_disable_vtx_attr_ptr(tex0_loc);
361  }
362  glBindBuffer(GL_ARRAY_BUFFER, 0);
363  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
364 }

394 static int
395 cache_compar(const void *a, const void *b)
396 {
397  const cache_entry_t *ca = a, *cb = b;
398 
399  if (ca->type < cb->type)
400  return (-1);
401  if (ca->type > cb->type)
402  return (1);
403  for (int i = 0; i < 2; i++) {
404  if (ca->buf_sz[i] < cb->buf_sz[i])
405  return (-1);
406  if (ca->buf_sz[i] > cb->buf_sz[i])
407  return (1);
408  }
409  for (int i = 0; i < 2; i++) {
410  int res = memcmp(ca->buf[i], cb->buf[i], cb->buf_sz[i]);
411 
412  if (res < 0)
413  return (-1);
414  if (res > 0)
415  return (1);
416  }
417 
418  return (0);
419 }

454 static void
455 free_cache_entry(cache_entry_t *ce)
456 {
457  ASSERT(ce != NULL);

459  switch (ce->type) {
460  case CACHE_ENTRY_2D_QUADS:

462  glutils_destroy_quads(&ce->quads);
463  break;
464  case CACHE_ENTRY_3D_LINES:
465  glutils_destroy_lines(&ce->lines);
466  break;
467  default:

469  }
470  free(ce->buf[0]);

479 void
480 glutils_cache_destroy(glutils_cache_t *cache)
481 {
482  cache_entry_t *ce;
483  void *cookie;
484 
485  if (cache == NULL)
486  return;
487 
488  /* The entries are retained in the tree, so we'll just empty the lru */
489  while ((ce = list_remove_head(&cache->lru)) != NULL)
490  ;
491  list_destroy(&cache->lru);
492 
493  cookie = NULL;
494  while ((ce = avl_destroy_nodes(&cache->tree, &cookie)) != NULL)
495  free_cache_entry(ce);
496  avl_destroy(&cache->tree);
497 
498  free(cache);
499 }

501 /*
502  * Takes a cache and trims its memory usage until it fits within its
503  * capacity limit. `extra_needed' specifies the extra cache storage
504  * space required to insert a new object, and thus adjusts the cache's
505  * storage target.
506  */
507 static void
508 trim_cache(glutils_cache_t *cache, size_t extra_needed)
509 {
510  ASSERT(cache != NULL);
511 
512  while (cache->sz + extra_needed > cache->cap) {
513  cache_entry_t *ce = list_remove_tail(&cache->lru);
514 
515  if (ce == NULL)
516  break;
517  avl_remove(&cache->tree, ce);
518  ASSERT3U(cache->sz, >=, ce->buf_sz[0] + ce->buf_sz[1]);
519  cache->sz -= (ce->buf_sz[0] + ce->buf_sz[1]);
520  free_cache_entry(ce);
521  }
522 }

524 /*
525  * Adds a new cache entry.
526  */
527 static cache_entry_t *
528 cache_add_entry(glutils_cache_t *cache, avl_index_t where,
529  cache_entry_type_t type, const void *buf0, size_t buf0_sz,
530  const void *buf1, size_t buf1_sz, size_t num_pts)
531 {
532  cache_entry_t *ce = safe_calloc(1, sizeof (*ce));
533 
534  ASSERT(cache != NULL);
535  ASSERT(buf0 != NULL);
536 
537  ce->type = type;
538 
539  ce->buf_sz[0] = buf0_sz;
540  ce->buf[0] = safe_malloc(buf0_sz);
541  memcpy(ce->buf[0], buf0, buf0_sz);
542  if (buf1 != NULL) {
543  ce->buf_sz[1] = buf1_sz;
544  ce->buf[1] = safe_malloc(buf1_sz);
545  memcpy(ce->buf[1], buf1, buf1_sz);
546  }
547 
548  switch (type) {
549  case CACHE_ENTRY_2D_QUADS:
550  glutils_init_2D_quads(&ce->quads, buf0, buf1, num_pts);
551  break;
552  case CACHE_ENTRY_3D_QUADS:
553  glutils_init_3D_quads(&ce->quads, buf0, buf1, num_pts);
554  break;
555  case CACHE_ENTRY_3D_LINES:
556  glutils_init_3D_lines(&ce->lines, buf0, num_pts);
557  break;
558  default:
559  VERIFY(0);
560  }
561  avl_insert(&cache->tree, ce, where);
562  list_insert_head(&cache->lru, ce);
563 
564  return (ce);
565 }

567 static void *
568 glutils_cache_get_common(glutils_cache_t *cache, cache_entry_type_t type,
569  const void *p, const void *t, size_t num_pts)
570 {
571  size_t p_sz = (type == CACHE_ENTRY_2D_QUADS ? sizeof (vect2_t) :
572  sizeof (vect3_t));
573  const cache_entry_t srch = {
574  .type = type,
575  .buf = { (void *)p, (void *)t },
576  .buf_sz = {
577  p_sz * num_pts,
578  t != NULL ? sizeof (vect2_t) * num_pts : 0
579  }
580  };
581  cache_entry_t *ce;
582  avl_index_t where;
583  size_t bytes = srch.buf_sz[0] + srch.buf_sz[1];
584 
585  ASSERT(cache != NULL);
586  ASSERT(p != NULL);
587  ASSERT(num_pts != 0);
588 
589  ce = avl_find(&cache->tree, &srch, &where);
590  if (ce == NULL) {
591  trim_cache(cache, bytes);
592  ce = cache_add_entry(cache, where, type, p, srch.buf_sz[0],
593  t, srch.buf_sz[1], num_pts);
594  } else {
595  list_remove(&cache->lru, ce);
596  list_insert_head(&cache->lru, ce);
597  }
598  if (type <= CACHE_ENTRY_3D_QUADS)
599  return (&ce->quads);
600  else
601  return (&ce->lines);
602 }

862 static inline void
863 texsz_incr(const char *token, const void *instance, const char *filename,
864  int line, int64_t bytes)
865 {
866  texsz_alloc_t srch = { .token = token };
867  texsz_alloc_t *ta;
868  avl_index_t where;
869 
870  mutex_enter(&texsz.lock);
871 
872  ASSERT_MSG(texsz.bytes + bytes >= 0, "Texture size accounting error "
873  "(incr %ld bytes)", (long)bytes);
874  texsz.bytes += bytes;
875  ta = avl_find(&texsz.allocs, &srch, &where);
876  if (ta == NULL) {
877  ta = safe_calloc(1, sizeof (*ta));
878  ta->token = token;
879  avl_create(&ta->instances, texsz_instance_compar,
880  sizeof (texsz_instance_t),
881  offsetof(texsz_instance_t, node));
882  avl_insert(&texsz.allocs, ta, where);
883  }
884  ASSERT_MSG(ta->bytes + bytes >= 0, "Texture size accounting zone "
885  "underflow error (incr %ld bytes in zone %s instance %p)",
886  (long)bytes, ta->token, instance);
887  ta->bytes += bytes;
888 
889  if (instance != NULL) {
890  texsz_instance_t srch_ti = { .instance = instance };
891  texsz_instance_t *ti;
892  avl_index_t where_ti;
893 
894  ti = avl_find(&ta->instances, &srch_ti, &where_ti);
895  if (ti == NULL) {
896  ASSERT_MSG(bytes >= 0, "Texture size accounting error "
897  "(incr %ld bytes in zone %s instance %p, but "
898  "instance is empty).", (long)bytes, ta->token,
899  instance);
900  ti = safe_calloc(1, sizeof (*ti));
901  ti->instance = instance;
902  avl_insert(&ta->instances, ti, where_ti);
903  }
904  ASSERT_MSG(ti->bytes + bytes >= 0, "Texture size accounting "
905  "instance underflow error (incr %ld bytes in zone %s "
906  "instance %p)", (long)bytes, ta->token, instance);
907  ti->bytes += bytes;
908  if (filename != NULL && snprintf(ti->allocd_at,
909  sizeof (ti->allocd_at), "%s:%d", filename, line) >=
910  (int)sizeof (ti->allocd_at)) {
911  int l = strlen(filename);
912  int off = MAX(l - 26, 0);
913  snprintf(ti->allocd_at, sizeof (ti->allocd_at),
914  "%s:%d", &filename[off], line);
915  }
916  if (ti->bytes == 0) {
917  avl_remove(&ta->instances, ti);
918  free(ti);
919  }
920  }
921 
922  mutex_exit(&texsz.lock);
923 }

925 static inline int64_t
926 texsz_bytes(GLenum format, GLenum type, unsigned w, unsigned h)
927 {
928  int channels, bpc;
929 
930  switch (format) {
931  case GL_RG:
932  case GL_RG_INTEGER:
933  channels = 2;
934  break;
935  case GL_RGB:
936  case GL_BGR:
937  case GL_RGB_INTEGER:
938  case GL_BGR_INTEGER:
939  channels = 3;
940  break;
941  case GL_RGBA:
942  case GL_BGRA:
943  case GL_RGBA_INTEGER:
944  case GL_BGRA_INTEGER:
945  channels = 4;
946  break;
947  default:
948  channels = 1;
949  break;
950  }
951 
952  switch (type) {
953  case GL_UNSIGNED_BYTE:
954  case GL_BYTE:
955  case GL_UNSIGNED_SHORT:
956  case GL_SHORT:
957  bpc = 2;
958  break;
959  case GL_UNSIGNED_INT:
960  case GL_INT:
961  case GL_FLOAT:
962  bpc = 4;
963  break;
964  default:
965  bpc = 1;
966  break;
967  }
968 
969  return (channels * bpc * w * h);
970 }
