157 /*
158  * This function is a shorthand for creating an IBO to triangulate an
159  * old-style GL_QUADS object (which isn't supported in modern OpenGL anymore).
160  * The quad is triangulated in this order: p0-p1-p2 & p0-p2-p3.
161  * The IBO is constructed for quads composed num_vtx vertices. This number
162  * must be a multiple of 4.
163  */
164 API_EXPORT GLuint
165 glutils_make_quads_IBO(size_t num_vtx)
166 {
167  GLuint idx_data[num_vtx * 2];
168  size_t i, n;
169  GLuint buf;
170 
171  ASSERT0(num_vtx & 3);
172 
173  for (i = 0, n = 0; i < num_vtx; i += 4, n += 6) {
174  idx_data[n + 0] = i + 0;
175  idx_data[n + 1] = i + 1;
176  idx_data[n + 2] = i + 2;
177 
178  idx_data[n + 3] = i + 0;
179  idx_data[n + 4] = i + 2;
180  idx_data[n + 5] = i + 3;
181  }
182 
183  glGenBuffers(1, &buf);
184  VERIFY(buf != 0);
185  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, buf);
186  glBufferData(GL_ELEMENT_ARRAY_BUFFER, n * sizeof (*idx_data),
187  idx_data, GL_STATIC_DRAW);
188 
189  return (buf);
190 }

246 API_EXPORT void
247 glutils_update_3D_quads_impl(glutils_quads_t *quads, const char *filename,
248  int line, const vect3_t *p, const vect2_t *t, size_t num_pts)
249 {
250  vtx_t *vtx_data = safe_calloc(num_pts, sizeof (*vtx_data));
251 
252  ASSERT(quads != NULL);
253  ASSERT(p != NULL);
254  ASSERT(glutils_quads_inited(quads));
255 
256  for (size_t i = 0; i < num_pts; i++) {
257  vtx_data[i].pos[0] = p[i].x;
258  vtx_data[i].pos[1] = p[i].y;
259  vtx_data[i].pos[2] = p[i].z;
260  if (t != NULL) {
261  vtx_data[i].tex0[0] = t[i].x;
262  vtx_data[i].tex0[1] = t[i].y;
263  }
264  }
265  glBindBuffer(GL_ARRAY_BUFFER, quads->vbo);
266  glBufferData(GL_ARRAY_BUFFER, num_pts * sizeof (vtx_t), vtx_data,
267  GL_STATIC_DRAW);
268  if (quads->num_vtx != num_pts || quads->ibo == 0) {
269  if (quads->ibo != 0)
270  glDeleteBuffers(1, &quads->ibo);
271  quads->ibo = glutils_make_quads_IBO(num_pts);
272  }
273  if (quads->num_vtx != num_pts) {
274  IF_TEXSZ(TEXSZ_FREE_BYTES_INSTANCE(glutils_quads_vbo, quads,
275  quads->num_vtx * sizeof (vtx_t)));
276  IF_TEXSZ(TEXSZ_ALLOC_BYTES_INSTANCE(glutils_quads_vbo, quads,
277  filename, line, num_pts * sizeof (vtx_t)));
278  }
279  quads->num_vtx = num_pts;
280 
281  free(vtx_data);
282 }

299 static void
300 glutils_draw_common(GLenum mode, GLuint vao, GLuint vbo, GLuint ibo,
301  bool_t *setup, size_t num_vtx, GLint prog)
302 {
303  GLint pos_loc = -1, tex0_loc = -1;
304 
305  ASSERT(vbo != 0);
306  ASSERT(prog != 0);
307 
308  if (vao != 0) {
309  /* Vertex arrays supported */
310  glBindVertexArray(vao);
311  }
312 
313  if (vao == 0 || !(*setup)) {
314  glBindBuffer(GL_ARRAY_BUFFER, vbo);
315  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo);
316 
317  pos_loc = glGetAttribLocation(prog, "vtx_pos");
318  tex0_loc = glGetAttribLocation(prog, "vtx_tex0");
319 
320  glutils_enable_vtx_attr_ptr(pos_loc, 3, GL_FLOAT, GL_FALSE,
321  sizeof (vtx_t), offsetof(vtx_t, pos));
322  glutils_enable_vtx_attr_ptr(tex0_loc, 2, GL_FLOAT, GL_FALSE,
323  sizeof (vtx_t), offsetof(vtx_t, tex0));
324  *setup = B_TRUE;
325  }
326 
327  if (ibo != 0)
328  glDrawElements(mode, num_vtx, GL_UNSIGNED_INT, NULL);
329  else
330  glDrawArrays(mode, 0, num_vtx);
331 
332  if (vao != 0) {
333  glBindVertexArray(0);
334  } else {
335  glutils_disable_vtx_attr_ptr(pos_loc);
336  glutils_disable_vtx_attr_ptr(tex0_loc);
337  }
338  glBindBuffer(GL_ARRAY_BUFFER, 0);
339  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
340 }

354 static int
355 cache_compar(const void *a, const void *b)
356 {
357  const cache_entry_t *ca = a, *cb = b;
358 
359  if (ca->type < cb->type)
360  return (-1);
361  if (ca->type > cb->type)
362  return (1);
363  for (int i = 0; i < 2; i++) {
364  if (ca->buf_sz[i] < cb->buf_sz[i])
365  return (-1);
366  if (ca->buf_sz[i] > cb->buf_sz[i])
367  return (1);
368  }
369  for (int i = 0; i < 2; i++) {
370  int res = memcmp(ca->buf[i], cb->buf[i], cb->buf_sz[i]);
371 
372  if (res < 0)
373  return (-1);
374  if (res > 0)
375  return (1);
376  }
377 
378  return (0);
379 }

381 /*
382  * Constructs a new object cache with a certain defined capacity in bytes.
383  * This cache lets you construct & cache gl_utils_quads_t and gl_utils_lines_t
384  * objects on the GPU. USe glutils_cache_get_* to store & retrieve cached
385  * objects. When the cache is destroyed, the stored objects are destroyed
386  * and their memory freed from the GPU.
387  *
388  * The cache is an LRU cache, so it remembers which objects were used most
389  * recently and releases the least-recently-used when it needs to trim its
390  * memory usage down.
391  *
392  * @param cap_bytes Defines the maximum size the cache is allowed to reach
393  * before it starts trimming least-recently-used objects.
394  */
395 glutils_cache_t *
396 glutils_cache_new(size_t cap_bytes)
397 {
398  glutils_cache_t *cache = safe_calloc(1, sizeof (*cache));
399 
400  ASSERT(cap_bytes != 0);
401 
402  avl_create(&cache->tree, cache_compar, sizeof (cache_entry_t),
403  offsetof(cache_entry_t, tree_node));
404  list_create(&cache->lru, sizeof (cache_entry_t),
405  offsetof(cache_entry_t, lru_node));
406  cache->cap = cap_bytes;
407 
408  return (cache);
409 }

411 /*
412  * Frees the storage associated with a glutils cache entry.
413  */
414 static void
415 free_cache_entry(cache_entry_t *ce)
416 {
417  ASSERT(ce != NULL);
418 
419  switch (ce->type) {
420  case CACHE_ENTRY_2D_QUADS:
421  case CACHE_ENTRY_3D_QUADS:
422  glutils_destroy_quads(&ce->quads);
423  break;
424  case CACHE_ENTRY_3D_LINES:
425  glutils_destroy_lines(&ce->lines);
426  break;
427  default:
428  VERIFY(0);
429  }
430  free(ce->buf[0]);
431  free(ce->buf[1]);
432  free(ce);
433 }

435 /*
436  * Destroys a glutils object cache. This frees all stored objects and
437  * their associated memory from both main memory and GPU memory.
438  */
439 void
440 glutils_cache_destroy(glutils_cache_t *cache)
441 {
442  cache_entry_t *ce;
443  void *cookie;
444 
445  if (cache == NULL)
446  return;
447 
448  /* The entries are retained in the tree, so we'll just empty the lru */
449  while ((ce = list_remove_head(&cache->lru)) != NULL)
450  ;
451  list_destroy(&cache->lru);
452 
453  cookie = NULL;
454  while ((ce = avl_destroy_nodes(&cache->tree, &cookie)) != NULL)
455  free_cache_entry(ce);
456  avl_destroy(&cache->tree);
457 
458  free(cache);
459 }

461 /*
462  * Takes a cache and trims its memory usage until it fits within its
463  * capacity limit. `extra_needed' specifies the extra cache storage
464  * space required to insert a new object, and thus adjusts the cache's
465  * storage target.
466  */
467 static void
468 trim_cache(glutils_cache_t *cache, size_t extra_needed)
469 {
470  ASSERT(cache != NULL);
471 
472  while (cache->sz + extra_needed > cache->cap) {
473  cache_entry_t *ce = list_remove_tail(&cache->lru);
474 
475  if (ce == NULL)
476  break;
477  avl_remove(&cache->tree, ce);
478  ASSERT3U(cache->sz, >=, ce->buf_sz[0] + ce->buf_sz[1]);
479  cache->sz -= (ce->buf_sz[0] + ce->buf_sz[1]);
480  free_cache_entry(ce);
481  }
482 }

484 /*
485  * Adds a new cache entry.
486  */
487 static cache_entry_t *
488 cache_add_entry(glutils_cache_t *cache, avl_index_t where,
489  cache_entry_type_t type, const void *buf0, size_t buf0_sz,
490  const void *buf1, size_t buf1_sz, size_t num_pts)
491 {
492  cache_entry_t *ce = safe_calloc(1, sizeof (*ce));
493 
494  ASSERT(cache != NULL);
495  ASSERT(buf0 != NULL);
496 
497  ce->type = type;
498 
499  ce->buf_sz[0] = buf0_sz;
500  ce->buf[0] = safe_malloc(buf0_sz);
501  memcpy(ce->buf[0], buf0, buf0_sz);
502  if (buf1 != NULL) {
503  ce->buf_sz[1] = buf1_sz;
504  ce->buf[1] = safe_malloc(buf1_sz);
505  memcpy(ce->buf[1], buf1, buf1_sz);
506  }
507 
508  switch (type) {
509  case CACHE_ENTRY_2D_QUADS:
510  glutils_init_2D_quads(&ce->quads, buf0, buf1, num_pts);
511  break;
512  case CACHE_ENTRY_3D_QUADS:
513  glutils_init_3D_quads(&ce->quads, buf0, buf1, num_pts);
514  break;
515  case CACHE_ENTRY_3D_LINES:
516  glutils_init_3D_lines(&ce->lines, buf0, num_pts);
517  break;
518  default:
519  VERIFY(0);
520  }
521  avl_insert(&cache->tree, ce, where);
522  list_insert_head(&cache->lru, ce);
523 
524  return (ce);
525 }

527 static void *
528 glutils_cache_get_common(glutils_cache_t *cache, cache_entry_type_t type,
529  const void *p, const void *t, size_t num_pts)
530 {
531  size_t p_sz = (type == CACHE_ENTRY_2D_QUADS ? sizeof (vect2_t) :
532  sizeof (vect3_t));
533  const cache_entry_t srch = {
534  .type = type,
535  .buf = { (void *)p, (void *)t },
536  .buf_sz = {
537  p_sz * num_pts,
538  t != NULL ? sizeof (vect2_t) * num_pts : 0
539  }
540  };
541  cache_entry_t *ce;
542  avl_index_t where;
543  size_t bytes = srch.buf_sz[0] + srch.buf_sz[1];
544 
545  ASSERT(cache != NULL);
546  ASSERT(p != NULL);
547  ASSERT(num_pts != 0);
548 
549  ce = avl_find(&cache->tree, &srch, &where);
550  if (ce == NULL) {
551  trim_cache(cache, bytes);
552  ce = cache_add_entry(cache, where, type, p, srch.buf_sz[0],
553  t, srch.buf_sz[1], num_pts);
554  } else {
555  list_remove(&cache->lru, ce);
556  list_insert_head(&cache->lru, ce);
557  }
558  if (type <= CACHE_ENTRY_3D_QUADS)
559  return (&ce->quads);
560  else
561  return (&ce->lines);
562 }

733 static inline void
734 texsz_incr(const char *token, const void *instance, const char *filename,
735  int line, int64_t bytes)
736 {
737  texsz_alloc_t srch = { .token = token };
738  texsz_alloc_t *ta;
739  avl_index_t where;
740 
741  mutex_enter(&texsz.lock);
742 
743  ASSERT_MSG(texsz.bytes + bytes >= 0, "Texture size accounting error "
744  "(incr %ld bytes)", (long)bytes);
745  texsz.bytes += bytes;
746  ta = avl_find(&texsz.allocs, &srch, &where);
747  if (ta == NULL) {
748  ta = safe_calloc(1, sizeof (*ta));
749  ta->token = token;
750  avl_create(&ta->instances, texsz_instance_compar,
751  sizeof (texsz_instance_t),
752  offsetof(texsz_instance_t, node));
753  avl_insert(&texsz.allocs, ta, where);
754  }
755  ASSERT_MSG(ta->bytes + bytes >= 0, "Texture size accounting zone "
756  "underflow error (incr %ld bytes in zone %s instance %p)",
757  (long)bytes, ta->token, instance);
758  ta->bytes += bytes;
759 
760  if (instance != NULL) {
761  texsz_instance_t srch_ti = { .instance = instance };
762  texsz_instance_t *ti;
763  avl_index_t where_ti;
764 
765  ti = avl_find(&ta->instances, &srch_ti, &where_ti);
766  if (ti == NULL) {
767  ASSERT_MSG(bytes >= 0, "Texture size accounting error "
768  "(incr %ld bytes in zone %s instance %p, but "
769  "instance is empty).", (long)bytes, ta->token,
770  instance);
771  ti = safe_calloc(1, sizeof (*ti));
772  ti->instance = instance;
773  avl_insert(&ta->instances, ti, where_ti);
774  }
775  ASSERT_MSG(ti->bytes + bytes >= 0, "Texture size accounting "
776  "instance underflow error (incr %ld bytes in zone %s "
777  "instance %p)", (long)bytes, ta->token, instance);
778  ti->bytes += bytes;
779  if (filename != NULL && snprintf(ti->allocd_at,
780  sizeof (ti->allocd_at), "%s:%d", filename, line) >=
781  (int)sizeof (ti->allocd_at)) {
782  int l = strlen(filename);
783  int off = MAX(l - 26, 0);
784  snprintf(ti->allocd_at, sizeof (ti->allocd_at),
785  "%s:%d", &filename[off], line);
786  }
787  if (ti->bytes == 0) {
788  avl_remove(&ta->instances, ti);
789  free(ti);
790  }
791  }
792 
793  mutex_exit(&texsz.lock);
794 }

796 static inline int64_t
797 texsz_bytes(GLenum format, GLenum type, unsigned w, unsigned h)
798 {
799  int channels, bpc;
800 
801  switch (format) {
802  case GL_RG:
803  case GL_RG_INTEGER:
804  channels = 2;
805  break;
806  case GL_RGB:
807  case GL_BGR:
808  case GL_RGB_INTEGER:
809  case GL_BGR_INTEGER:
810  channels = 3;
811  break;
812  case GL_RGBA:
813  case GL_BGRA:
814  case GL_RGBA_INTEGER:
815  case GL_BGRA_INTEGER:
816  channels = 4;
817  break;
818  default:
819  channels = 1;
820  break;
821  }
822 
823  switch (type) {
824  case GL_UNSIGNED_BYTE:
825  case GL_BYTE:
826  case GL_UNSIGNED_SHORT:
827  case GL_SHORT:
828  bpc = 2;
829  break;
830  case GL_UNSIGNED_INT:
831  case GL_INT:
832  case GL_FLOAT:
833  bpc = 4;
834  break;
835  default:
836  bpc = 1;
837  break;
838  }
839 
840  return (channels * bpc * w * h);
841 }
