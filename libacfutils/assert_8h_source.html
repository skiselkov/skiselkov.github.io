52 /*
53  * ASSERT() and VERIFY() are assertion test macros. If the condition
54  * expression provided as the argument to the macro evaluates as non-true,
55  * the program prints a debug message specifying exactly where and what
56  * condition was violated, a stack backtrace and a dumps core by
57  * calling abort().
58  *
59  * The difference between ASSERT and VERIFY is that ASSERT compiles to
60  * a no-op unless -DDEBUG is provided to the compiler. VERIFY always
61  * checks its condition and dumps if it is non-true.
62  */
63 
64 #define VERIFY_MSG(x, fmt, ...) \
65  do { \
66  if (COND_UNLIKELY(!(x))) { \
67  log_impl(log_basename(__FILE__), __LINE__, \
68  "assertion \"%s\" failed: " fmt, #x, __VA_ARGS__); \
69  LACF_CRASH(); \
70  } \
71  } while (0)

73 #define VERIFY(x) VERIFY_MSG(x, "%s", "")
74 
75 #define VERIFY3_impl(x, op, y, type, fmt) \
76  do { \
77  type tmp_x = (type)(x); \
78  type tmp_y = (type)(y); \
79  if (COND_UNLIKELY(!(tmp_x op tmp_y))) { \
80  log_impl(log_basename(__FILE__), __LINE__, \
81  "assertion %s %s %s failed (" fmt " %s " \
82  fmt ")", #x, #op, #y, tmp_x, #op, tmp_y); \
83  LACF_CRASH(); \
84  } \
85  } while (0)
86 #define VERIFY3S(x, op, y) VERIFY3_impl(x, op, y, long, "%lu")
87 #if IBM
88 #define VERIFY3U(x, op, y) \
89  VERIFY3_impl(x, op, y, unsigned long long, "0x%I64x")
90 #else /* !IBM */
91 #define VERIFY3U(x, op, y) \
92  VERIFY3_impl(x, op, y, unsigned long long, "0x%llx")
93 #endif /* !IBM */
94 #define VERIFY3F(x, op, y) VERIFY3_impl(x, op, y, double, "%f")
95 #define VERIFY3P(x, op, y) VERIFY3_impl(x, op, y, void *, "%p")
96 #define VERIFY0(x) VERIFY3S((x), ==, 0)
97 #define VERIFY_FAIL() \
98  do { \
99  log_impl(log_basename(__FILE__), __LINE__, "Internal error"); \
100  LACF_CRASH(); \
101  } while (0)
