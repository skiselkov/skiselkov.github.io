130  * When X-Plane is in a modal UI window (such as the weather setup

135 static int
136 elec_draw_cb(XPLMDrawingPhase phase, int before, void *refcon)
137 {
138  elec_sys_t *sys;
139  double sim_time, time_factor;
140 
141  UNUSED(phase);
142  UNUSED(before);
143  ASSERT(refcon != NULL);
144  sys = refcon;
145 
146  sim_time = dr_getf_prot(&sys->drs.sim_time);
147  /* Caution: sim_speed_act likes to contain a NAN while paused */
148  time_factor = round(dr_getf(&sys->drs.sim_speed_act) * 10) / 10;
149  if (sys->prev_sim_time >= sim_time || dr_geti(&sys->drs.replay) != 0 ||
150  dr_geti(&sys->drs.paused) != 0 || time_factor == 0) {
151  /* Reset the worker interval to default */
152  if (sys->started)
153  worker_set_interval_nowake(&sys->worker, EXEC_INTVAL);
154  mutex_enter(&sys->paused_lock);
155  sys->paused = true;
156  sys->time_factor = 0;
157  mutex_exit(&sys->paused_lock);
158  return (1);
159  }
160  sys->prev_sim_time = sim_time;
161  /*
162  * If the time factor returns to 1.0x, we want to make sure we
163  * go back to 1.0x exactly, instead of some fractional 1.04x thing
164  * in case the sim was only ever so slightly accelerated/decelerated.
165  */
166  if ((time_factor != sys->time_factor ||
167  (time_factor == 1 && sys->time_factor != 1)) && sys->started) {
168  worker_set_interval_nowake(&sys->worker,
169  EXEC_INTVAL / time_factor);
170  }
171  mutex_enter(&sys->paused_lock);
172  sys->paused = false;
173  sys->time_factor = time_factor;
174  mutex_exit(&sys->paused_lock);
175 
176  return (1);
177 }

296 static bool
297 resolve_bus_links(elec_sys_t *sys, elec_comp_t *bus)
298 {
299  ASSERT(sys != NULL);
300  ASSERT(bus != NULL);
301  ASSERT(bus->info != NULL);
302  ASSERT3U(bus->info->type, ==, ELEC_BUS);
303 
304  bus->n_links = bus->info->bus.n_comps;
305  bus->links = safe_calloc(bus->n_links, sizeof (*bus->links));
306 
307 #define CHECK_COMP(cond, reason) \
308  do { \
309  if (!(cond)) { \
310  logMsg("%s (%s:%d): %s", comp->info->name, \
311  sys->conf_filename, comp->info->parse_linenum, \
312  (reason)); \
313  return (false); \
314  } \
315  } while (0)
316 #define CHECK_COMP_V(cond, reason, ...) \
317  do { \
318  if (!(cond)) { \
319  logMsg("%s (%s:%d): " reason, comp->info->name, \
320  sys->conf_filename, comp->info->parse_linenum, \
321  __VA_ARGS__); \
322  return (false); \
323  } \
324  } while (0)
325 
326  for (size_t i = 0; i < bus->info->bus.n_comps; i++) {
327  elec_comp_t *comp = find_comp(sys, bus->info->bus.comps[i],
328  bus);
329 
330  bus->links[i].comp = comp;
331  ASSERT(comp->info != NULL);
332  switch (comp->info->type) {
333  case ELEC_BATT:
334  /* Batteries are DC-only devices! */
335  CHECK_COMP(!bus->info->bus.ac, "batteries cannot "
336  "connect to AC buses (batteries are inherently "
337  "DC-only devices)");
338  ASSERT3U(comp->n_links, ==, 1);
339  ASSERT(comp->links != NULL);
340  comp->links[0].comp = bus;
341  break;
342  case ELEC_GEN:
343  /* Generators can be DC or AC */
344  CHECK_COMP_V(bus->info->bus.ac == src_is_AC(comp->info),
345  "AC/DC status is mismatched between the generator "
346  "and its output bus %s", bus->info->name);
347  ASSERT3U(comp->n_links, ==, 1);
348  ASSERT(comp->links != NULL);
349  comp->links[0].comp = bus;
350  break;
351  case ELEC_TRU:
352  ASSERT3U(comp->n_links, ==, 2);
353  ASSERT(comp->links != NULL);
354  if (comp->info->tru.ac == bus->info) {
355  CHECK_COMP_V(bus->info->bus.ac, "input to the "
356  "TRU must connect to an AC bus, but "
357  "%s is DC", bus->info->name);
358  comp->links[0].comp = bus;
359  } else {
360  ASSERT3P(comp->info->tru.dc, ==, bus->info);
361  CHECK_COMP_V(!bus->info->bus.ac, "output of "
362  "the TRU must connect to a DC bus, "
363  "but %s is AC", bus->info->name);
364  comp->links[1].comp = bus;
365  }
366  break;
367  case ELEC_INV:
368  ASSERT3U(comp->n_links, ==, 2);
369  ASSERT(comp->links != NULL);
370  if (comp->info->tru.dc == bus->info) {
371  CHECK_COMP_V(!bus->info->bus.ac, "input to "
372  "the inverter must connect to a DC bus, "
373  "but %s is AC", bus->info->name);
374  comp->links[0].comp = bus;
375  } else {
376  ASSERT3P(comp->info->tru.ac, ==, bus->info);
377  CHECK_COMP_V(bus->info->bus.ac, "output of "
378  "the the inverter must connect to an "
379  "AC bus, but %s is DC", bus->info->name);
380  comp->links[1].comp = bus;
381  }
382  break;
383  case ELEC_LOAD:
384  CHECK_COMP_V(bus->info->bus.ac == comp->info->load.ac,
385  "cannot connect %s load to %s bus",
386  comp->info->load.ac ? "AC" : "DC",
387  bus->info->bus.ac ? "AC" : "DC");
388  ASSERT3U(comp->n_links, ==, 1);
389  ASSERT(comp->links != NULL);
390  comp->links[0].comp = bus;
391  break;
392  case ELEC_BUS:
393  CHECK_COMP_V(false, "Invalid link: cannot connect "
394  "bus %s directly to bus %s", bus->info->name,
395  comp->info->name);
396  break;
397  case ELEC_CB:
398  case ELEC_SHUNT:
399  /* 3-phase breakers are only allowed on AC buses */
400  if (comp->info->type == ELEC_CB) {
401  CHECK_COMP(!comp->info->cb.triphase ||
402  bus->info->bus.ac, "3-phase breakers "
403  "cannot be connected to DC buses");
404  }
405  ASSERT3U(comp->n_links, ==, 2);
406  ASSERT(comp->links != NULL);
407  if (comp->links[0].comp == NULL) {
408  comp->links[0].comp = bus;
409  } else {
410  elec_comp_t *other_bus = comp->links[0].comp;
411 
412  CHECK_COMP(comp->links[1].comp == NULL,
413  "too many connections");
414  comp->links[1].comp = bus;
415  CHECK_COMP_V(bus->info->bus.ac ==
416  other_bus->info->bus.ac, "cannot link "
417  "two buses of incompatible type (%s is "
418  "%s and %s is %s)",
419  bus->info->name,
420  bus->info->bus.ac ? "AC" : "DC",
421  other_bus->info->name,
422  other_bus->info->bus.ac ? "AC" : "DC");
423  }
424  break;
425  case ELEC_TIE:
426  comp->n_links++;
427  comp->links = safe_realloc(comp->links,
428  comp->n_links * sizeof (*comp->links));
429  comp->links[comp->n_links - 1].comp = bus;
430  free(comp->tie.cur_state);
431  comp->tie.cur_state = safe_calloc(comp->n_links,
432  sizeof (*comp->tie.cur_state));
433  free(comp->tie.wk_state);
434  comp->tie.wk_state = safe_calloc(comp->n_links,
435  sizeof (*comp->tie.wk_state));
436  break;
437  case ELEC_DIODE:
438  /*
439  * Diodes are DC-only devices. Don't attempt to build
440  * rectifiers, use a TRU for that!
441  */
442  CHECK_COMP_V(!bus->info->bus.ac, "cannot connect "
443  "diode %s to an AC bus (libelec cannot be used "
444  "to build a bridge rectifier, use a \"TRU\" "
445  "component for that)", comp->info->name);
446  ASSERT3U(comp->n_links, ==, 2);
447  ASSERT(comp->links != NULL);
448  if (comp->info->diode.sides[0] == bus->info) {
449  comp->links[0].comp = bus;
450  } else {
451  ASSERT3P(comp->info->diode.sides[1], ==,
452  bus->info);
453  comp->links[1].comp = bus;
454  }
455  break;
456  default:
457  VERIFY_FAIL();
458  }
459  }
460 #undef CHECK_COMP
461 #undef CHECK_COMP_V
462  return (true);
463 }

571 elec_sys_t *
572 libelec_new(const char *filename)
573 {
574  elec_sys_t *sys = safe_calloc(1, sizeof (*sys));
575  unsigned src_i = 0, comp_i = 0;
576  void *buf;
577  size_t bufsz;
578 
579  ASSERT(filename != NULL);
580 
581  buf = file2buf(filename, &bufsz);
582  if (buf == NULL) {
583  logMsg("Can't open %s: %s", filename, strerror(errno));
584  ZERO_FREE(sys);
585  return (NULL);
586  }
587  sys->conf_filename = safe_strdup(filename);
588  sys->conf_crc = crc64(buf, bufsz);
589  free(buf);
590 
591  sys->comp_infos = infos_parse(filename, &sys->num_infos);
592  if (sys->comp_infos == NULL) {
593  ZERO_FREE(sys);
594  return (NULL);
595  }
596  list_create(&sys->comps, sizeof (elec_comp_t),
597  offsetof(elec_comp_t, comps_node));
598  list_create(&sys->gens_batts, sizeof (elec_comp_t),
599  offsetof(elec_comp_t, gens_batts_node));
600  list_create(&sys->ties, sizeof (elec_comp_t),
601  offsetof(elec_comp_t, ties_node));
602  avl_create(&sys->info2comp, info2comp_compar, sizeof (elec_comp_t),
603  offsetof(elec_comp_t, info2comp_node));
604  avl_create(&sys->name2comp, name2comp_compar, sizeof (elec_comp_t),
605  offsetof(elec_comp_t, name2comp_node));
606 
607  mutex_init(&sys->user_cbs_lock);
608  avl_create(&sys->user_cbs, user_cb_info_compar,
609  sizeof (user_cb_info_t), offsetof(user_cb_info_t, node));
610  mutex_init(&sys->worker_interlock);
611  mutex_init(&sys->paused_lock);
612  sys->time_factor = 1;
613 
614  for (size_t i = 0; i < sys->num_infos; i++) {
615  elec_comp_t *comp = safe_calloc(1, sizeof (*comp));
616  avl_index_t where;
617  elec_comp_info_t *info = &sys->comp_infos[i];
618 
619  comp->sys = sys;
620  comp->info = info;
621  mutex_init(&comp->rw_ro_lock);
622  VERIFY_MSG(avl_find(&sys->info2comp, comp, &where) == NULL,
623  "Duplicate elec info usage %s", info->name);
624  avl_insert(&sys->info2comp, comp, where);
625  list_insert_tail(&sys->comps, comp);
626 
627  VERIFY_MSG(avl_find(&sys->name2comp, comp, &where) ==
628  NULL, "Duplicate info name %s", info->name);
629  avl_insert(&sys->name2comp, comp, where);
630 
631  /* Initialize component */
632  if (comp->info->type == ELEC_LOAD) {
633  comp->load.random_load_factor = 1;
634  } else if (comp->info->type == ELEC_BATT ||
635  comp->info->type == ELEC_GEN ||
636  comp->info->type == ELEC_TRU ||
637  comp->info->type == ELEC_INV) {
638  comp->src_idx = src_i;
639  src_i++;
640  }
641  if (comp->info->type == ELEC_BATT ||
642  comp->info->type == ELEC_GEN) {
643  list_insert_tail(&sys->gens_batts, comp);
644  } else if (comp->info->type == ELEC_TIE) {
645  list_insert_tail(&sys->ties, comp);
646  }
647  /* Initialize links */
648  switch (comp->info->type) {
649  case ELEC_BATT:
650  case ELEC_GEN:
651  case ELEC_LOAD:
652  comp->n_links = 1;
653  break;
654  case ELEC_TRU:
655  case ELEC_INV:
656  case ELEC_CB:
657  case ELEC_SHUNT:
658  case ELEC_DIODE:
659  comp->n_links = 2;
660  break;
661  case ELEC_BUS:
662  case ELEC_LABEL_BOX:
663  case ELEC_TIE:

751 /*
752  * Validates the elec_comp_info_t's we've parsed out of the file.
753  * This needs to run after the parsing phase, but before the network
754  * is fully ready to operate. The user might want to set up info
755  * struct callbacks before this is done, so we shouldn't error out
756  * on missing mandatory callbacks. That check is done in
757  * validate_elec_comp_infos_start() just before the network is started.
758  */
759 static bool
760 validate_elec_comp_infos_parse(const elec_comp_info_t *infos, size_t n_infos,
761  const char *filename)
762 {
763  ASSERT(infos != NULL || n_infos == 0);
764  ASSERT(filename != NULL);
765 
766 #define CHECK_COMP(cond, reason) \
767  do { \
768  if (!(cond)) { \
769  logMsg("%s (%s:%d): %s", info->name, filename, \
770  info->parse_linenum, (reason)); \
771  return (false); \
772  } \
773  } while (0)
774 
775  for (size_t i = 0; i < n_infos; i++) {
776  const elec_comp_info_t *info = &infos[i];
777 
778  switch (info->type) {
779  case ELEC_BATT:
780  CHECK_COMP(info->batt.volts > 0,
781  "missing required \"VOLTS\" parameter");
782  CHECK_COMP(info->batt.max_pwr > 0,
783  "missing required \"MAX_PWR\" parameter");
784  CHECK_COMP(info->batt.capacity >= 0,
785  "missing required \"CAPACITY\" parameter");
786  CHECK_COMP(info->batt.chg_R > 0,
787  "missing required \"CHG_R\" parameter");
788  break;
789  case ELEC_GEN:
790  CHECK_COMP(info->gen.volts > 0,
791  "missing required \"VOLTS\" parameter");
792  CHECK_COMP(info->gen.exc_rpm <= info->gen.min_rpm,
793  "\"EXC_RPM\" parameter must be less than or "
794  "equal to \"MIN_RPM\"");
795  CHECK_COMP(info->gen.min_rpm > 0,
796  "missing required \"MIN_RPM\" parameter");
797  CHECK_COMP(info->gen.max_rpm > 0,
798  "missing required \"MAX_RPM\" parameter");
799  CHECK_COMP(info->gen.min_rpm < info->gen.max_rpm,
800  "\"MIN_RPM\" must be lower than \"MAX_RPM\"");
801  CHECK_COMP(info->gen.eff_curve != NULL, "generators "
802  "require at least two \"CURVEPT EFF\" parameters");
803  break;
804  case ELEC_TRU:
805  case ELEC_INV:
806  CHECK_COMP(info->tru.in_volts > 0,
807  "missing required \"IN_VOLTS\" parameter");
808  CHECK_COMP(info->tru.out_volts > 0,
809  "missing required \"OUT_VOLTS\" parameter");
810  if (info->type == ELEC_INV) {
811  CHECK_COMP(info->tru.out_freq > 0,
812  "missing required \"OUT_FREQ\" parameter");
813  }
814  CHECK_COMP(info->tru.eff_curve != NULL,
815  "at least two \"CURVEPT EFF\" parameters required");
816  CHECK_COMP(info->tru.ac != NULL,
817  "AC side not connected");
818  CHECK_COMP(info->tru.dc != NULL,
819  "DC side not connected");
820  break;
821  case ELEC_LOAD:
822  /*
823  * Stabilized loads MUST declare a minimum voltage.
824  * Non-stabilized loads don't need to.
825  */
826  CHECK_COMP(info->load.min_volts > 0 ||
827  !info->load.stab, "loads must specify "
828  "a \"MIN_VOLTS\" when \"STAB\" is set to TRUE");
829  ASSERT3F(info->load.incap_C, >=, 0);
830  if (info->load.incap_C > 0)
831  ASSERT3F(info->load.incap_R, >, 0);
832  break;
833  case ELEC_BUS:
834  CHECK_COMP(info->bus.comps != NULL,
835  "buses must connect to at least 1 component");
836  break;
837  case ELEC_CB:
838  case ELEC_SHUNT:
839  case ELEC_TIE:
840  break;
841  case ELEC_DIODE:
842  CHECK_COMP(info->diode.sides[0] != NULL &&
843  info->diode.sides[1] != NULL,
844  "diodes need to have both end points connected");
845  break;
846  case ELEC_LABEL_BOX:
847  break;
848  }
849  }
850 
851 #undef CHECK_COMP
852 
853  return (true);
854 }

2918 static void
2919 network_update_gen(elec_comp_t *gen, double d_t)
2920 {
2921  double rpm;
2922 
2923  ASSERT(gen != NULL);
2924  ASSERT(gen->info != NULL);
2925  ASSERT3U(gen->info->type, ==, ELEC_GEN);
2926  ASSERT(gen->info->gen.get_rpm != NULL);
2927 
2928  rpm = gen->info->gen.get_rpm(gen, gen->info->userinfo);
2929  ASSERT(!isnan(rpm));
2930  rpm = MAX(rpm, 1e-3);
2931  gen->gen.rpm = rpm;
2932  if (rpm <= 1e-3) {
2933  gen->gen.stab_factor_U = 1;
2934  gen->gen.stab_factor_f = 1;
2935  gen->rw.in_volts = 0;
2936  gen->rw.in_freq = 0;
2937  gen->rw.out_volts = 0;
2938  gen->rw.out_freq = 0;
2939  return;
2940  }
2941  /*
2942  * Gradual voltage & frequency stabilization adjustment, to simulate
2943  * that the CSD takes a little time to adjust to rpm changes.
2944  */
2945  if (gen->info->gen.stab_rate_U > 0) {
2946  double stab_factor_U = clamp(gen->gen.ctr_rpm / rpm,
2947  gen->gen.min_stab_U, gen->gen.max_stab_U);
2948  double stab_rate_mod =
2949  clamp(1 + crc64_rand_normal(0.1), 0.1, 10);
2950  FILTER_IN(gen->gen.stab_factor_U, stab_factor_U, d_t,

3558 static double
3559 network_load_integrate_load(const elec_comp_t *src, elec_comp_t *comp,
3560  unsigned depth, double d_t)
3561 {
3562  double load_WorI, load_I, in_volts_net, incap_I, src_fract;
3563  const elec_comp_info_t *info;
3564 
3565  /* src can be NULL */
3566  ASSERT(comp != NULL);
3567  ASSERT(comp->info != NULL);
3568  ASSERT3U(comp->info->type, ==, ELEC_LOAD);
3569  UNUSED(depth);
3570 
3571  info = comp->info;
3572  /*
3573  * If the input voltage is lower than our input capacitance voltage,
3574  * it will be our input capacitance powering the load, not the input.
3575  */
3576  in_volts_net = MAX(comp->rw.in_volts, comp->load.incap_U);
3577  /*
3578  * Only ask the load if we are receiving sufficient volts.
3579  */
3580  if (in_volts_net >= comp->info->load.min_volts) {
3581  load_WorI = info->load.std_load;
3582  if (info->load.get_load != NULL) {
3583  load_WorI += info->load.get_load(comp,
3584  comp->info->userinfo);
3585  }
3586  } else {
3587  load_WorI = 0;
3588  }
3589  ASSERT3F(load_WorI, >=, 0);
3590 
3591  load_WorI *= comp->load.random_load_factor;
3592  /*
3593  * If the load use a stabilized power supply, the load value is
3594  * in Watts. Calculate the effective current.
3595  */
3596  if (info->load.stab) {
3597  double volts = MAX(in_volts_net, info->load.min_volts);
3598  ASSERT3F(volts, >, 0);
3599  load_I = load_WorI / volts;
3600  } else {
3601  load_I = load_WorI;
3602  }
3603  if (comp->rw.shorted) {
3604  /*
3605  * Shorted components boost their current draw.
3606  */
3607  ASSERT3F(comp->rw.leak_factor, <, 1);
3608  load_I /= (1 - comp->rw.leak_factor);
3609  } else if (comp->rw.failed) {
3610  /*
3611  * Failed components just drop their power consumption to zero
3612  */
3613  load_I = 0;
3614  }
3615  /*
3616  * When the input voltage is greater than the input capacitance
3617  * voltage, we will be charging up the input capacitance.
3618  */
3619  if (info->load.incap_C > 0 &&
3620  comp->rw.in_volts > comp->load.incap_U + 0.01) {
3621  /*
3622  * Capacitor voltage U_c is:
3623  *
3624  * U_c = U_in * (1 - e^(-t / RC))
3625  *
3626  * Where:
3627  * U_in - the input charging voltage
3628  * t - amount of time the cap has been charging in seconds
3629  * R - a series resistance to limit charge current in Ohms
3630  * C - the capacitance in Farad
3631  *
3632  * Since we operate in a fixed time simulation steps, we
3633  * want to instead compute the amount of change in charge
3634  * in one simulation step. So we start with the capacitor
3635  * voltage from the previous simulation loop, U_c_old, and
3636  * subtract it from the input voltage. We can then compute
3637  * the new capacitor voltage using a stepped algorithm:
3638  *
3639  * U_c_new = U_c_old + ((U_in - U_c_old) * (1 - e^(-t / RC)))
3640  */
3641  double U_in = comp->rw.in_volts;
3642  double U_c_old = comp->load.incap_U;
3643  double R = info->load.incap_R;
3644  double C = info->load.incap_C;
3645  double incap_U_new = U_c_old +
3646  ((U_in - U_c_old) * (1 - exp(-d_t / (R * C))));
3647  /*
3648  * Next we convert the previous and new capacitor voltages
3649  * to a total charge value and figure out the net change
3650  * in charge.
3651  *
3652  * Q_old = U_c_old * C
3653  * Q_new = U_c_new * C
3654  * Q_delta = Q_new - Q_old
3655  *
3656  * Where Q_old and Q_new are the old and new capacitor
3657  * charge states (in Coulomb). To calculate the charge
3658  * current, we then simply divide the change in charge
3659  * Q_delta by the time quantum:
3660  *
3661  * I = Q_delta / t
3662  */
3663  double Q_old = comp->load.incap_U * info->load.incap_C;
3664  double Q_new = incap_U_new * info->load.incap_C;
3665  double Q_delta = Q_new - Q_old;
3666 
3667  incap_I = Q_delta / d_t;
3668  } else {
3669  incap_I = 0;
3670  }
3671  /*
3672  * If the input capacitance has a greater voltage than the network
3673  * input voltage, then we need to calculate how much of the charge
3674  * will be provided by the input capacitance. The amount of charge
3675  * Q_load that the load will draw in one time step is:
3676  *
3677  * Q_load = I_load * t
3678  *
3679  * The amount of charge that the capacitor Q_c can provide is
3680  * dependent upon the delta between capacitor charge and network
3681  * input voltage:
3682  *
3683  * Q_c = (U_c - U_in) * C
3684  *
3685  * After draining this amount of charge, the capacitor's voltage
3686  * will be lower than the input voltage and so no more charge can
3687  * be drawn from it.
3688  */
3689  if (comp->load.incap_U > comp->rw.in_volts) {
3690  /* Amount of charge requested by the load in this time step */
3691  double load_Q = load_I * d_t;
3692  /* Amount of charge that can be drawn from the incap */
3693  double avail_Q = (comp->load.incap_U - comp->rw.in_volts) *
3694  info->load.incap_C;
3695  /* Amount of charge actually drawn from the incap */
3696  double used_Q = MIN(load_Q, avail_Q);
3697  /*
3698  * Subtract the charge provided by the incap from the
3699  * network-demanded charge.
3700  */
3701  load_Q -= used_Q;
3702  /*
3703  * Actual network current is the delta vs what the incap
3704  * can provide.
3705  */
3706  comp->rw.in_amps = load_Q / d_t;
3707  comp->rw.out_amps = load_I;
3708  if (comp->load.incap_U >= comp->info->load.min_volts)
3709  comp->rw.out_volts = comp->load.incap_U;
3710  else
3711  comp->rw.out_volts = 0;
3712  comp->load.incap_d_Q = -used_Q;
3713  } else {
3714  /*
3715  * Don't forget to add the input capacitance charging
3716  * current to the network current draw.
3717  */
3718  comp->rw.in_amps = load_I + incap_I;
3719  comp->rw.out_amps = load_I;
3720  comp->rw.out_volts = comp->rw.in_volts;
3721  comp->rw.out_freq = comp->rw.in_freq;
3722  comp->load.incap_d_Q = incap_I * d_t;
3723  }
3724  ASSERT(!isnan(comp->rw.out_amps));
3725  ASSERT(!isnan(comp->rw.out_volts));
3726  comp->load.seen = true;
3727  if (src != NULL) {
3728  src_fract = get_src_fract(comp, src);
3729  comp->links[0].out_amps[src->src_idx] =
3730  NO_NEG_ZERO(-comp->rw.in_amps * src_fract);
3731  } else {
3732  src_fract = 1;
3733  }
3734 
3735  return (comp->rw.in_amps * src_fract);
3736 }
