13 /*
14  * ## USAGE NOTES
15  *
16  * The XPWidgetUtils library contains useful functions that make writing and
17  * using widgets less of a pain.
18  *
19  * One set of functions are the widget behavior functions. These functions
20  * each add specific useful behaviors to widgets. They can be used in two
21  * manners:
22  *
23  * 1. You can add a widget behavior function to a widget as a callback proc
24  * using the XPAddWidgetCallback function. The widget will gain that
25  * behavior. Remember that the last function you add has highest priority.
26  * You can use this to change or augment the behavior of an existing
27  * finished widget.
28  * 2. You can call a widget function from inside your own widget function.
29  * This allows you to include useful behaviors in custom-built widgets. A
30  * number of the standard widgets get their behavior from this library. To
31  * do this, call the behavior function from your function first. If it
32  * returns 1, that means it handled the event and you don't need to; simply
33  * return 1.
34  *
35  */

72 /*
73  * XPWidgetCreate_t
74  *
75  * This structure contains all of the parameters needed to create a wiget. It
76  * is used with XPUCreateWidgets to create widgets in bulk from an array. All
77  * parameters correspond to those of XPCreateWidget except for the container
78  * index.
79  *
80  * If the container index is equal to the index of a widget in the array, the
81  * widget in the array passed to XPUCreateWidgets is used as the parent of
82  * this widget. Note that if you pass an index greater than your own position
83  * in the array, the parent you are requesting will not exist yet.
84  *
85  * If the container index is NO_PARENT, the parent widget is specified as
86  * NULL. If the container index is PARAM_PARENT, the widget passed into
87  * XPUCreateWidgets is used.
88  *
89  */

92  int top;
93  int right;
94  int bottom;
95  int visible;
96  const char * descriptor;
97  /* Whether ethis widget is a root wiget */
98  int isRoot;
99  /* The index of the widget to contain within, or a constant */
100  int containerIndex;
101  XPWidgetClass widgetClass;
102 } XPWidgetCreate_t;
