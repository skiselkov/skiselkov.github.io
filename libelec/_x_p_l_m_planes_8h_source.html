79 /*
80  * XPLMPlaneDrawState_t
81  *
82  * This structure contains additional plane parameter info to be passed to
83  * draw plane. Make sure to fill in the size of the structure field with
84  * sizeof(XPLMDrawPlaneState_t) so that the XPLM can tell how many fields you
85  * knew about when compiling your plugin (since more fields may be added
86  * later).
87  *
88  * Most of these fields are ratios from 0 to 1 for control input. X-Plane
89  * calculates what the actual controls look like based on the .acf file for
90  * that airplane. Note for the yoke inputs, this is what the pilot of the
91  * plane has commanded (post artificial stability system if there were one)
92  * and affects aelerons, rudder, etc. It is not necessarily related to the
93  * actual position of the plane!
94  *
95  */
96 typedef struct {
97  /* The size of the draw state struct. */
98  int structSize;
99  /* A ratio from [0..1] describing how far the landing gear is extended. */
100  float gearPosition;
101  /* Ratio of flap deployment, 0 = up, 1 = full deploy. */
102  float flapRatio;
103  /* Ratio of spoiler deployment, 0 = none, 1 = full deploy. */
104  float spoilerRatio;
105  /* Ratio of speed brake deployment, 0 = none, 1 = full deploy. */
106  float speedBrakeRatio;
107  /* Ratio of slat deployment, 0 = none, 1 = full deploy. */
108  float slatRatio;
109  /* Wing sweep ratio, 0 = forward, 1 = swept. */
110  float wingSweep;
111  /* Thrust power, 0 = none, 1 = full fwd, -1 = full reverse. */
112  float thrust;
113  /* Total pitch input for this plane. */
114  float yokePitch;
115  /* Total Heading input for this plane. */
116  float yokeHeading;
117  /* Total Roll input for this plane. */
118  float yokeRoll;
119 } XPLMPlaneDrawState_t;
120 #endif /* XPLM_DEPRECATED */
